4 const std::vector<size_t> &quadVertices) const {
5
6 std::vector<float> sum(vertexDistance_[*quadVertices.begin()].size(),
7 std::numeric_limits<float>::infinity());
8
9#ifdef TTK_ENABLE_OPENMP
10#pragma omp parallel for num_threads(threadNumber_)
11#endif // TTK_ENABLE_OPENMP
12 for(size_t i = 0; i < sum.size(); ++i) {
13
14 // skip following computation if too far from any parent quad vertex
15 bool skip = false;
16
17 for(const auto vert : quadVertices) {
18 if(vertexDistance_[vert][i] == std::numeric_limits<float>::infinity()) {
19 skip = true;
20 break;
21 }
22 }
23
24 if(skip) {
25 continue;
26 }
27
28 float m = vertexDistance_[quadVertices[0]][i];
29 float n = vertexDistance_[quadVertices[1]][i];
30 float o = vertexDistance_[quadVertices[2]][i];
31 float p = vertexDistance_[quadVertices[3]][i];
32
33 // try to be "near" the four parent vertices
34 sum[i] = m + n + o + p;
35
36 // try to be on the diagonals intersection
37 sum[i] += std::abs(m - o);
38 sum[i] += std::abs(n - p);
39 }
40
41 return std::min_element(sum.begin(), sum.end()) - sum.begin();

45 ttk::QuadrangulationSubdivision::getQuadNormal(const size_t a) const {
46 Point quadNormal{};
47
48 // current vertex 3d coordinates
49 Point pa = outputPoints_[a];
50
51 // find all quads that have a as vertex
52 std::vector<Quad> quads{};
53 for(auto &q : outputQuads_) {
54 auto _a = static_cast<LongSimplexId>(a);
55 if(q[0] == _a || q[1] == _a || q[2] == _a || q[3] == _a) {
56 quads.emplace_back(q);
57 }
58 }
59
60 // store for each quad around a its two direct neighbors
61 std::set<std::array<size_t, 2>> couples{};
62
63 // find couple of neighbors of a sharing a quad
64 for(auto &q : quads) {
65 auto _a = static_cast<LongSimplexId>(a);
66 std::array<size_t, 2> tmp{};
67 if(q[0] == _a) {
68 tmp[0] = static_cast<size_t>(q[3]);
69 tmp[1] = static_cast<size_t>(q[1]);
70 } else if(q[1] == _a) {
71 tmp[0] = static_cast<size_t>(q[0]);
72 tmp[1] = static_cast<size_t>(q[2]);
73 } else if(q[2] == _a) {
74 tmp[0] = static_cast<size_t>(q[1]);
75 tmp[1] = static_cast<size_t>(q[3]);
76 } else if(q[3] == _a) {
77 tmp[0] = static_cast<size_t>(q[2]);
78 tmp[1] = static_cast<size_t>(q[0]);
79 }
80 couples.insert(tmp);
81 }
82
83 // triangle normals around current quadrangulation vertex
84 std::vector<Point> normals{};
85
86 for(auto &t : couples) {
87 Point pb = outputPoints_[t[0]];
88 Point pc = outputPoints_[t[1]];
89
90 // triangle normal: cross product of two edges
91 Point crossP{};
92 // ab, ac vectors
93 Point ab = pb - pa;
94 Point ac = pc - pa;
95 // compute ab ^ ac
96 Geometry::crossProduct(&ab.x, &ac.x, &crossP.x);
97 // magnitude
98 auto mag = Geometry::magnitude(&crossP.x);
99 // ensure normal not null
100 if(mag > powf(10, -FLT_DIG)) {
101 // unitary normal vector
102 normals.emplace_back(crossP / mag);
103 }
104 }
105
106 // ensure normals have same direction
107 for(size_t i = 1; i < normals.size(); ++i) {
108 auto dotProd = Geometry::dotProduct(&normals[0].x, &normals[i].x);
109 if(dotProd < 0.0F) {
110 normals[i] = normals[i] * -1.0F;
111 }
112 }
113
114 if(!normals.empty()) {
115 // compute mean of normals
116 quadNormal = std::accumulate(
117 normals.begin(), normals.end(), Point{}, std::plus<Point>());
118 quadNormal = quadNormal / static_cast<float>(normals.size());
119 } else {
120 // set error value directly in output variable...
121 quadNormal.x = NAN;
122 }
123
124 return quadNormal;
125}

171int ttk::QuadrangulationSubdivision::relax(const std::set<size_t> &filtered) {
172 Timer tm;
173
174 // temp storage for relaxed points
175 std::vector<Point> tmp(outputPoints_.size());
176
177 // loop over output points, do not touch input MSC critical points
178#ifdef TTK_ENABLE_OPENMP
179#pragma omp parallel for num_threads(threadNumber_)
180#endif // TTK_ENABLE_OPENMP
181 for(size_t i = 0; i < outputPoints_.size(); i++) {
182
183 // skip computation if i in filtered
184 if(filtered.find(i) != filtered.end()) {
185 tmp[i] = outputPoints_[i];
186 continue;
187 }
188
189 // barycenter of curr neighbors
190 Point relax{};
191 for(auto &neigh : quadNeighbors_[i]) {
192 relax = relax + outputPoints_[neigh];
193 }
194 relax = relax * (1.0F / static_cast<float>(quadNeighbors_[i].size()));
195
196 tmp[i] = relax;
197 }

199 outputPoints_ = std::move(tmp);
