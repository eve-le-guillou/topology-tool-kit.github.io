242 // subtree. When a node have only one way out : enqueue it to continue

248 std::tie(curNodeId, needToGoUp) = node2see.front();

272 markThisArc(subtreeUF, curNodeId, mergingArcId, parentNodeId);
273
274 // if we have processed all but one arc of this node, we nee to continue
275 // traversall
276 // throug it
277 if(valenceOffset[parentNodeId].first
278 + valenceOffset[parentNodeId].second + 1
279 == getNode(parentNodeId)->getValence()) {
280 // only one way out, is it up ?
281 node2see.emplace(
282 parentNodeId, valenceOffset[parentNodeId].second + 1 ==
283
284 getNode(parentNodeId)->getUpValence());
285 if(DEBUG) {
286 std::cout << " add to see " << printNode(parentNodeId) << std::endl;

291 // for each node valenceOffset is the number of arc attached to this node
292 // that will merge

294 // Debug print
295 if(DEBUG) {
296 std::cout << "node subtrees before creating receptarc " << std::endl;

309 //}

312 //-----------
313 //{
314
315 // Add the origin of all pairs that need to merge
316 for(const auto &pp : sortedPairs) {
317 if(std::get<2>(pp) < params_->simplifyThreshold) {

324 || scalars_->sosOffsets[thisOriginVert] >= posSeed1
325 || scalars_->sosOffsets[thisEndVert] <= posSeed0
326 || scalars_->sosOffsets[thisEndVert] >= posSeed1) {
327 continue;
328 }
329
330 if(subtreeUF[thisOriginId]->find()->getData() < 0) {
331 // create receptarc
332 const idNode &subtreeRoot
333 = -subtreeUF[thisOriginId]->find()->getData() - 1;
334 // The id of the next arc to be created : NOT PARALLEL
335 const idSuperArc receptArcId = treeData_.superArcs.size();
336 // down , up, segmentation size
337 // create the receptacle arc and merge arc not in sub-tree in it
338 const std::tuple<idNode, idNode, SimplexId> &receptArc
339 = createReceptArc(
340 subtreeRoot, receptArcId, subtreeUF, valenceOffset);
341
342 // make superArc and do the makeAlloc on it
343 const bool overlapB =
344
345 scalars_
346 ->sosOffsets[getNode(std::get<0>(receptArc))->getVertexId()]
347 < posSeed0;
348 const bool overlapA =
349
350 scalars_
351 ->sosOffsets[getNode(std::get<1>(receptArc))->getVertexId()]
352 >= posSeed1;
353 const idSuperArc na
354 = makeSuperArc(std::get<0>(receptArc), std::get<1>(receptArc),

357 if(overlapB) {
358 treeData_.arcsCrossingBelow.emplace_back(na);
359 }
360
361 if(overlapA) {
362 treeData_.arcsCrossingAbove.emplace_back(na);
363 }
364
365 subtreeUF[thisOriginId]->find()->setData(receptArcId);
366 getSuperArc(receptArcId)->makeAllocGlobal(std::get<2>(receptArc));
367
368 if(DEBUG) {
369 std::cout << "create arc : " << printArc(receptArcId)
370 << " with segm : " << std::get<2>(receptArc)
371 << std::endl;
372 }
373 }
374 } else
375 break;
376 }
377
378 //}
379 //----------
380 // Merge these subtrees
381 //-----------
382 //{
383
384 // The merge is done after because we don't want to forget arcs parallel
385 // to the recept'arc but we cannot make the difference before the former
386 // are created
387
388 // nbArcs is before the insertion of receptarcs so they will no be crossed
389 // here
390 for(idSuperArc arc = 0; arc < nbArcs; arc++) {
391
392 if(getSuperArc(arc)->isMerged()) {
393 const idSuperArc &receptacleArcId

395 // take care of connectivity

398 std::cout << " parallel merge in " << printArc(receptacleArcId)

403 }
404 getSuperArc(receptacleArcId)
405 ->addSegmentationGlobal(
406 getSuperArc(arc)->getVertList(), getSuperArc(arc)->getVertSize());
407 } else {
408 const idNode &downNode = getSuperArc(arc)->getDownNodeId();
409 const idNode &upNode = getSuperArc(arc)->getUpNodeId();
410
411 if(!(subtreeUF[downNode] && subtreeUF[upNode]))
412 continue;
413
414 if(subtreeUF[downNode] && subtreeUF[upNode]
415 && subtreeUF[downNode]->find() != subtreeUF[upNode]->find()) {
416 if(DEBUG) {
417 std::cout << "Arc between 2 degenerate with mergin "
418 << printArc(arc) << std::endl;
419 std::cout << "below recept : "
420 << printArc(subtreeUF[downNode]->find()->getData());
421 std::cout << std::endl;
422 std::cout << "Above recept : "
423 << printArc(subtreeUF[upNode]->find()->getData())
424 << std::endl;
425 std::cout << std::endl;
426 }
427
428 continue;
429 }
430
431 ExtendedUnionFind *curUF = (subtreeUF[upNode])
432 ? subtreeUF[upNode]->find()
433 : subtreeUF[downNode]->find();
434
435 const idSuperArc &receptacleArcId = curUF->getData();
436
437 if(DEBUG) {
438 std::cout << "merge in " << printArc(receptacleArcId) << std::endl;
439 std::cout << " arc " << printArc(arc)
440 << " size : " << getSuperArc(arc)->getVertSize();
441 std::cout << std::endl;
442 }
443
444 getSuperArc(arc)->merge(receptacleArcId);
445 if(getSuperArc(arc)->getVertSize()) {
446 getSuperArc(receptacleArcId)
447 ->addSegmentationGlobal(getSuperArc(arc)->getVertList(),
448 getSuperArc(arc)->getVertSize());
449
450 getSuperArc(receptacleArcId)
451 ->addSegmentationGlobal(getNode(downNode)->getVertexId());
452
453 getSuperArc(receptacleArcId)
454 ->addSegmentationGlobal(getNode(upNode)->getVertexId());
455 }
456
457 // Tree topology
458 getNode(downNode)->removeUpSuperArc(arc);
459 getNode(downNode)->decUpValence();
460 getNode(upNode)->removeDownSuperArc(arc);
461 getNode(upNode)->decDownValence();
462
463 if(!getNode(downNode)->getNumberOfUpSuperArcs())
464 hideNode(downNode);
465 if(!getNode(upNode)->getNumberOfDownSuperArcs())
466 hideNode(upNode);
467 }
468 } // end for arcs
469
470 //}
471
472 return nbPairMerged;
473 }

547 template <typename scalarType, typename triangulationType>
548 void MergeTree::recoverMTPairs(

551 std::vector<std::tuple<SimplexId, SimplexId, scalarType, bool>> &pairsST,

643 // std::cout << "pending for " << further << " is " << v <<

661#ifdef TTK_ENABLE_OPENMP
662#pragma omp section
663#endif
664 {
665 pairsST.reserve(treeData_.leaves.size());

667 std::map<SimplexId, SimplexId> pendingMinMax;
668
669 for(auto it = sortedNodes.crbegin(); it != sortedNodes.crend();
670 ++it) {
671 const idNode &n = *it;
672 const SimplexId &v = getNode(n)->getVertexId();
673
674 const auto &nbUp = getNode(n)->getNumberOfUpSuperArcs();
675 const auto &nbDown = getNode(n)->getNumberOfDownSuperArcs();

677 if(nbUp == 0) {
678 // leaf
679 vect_SplitUF[n]
680 = new ExtendedUnionFind(getNode(n)->getVertexId());

682 // std::cout << " st origin : " << v << std::endl;

684 // first descendant

686 const SuperArc *firstSA = getSuperArc(firstSaId);
687 const idNode &firstChildNodeId = firstSA->getUpNodeId();
688
689 ExtendedUnionFind *merge = vect_SplitUF[firstChildNodeId]->find();
690 SimplexId further = merge->getOrigin();
691 idSuperArc furtherI = 0;

693 for(idSuperArc ni = 1; ni < nbUp; ++ni) {
694 // find the more persistant way
695 const idSuperArc curSaId = getNode(n)->getUpSuperArcId(ni);

697 // Ignore hidden / simplified arc
698 if(!curSA->isVisible())
699 continue;

702 // fix
703 if(neigh == n)

914 const SimplexId currentVertex = overlapAfter[sortedNode];

916 const bool overlapA = isJT;

918 currentVertex, vect_baseUF, overlapB, overlapA, mesh, timerBegin);
919 } // foreach node

921 // }
922 // ---------------
923 // Close root arcs
924 // ---------------
925 // {

927 // Closing step for openedSuperArc.
928 // Not aesthetic but efficient
929 // More efficient that using nullity of the neighborhood
930 // to detect extrema of the opponent tree
931 idNode rootNode;
932 SimplexId corrVertex, origin;
933 idSuperArc tmp_sa;

991 template <typename triangulationType>
992 void MergeTree::processVertex(const SimplexId &currentVertex,
993 std::vector<ExtendedUnionFind *> &vect_baseUF,
994 const bool overlapB,
995 const bool overlapA,
996 const triangulationType &mesh,
997 DebugTimer &begin) {
998 std::vector<ExtendedUnionFind *> vect_neighUF;
999 ExtendedUnionFind *seed = nullptr, *tmpseed;
1000
1001 SimplexId neighSize;
1002 const SimplexId neighborNumber
1003 = mesh->getVertexNeighborNumber(currentVertex);
1004 const bool isJT = treeData_.treeType == TreeType::Join;
1005
1006 idSuperArc currentArc;
1007 idNode closingNode, currentNode;
1008 SimplexId neighbor{-1};
1009
1010 // Check UF in neighborhood
1011 for(SimplexId n = 0; n < neighborNumber; ++n) {
1012 mesh->getVertexNeighbor(currentVertex, n, neighbor);
1013 // if the vertex is out: consider it null
1014 tmpseed = vect_baseUF[neighbor];
1015 // unvisited vertex, we continue.
1016 if(tmpseed == nullptr) {
1017 continue;
1018 }
1019
1020 tmpseed = tmpseed->find();
1021
1022 // get all different UF in neighborhood
1023 if(find(vect_neighUF.cbegin(), vect_neighUF.cend(), tmpseed)
1024 == vect_neighUF.end()) {
1025 vect_neighUF.emplace_back(tmpseed);
1026 seed = tmpseed;
1027 }
1028 }
1029
1030 (neighSize = vect_neighUF.size());
1031
1032 // SimplexId test = 1;
1033 // if (currentVertex == test)
1034 // cout << test << " : " << vect_neighUF.size() << " " <<
1035 // vect_interfaceUF.size() << endl; Make output
1036 if(!neighSize) {
1037 // we are on a real extrema we have to create a new UNION FIND and a
1038 // branch a real extrema can't be a virtual extrema
1039
1040 seed = new ExtendedUnionFind(currentVertex);
1041 // When creating an extrema we create a pair ending on this node.
1042 currentNode = makeNode(currentVertex);
1043 getNode(currentNode)->setOrigin(currentNode);
1044 currentArc = openSuperArc(currentNode, overlapB, overlapA);
1045 // if(overlap && partition_ == 1) cout << currentVertex << endl;
1046 treeData_.leaves.emplace_back(currentNode);
1047
1048 if(params_->debugLevel >= static_cast<int>(debug::Priority::DETAIL)) {
1049 if(isJT) {
1050 this->printMsg("Min node id: " + std::to_string(currentVertex), 1.0,
1051 begin.getElapsedTime(), this->threadNumber_);
1052 } else {
1053 this->printMsg("Max node id: " + std::to_string(currentVertex), 1.0,
1054 begin.getElapsedTime(), this->threadNumber_);
1055 }
1056 }
1057 } else if(neighSize > 1) {
1058 // Is a saddle if have more than one UF in neighborhood
1059 // Or is linked to an interface (virtual extrema -> leaves or
1060 // interface cc representant : regular node )
1061
1062 // Merge operation => close all arriving SuperArc (ignoring UF from
1063 // interface) And open a new one
1064 closingNode = makeNode(currentVertex);
1065 currentArc = openSuperArc(closingNode, overlapB, overlapA);
1066
1067 SimplexId farOrigin = vect_neighUF[0]->find()->getOrigin();
1068
1069 // close each SuperArc finishing here
1070 for(auto *neigh : vect_neighUF) {
1071 closeSuperArc((idSuperArc)neigh->find()->getData(), closingNode,
1072 overlapB, overlapA);
1073 // persistance pair closing here.
1074 // For the one who will continue, it will be overide later
1075 vertex2Node(neigh->find()->getOrigin())->setTerminaison(closingNode);
1076
1077 // cout <<
1078 // getNode(getCorrespondingNode(neigh->find()->getOrigin()))->getVertexId()
1079 //<< " terminate on " << getNode(closingNode)->getVertexId() << endl;
1080
1081 if((isJT && isLower(neigh->find()->getOrigin(), farOrigin))
1082 || (!isJT && isHigher(neigh->find()->getOrigin(), farOrigin))) {
1083 // here we keep the continuing the most persitant pair.
1084 // It means a pair end when a parent have another origin thant the
1085 // current leaf (or is the root) It might be not intuitive but it is
1086 // more convenient for degenerate cases
1087 farOrigin = neigh->find()->getOrigin();
1088 // cout << "find origin " << farOrigin << " for " << currentVertex
1089 // << " " << isJT
1090 //<< endl;
1091 }
1092 }
1093
1094 // Union correspond to the merge
1095 seed = ExtendedUnionFind::makeUnion(vect_neighUF);
1096 if(seed == nullptr) {
1097 return;
1098 }
1099 seed->setOrigin(farOrigin);
1100 getNode(closingNode)->setOrigin(getCorrespondingNodeId(farOrigin));
1101
1102 // cout << " " << getNode(closingNode)->getVertexId() << " have origin
1103 // at "
1104 //<< getNode(getCorrespondingNode(farOrigin))->getVertexId() << endl;
1105
1106 this->printMsg("Saddle node id: " + std::to_string(currentVertex),
1107 debug::Priority::DETAIL);
1108
1109 } else {
1110#ifndef TTK_ENABLE_KAMIKAZE
1111 if(seed == nullptr) {
1112 return;
1113 }
1114#endif // TTK_ENABLE_KAMIKAZE
1115 // regular node
1116 currentArc = (idSuperArc)seed->find()->getData();
1117 updateCorrespondingArc(currentVertex, currentArc);
1118 }
1119 // common
1120 seed->setData((ufDataType)currentArc);
1121 getSuperArc(currentArc)->setLastVisited(currentVertex);
1122 vect_baseUF[currentVertex] = seed;
1123 }
