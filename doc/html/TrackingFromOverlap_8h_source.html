75 CoordinateComparator(const float *coords) : coordinates(coords) {

78 inline bool operator()(const size_t &i, const size_t &j) {
79 size_t ic = i * 3;
80 size_t jc = j * 3;
81 return coordinates[ic] == coordinates[jc]
82 ? coordinates[ic + 1] == coordinates[jc + 1]
83 ? coordinates[ic + 2] < coordinates[jc + 2]
84 : coordinates[ic + 1] < coordinates[jc + 1]
85 : coordinates[ic] < coordinates[jc];
86 }

109 int computeBranches(std::vector<Edges> &timeEdgesMap,
110 std::vector<Nodes> &timeNodesMap) const {
111 printMsg("Computing branches ... ", debug::Priority::PERFORMANCE);
112 Timer tm;
113
114 size_t nT = timeNodesMap.size();
115
116 // Compute max pred and succ
117 for(size_t t = 1; t < nT; t++) {
118 auto &nodes0 = timeNodesMap[t - 1];
119 auto &nodes1 = timeNodesMap[t];
120 auto &edges = timeEdgesMap[t - 1];
121
122 size_t nE = edges.size();
123
124 for(size_t i = 0; i < nE; i += 4) {
125 auto n0Index = edges[i];
126 auto n1Index = edges[i + 1];
127 auto &n0 = nodes0[n0Index];
128 auto &n1 = nodes1[n1Index];
129
130 sizeType n0MaxSuccSize
131 = n0.maxSuccID != -1 ? nodes1[n0.maxSuccID].size : 0;
132 sizeType n1MaxPredSize
133 = n1.maxPredID != -1 ? nodes0[n1.maxPredID].size : 0;
134 if(n0MaxSuccSize < n1.size)
135 n0.maxSuccID = n1Index;
136 if(n1MaxPredSize < n0.size)
137 n1.maxPredID = n0Index;
138 }
139 }
140
141 // Label first nodes of branches
142 idType branchCounter = 0;
143
144 for(size_t t = 0; t < nT; t++)
145 for(auto &n : timeNodesMap[t])
146 n.branchID = n.maxPredID == -1 ? branchCounter++ : -1;
147
148 for(size_t t = 1; t < nT; t++) {
149 auto &nodes0 = timeNodesMap[t - 1];
150 auto &nodes1 = timeNodesMap[t];
151
152 for(size_t i = 0; i < nodes1.size(); i++) {
153 auto &n1 = nodes1[i];
154 if(n1.maxPredID != -1
155 && ((idType)i) != nodes0[n1.maxPredID].maxSuccID)
156 n1.branchID = branchCounter++;
157 }
158 }
159
160 // Propagate branch labels
161 for(size_t t = 1; t < nT; t++) {
162 auto &nodes0 = timeNodesMap[t - 1];
163 auto &nodes1 = timeNodesMap[t];
164 auto &edges = timeEdgesMap[t - 1];
165
166 size_t nE = edges.size();
167
168 for(size_t i = 0; i < nE; i += 4) {
169 auto n0Index = edges[i];
170 auto n1Index = edges[i + 1];
171 auto &n0 = nodes0[n0Index];
172 auto &n1 = nodes1[n1Index];
173
174 if(n1.branchID == -1 && n0Index == n1.maxPredID)
175 n1.branchID = n0.branchID;
176 }
177 }
178
179 // Label edges
180 for(size_t t = 1; t < nT; t++) {
181 auto &nodes0 = timeNodesMap[t - 1];
182 auto &nodes1 = timeNodesMap[t];
183 auto &edges = timeEdgesMap[t - 1];
184
185 size_t nE = edges.size();
186
187 for(size_t i = 0; i < nE; i += 4) {
188 auto n0Index = edges[i];
189 auto n1Index = edges[i + 1];
190 auto &n0 = nodes0[n0Index];
191 auto &n1 = nodes1[n1Index];
192
193 edges[i + 3] = n0.branchID == n1.branchID ? n0.branchID
194 : n0.maxSuccID == n1Index ? n0.branchID
195 : n1.branchID;
196 }
197 }
198
199 std::stringstream msg;
200 msg << "done (" << tm.getElapsedTime() << " s).";
201 printMsg(msg.str(), debug::Priority::PERFORMANCE);
202
203 return 1;
204 }

279
280 for(size_t i = 0; i < nNodes; i++) {
281 Node &n = nodes[i];

283 n.x /= size;
284 n.y /= size;
285 n.z /= size;

301template <typename labelType>
302int ttk::TrackingFromOverlap::computeOverlap(const float *pointCoordinates0,
303 const float *pointCoordinates1,
304 const labelType *pointLabels0,
305 const labelType *pointLabels1,
306 const size_t nPoints0,
307 const size_t nPoints1,
308
309 Edges &edges) const {
310 // -------------------------------------------------------------------------
311 // Compute labelIndexMaps
312 // -------------------------------------------------------------------------
313 std::map<labelType, size_t> labelIndexMap0;
314 std::map<labelType, size_t> labelIndexMap1;
315 this->computeLabelIndexMap<labelType>(pointLabels0, nPoints0, labelIndexMap0);
316 this->computeLabelIndexMap<labelType>(pointLabels1, nPoints1, labelIndexMap1);
317
318 // -------------------------------------------------------------------------
319 // Sort coordinates
320 // -------------------------------------------------------------------------
321 std::vector<size_t> sortedIndicies0;
322 std::vector<size_t> sortedIndicies1;
323 this->sortCoordinates(pointCoordinates0, nPoints0, sortedIndicies0);
324 this->sortCoordinates(pointCoordinates1, nPoints1, sortedIndicies1);
325
326 // -------------------------------------------------------------------------
327 // Track Nodes
328 // -------------------------------------------------------------------------
329 printMsg("Tracking .............. ", debug::Priority::PERFORMANCE);
330 Timer t;
331
332 /* Function that determines configuration of point p0 and p1:
333 0: p0Coords = p1Coords
334 >0: p0Coords < p1Coords
335 <0: p0Coords > p1Coords
336 */
337 auto compare = [&](size_t p0, size_t p1) {
338 size_t p0CoordIndex = p0 * 3;
339 size_t p1CoordIndex = p1 * 3;
340
341 float p0_X = pointCoordinates0[p0CoordIndex++];
342 float p0_Y = pointCoordinates0[p0CoordIndex++];
343 float p0_Z = pointCoordinates0[p0CoordIndex];
344
345 float p1_X = pointCoordinates1[p1CoordIndex++];
346 float p1_Y = pointCoordinates1[p1CoordIndex++];
347 float p1_Z = pointCoordinates1[p1CoordIndex];
348
349 return p0_X == p1_X ? p0_Y == p1_Y ? p0_Z == p1_Z ? 0
350 : p0_Z < p1_Z ? -1
351 : 1
352 : p0_Y < p1_Y ? -1
353 : 1
354 : p0_X < p1_X ? -1
355 : 1;
356 };
357
358 size_t i = 0; // iterator for 0
359 size_t j = 0; // iterator for 1
360
361 size_t nEdges = 0;
362 std::unordered_map<size_t, std::unordered_map<size_t, size_t>> edgesMap;
363 // Iterate over both point sets synchronously using comparison function
364 while(i < nPoints0 && j < nPoints1) {
365 size_t pointIndex0 = sortedIndicies0[i];
366 size_t pointIndex1 = sortedIndicies1[j];
367
368 // Determine point configuration
369 int c = compare(pointIndex0, pointIndex1);
370
371 if(c == 0) { // Points have same coordinates -> track
372 labelType label0 = pointLabels0[pointIndex0];
373 labelType label1 = pointLabels1[pointIndex1];
374
375 size_t &nodeIndex0 = labelIndexMap0[label0];
376 size_t &nodeIndex1 = labelIndexMap1[label1];
377
378 // Find edge and increase overlap counter
379 auto edges0 = edgesMap.find(nodeIndex0); // Edges from label0 to nodes1
380
381 // If map does not exist then create it
382 if(edges0 == edgesMap.end()) {
383 edgesMap[nodeIndex0] = std::unordered_map<size_t, size_t>();
384 edges0 = edgesMap.find(nodeIndex0);
385 }
386
387 // Find edge label0 -> label1
388 auto edge = edges0->second.find(nodeIndex1);
389
390 // If edge does not exist then create it
391 if(edge == edges0->second.end()) {
392 edges0->second[nodeIndex1] = 0;
393 edge = edges0->second.find(nodeIndex1);
394 nEdges++;
395 }
396
397 // Increase overlap
398 edge->second++;
399
400 i++;
401 j++;
402 } else if(c > 0) { // p0 in front of p1 -> let p1 catch up
403 j++;
404 } else { // p1 in front of p0 -> let p0 catch up
405 i++;
406 }
407 }
408
409 // -------------------------------------------------------------------------
410 // Pack Output
411 // -------------------------------------------------------------------------
412 {
413 edges.resize(nEdges * 4);
414 size_t q = 0;
415 for(auto &it0 : edgesMap) {
416 for(auto &it1 : it0.second) {
417 edges[q++] = it0.first;
418 edges[q++] = it1.first;
419 edges[q++] = it1.second;
420 edges[q++] = -1;
421 }
422 }
423 }
424
425 // Print Status
426 {
427 std::stringstream msg;
428 msg << "done (#" << nEdges << " in " << t.getElapsedTime() << " s).";
429 printMsg(msg.str(), debug::Priority::PERFORMANCE);
430 }
431
432 return 0;
433}
