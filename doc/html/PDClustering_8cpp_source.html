944void ttk::PDClustering::initializeCentroidsKMeanspp() {
945 std::vector<int> indexes_clusters;
946 int random_idx = deterministic_ ? 0 : rand() % numberOfInputs_;
947 indexes_clusters.emplace_back(random_idx);
948
949 if(do_min_) {
950 GoodDiagram centroid_min
951 = diagramToCentroid(current_bidder_diagrams_min_[random_idx]);
952 centroids_min_.emplace_back(centroid_min);
953 }
954 if(do_sad_) {
955 GoodDiagram centroid_sad
956 = diagramToCentroid(current_bidder_diagrams_saddle_[random_idx]);
957 centroids_saddle_.emplace_back(centroid_sad);
958 }
959 if(do_max_) {
960 GoodDiagram centroid_max
961 = diagramToCentroid(current_bidder_diagrams_max_[random_idx]);
962 centroids_max_.emplace_back(centroid_max);
963 }
964
965 while((int)indexes_clusters.size() < k_) {
966 std::vector<double> min_distance_to_centroid(numberOfInputs_);
967 std::vector<double> probabilities(numberOfInputs_);
968
969 // Uncomment for a deterministic algorithm
970 double maximal_distance = 0;
971 int candidate_centroid = 0;
972
973 for(int i = 0; i < numberOfInputs_; i++) {
974
975 min_distance_to_centroid[i] = std::numeric_limits<double>::max();
976 if(std::find(indexes_clusters.begin(), indexes_clusters.end(), i)
977 != indexes_clusters.end()) {
978
979 min_distance_to_centroid[i] = 0;
980 } else {
981
982 for(size_t j = 0; j < indexes_clusters.size(); ++j) {
983
984 double distance = 0;
985 if(do_min_) {
986 GoodDiagram centroid_min
987 = centroidWithZeroPrices(centroids_min_[j]);
988 distance += computeDistance(
989 current_bidder_diagrams_min_[i], centroid_min, 0.01);
990 }
991 if(do_sad_) {
992 GoodDiagram centroid_saddle
993 = centroidWithZeroPrices(centroids_saddle_[j]);
994 distance += computeDistance(
995 current_bidder_diagrams_saddle_[i], centroid_saddle, 0.01);
996 }
997 if(do_max_) {
998 GoodDiagram centroid_max
999 = centroidWithZeroPrices(centroids_max_[j]);
1000 distance += computeDistance(
1001 current_bidder_diagrams_max_[i], centroid_max, 0.01);
1002 }
1003 if(distance < min_distance_to_centroid[i]) {
1004 min_distance_to_centroid[i] = distance;
1005 }
1006 }
1007 }
1008 probabilities[i] = Geometry::pow(min_distance_to_centroid[i], 2);
1009
1010 // The following block is useful in case of need for a deterministic
1011 // algoritm
1012 if(deterministic_ && min_distance_to_centroid[i] > maximal_distance) {
1013 maximal_distance = min_distance_to_centroid[i];
1014 candidate_centroid = i;
1015 }
1016 }
1017 // Comment the following four lines to make it deterministic
1018 std::random_device rd;
1019 std::mt19937 gen(rd());
1020 std::discrete_distribution<int> distribution(
1021 probabilities.begin(), probabilities.end());
1022
1023 if(!deterministic_) {
1024 candidate_centroid = distribution(gen);
1025 }
1026
1027 indexes_clusters.emplace_back(candidate_centroid);
1028 if(do_min_) {
1029 GoodDiagram centroid_min
1030 = diagramToCentroid(current_bidder_diagrams_min_[candidate_centroid]);
1031 centroids_min_.emplace_back(centroid_min);
1032 }
1033 if(do_sad_) {
1034 GoodDiagram centroid_sad = diagramToCentroid(
1035 current_bidder_diagrams_saddle_[candidate_centroid]);
1036 centroids_saddle_.emplace_back(centroid_sad);
1037 }
1038 if(do_max_) {
1039 GoodDiagram centroid_max
1040 = diagramToCentroid(current_bidder_diagrams_max_[candidate_centroid]);
1041 centroids_max_.emplace_back(centroid_max);
1042 }
1043 }

1178 std::vector<std::vector<double>> distance_matrix = getDistanceMatrix();
1179 old_clustering_ = clustering_;
1180 invertClusters();
1181 initializeEmptyClusters();
1182
1183 for(int i = 0; i < numberOfInputs_; ++i) {
1184 double min_distance_to_centroid = std::numeric_limits<double>::max();
1185 int cluster = -1;
1186 for(int c = 0; c < k_; ++c) {
1187 if(distance_matrix[i][c] < min_distance_to_centroid) {
1188 min_distance_to_centroid = distance_matrix[i][c];
1189 cluster = c;
1190 }
1191 }
1192
1193 clustering_[cluster].emplace_back(i);
1194 if(cluster != inv_clustering_[i]) {
1195 // New centroid attributed to this diagram
1196 resetDosToOriginalValues();
1197 barycenter_inputs_reset_flag = true;
1198 if(do_min_) {
1199 centroids_with_price_min_[i]
1200 = centroidWithZeroPrices(centroids_min_[cluster]);
1201 }
1202 if(do_sad_) {
1203 centroids_with_price_saddle_[i]
1204 = centroidWithZeroPrices(centroids_saddle_[cluster]);
1205 }
1206 if(do_max_) {
1207 centroids_with_price_max_[i]
1208 = centroidWithZeroPrices(centroids_max_[cluster]);
1209 }
1210 inv_clustering_[i] = cluster;
1211 }
1212 }
1213
1214 } else {
1215 old_clustering_ = clustering_;
1216 invertClusters();
1217 initializeEmptyClusters();
1218
1219 for(int i = 0; i < numberOfInputs_; i++) {
1220 clustering_[0].emplace_back(i);
