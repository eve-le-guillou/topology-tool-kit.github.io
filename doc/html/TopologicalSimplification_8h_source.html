127 enum class BACKEND { LEGACY, LTS, TO };
128 /*
129 * Either execute this file "legacy" algorithm, or the

131 * Default is lts (localized).
132 */

140 const bool addPerturbation,
141 triangulationType &triangulation,

144 inline void setBackend(const BACKEND arg) {

148 inline int preconditionTriangulation(AbstractTriangulation *triangulation) {
149 switch(backend_) {
150 case BACKEND::LEGACY:
151 legacyObject_.setDebugLevel(debugLevel_);
152 legacyObject_.setThreadNumber(threadNumber_);
153 legacyObject_.preconditionTriangulation(triangulation);
154 break;
155
156 case BACKEND::LTS:
157 ltsObject_.setDebugLevel(debugLevel_);
158 ltsObject_.setThreadNumber(threadNumber_);
159 ltsObject_.preconditionTriangulation(triangulation);
160 break;
161
162 case BACKEND::TO:
163 topologyOptimizer_.setDebugLevel(debugLevel_);
164 topologyOptimizer_.setThreadNumber(threadNumber_);
165 topologyOptimizer_.preconditionTriangulation(triangulation);
166 break;
167
168 default:
169 this->printErr(
170 "Error, the backend for topological simplification is invalid");
171 return -1;
172 }
173 return 0;
174 }
175
176 protected:
177 BACKEND backend_{BACKEND::LTS};
178 LegacyTopologicalSimplification legacyObject_;
179 lts::LocalizedTopologicalSimplification ltsObject_;
180 ttk::TopologicalOptimization topologyOptimizer_;
181
182 SimplexId vertexNumber_{};
183 bool UseFastPersistenceUpdate{true};
184 bool FastAssignmentUpdate{true};
185 int EpochNumber{1000};
186
187 // if PDCMethod == 0 then we use Progressive approach
188 // if PDCMethod == 1 then we use Classical Auction approach
189 int PDCMethod{1};
190
191 // if MethodOptimization == 0 then we use direct optimization
192 // if MethodOptimization == 1 then we use Adam
193 int MethodOptimization{0};
194
195 // if FinePairManagement == 0 then we let the algorithm choose
196 // if FinePairManagement == 1 then we fill the domain
197 // if FinePairManagement == 2 then we cut the domain
198 int FinePairManagement{0};
199
200 // Adam
201 bool ChooseLearningRate{false};
202 double LearningRate{0.0001};
203
204 // Direct Optimization : Gradient Step Size
205 double Alpha{0.5};
206
207 // Stopping criterion: when the loss becomes less than a percentage (e.g.
208 // 1%) of the original loss (between input diagram and simplified diagram)
209 double CoefStopCondition{0.01};
210
211 //
212 bool OptimizationWithoutMatching{false};
213 int ThresholdMethod{1};
214 double Threshold{0.01};
215 int LowerThreshold{-1};
216 int UpperThreshold{2};
217 int PairTypeToDelete{1};
218
219 bool ConstraintAveraging{true};
220
221 int PrintFrequency{10};

225template <typename dataType, typename triangulationType>
226int ttk::TopologicalSimplification::execute(
227 const dataType *const inputScalars,
228 dataType *const outputScalars,
229 const SimplexId *const identifiers,
230 const SimplexId *const inputOffsets,
231 SimplexId *const offsets,
232 const SimplexId constraintNumber,
233 const bool addPerturbation,
234 triangulationType &triangulation,
235 const ttk::DiagramType &constraintDiagram) {
236 switch(backend_) {
237 case BACKEND::LTS:
238 return ltsObject_
239 .removeUnauthorizedExtrema<dataType, SimplexId, triangulationType>(
240 outputScalars, offsets, &triangulation, identifiers, constraintNumber,
241 addPerturbation);
242 case BACKEND::LEGACY:
243 return legacyObject_.execute(inputScalars, outputScalars, identifiers,
244 inputOffsets, offsets, constraintNumber,
245 triangulation);
246
247 case BACKEND::TO:
248 topologyOptimizer_.setUseFastPersistenceUpdate(UseFastPersistenceUpdate);
249 topologyOptimizer_.setFastAssignmentUpdate(FastAssignmentUpdate);
250 topologyOptimizer_.setEpochNumber(EpochNumber);
251 topologyOptimizer_.setPDCMethod(PDCMethod);
252 topologyOptimizer_.setMethodOptimization(MethodOptimization);
253 topologyOptimizer_.setFinePairManagement(FinePairManagement);
254 topologyOptimizer_.setChooseLearningRate(ChooseLearningRate);
255 topologyOptimizer_.setLearningRate(LearningRate);
256 topologyOptimizer_.setAlpha(Alpha);
257 topologyOptimizer_.setCoefStopCondition(CoefStopCondition);
258 topologyOptimizer_.setOptimizationWithoutMatching(
259 OptimizationWithoutMatching);
260 topologyOptimizer_.setThresholdMethod(ThresholdMethod);
261 topologyOptimizer_.setThresholdPersistence(Threshold);
262 topologyOptimizer_.setLowerThreshold(LowerThreshold);
263 topologyOptimizer_.setUpperThreshold(UpperThreshold);
264 topologyOptimizer_.setPairTypeToDelete(PairTypeToDelete);
265 topologyOptimizer_.setConstraintAveraging(ConstraintAveraging);
266 topologyOptimizer_.setPrintFrequency(PrintFrequency);
267
268 return topologyOptimizer_.execute(inputScalars, outputScalars, offsets,
269 &triangulation, constraintDiagram);
270 default:
271 this->printErr(
272 "Error, the backend for topological simplification is invalid");
273 return -1;
274 }
