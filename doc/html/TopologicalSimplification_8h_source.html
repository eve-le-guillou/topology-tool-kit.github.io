131 inline int preconditionTriangulation(AbstractTriangulation *triangulation) {
132 switch(backend_) {
133 case BACKEND::LEGACY:
134 legacyObject_.setDebugLevel(debugLevel_);
135 legacyObject_.setThreadNumber(threadNumber_);
136 legacyObject_.preconditionTriangulation(triangulation);
137 break;
138
139 case BACKEND::LTS:
140 ltsObject_.setDebugLevel(debugLevel_);
141 ltsObject_.setThreadNumber(threadNumber_);
142 ltsObject_.preconditionTriangulation(triangulation);
143 break;
144
145 default:
146 this->printErr(
147 "Error, the backend for topological simplification is invalid");
148 return -1;
149 }
150 return 0;
151 }
152
153 protected:
154 BACKEND backend_{BACKEND::LTS};
155 LegacyTopologicalSimplification legacyObject_;
156 lts::LocalizedTopologicalSimplification ltsObject_;
157 };
