204void ttkPersistenceDiagramClustering::VTUToDiagram(
205 diagramType &diagram, vtkUnstructuredGrid *vtu) const {
206
207 const auto pd = vtu->GetPointData();
208 const auto cd = vtu->GetCellData();
209
210 if(pd == nullptr) {
211 this->printErr("VTU diagram with NULL Point Data");
212 return;
213 }
214 if(cd == nullptr) {
215 this->printErr("VTU diagram with NULL Cell Data");
216 return;
217 }
218
219 // cell data
220 const auto pairId = vtkIntArray::SafeDownCast(cd->GetArray("PairIdentifier"));
221 const auto pairType = vtkIntArray::SafeDownCast(cd->GetArray("PairType"));
222 const auto pairPers = cd->GetArray("Persistence");
223 const auto birthScalars = cd->GetArray("Birth");
224
225 // point data
226 const auto vertexId
227 = vtkIntArray::SafeDownCast(pd->GetArray(ttk::VertexScalarFieldName));
228 const auto critType = vtkIntArray::SafeDownCast(pd->GetArray("CriticalType"));
229 const auto coords = vtkFloatArray::SafeDownCast(pd->GetArray("Coordinates"));
230
231 const auto points = vtu->GetPoints();
232
233 const bool embed = coords == nullptr;
234
235 int nPairs = pairId->GetNumberOfTuples();
236
237 // compact pairIds in [0, nPairs - 1] (diagonal excepted)
238 for(int i = 0; i < nPairs; i++) {
239 if(pairId->GetTuple1(i) != -1) {
240 pairId->SetTuple1(i, i);
241 }
242 }
243
244 // skip diagram diagonal if present (assuming it's the last pair in the
245 // diagram)
246 if(pairId->GetTuple1(nPairs - 1) == -1)
247 nPairs -= 1;
248
249 if(nPairs < 1 || vertexId == nullptr || pairId == nullptr
250 || critType == nullptr || pairPers == nullptr || pairType == nullptr
251 || points == nullptr) {
252 this->printErr("Either no pairs in diagram or some array is NULL");
253 return;
254 }
255
256 if(NumberOfClusters == 1) {
257 diagram.resize(nPairs);
258 } else {
259 diagram.resize(nPairs + 1);
260 }
261
262 // count the number of pairs whose index is >= nPairs
263 int nbNonCompact = 0;
264
265 // skip diagonal cell (corresponding points already dealt with)
266 for(int i = 0; i < nPairs; ++i) {
267
268 const int v0 = vertexId->GetValue(2 * i);
269 const int v1 = vertexId->GetValue(2 * i + 1);
270 const int ct0 = critType->GetValue(2 * i);
271 const int ct1 = critType->GetValue(2 * i + 1);
272
273 const int pId = pairId->GetValue(i);
274 const int pType = pairType->GetValue(i);
275 const auto pers = pairPers->GetTuple1(i);
276 const auto birth = birthScalars->GetTuple1(i);
277 const auto death = birth + pers;
278
279 std::array<double, 3> coordsBirth{}, coordsDeath{};
280
281 if(embed) {
282 points->GetPoint(2 * i + 0, coordsBirth.data());
283 points->GetPoint(2 * i + 1, coordsDeath.data());
284 } else {
285 coords->GetTuple(2 * i + 0, coordsBirth.data());
286 coords->GetTuple(2 * i + 1, coordsDeath.data());
287 }
288
289 if(pId != -1 && pId < nPairs) {
290
291 if(pId == 0) {
292 // deal with the global min-max pair separately
293 // (what do we do with other infinite pairs?)
294
295 if(NumberOfClusters == 1) {
296 diagram[0] = std::make_tuple(
297 v0, CriticalType::Local_minimum, v1, CriticalType::Local_maximum,
298 pers, pType, birth, coordsBirth[0], coordsBirth[1], coordsBirth[2],
299 death, coordsDeath[0], coordsDeath[1], coordsDeath[2]);
300 } else {
301 // duplicate the global min-max pair into two: one min-saddle pair and
302 // one saddle-max pair
303 diagram[0] = std::make_tuple(
304 v0, CriticalType::Local_minimum, v1, CriticalType::Saddle1, pers,
305 pType, birth, coordsBirth[0], coordsBirth[1], coordsBirth[2], death,
306 coordsDeath[0], coordsDeath[1], coordsDeath[2]);
307 // store the saddle max pair at the vector end
308 diagram[nPairs] = std::make_tuple(
309 v0, CriticalType::Saddle1, v1, CriticalType::Local_maximum, pers,
310 pType, birth, coordsBirth[0], coordsBirth[1], coordsBirth[2], death,
311 coordsDeath[0], coordsDeath[1], coordsDeath[2]);
312 }
313
314 } else {
315 // all other pairs
316 diagram[pId] = std::make_tuple(
317 v0, static_cast<CriticalType>(ct0), v1,
318 static_cast<CriticalType>(ct1), pers, pType, birth, coordsBirth[0],
319 coordsBirth[1], coordsBirth[2], death, coordsDeath[0], coordsDeath[1],
320 coordsDeath[2]);
321 }
322 }
323
324 if(pId >= nPairs) {
325 nbNonCompact++;
326 }
327 }
328
329 if(nbNonCompact > 0) {
330 this->printWrn("Missed " + std::to_string(nbNonCompact)
331 + " pairs due to non-compactness.");
332 }

335void ttkPersistenceDiagramClustering::diagramToVTU(
336 vtkUnstructuredGrid *output,
337 const diagramType &diagram,
338 const int cid,
339 const double max_persistence) const {
340

347 vtkNew<vtkPoints> points{};

349 output->SetPoints(points);
350
351 // point data
352 vtkNew<vtkIntArray> critType{};
353 critType->SetName("CriticalType");
354 critType->SetNumberOfTuples(nPoints);
355 output->GetPointData()->AddArray(critType);
356
357 vtkNew<vtkIntArray> clusterId{};
358 clusterId->SetName("ClusterID");
359 clusterId->SetNumberOfComponents(1);
360 clusterId->SetNumberOfTuples(nPoints);
361 clusterId->Fill(cid);
362 output->GetPointData()->AddArray(clusterId);
363
364 vtkNew<vtkFloatArray> coords{};
365 coords->SetNumberOfComponents(3);
366 coords->SetName("Coordinates");
367 coords->SetNumberOfTuples(nPoints);
368 output->GetPointData()->AddArray(coords);
369
370 vtkNew<vtkDoubleArray> pointPers{};
371 pointPers->SetName("Persistence");
372 pointPers->SetNumberOfTuples(nPoints);
373 output->GetPointData()->AddArray(pointPers);

375 vtkNew<ttkSimplexIdTypeArray> vsf{};
376 vsf->SetName(ttk::VertexScalarFieldName);
377 vsf->SetNumberOfTuples(nPoints);
378 output->GetPointData()->AddArray(vsf);
379
380 // cell data
381 vtkNew<vtkIntArray> pairId{};
382 pairId->SetName("PairIdentifier");
383 pairId->SetNumberOfTuples(diagram.size() + 1);
384 output->GetCellData()->AddArray(pairId);
385
386 vtkNew<vtkIntArray> pairType{};
387 pairType->SetName("PairType");
388 pairType->SetNumberOfTuples(diagram.size() + 1);
389 output->GetCellData()->AddArray(pairType);
390
391 vtkNew<vtkDoubleArray> pairPers{};
392 pairPers->SetName("Persistence");
393 pairPers->SetNumberOfTuples(diagram.size() + 1);
394 output->GetCellData()->AddArray(pairPers);
395
396 vtkNew<vtkDoubleArray> birthScalars{};
397 birthScalars->SetName(ttk::PersistenceBirthName);
398 birthScalars->SetNumberOfTuples(diagram.size() + 1);
399 output->GetCellData()->AddArray(birthScalars);
400
401 vtkNew<vtkUnsignedCharArray> isFinite{};
402 isFinite->SetName(ttk::PersistenceIsFinite);
403 isFinite->SetNumberOfTuples(diagram.size() + 1);
404 output->GetCellData()->AddArray(isFinite);
405
406 for(size_t j = 0; j < diagram.size(); ++j) {
407 const auto &pair{diagram[j]};
408 const auto birth{std::get<6>(pair)};
409 const auto death{std::get<10>(pair)};
410 const auto birtVertId{std::get<0>(pair)};
411 const auto deathVertId{std::get<2>(pair)};
412 const auto pType{std::get<5>(pair)};
413 const auto birthType{std::get<1>(pair)};
414 const auto deathType{std::get<3>(pair)};
415 std::array<float, 3> coordsBirth{
416 std::get<7>(pair), std::get<8>(pair), std::get<9>(pair)};
417 std::array<float, 3> coordsDeath{
418 std::get<11>(pair), std::get<12>(pair), std::get<13>(pair)};
419
420 // cell data
421 pairId->SetTuple1(j, j);
422 pairType->SetTuple1(j, pType);
423 pairPers->SetTuple1(j, death - birth);
424 birthScalars->SetTuple1(j, birth);
425 isFinite->SetTuple1(j, j == 0 ? 0 : 1);
426
427 // point data
428 coords->SetTuple(2 * j + 0, coordsBirth.data());
429 coords->SetTuple(2 * j + 1, coordsDeath.data());
430 pointPers->SetTuple1(2 * j + 0, death - birth);
431 pointPers->SetTuple1(2 * j + 1, death - birth);
432 critType->SetTuple1(2 * j + 0, static_cast<int>(birthType));
433 critType->SetTuple1(2 * j + 1, static_cast<int>(deathType));
434 vsf->SetTuple1(2 * j + 0, birtVertId);
435 vsf->SetTuple1(2 * j + 1, deathVertId);
436
437 points->SetPoint(2 * j + 0, birth, birth, 0);
438 points->SetPoint(2 * j + 1, birth, death, 0);
439
440 const std::array<vtkIdType, 2> ids{
441 2 * static_cast<vtkIdType>(j) + 0,
442 2 * static_cast<vtkIdType>(j) + 1,
443 };
444 output->InsertNextCell(VTK_LINE, 2, ids.data());
445 }
446
447 // add diagonal
448 const auto minmax_birth = std::minmax_element(
449 diagram.begin(), diagram.end(), [](const pairTuple &a, const pairTuple &b) {
450 return std::get<6>(a) < std::get<6>(b);
451 });
452 const std::array<vtkIdType, 2> ids{
453 2 * (minmax_birth.first - diagram.begin()),
454 2 * (minmax_birth.second - diagram.begin()),
455 };
456 output->InsertNextCell(VTK_LINE, 2, ids.data());
457 pairId->SetTuple1(diagram.size(), -1);
458 pairType->SetTuple1(diagram.size(), -1);
459 // use twice the max persistence of all input diagrams...
460 pairPers->SetTuple1(diagram.size(), 2.0 * max_persistence);
461 birthScalars->SetTuple1(diagram.size(), std::get<6>(*minmax_birth.first));
462 isFinite->SetTuple1(diagram.size(), 0);
463}

465void ttkPersistenceDiagramClustering::outputClusteredDiagrams(
466 vtkMultiBlockDataSet *output,
467 const std::vector<vtkUnstructuredGrid *> &diags,
468 const std::vector<int> &inv_clustering,
469 const DISPLAY dm,
470 const double spacing,
471 const double max_persistence) const {
472
473 // index of diagram in its cluster
474 std::vector<int> diagIdInClust{};
475 // number of diagrams per cluster
476 std::vector<int> clustSize{};
477
478 // prep work for displaying diagrams as cluster stars
479 if(dm == DISPLAY::STARS) {
480 // total number of clusters
481 const auto nClusters
482 = 1 + *std::max_element(inv_clustering.begin(), inv_clustering.end());
483 clustSize.resize(nClusters, 0);
484 diagIdInClust.resize(diags.size());
485 for(size_t i = 0; i < inv_clustering.size(); ++i) {
486 auto &diagsInClust = clustSize[inv_clustering[i]];
487 diagIdInClust[i] = diagsInClust;
488 diagsInClust++;
489 }
490 }
491
492 output->SetNumberOfBlocks(diags.size());
493
494 for(size_t i = 0; i < diags.size(); ++i) {
495 vtkNew<vtkUnstructuredGrid> vtu{};
496 vtu->ShallowCopy(diags[i]);
497
498 vtkNew<vtkIntArray> clusterId{};
499 clusterId->SetName("ClusterID");
500 clusterId->SetNumberOfComponents(1);
501 clusterId->SetNumberOfTuples(vtu->GetNumberOfPoints());
502 clusterId->Fill(inv_clustering[i]);
503 vtu->GetPointData()->AddArray(clusterId);
504
505 // add Persistence data array on vertices
506 vtkNew<vtkDoubleArray> pointPers{};
507 pointPers->SetName("Persistence");
508 pointPers->SetNumberOfTuples(vtu->GetNumberOfPoints());
509 vtu->GetPointData()->AddArray(pointPers);
510
511 // diagonal uses two existing points
512 for(int j = 0; j < vtu->GetNumberOfCells() - 1; ++j) {
513 const auto persArray = vtu->GetCellData()->GetArray("Persistence");
514 const auto pers = persArray->GetTuple1(j);
515 pointPers->SetTuple1(2 * j + 0, pers);
516 pointPers->SetTuple1(2 * j + 1, pers);
517 }
518
519 // translate diagram back to canonical representation
520 ResetDiagramPosition(vtu, *this);
521
522 if(dm == DISPLAY::MATCHINGS && spacing > 0) {
523 // translate diagrams along the Z axis
524 const std::array<double, 3> trans{0, 0, i == 0 ? -spacing : spacing};
525 TranslateDiagram(vtu, trans);
526 output->SetBlock(i, vtu);
527 } else if(dm == DISPLAY::STARS && spacing > 0) {
528 const auto c = inv_clustering[i];
529 const auto angle = 2.0 * M_PI * static_cast<double>(diagIdInClust[i])
530 / static_cast<double>(clustSize[c]);
531 // translate diagrams in the XY plane
532 const std::array<double, 3> trans{
533 3.0 * (spacing + 0.2) * max_persistence * c
534 + spacing * max_persistence * std::cos(angle) + 0.2,
535 spacing * max_persistence * std::sin(angle), 0};
536 TranslateDiagram(vtu, trans);
537 output->SetBlock(i, vtu);
538
539 } else {
540 // add diagram to output multi-block dataset
541 output->SetBlock(i, vtu);
542 }
543 }
544}
