109 std::vector<int> PersistenceDiagramClustering::execute(
110 std::vector<std::vector<diagramTuple>> &intermediateDiagrams,
111 std::vector<std::vector<diagramTuple>> &final_centroids,
112 std::vector<std::vector<std::vector<matchingTuple>>> &all_matchings) {
113
114 const int numberOfInputs_ = intermediateDiagrams.size();
115 Timer tm;
116 {
117 printMsg("Clustering " + std::to_string(numberOfInputs_) + " diagrams in "
118 + std::to_string(NumberOfClusters) + " cluster(s).");
119 }
120
121 std::vector<std::vector<diagramTuple>> data_min(numberOfInputs_);
122 std::vector<std::vector<diagramTuple>> data_sad(numberOfInputs_);
123 std::vector<std::vector<diagramTuple>> data_max(numberOfInputs_);
124
125 std::vector<std::vector<int>> data_min_idx(numberOfInputs_);
126 std::vector<std::vector<int>> data_sad_idx(numberOfInputs_);
127 std::vector<std::vector<int>> data_max_idx(numberOfInputs_);
128
129 std::vector<int> inv_clustering(numberOfInputs_);
130
131 bool do_min = false;
132 bool do_sad = false;
133 bool do_max = false;
134
135 // Create diagrams for min, saddle and max persistence pairs
136 for(int i = 0; i < numberOfInputs_; i++) {
137 std::vector<diagramTuple> &CTDiagram = intermediateDiagrams[i];
138
139 for(size_t j = 0; j < CTDiagram.size(); ++j) {
140 diagramTuple t = CTDiagram[j];
141
142 BNodeType nt1 = std::get<1>(t);
143 BNodeType nt2 = std::get<3>(t);
144
145 dataType dt = std::get<4>(t);
146 // if (abs<dataType>(dt) < zeroThresh) continue;
147 if(dt > 0) {
148 if(nt1 == BLocalMin && nt2 == BLocalMax) {
149 if(PairTypeClustering == 2) {
150 data_max[i].push_back(t);
151 data_max_idx[i].push_back(j);
152 do_max = true;
153 } else {
154 data_min[i].push_back(t);
155 data_min_idx[i].push_back(j);
156 do_min = true;
157 }
158 } else {
159 if(nt1 == BLocalMax || nt2 == BLocalMax) {
160 data_max[i].push_back(t);
161 data_max_idx[i].push_back(j);
162 do_max = true;
163 }
164 if(nt1 == BLocalMin || nt2 == BLocalMin) {
165 data_min[i].push_back(t);
166 data_min_idx[i].push_back(j);
167 do_min = true;
168 }
169 if((nt1 == BSaddle1 && nt2 == BSaddle2)
170 || (nt1 == BSaddle2 && nt2 == BSaddle1)) {
171 data_sad[i].push_back(t);
172 data_sad_idx[i].push_back(j);
173 do_sad = true;
174 }
175 }
176 }
177 }
178 }
179
180 {
181 std::stringstream msg;
182 switch(PairTypeClustering) {
183 case(0):
184 msg << "Only MIN-SAD Pairs";
185 do_max = false;
186 do_sad = false;
187 break;
188 case(1):
189 msg << "Only SAD-SAD Pairs";
190 do_max = false;
191 do_min = false;
192 break;
193 case(2):
194 msg << "Only SAD-MAX Pairs";
195 do_min = false;
196 do_sad = false;
197 break;
198 default:
199 msg << "All critical pairs: "
200 "global clustering";
201 break;
202 }
203 printMsg(msg.str());
204 }
205
206 std::vector<std::vector<std::vector<std::vector<matchingTuple>>>>
207 all_matchings_per_type_and_cluster;
208 PDClustering<dataType> KMeans = PDClustering<dataType>();
209 KMeans.setNumberOfInputs(numberOfInputs_);
210 KMeans.setWasserstein(WassersteinMetric);
211 KMeans.setUseProgressive(UseProgressive);
212 KMeans.setAccelerated(UseAccelerated);
213 KMeans.setUseKDTree(true);
214 KMeans.setTimeLimit(TimeLimit);
215 KMeans.setGeometricalFactor(Alpha);
216 KMeans.setLambda(Lambda);
217 KMeans.setDeterministic(Deterministic);
218 KMeans.setForceUseOfAlgorithm(ForceUseOfAlgorithm);
219 KMeans.setDebugLevel(debugLevel_);
220 KMeans.setDeltaLim(DeltaLim);
221 KMeans.setUseDeltaLim(UseAdditionalPrecision);
222 KMeans.setDistanceWritingOptions(DistanceWritingOptions);
223 KMeans.setKMeanspp(UseKmeansppInit);
224 KMeans.setK(NumberOfClusters);
225 KMeans.setDiagrams(&data_min, &data_sad, &data_max);
226 KMeans.setDos(do_min, do_sad, do_max);
227 inv_clustering
228 = KMeans.execute(final_centroids, all_matchings_per_type_and_cluster);
229 std::vector<std::vector<int>> centroids_sizes
230 = KMeans.get_centroids_sizes();
231
232 this->distances = KMeans.getDistances();
233
235 //
236 std::vector<int> cluster_size;
237 std::vector<int> idxInCluster(numberOfInputs_);
238
239 for(int j = 0; j < numberOfInputs_; ++j) {
240 unsigned int c = inv_clustering[j];
241 if(c + 1 > cluster_size.size()) {
242 cluster_size.resize(c + 1);
243 cluster_size[c] = 1;
244 idxInCluster[j] = 0;
245 } else {
246 cluster_size[c]++;
247 idxInCluster[j] = cluster_size[c] - 1;
248 }
249 }
250
251 bool removeDuplicateGlobalPair = false;
252 if(NumberOfClusters > 1 and do_min and do_max) {
253 removeDuplicateGlobalPair = true;
254 }
255
256 all_matchings.resize(NumberOfClusters);
257 for(int c = 0; c < NumberOfClusters; c++) {
258 all_matchings[c].resize(numberOfInputs_);
259 if(removeDuplicateGlobalPair) {
260 centroids_sizes[c][0] -= 1;
261 }
262 }
263 for(int i = 0; i < numberOfInputs_; i++) {
264 unsigned int c = inv_clustering[i];
265
266 if(do_min) {
267 for(unsigned int j = 0;
268 j
269 < all_matchings_per_type_and_cluster[c][0][idxInCluster[i]].size();
270 j++) {
271 matchingTuple t
272 = all_matchings_per_type_and_cluster[c][0][idxInCluster[i]][j];
273 int bidder_id = std::get<0>(t);
274 if(bidder_id < (int)data_min[i].size()) {
275 if(bidder_id < 0) { // matching with a diagonal point
276 std::get<0>(t) = -1;
277 } else {
278 std::get<0>(t) = data_min_idx[i][bidder_id];
279 }
280 // cout<<" IDS : "<<bidder_id<<" "<<std::get<0>(t)<<endl;
281 if(std::get<1>(t) < 0) {
282 std::get<1>(t) = -1;
283 }
284 all_matchings[inv_clustering[i]][i].push_back(t);
285 }
286 }
287 }
288
289 if(do_sad) {
290 for(unsigned int j = 0;
291 j
292 < all_matchings_per_type_and_cluster[c][1][idxInCluster[i]].size();
293 j++) {
294 matchingTuple t
295 = all_matchings_per_type_and_cluster[c][1][idxInCluster[i]][j];
296 int bidder_id = std::get<0>(t);
297 if(bidder_id < (int)data_sad[i].size()) {
298 if(bidder_id < 0) { // matching with a diagonal point
299 std::get<0>(t) = -1;
300 } else {
301 std::get<0>(t) = data_sad_idx[i][bidder_id];
302 }
303 if(std::get<1>(t) >= 0) {
304 std::get<1>(t) = std::get<1>(t) + centroids_sizes[c][0];
305 } else {
306 std::get<1>(t) = -1;
307 }
308 all_matchings[inv_clustering[i]][i].push_back(t);
309 }
310 }
311 }
312
313 if(do_max) {
314 for(unsigned int j = 0;
315 j
316 < all_matchings_per_type_and_cluster[c][2][idxInCluster[i]].size();
317 j++) {
318 matchingTuple t
319 = all_matchings_per_type_and_cluster[c][2][idxInCluster[i]][j];
320 int bidder_id = std::get<0>(t);
321 if(bidder_id < (int)data_max[i].size()) {
322 if(bidder_id < 0) { // matching with a diagonal point
323 std::get<0>(t) = -1;
324 } else {
325 std::get<0>(t) = data_max_idx[i][bidder_id];
326 }
327 // std::get<0>(t) = data_max_idx[i][bidder_id];
328 if(std::get<1>(t) > 0) {
329 std::get<1>(t) = std::get<1>(t) + centroids_sizes[c][0]
330 + centroids_sizes[c][1];
331 } else if(std::get<1>(t) == 0) {
332 if(!removeDuplicateGlobalPair) {
333 std::get<1>(t) = std::get<1>(t) + centroids_sizes[c][0]
334 + centroids_sizes[c][1];
335 }
336 } else {
337 std::get<1>(t) = -1;
338 }
339 all_matchings[inv_clustering[i]][i].push_back(t);
340 }
341 }
342 }
343 }
344
345 printMsg("Complete", 1, tm.getElapsedTime(), threadNumber_);
346 return inv_clustering;
347 }
