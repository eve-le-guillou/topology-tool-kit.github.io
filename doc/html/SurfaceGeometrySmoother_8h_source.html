147 template <typename triangulationType>
148 inline Point
149 relax(const SimplexId a,
150 std::vector<ttk::SurfaceGeometrySmoother::Point> &outputPoints,
151 const triangulationType &triangulationToSmooth) const {
152 Point relaxed{outputPoints[a]};
153 const auto nneigh{triangulationToSmooth.getVertexNeighborNumber(a)};
154 for(SimplexId i = 0; i < nneigh; ++i) {
155 SimplexId neigh{};
156 triangulationToSmooth.getVertexNeighbor(a, i, neigh);
157 relaxed = relaxed + outputPoints[neigh];
158 }
159 return relaxed * (1.0F / static_cast<float>(nneigh + 1));
160 }

170 inline Point projectOnTrianglePlane(const Point &p,
171 const Point &a,
172 const Point &normTri) const {
173 const auto ap{p - a};
174 return p - normTri * Geometry::dotProduct(normTri.data(), ap.data());
175 }
176
185 inline Point
186 projectOnEdge(const Point &p, const Point &a, const Point &b) const {
187 const auto ab{b - a};
188 const auto ap{p - a};
189 return a
190 + ab * Geometry::dotProduct(ap.data(), ab.data())
191 / Geometry::dotProduct(ab.data(), ab.data());
192 }

202 template <typename triangulationType>
203 inline SimplexId
204 getNearestSurfaceVertex(const Point &pa,
205 std::vector<float> &dists,
206 const triangulationType &triangulation) const {
207 for(SimplexId i = 0; i < triangulation.getNumberOfVertices(); ++i) {
208 Point pv{};
209 triangulation.getVertexPoint(i, pv[0], pv[1], pv[2]);
210 dists[i] = Geometry::distance(pa.data(), pv.data());
211 }
212 return std::min_element(dists.begin(), dists.end()) - dists.begin();
213 }

218template <typename triangulationType>
219ttk::SurfaceGeometrySmoother::ProjectionResult
220 ttk::SurfaceGeometrySmoother::findProjection(
221 const ProjectionInput &pi,
222 VisitedMask &trianglesTested,
223 std::vector<float> &dists,
224 std::stack<SimplexId> &trianglesToTest,
225 const triangulationType &triangulation) const {
226
227 ProjectionResult res{pi.pt, pi.nearestVertex, 0, false};
228
229 // clean trianglesToTest
230 while(!trianglesToTest.empty()) {
231 trianglesToTest.pop();
232 }
233
234 // init pipeline by checking in the first triangle around selected vertex
235 if(triangulation.getVertexTriangleNumber(res.nearestVertex) > 0) {
236 SimplexId next{};
237 triangulation.getVertexTriangle(res.nearestVertex, 0, next);
238 trianglesToTest.push(next);
239 }
240
241 while(!trianglesToTest.empty()) {
242 const auto curr = trianglesToTest.top();
243 trianglesToTest.pop();
244
245 // skip if already tested
246 if(trianglesTested.isVisited_[curr]) {
247 continue;
248 }
249
250 // get triangle vertices
251 std::array<SimplexId, 3> tverts{};
252 triangulation.getTriangleVertex(curr, 0, tverts[0]);
253 triangulation.getTriangleVertex(curr, 1, tverts[1]);
254 triangulation.getTriangleVertex(curr, 2, tverts[2]);
255
256 // get coordinates of triangle vertices (lets name is MNO)
257 std::array<Point, 3> mno{}; // [0] is M, [1] is N and [2] is O
258 triangulation.getVertexPoint(tverts[0], mno[0][0], mno[0][1], mno[0][2]);
259 triangulation.getVertexPoint(tverts[1], mno[1][0], mno[1][1], mno[1][2]);
260 triangulation.getVertexPoint(tverts[2], mno[2][0], mno[2][1], mno[2][2]);
261
262 // triangle normal: cross product of two edges
263 Point crossP{};
264 // mn, mo vectors
265 const Point mn = mno[1] - mno[0];
266 const Point mo = mno[2] - mno[0];
267 // compute mn ^ mo
268 Geometry::crossProduct(mn.data(), mo.data(), crossP.data());
269 // unitary normal vector
270 const Point normTri = crossP / Geometry::magnitude(crossP.data());
271
272 res.pt = this->projectOnTrianglePlane(pi.pt, mno[0], normTri);
273
274 // compute barycentric coords of projection
275 Point baryCoords{};
276 Geometry::computeBarycentricCoordinates(
277 mno[0].data(), mno[1].data(), mno[2].data(), res.pt.data(), baryCoords);
278
279 // check if projection in triangle
280 bool inTriangle = true;
281 static const float PREC_FLT{powf(10, -FLT_DIG)};
282
283 for(auto &coord : baryCoords) {
284 if(coord < -PREC_FLT) {
285 inTriangle = false;
286 }
287 if(coord > 1 + PREC_FLT) {
288 inTriangle = false;
289 }
290 }
291
292 // mark triangle as tested
293 trianglesTested.insert(curr);
294 res.trianglesChecked++;
295
296 if(inTriangle) {
297 res.projSuccess = true;
298 // should we check if we have the nearest triangle?
299 break;
300 }
301
302 // extrema values in baryCoords
303 const auto extrema
304 = std::minmax_element(baryCoords.begin(), baryCoords.end());
305
306 // find the nearest triangle vertices local ids (with the
307 // highest/positive values in baryCoords) from proj
308 std::array<SimplexId, 2> vid{
309 static_cast<SimplexId>(extrema.second - baryCoords.begin()), 0};
310 for(size_t j = 0; j < baryCoords.size(); j++) {
311 if(j != static_cast<size_t>(extrema.first - baryCoords.begin())
312 && j != static_cast<size_t>(extrema.second - baryCoords.begin())) {
313 vid[1] = j;
314 break;
315 }
316 }
317
318 // store vertex with highest barycentric coordinate
319 res.nearestVertex = tverts[vid[0]];
320 const auto secondNearVert{tverts[vid[1]]};
321
322 // get the triangle edge with the two vertices
323 SimplexId edge{};
324 const auto nEdges{triangulation.getVertexEdgeNumber(res.nearestVertex)};
325 for(SimplexId i = 0; i < nEdges; ++i) {
326 triangulation.getVertexEdge(res.nearestVertex, i, edge);
327 SimplexId v{};
328 triangulation.getEdgeVertex(edge, 0, v);
329 if(v == res.nearestVertex) {
330 triangulation.getEdgeVertex(edge, 1, v);
331 }
332 if(v == secondNearVert) {
333 break;
334 }
335 }
336
337 // next triangle to visit
338 SimplexId next{};
339 const auto nTri{triangulation.getEdgeTriangleNumber(edge)};
340 for(SimplexId i = 0; i < nTri; ++i) {
341 triangulation.getEdgeTriangle(edge, i, next);
342 if(next != curr) {
343 break;
344 }
345 }
346
347 const auto nVisited{trianglesTested.visitedIds_.size()};
348 if(nVisited > 1 && next == trianglesTested.visitedIds_[nVisited - 2]) {
349 // the relaxed point is probably over the edge separating the
350 // current and the last visited triangles
351 res.pt = this->projectOnEdge(pi.pt, mno[vid[0]], mno[vid[1]]);
352 res.projSuccess = true;
353 break;
354 }
355
356 if(!trianglesTested.isVisited_[next]) {
357 trianglesToTest.push(next);
358 }
359 }
360
361 const size_t maxTrChecked = 100;
362
363 if(res.projSuccess && res.trianglesChecked > maxTrChecked) {
364 res.projSuccess = false;
365 }
366
367 if(!res.projSuccess) {
368 // replace proj by the nearest vertex?
369 res.nearestVertex
370 = this->getNearestSurfaceVertex(pi.pt, dists, triangulation);
371 triangulation.getVertexPoint(
372 res.nearestVertex, res.pt[0], res.pt[1], res.pt[2]);
373 }
374
375 return res;
376}

378template <typename triangulationType0, typename triangulationType1>
379int ttk::SurfaceGeometrySmoother::relaxProject(
380 std::vector<ttk::SurfaceGeometrySmoother::Point> &outputPoints,
381 std::vector<ttk::SurfaceGeometrySmoother::Point> &tmpStorage,
382 std::vector<SimplexId> &nearestVertexId,
383 std::vector<bool> &trianglesTested,
384 std::vector<SimplexId> &visitedTriangles,
385 std::vector<float> &dists,
386 const char *const mask,
387 const triangulationType0 &triangulationToSmooth,
388 const triangulationType1 &triangulationSurface) const {
389
390 Timer tm;
391 std::stack<SimplexId> trianglesToTest{};
392
393 // main loop
394#ifdef TTK_ENABLE_OPENMP
395#pragma omp parallel for num_threads(threadNumber_) \
396 firstprivate(trianglesTested, visitedTriangles, dists, trianglesToTest)
397#endif // TTK_ENABLE_OPENMP
398 for(size_t i = 0; i < outputPoints.size(); i++) {
399
400 // skip computation if i in filtered
401 if(mask != nullptr && mask[i] == 0) {
402 tmpStorage[i] = outputPoints[i];
403 continue;
404 }
405 tmpStorage[i] = this->relax(i, outputPoints, triangulationToSmooth);
406
407 VisitedMask vm{trianglesTested, visitedTriangles};
408
409 // replace curr in outputPoints_ by its projection
410 const auto res
411 = this->findProjection(ProjectionInput{tmpStorage[i], nearestVertexId[i]},
412 vm, dists, trianglesToTest, triangulationSurface);
413
414 tmpStorage[i] = res.pt;
415 nearestVertexId[i] = res.nearestVertex;
416 }
417
418 std::swap(outputPoints, tmpStorage);
419
420 this->printMsg("Projected " + std::to_string(outputPoints.size()) + " points",
421 1.0, tm.getElapsedTime(), this->threadNumber_,
422 debug::LineMode::NEW, debug::Priority::DETAIL);
423
424 return 0;
425}

427template <typename triangulationType0, typename triangulationType1>
428int ttk::SurfaceGeometrySmoother::execute(
429 float *const outputCoords,
430 const float *const inputCoords,
431 const char *const mask,
432 const SimplexId *const vertsId,
433 const int nIter,
434 const triangulationType0 &triangulationToSmooth,
435 const triangulationType1 &triangulationSurface) const {
436
437 const auto nPoints{triangulationToSmooth.getNumberOfVertices()};
438 if(triangulationSurface.getDimensionality() != 2) {
439 this->printErr("Can only project onto a surface");
440 return -1;
441 }
442
443 if(triangulationToSmooth.getDimensionality() < 1
444 || triangulationToSmooth.getDimensionality() > 2) {
445 this->printErr("Can only project a 1D or a 2D triangulated object");
446 return -1;
447 }
448
449 Timer tm{};
450 this->printMsg("Smoothing " + std::to_string(nPoints) + " in "
451 + std::to_string(nIter) + " iterations...");
452
453 // list of triangle IDs already tested
454 // (takes more memory to reduce computation time)
455 std::vector<bool> trianglesTested(
456 triangulationSurface.getNumberOfTriangles(), false);
457 std::vector<SimplexId> visitedTriangles{};
458 // distance between every mesh point and current point
459 std::vector<float> dists(triangulationSurface.getNumberOfVertices());
460
461 // temporary storage
462 std::vector<ttk::SurfaceGeometrySmoother::Point> outputPoints(nPoints),
463 tmpStorage(nPoints);
464 std::vector<SimplexId> nearestVertexId(nPoints);
465
466 // copy input
467#ifdef TTK_ENABLE_OPENMP
468#pragma omp parallel for num_threads(threadNumber_)
469#endif // TTK_ENABLE_OPENMP
470 for(SimplexId i = 0; i < nPoints; ++i) {
471 outputPoints[i][0] = inputCoords[3 * i + 0];
472 outputPoints[i][1] = inputCoords[3 * i + 1];
473 outputPoints[i][2] = inputCoords[3 * i + 2];
474 }
475
476 // ttkVertexScalarField is optional (helps for instance with
477 // MorseSmaleComplex 1-separatrices)
478 if(vertsId != nullptr) {
479#ifdef TTK_ENABLE_OPENMP
480#pragma omp parallel for num_threads(threadNumber_)
481#endif // TTK_ENABLE_OPENMP
482 for(SimplexId i = 0; i < nPoints; ++i) {
483 nearestVertexId[i] = vertsId[i];
484 }
485 } else {
486 // generate a ttkVertexScalarField-like point data array using raw
487 // euclidian distance between the points to smooth and every
488 // vertex of the surface
489 Timer tm_nv{};
490 this->printMsg("Computing nearest vertices...", debug::Priority::INFO,
491 debug::LineMode::REPLACE);
492#ifdef TTK_ENABLE_OPENMP
493#pragma omp parallel for num_threads(threadNumber_) firstprivate(dists)
494#endif // TTK_ENABLE_OPENMP
495 for(SimplexId i = 0; i < nPoints; ++i) {
496 nearestVertexId[i] = this->getNearestSurfaceVertex(
497 outputPoints[i], dists, triangulationSurface);
498 }
499 this->printMsg("Computed nearest vertices", 1.0, tm_nv.getElapsedTime(),
500 this->threadNumber_);
501 }
502
503 for(int i = 0; i < nIter; ++i) {
504 this->relaxProject(outputPoints, tmpStorage, nearestVertexId,
505 trianglesTested, visitedTriangles, dists, mask,
506 triangulationToSmooth, triangulationSurface);
507 }
508
509 // copy output
510#ifdef TTK_ENABLE_OPENMP
511#pragma omp parallel for num_threads(threadNumber_)
512#endif // TTK_ENABLE_OPENMP
513 for(SimplexId i = 0; i < nPoints; ++i) {
514 outputCoords[3 * i + 0] = outputPoints[i][0];
515 outputCoords[3 * i + 1] = outputPoints[i][1];
516 outputCoords[3 * i + 2] = outputPoints[i][2];
517 }
518
519 this->printMsg("Smoothed " + std::to_string(nPoints) + " points", 1.0,
520 tm.getElapsedTime(), this->threadNumber_);
521
522 return 0;
523}
