106 return -1;

109 };

139 }
140 }

336int ttkAlgorithm::RequestDataObject(vtkInformation *ttkNotUsed(request),
337 vtkInformationVector **inputVector,
338 vtkInformationVector *outputVector) {
339 // for each output
340 for(int i = 0; i < this->GetNumberOfOutputPorts(); ++i) {
341 auto outInfo = outputVector->GetInformationObject(i);
342 if(!outInfo) {
343 this->printErr("Unable to retrieve output vtkDataObject at port "
344 + std::to_string(i));
345 return 0;
346 }
347
348 auto outputPortInfo = this->GetOutputPortInformation(i);
349
350 // always request output type again for dynamic filter outputs
351 if(!this->FillOutputPortInformation(i, outputPortInfo)) {
352 this->printErr("Unable to fill output port information at port "
353 + std::to_string(i));
354 return 0;
355 }
356
357 if(outputPortInfo->Has(ttkAlgorithm::SAME_DATA_TYPE_AS_INPUT_PORT())) {
358 // Set output data type to input data type at specified port
359 auto inPortIndex
360 = outputPortInfo->Get(ttkAlgorithm::SAME_DATA_TYPE_AS_INPUT_PORT());
361 if(inPortIndex < 0 || inPortIndex >= this->GetNumberOfInputPorts()) {
362 this->printErr("Input port index " + std::to_string(inPortIndex)
363 + " specified by 'SAME_DATA_TYPE_AS_INPUT_PORT' key of "
364 "output port is out of range ("
365 + std::to_string(this->GetNumberOfInputPorts())
366 + " input ports).");
367 return 0;
368 }
369 auto inInfo = inputVector[inPortIndex]->GetInformationObject(0);
370 if(!inInfo) {
371 this->printErr(
372 "No information object at port " + std::to_string(inPortIndex)
373 + " specified by 'SAME_DATA_TYPE_AS_INPUT_PORT' key of output port.");
374 return 0;
375 }
376
377 auto input = vtkDataObject::GetData(inInfo);
378 auto output = vtkDataObject::GetData(outInfo);
379
380 if(!output || !output->IsA(input->GetClassName())) {
381 auto newOutput
382 = vtkSmartPointer<vtkDataObject>::Take(input->NewInstance());
383 outputPortInfo->Set(
384 vtkDataObject::DATA_TYPE_NAME(), input->GetClassName());
385 outInfo->Set(vtkDataObject::DATA_OBJECT(), newOutput);
386 }
387 } else {
388 // Explicitly create output by data type name
389 if(!outputPortInfo->Has(vtkDataObject::DATA_TYPE_NAME())) {
390 this->printErr("DATA_TYPE_NAME of output port " + std::to_string(i)
391 + " not specified");
392 return 0;
393 }
394 std::string outputType
395 = outputPortInfo->Get(vtkDataObject::DATA_TYPE_NAME());
396
397 if(outputType == "vtkUnstructuredGrid") {
398 prepOutput<vtkUnstructuredGrid>(outInfo, outputType);
399 } else if(outputType == "vtkPolyData") {
400 prepOutput<vtkPolyData>(outInfo, outputType);
401 } else if(outputType == "vtkMultiBlockDataSet") {
402 prepOutput<vtkMultiBlockDataSet>(outInfo, outputType);
403 } else if(outputType == "vtkTable") {
404 prepOutput<vtkTable>(outInfo, outputType);
405 } else if(outputType == "vtkImageData") {
406 prepOutput<vtkImageData>(outInfo, outputType);
407 } else {
408 this->printErr("Unsupported data type for output[" + std::to_string(i)
409 + "]: " + outputType);
410 return 0;
411 }
412 }
413
414 this->printMsg(
415 "Created '"
416 + std::string(outputPortInfo->Get(vtkDataObject::DATA_TYPE_NAME()))
417 + "' at output port " + std::to_string(i),
418 ttk::debug::Priority::VERBOSE);
419 }
420
421 return 1;
422}

425void ttkAlgorithm::MPIPipelinePreconditioning(vtkDataSet *input) {
426
427 vtkNew<vtkGenerateGlobalIds> globalIds;
428 // If the global point id array doesn't exist, it is created
429 if(input->GetPointData()->GetGlobalIds() == nullptr) {
430 printWrn("Global ids haven't been produced in sequential, the parallel "
431 "result may be different");
432 globalIds->SetInputData(input);
433 globalIds->Update();
434 input->ShallowCopy(globalIds->GetOutputDataObject(0));
435 }
436
437 vtkNew<vtkGhostCellsGenerator> generator;
438 if(!input->HasAnyGhostCells()) {
439 generator->SetInputData(input);
440 generator->BuildIfRequiredOff();
441 generator->SetNumberOfGhostLayers(2);
442 generator->Update();
443 input->ShallowCopy(generator->GetOutputDataObject(0));
444 input->GetPointData()->AddArray(
445 generator->GetOutputDataObject(0)->GetGhostArray(0));
446 input->GetCellData()->AddArray(
447 generator->GetOutputDataObject(0)->GetGhostArray(1));
448 }
449
450 // If the RankArray array doesn't exist for pointdata, it is created
451 if(input->GetPointData()->GetArray("RankArray") == nullptr) {
452 int vertexNumber = input->GetNumberOfPoints();
453 std::vector<int> rankArray(vertexNumber, 0);
454 double *boundingBox = input->GetBounds();
455 ttk::produceRankArray(rankArray,
456 static_cast<long int *>(ttkUtils::GetVoidPointer(
457 input->GetPointData()->GetGlobalIds())),
458 static_cast<unsigned char *>(ttkUtils::GetVoidPointer(
459 input->GetPointData()->GetArray("vtkGhostType"))),
460 vertexNumber, boundingBox);
461
462 vtkNew<vtkIntArray> vtkRankArray{};
463 vtkRankArray->SetName("RankArray");
464 vtkRankArray->SetNumberOfComponents(1);
465 vtkRankArray->SetNumberOfTuples(vertexNumber);
466
467 for(int i = 0; i < vertexNumber; i++) {
468 vtkRankArray->SetComponent(i, 0, rankArray[i]);
469 }
470
471 input->GetPointData()->AddArray(vtkRankArray);
472 }
473
474 // If the RankArray array doesn't exist for celldata, it is created
475 if(input->GetCellData()->GetArray("RankArray") == nullptr) {
476 int cellNumber = input->GetNumberOfCells();
477 std::vector<int> cellsRankArray(cellNumber, 0);
478 double *boundingBox = input->GetBounds();
479 ttk::produceRankArray(cellsRankArray,
480 static_cast<long int *>(ttkUtils::GetVoidPointer(
481 input->GetCellData()->GetGlobalIds())),
482 static_cast<unsigned char *>(ttkUtils::GetVoidPointer(
483 input->GetCellData()->GetArray("vtkGhostType"))),
484 cellNumber, boundingBox);
485
486 vtkNew<vtkIntArray> vtkCellsRankArray{};
487 vtkCellsRankArray->SetName("RankArray");
488 vtkCellsRankArray->SetNumberOfComponents(1);
489 vtkCellsRankArray->SetNumberOfTuples(cellNumber);
490
491 for(int i = 0; i < cellNumber; i++) {
492 vtkCellsRankArray->SetComponent(i, 0, cellsRankArray[i]);
493 }
494
495 input->GetCellData()->AddArray(vtkCellsRankArray);
496 }
497}

524 vtkInformationVector **inputVector,
525 vtkInformationVector *outputVector) {
526 // 1. Pass
527 if(request->Has(vtkCompositeDataPipeline::REQUEST_DATA_OBJECT())) {
528 this->printMsg(
529 "Processing REQUEST_DATA_OBJECT", ttk::debug::Priority::VERBOSE);
530 return this->RequestDataObject(request, inputVector, outputVector);
531 }
532
533 // 2. Pass
534 if(request->Has(vtkCompositeDataPipeline::REQUEST_INFORMATION())) {
535 this->printMsg(
536 "Processing REQUEST_INFORMATION", ttk::debug::Priority::VERBOSE);
537 return this->RequestInformation(request, inputVector, outputVector);
538 }
539
540 // 3. Pass
541 if(request->Has(vtkCompositeDataPipeline::REQUEST_UPDATE_TIME())) {
542 this->printMsg(
543 "Processing REQUEST_UPDATE_TIME", ttk::debug::Priority::VERBOSE);
544 return this->RequestUpdateTime(request, inputVector, outputVector);
545 }
546
547 // 4. Pass
548 if(request->Has(
549 vtkCompositeDataPipeline::REQUEST_TIME_DEPENDENT_INFORMATION())) {
550 this->printMsg("Processing REQUEST_TIME_DEPENDENT_INFORMATION",
551 ttk::debug::Priority::VERBOSE);
552 return this->RequestUpdateTimeDependentInformation(
553 request, inputVector, outputVector);
554 }
555
556 // 5. Pass
557 if(request->Has(vtkCompositeDataPipeline::REQUEST_UPDATE_EXTENT())) {
558 this->printMsg(
559 "Processing REQUEST_UPDATE_EXTENT", ttk::debug::Priority::VERBOSE);
560 return this->RequestUpdateExtent(request, inputVector, outputVector);
561 }
562
563 // 6. Pass
564 if(request->Has(vtkCompositeDataPipeline::REQUEST_DATA_NOT_GENERATED())) {
565 this->printMsg(
566 "Processing REQUEST_DATA_NOT_GENERATED", ttk::debug::Priority::VERBOSE);
567 return this->RequestDataNotGenerated(request, inputVector, outputVector);
568 }
569
570 // 7. Pass
571 if(request->Has(vtkCompositeDataPipeline::REQUEST_DATA())) {
572 this->printMsg("Processing REQUEST_DATA", ttk::debug::Priority::VERBOSE);
573 this->printMsg(ttk::debug::Separator::L0);
574 return this->RequestData(request, inputVector, outputVector);
575 }
576
577 this->printErr("Unsupported pipeline pass:");
578 request->Print(cout);
579
580 return 0;
