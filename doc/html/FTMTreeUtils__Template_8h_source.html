16 template <class dataType>
17 bool FTMTree_MT::isJoinTree() {
18 auto root = this->getRoot();
19 std::vector<idNode> rootChildren;
20 this->getChildren(root, rootChildren);
21 idNode child = rootChildren[0];
22 if(this->isFullMerge()) {
23 dataType min = std::numeric_limits<dataType>::max();
24 for(unsigned int i = 0; i < this->getNumberOfNodes(); ++i) {
25 dataType value = this->getValue<dataType>(i);
26 if(not this->isNodeAlone(i) and value < min) {
27 min = value;
28 child = i;
29 }
30 }
31 }
32 return this->getValue<dataType>(root) > this->getValue<dataType>(child);
33 }

35 template <class dataType>
36 bool FTMTree_MT::isImportantPair(idNode nodeId,
37 double threshold,
38 std::vector<double> &excludeLower,
39 std::vector<double> &excludeHigher) {
40 dataType rootPers = this->getNodePersistence<dataType>(this->getRoot());
41 if(threshold > 1)
42 threshold /= 100.0;
43 threshold = rootPers * threshold;
44 auto pers = this->getNodePersistence<dataType>(nodeId);
45
46 // Excluded pairs
47 bool isExcluded = false;
48 if(excludeLower.size() == excludeHigher.size())
49 for(unsigned i = 0; i < excludeLower.size(); ++i) {
50 isExcluded |= (pers > rootPers * excludeLower[i] / 100.0
51 and pers < rootPers * excludeHigher[i] / 100.0);
52 }
53
54 return pers > threshold and not isExcluded;
55 }

65 bool FTMTree_MT::isParentInconsistent(idNode nodeId) {
66 auto parentBirthDeath
67 = this->getBirthDeath<dataType>(this->getParentSafe(nodeId));
68 dataType parentBirth = std::get<0>(parentBirthDeath);
69 dataType parentDeath = std::get<1>(parentBirthDeath);
70 auto birthDeath = this->getBirthDeath<dataType>(nodeId);
71 dataType birth = std::get<0>(birthDeath);
72 dataType death = std::get<1>(birthDeath);
73 bool parentInconsistent = parentDeath < death or parentBirth > birth;
74 return parentInconsistent;

77 template <class dataType>
78 bool FTMTree_MT::verifyBranchDecompositionInconsistency() {
79 bool inconsistency = false;
80 std::queue<idNode> queue;
81 queue.emplace(this->getRoot());
82 while(!queue.empty()) {
83 idNode node = queue.front();
84 queue.pop();
85 if(!this->isRoot(node) and this->isParentInconsistent<dataType>(node)) {
86 printErr("inconsistency");
87 this->printNode2<dataType>(node);
88 this->printNode2<dataType>(this->getParentSafe(node));
89 inconsistency = true;

91 std::vector<idNode> children;
92 this->getChildren(node, children);
93 for(idNode child : children)
94 queue.emplace(child);
95 }

105 int maxIndex = -1;
106 auto root = this->getRoot();

120 template <class dataType>
121 idNode FTMTree_MT::getLowestNode(idNode nodeStart) {
122 idNode lowestNode = nodeStart;
123 bool isJT = this->isJoinTree<dataType>();
124 dataType bestVal = isJT ? std::numeric_limits<dataType>::max()
125 : std::numeric_limits<dataType>::lowest();
126 std::queue<idNode> queue;
127 queue.emplace(nodeStart);
128 while(!queue.empty()) {
129 idNode node = queue.front();
130 queue.pop();
131 dataType val = this->getValue<dataType>(node);
132 if((val < bestVal and isJT) or (val > bestVal and not isJT)) {
133 lowestNode = node;
134 bestVal = val;
135 }
136 std::vector<idNode> children;
137 this->getChildren(node, children);
138 for(idNode child : children)
139 queue.emplace(child);
140 }
141 return lowestNode;
142 }

216 template <class dataType>
217 dataType FTMTree_MT::getMaximumPersistence() {
218 idNode root = this->getRoot();
219 bool fullMerge = this->isFullMerge();
220
221 // Classic case
222 if(not fullMerge)
223 return this->getNodePersistence<dataType>(this->getRoot());
224
225 // Full merge case
226 dataType maxPers = std::numeric_limits<dataType>::lowest();
227 for(unsigned int i = 0; i < this->getNumberOfNodes(); ++i)
228 if(/*not this->isNodeAlone(i) and*/ this->isNodeOriginDefined(i)
229 and this->getNode(i)->getOrigin() == (int)root)
230 maxPers = std::max(maxPers, this->getNodePersistence<dataType>(i));
231
232 return maxPers;
233 }

264 void FTMTree_MT::getPersistencePairsFromTree(
265 std::vector<std::tuple<idNode, idNode, dataType>> &pairs, bool useBD) {
266 std::vector<idNode> nodes;
267 if(useBD) {
268 for(unsigned int i = 0; i < this->getNumberOfNodes(); ++i)
269 if(!this->isNodeAlone(i) and this->getNode(i)->getOrigin() != (int)i)
270 nodes.push_back(i);
271 } else
272 this->getLeavesFromTree(nodes);
273 for(auto node : nodes) {
274 auto pers = this->getNodePersistence<dataType>(node);
275 pairs.push_back(
276 std::make_tuple(node, this->getNode(node)->getOrigin(), pers));
277 }
278 auto comp = [&](const std::tuple<idNode, idNode, dataType> a,
279 const std::tuple<idNode, idNode, dataType> b) {
280 return std::get<2>(a) < std::get<2>(b);
281 };
282 sort(pairs.begin(), pairs.end(), comp);
283 }

286 std::vector<idNode> FTMTree_MT::getMultiPersOrigins(bool useBD) {
287 std::vector<idNode> multiPersOrigins;
288
289 std::vector<std::tuple<idNode, idNode, dataType>> pairs;
290 this->getPersistencePairsFromTree(pairs, useBD);
291 // std::vector<idNode> origins(this->getNumberOfNodes(), -1);
292 std::vector<std::vector<idNode>> origins(this->getNumberOfNodes());
293 std::vector<bool> birthFound(this->getNumberOfNodes(), false);
294 for(auto pair : pairs) {
295 idNode nodeBirth = std::get<0>(pair);
296 idNode nodeDeath = std::get<1>(pair);
297
298 origins[nodeDeath].push_back(nodeBirth);
299 birthFound[nodeBirth] = true;
300 }
301
302 for(unsigned int i = 0; i < origins.size(); ++i)
303 if(birthFound[i])
304 for(auto node : origins[i])
305 multiPersOrigins.push_back(node);

307 return multiPersOrigins;
