429template <typename dataType>
430dataType ttk::BottleneckDistance::buildMappings(
431 const std::vector<matchingTuple> &inputMatchings,
432 const bool transposeGlobal,
433 const bool transposeLocal,
434 std::vector<matchingTuple> &outputMatchings,
435 const std::vector<int> &m1,
436 const std::vector<int> &m2,
437 int wasserstein) {
438 // Input map permutation (so as to ignore transposition later on)
439 const std::vector<int> map1 = transposeLocal ? m2 : m1;
440 const std::vector<int> map2 = transposeLocal ? m1 : m2;
441
442 std::stringstream msg;
443 dataType addedPersistence = 0;
444 for(int i = 0, s = (int)inputMatchings.size(); i < s; ++i) {
445 matchingTuple t = inputMatchings.at(i);
446 dataType val = abs<dataType>(std::get<2>(t));
447
448 int p1 = std::get<0>(t);
449 int p2 = std::get<1>(t);
450
451 if(p1 >= (int)map1.size() || p1 < 0 || p2 >= (int)map2.size() || p2 < 0) {
452 addedPersistence = (wasserstein > 0 ? addedPersistence + val
453 : std::max(val, addedPersistence));
454 } else {
455 int point1 = map1.at((unsigned long)p1);
456 int point2 = map2.at((unsigned long)p2);
457 bool doTranspose = transposeGlobal ^ transposeLocal;
458
459 matchingTuple newT = doTranspose ? std::make_tuple(point2, point1, val)
460 : std::make_tuple(point1, point2, val);
461
462 outputMatchings.push_back(newT);
463 }
464 }
