587int ExplicitTriangulation::preconditionDistributedCells() {
588 if(this->hasPreconditionedDistributedCells_) {
589 return 0;
590 }
591 if(!ttk::isRunningWithMPI()) {
592 return -1;
593 }
594 if(this->cellGid_ == nullptr) {
595 this->printWrn("Missing global identifiers on cells");
596 return -2;
597 }
598 if(this->ghostCellMask_ == nullptr) {
599 this->printWrn("Missing ghost mask on cells");
600 return -3;
601 }
602
603 // number of local cells (with ghost cells...)
604 const auto nLocCells{this->getNumberOfCells()};
605
606 // global cell id -> local cell id (reverse of this->cellGid_)
607 this->cellGidToLid_.reserve(nLocCells);
608 for(LongSimplexId lcid = 0; lcid < nLocCells; ++lcid) {
609 this->cellGidToLid_[this->cellGid_[lcid]] = lcid;
610 }
611
612 // global cell ids for cell owned by current rank
613 std::vector<LongSimplexId> globalCellsInRank{};
614 globalCellsInRank.reserve(nLocCells);
615
616 for(LongSimplexId lcid = 0; lcid < nLocCells; ++lcid) {
617 if(this->ghostCellMask_[lcid] == 0) {
618 globalCellsInRank.emplace_back(this->cellGid_[lcid]);
619 }
620 }
621
622 // global cell ids per rank (used by rank 0)
623 std::vector<std::vector<LongSimplexId>> cellsRankPerProc{};
624
625 gatherVectors(cellsRankPerProc, globalCellsInRank, 0);
626
627 if(ttk::MPIrank_ == 0) {
628 // global number of cells
629 auto nGlobalCells = cellsRankPerProc[0].size();
630 for(int i = 1; i < MPIsize_; ++i) {
631 nGlobalCells += cellsRankPerProc[i].size();
632 }
633
634 this->cellGidToRank_.resize(nGlobalCells, -1);
635 for(int i = 0; i < MPIsize_; ++i) {
636 for(const auto gcid : cellsRankPerProc[i]) {
637 this->cellGidToRank_[gcid] = i;
638 }
639 }
640 }
641
642 this->hasPreconditionedDistributedCells_ = true;
643
644 return 0;
645}

647template <typename Func>
648int preconditionDistributedIntermediate(size_t &globalCount,
649 const std::vector<int> &cellGidToRank,
650 const Func &processCells) {
651
652 if(!ttk::isRunningWithMPI()) {
653 return -1;
654 }
655
656 const int COMPUTE = 1, STOP = 2;
657
658 if(ttk::MPIrank_ == 0) {
659 size_t gcid{};
660 while(gcid < cellGidToRank.size()) {
661 const auto currRank{cellGidToRank[gcid]};
662 size_t endCurrRank{gcid + 1};
663 if(gcid < cellGidToRank.size() - 1) {
664 for(size_t j = gcid + 1; j < cellGidToRank.size(); ++j) {
665 if(cellGidToRank[j] != currRank) {
666 endCurrRank = j;
667 break;
668 }
669 }
670 if(endCurrRank == gcid + 1 && cellGidToRank[endCurrRank] == currRank) {
671 endCurrRank = cellGidToRank.size();
672 }
673 }
674
675 if(currRank == 0) {
676 // process local edges
677 processCells(gcid, endCurrRank, globalCount);
678 } else {
679 // send to relevant process
680 std::array<unsigned long, 3> toSend{globalCount, gcid, endCurrRank};
681 MPI_Send(toSend.data(), 3, MPI_UNSIGNED_LONG, currRank, COMPUTE,
682 ttk::MPIcomm_);
683 // receive updated edgeCount
684 MPI_Recv(&globalCount, 1, MPI_UNSIGNED_LONG, currRank, MPI_ANY_TAG,
685 ttk::MPIcomm_, MPI_STATUS_IGNORE);
686 }
687
688 gcid = endCurrRank;
689 }
690
691 // send STOP signal to break infinite loop
692 std::array<unsigned long, 3> dummy{};
693 for(int i = 1; i < ttk::MPIsize_; ++i) {
694 MPI_Send(dummy.data(), 3, MPI_UNSIGNED_LONG, i, STOP, ttk::MPIcomm_);
695 }
696
697 } else {
698 while(true) {
699 // receive data
700 std::array<unsigned long, 3> toRecv{};
701 MPI_Status status;
702 MPI_Recv(toRecv.data(), 3, MPI_UNSIGNED_LONG, 0, MPI_ANY_TAG,
703 ttk::MPIcomm_, &status);
704 if(status.MPI_TAG == STOP) {
705 break;
706 }
707 // process local edges
708 globalCount = toRecv[0];
709 processCells(toRecv[1], toRecv[2], globalCount);
710 // send back updated edgeCount to rank 0
711 MPI_Send(&globalCount, 1, MPI_UNSIGNED_LONG, 0, 1, ttk::MPIcomm_);
712 }
713 }
714
715 return 0;
716}

718int ExplicitTriangulation::preconditionDistributedEdges() {
719 if(this->hasPreconditionedDistributedEdges_) {
720 return 0;
721 }
722 if(!ttk::isRunningWithMPI()) {
723 return -1;
724 }
725 if(this->cellGid_ == nullptr) {
726 this->printWrn("Missing global identifiers on cells");
727 return -2;
728 }
729
730 if(this->getDimensionality() != 2 && this->getDimensionality() != 3) {
731 return -3;
732 }
733
734 this->preconditionDistributedCells();
735
736 // allocate memory
737 this->edgeLidToGid_.resize(this->getNumberOfEdgesInternal(), -1);
738 this->edgeGidToLid_.reserve(this->getNumberOfEdgesInternal());
739
740 const auto processEdge = [this](const SimplexId leid, const SimplexId lcid,
741 size_t &edgeCount) {
742 bool alreadyProcessed{false};
743 const auto nStar{this->TTK_TRIANGULATION_INTERNAL(getEdgeStarNumber)(leid)};
744 for(SimplexId i = 0; i < nStar; ++i) {
745 SimplexId sid{-1};
746 this->TTK_TRIANGULATION_INTERNAL(getEdgeStar)(leid, i, sid);
747 if(sid == -1 || sid == lcid) {
748 continue;
749 }
750 // rule: an edge is owned by the cell in its star with the
751 // lowest global id
752 if(this->cellGid_[sid] < this->cellGid_[lcid]) {
753 alreadyProcessed = true;
754 break;
755 }
756 }
757 if(!alreadyProcessed) {
758 this->edgeLidToGid_[leid] = edgeCount;
759 this->edgeGidToLid_[edgeCount] = leid;
760 edgeCount++;
761 }
762 };
763
764 const auto processCellsEdges
765 = [this, &processEdge](
766 const size_t gcid, const size_t endCurrRank, size_t &edgeCount) {
767 for(size_t j = gcid; j < endCurrRank; ++j) {
768 // local cell id
769 const auto lcid{this->cellGidToLid_[j]};
770 for(SimplexId i = 0; i < this->getCellEdgeNumberInternal(lcid); ++i) {
771 SimplexId leid{-1};
772 this->getCellEdgeInternal(lcid, i, leid);
773 processEdge(leid, lcid, edgeCount);
774 }
775 }
776 };
777
778 size_t edgeCount{};
779 preconditionDistributedIntermediate(
780 edgeCount, this->cellGidToRank_, processCellsEdges);
781
782 if(MPIrank_ == 0) {
783 this->printMsg("Domain contains " + std::to_string(edgeCount) + " edges");
784 }
785
786 this->hasPreconditionedDistributedEdges_ = true;
787
788 return 0;

791int ExplicitTriangulation::preconditionDistributedTriangles() {
792 if(this->hasPreconditionedDistributedTriangles_) {
793 return 0;
794 }
795 if(!ttk::isRunningWithMPI()) {
796 return -1;
797 }
798 if(this->cellGid_ == nullptr) {
799 this->printWrn("Missing global identifiers on cells");
800 return -2;
801 }
802
803 if(this->getDimensionality() != 3) {
804 return -3;
805 }
806
807 this->preconditionDistributedCells();
808
809 // allocate memory
810 this->triangleLidToGid_.resize(this->getNumberOfTrianglesInternal(), -1);
811 this->triangleGidToLid_.reserve(this->getNumberOfTrianglesInternal());
812
813 const auto processTriangle =
814 [this](const SimplexId ltid, const SimplexId lcid, size_t &triangleCount) {
815 bool alreadyProcessed{false};
816 const auto nStar{
817 this->TTK_TRIANGULATION_INTERNAL(getTriangleStarNumber)(ltid)};
818 for(SimplexId i = 0; i < nStar; ++i) {
819 SimplexId sid{-1};
820 this->TTK_TRIANGULATION_INTERNAL(getTriangleStar)(ltid, i, sid);
821 if(sid == -1 || sid == lcid) {
822 continue;
823 }
824 // rule: an edge is owned by the cell in its star with the
825 // lowest global id
826 if(this->cellGid_[sid] < this->cellGid_[lcid]) {
827 alreadyProcessed = true;
828 break;
829 }
830 }
831 if(!alreadyProcessed) {
832 this->triangleLidToGid_[ltid] = triangleCount;
833 this->triangleGidToLid_[triangleCount] = ltid;
834 triangleCount++;
835 }
836 };
837
838 const auto processCellsTriangles
839 = [this, &processTriangle](
840 const size_t gcid, const size_t endCurrRank, size_t &triangleCount) {
841 for(size_t j = gcid; j < endCurrRank; ++j) {
842 // local cell id
843 const auto lcid{this->cellGidToLid_[j]};
844 for(SimplexId i = 0; i < this->getCellTriangleNumberInternal(lcid);
845 ++i) {
846 SimplexId ltid{-1};
847 this->getCellTriangleInternal(lcid, i, ltid);
848 processTriangle(ltid, lcid, triangleCount);
849 }
850 }
851 };
852
853 size_t triangleCount{};
854 preconditionDistributedIntermediate(
855 triangleCount, this->cellGidToRank_, processCellsTriangles);
856
857 if(MPIrank_ == 0) {
858 this->printMsg("Domain contains " + std::to_string(triangleCount)
859 + " triangles");
860 }
861
862 this->hasPreconditionedDistributedTriangles_ = true;
863
864 return 0;
865}

916int ExplicitTriangulation::writeToFile(std::ofstream &stream) const {
917
918 // 1. magic bytes (char *)
919 stream.write(this->magicBytes_, std::strlen(this->magicBytes_));
920 // 2. format version (unsigned long)
921 writeBin(stream, this->formatVersion_);
922 // 3. dimensionality (int)
923 const auto dim = this->getDimensionality();
924 writeBin(stream, dim);
925 // 4. number of vertices (SimplexId)
926 const auto nVerts = this->getNumberOfVertices();
927 writeBin(stream, nVerts);
928 // 5. number of edges (SimplexId)
929 const auto edgesNumber = [this, dim]() -> SimplexId {
930 if(dim == 1) {
931 return this->getNumberOfCells();
932 } else if(dim > 1) {
933 return this->edgeList_.size();
934 }
935 return 0;
936 };
937 const auto nEdges = edgesNumber();
938 writeBin(stream, nEdges);
939 // 6. number of triangles (SimplexId, 0 in 1D)
940 const auto trianglesNumber = [this, dim]() -> SimplexId {
941 if(dim == 2) {
942 return this->getNumberOfCells();
943 } else if(dim == 3) {
944 return this->triangleList_.size();
945 }
946 return 0;
947 };
948 const auto nTriangles = trianglesNumber();
949 writeBin(stream, nTriangles);
950 // 7. number of tetrahedron (SimplexId, 0 in 2D)
951 const auto nTetras = dim > 2 ? this->getNumberOfCells() : 0;
952 writeBin(stream, nTetras);
953
954 // only write buffers oustside this->cellArray_ (cellVertex, vertexCoords),
955 // those ones will be provided by VTK
956
957 // fixed-size arrays (in AbstractTriangulation.h)
958
959#define WRITE_FIXED(ARRAY) \
960 if(ARRAY.empty()) { \
961 writeBin(stream, char{0}); \
962 } else { \
963 writeBin(stream, char{1}); \
964 writeBinArray(stream, ARRAY.data(), ARRAY.size()); \
965 }
966
967 // 8. edgeList (SimplexId array)
968 WRITE_FIXED(this->edgeList_);
969 // 9. triangleList (SimplexId array)
970 WRITE_FIXED(this->triangleList_);
971 // 10. triangleEdgeList (SimplexId array)
972 WRITE_FIXED(this->triangleEdgeList_);
973 // 11. tetraEdgeList (SimplexId array)
974 WRITE_FIXED(this->tetraEdgeList_);
975 // 12. tetraTriangleList (SimplexId array)
976 WRITE_FIXED(this->tetraTriangleList_);
977
978 // variable-size arrays (FlatJaggedArray in ExplicitTriangulation.h)
979
980#define WRITE_GUARD(ARRAY) \
981 if(ARRAY.empty()) { \
982 writeBin(stream, char{0}); \
983 return; \
984 } else { \
985 writeBin(stream, char{1}); \
986 }
987
988 const auto write_variable = [&stream](const FlatJaggedArray &arr) {
989 // empty array guard
990 WRITE_GUARD(arr);
991 writeBinArray(stream, arr.offset_ptr(), arr.size() + 1);
992 writeBinArray(stream, arr.get_ptr(0, 0), arr.dataSize());
993 };
994
995 // 13. vertexNeighbors (SimplexId array, offsets then data)
996 write_variable(this->vertexNeighborData_);
997 // 14. cellNeighbors (SimplexId array, offsets then data)
998 write_variable(this->cellNeighborData_);
999 // 15. vertexEdges (SimplexId array, offsets then data)
1000 write_variable(this->vertexEdgeData_);
1001 // 16. vertexTriangles (SimplexId array, offsets then data)
1002 write_variable(this->vertexTriangleData_);
1003 // 17. edgeTriangles (SimplexId array, offsets then data)
1004 write_variable(this->edgeTriangleData_);
1005 // 18. vertexStars (SimplexId array, offsets then data)
1006 write_variable(this->vertexStarData_);
1007 // 19. edgeStars (SimplexId array, offsets then data)
1008 write_variable(this->edgeStarData_);
1009 // 20. triangleStars (SimplexId array, offsets then data)
1010 write_variable(this->triangleStarData_);
1011 // 21. vertexLinks (SimplexId array, offsets then data)
1012 write_variable(this->vertexLinkData_);
1013 // 22. edgeLinks (SimplexId array, offsets then data)
1014 write_variable(this->edgeLinkData_);
1015 // 23. triangleLinks (SimplexId array, offsets then data)
1016 write_variable(this->triangleLinkData_);
1017
1018 const auto write_bool = [&stream](const std::vector<bool> &arr) {
1019 // empty array guard
1020 WRITE_GUARD(arr);
1021 for(size_t i = 0; i < arr.size(); ++i) {
1022 const auto b = static_cast<char>(arr[i]);
1023 writeBin(stream, b);
1024 }
1025 };
1026
1027 // 24. boundary vertices (bool array)
1028 write_bool(this->boundaryVertices_);
1029 // 25. boundary edges (bool array)
1030 write_bool(this->boundaryEdges_);
1031 // 26. boundary triangles (bool array)
1032 write_bool(this->boundaryTriangles_);
1033
1034 return 0;
1035}

1047int ExplicitTriangulation::readFromFile(std::ifstream &stream) {
1048
1049 // 1. magic bytes (char *)
1050 const auto magicBytesLen = std::strlen(this->magicBytes_);
1051 std::vector<char> mBytes(magicBytesLen + 1);
1052 stream.read(mBytes.data(), magicBytesLen);
1053 const auto hasMagicBytes = std::strcmp(mBytes.data(), this->magicBytes_) == 0;
1054 if(!hasMagicBytes) {
1055 this->printErr("Could not find magic bytes in input files!");
1056 this->printErr("Aborting...");
1057 return 0;
1058 }
1059 // 2. format version (unsigned long)
1060 unsigned long version{};
1061 readBin(stream, version);
1062 if(version != this->formatVersion_) {
1063 this->printWrn("File format version (" + std::to_string(version)
1064 + ") and software version ("
1065 + std::to_string(this->formatVersion_) + ") are different!");
1066 }
1067
1068 int dim{};
1069 SimplexId nVerts{}, nEdges{}, nTriangles{}, nTetras{};
1070
1071 // 3. dimensionality (int)
1072 readBin(stream, dim);
1073 // 4. number of vertices (SimplexId)
1074 readBin(stream, nVerts);
1075 // 5. number of edges (SimplexId)
1076 readBin(stream, nEdges);
1077 // 6. number of triangles (SimplexId, 0 in 1D)
1078 readBin(stream, nTriangles);
1079 // 7. number of tetrahedron (SimplexId, 0 in 2D)
1080 readBin(stream, nTetras);
1081
1082 if(dim != this->getDimensionality()) {
1083 this->printErr("Incorrect dimension!");
1084 return 0;
1085 }
1086 if(nVerts != this->getNumberOfVertices()) {
1087 this->printErr("Incorrect number of vertices!");
1088 return 0;
1089 }
1090 if((dim == 2 && nTriangles != this->getNumberOfCells())
1091 || (dim == 3 && nTetras != this->getNumberOfCells())) {
1092 this->printErr("Incorrect number of cells!");
1093 return 0;
1094 }
1095
1096 // fixed-size arrays (in AbstractTriangulation.h)
1097
1098 const auto read_guard = [&stream]() {

1100 readBin(stream, g);
1101 if(g == 0) {
1102 return true;
1103 }
1104 return false;
1105 };
1106
1107#define READ_FIXED(ARRAY, N_ITEMS) \

1109 ARRAY.resize(N_ITEMS); \
1110 readBinArray(stream, ARRAY.data(), N_ITEMS); \
1111 }
1112
1113 // 8. edgeList (SimplexId array)
1114 READ_FIXED(this->edgeList_, nEdges);
1115 // 9. triangleList (SimplexId array)
1116 READ_FIXED(this->triangleList_, nTriangles);
1117 // 10. triangleEdgeList (SimplexId array)
1118 READ_FIXED(this->triangleEdgeList_, nTriangles);
1119 // 11. tetraEdgeList (SimplexId array)
1120 READ_FIXED(this->tetraEdgeList_, nTetras);
1121 // 12. tetraTriangleList (SimplexId array)
1122 READ_FIXED(this->tetraTriangleList_, nTetras);
1123
1124 // variable-size arrays (FlagJaggedArrays in ExplicitTriangulation.h)
1125
1126 const auto read_variable
1127 = [&stream, &read_guard](FlatJaggedArray &arr, const SimplexId n_items) {
1128 // empty array guard
1129 if(read_guard()) {
1130 return;
1131 }
1132 std::vector<SimplexId> offsets{}, data{};
1133 offsets.resize(n_items + 1);
1134 readBinArray(stream, offsets.data(), offsets.size());
1135 data.resize(offsets.back());
1136 readBinArray(stream, data.data(), data.size());
1137 arr.setData(std::move(data), std::move(offsets));
1138 };
1139
1140 // 13. vertexNeighbors (SimplexId array, offsets then data)
1141 read_variable(this->vertexNeighborData_, nVerts);
1142 // 14. cellNeighbors (SimplexId array, offsets then data)
1143 read_variable(this->cellNeighborData_, this->getNumberOfCells());
1144 // 15. vertexEdges (SimplexId array, offsets then data)
1145 read_variable(this->vertexEdgeData_, nVerts);
1146 // 16. vertexTriangles (SimplexId array, offsets then data)
1147 read_variable(this->vertexTriangleData_, nVerts);
1148 // 17. edgeTriangles (SimplexId array, offsets then data)
1149 read_variable(this->edgeTriangleData_, nEdges);
1150 // 18. vertexStars (SimplexId array, offsets then data)
1151 read_variable(this->vertexStarData_, nVerts);
1152 // 19. edgeStars (SimplexId array, offsets then data)
1153 read_variable(this->edgeStarData_, nEdges);
1154 // 20. triangleStars (SimplexId array, offsets then data)
1155 read_variable(this->triangleStarData_, nTriangles);
1156 // 21. vertexLinks (SimplexId array, offsets then data)
1157 read_variable(this->vertexLinkData_, nVerts);
1158 // 22. edgeLinks (SimplexId array, offsets then data)
1159 read_variable(this->edgeLinkData_, nEdges);
1160 // 23. triangleLinks (SimplexId array, offsets then data)
1161 read_variable(this->triangleLinkData_, nTriangles);
1162
1163 const auto read_bool
1164 = [&stream, &read_guard](std::vector<bool> &arr, const SimplexId n_items) {
1165 // empty array guard
1166 if(read_guard()) {
1167 return;
1168 }
1169 // resize vector
1170 arr.resize(n_items);
1171 for(SimplexId i = 0; i < n_items; ++i) {
1172 char b{};
1173 stream.read(&b, sizeof(b));
1174 arr[i] = static_cast<bool>(b);
1175 }
1176 };
1177
1178 // 24. boundary vertices (bool array)
1179 read_bool(this->boundaryVertices_, nVerts);
1180 // 25. boundary edges (bool array)
1181 read_bool(this->boundaryEdges_, nEdges);
1182 // 26. boundary triangles (bool array)
1183 read_bool(this->boundaryTriangles_, nTriangles);
1184
1185 return 0;
1186}
