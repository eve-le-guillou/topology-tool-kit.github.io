70template <class dataType>
71int ttk::LDistance::execute(const dataType *const inputData1,
72 const dataType *const inputData2,
73 dataType *const outputData,
74 const std::string &distanceType,
75 const SimplexId vertexNumber) {
76
77 Timer t;
78 int status;
79
80// Check variables consistency
81#ifndef TTK_ENABLE_KAMIKAZE
82 if(inputData1 == nullptr || inputData2 == nullptr) {
83 return -1;
84 }
85#endif
86
87 if(distanceType == "inf") {
88 status = computeLinf(inputData1, inputData2, outputData, vertexNumber);
89 } else {
90 int n = stoi(distanceType);
91 if(n < 1)
92 return -4;
93
94 status = computeLn(inputData1, inputData2, outputData, n, vertexNumber);
95 }
96
97 if(this->printRes) {
98 this->printMsg(
99 "Data-set processed", 1.0, t.getElapsedTime(), this->threadNumber_);
100 }
101
102 return status;
103}

105template <class dataType>
106int ttk::LDistance::computeLn(const dataType *const input1,
107 const dataType *const input2,
108 dataType *const output,
109 const int n,
110 const ttk::SimplexId vertexNumber) {
111 dataType sum = 0;
112
113// Compute difference for each point.
114#ifdef TTK_ENABLE_OPENMP
115#pragma omp parallel for num_threads(threadNumber_) reduction(+ : sum)
116#endif
117 for(ttk::SimplexId i = 0; i < vertexNumber; ++i) {
118 const dataType diff = abs_diff<dataType>(input1[i], input2[i]);
119 const dataType power = Geometry::pow(diff, n);
120
121 // Careful: huge dataset + huge values
122 // may exceed double capacity.
123 sum += power;
124
125 // Store difference.
126 if(output)
127 output[i] = power;
128 }
129
130 sum = Geometry::pow(sum, 1.0 / (double)n);
131
132 // Affect result.
133 result = (double)sum;
134 if(this->printRes) {
135 this->printMsg("L" + std::to_string(n)
136 + "-distance: " + std::to_string(result));
137 }
138
139 return 0;
140}

142template <class dataType>
143int ttk::LDistance::computeLinf(const dataType *const input1,
144 const dataType *const input2,
145 dataType *const output,
146 const ttk::SimplexId vertexNumber) {
147 if(vertexNumber < 1)
148 return 0;
149
150 dataType maxValue = abs_diff<dataType>(input1[0], input2[0]);
151
152// Compute difference for each point.
153#ifdef TTK_ENABLE_OPENMP
154#pragma omp parallel for num_threads(threadNumber_) reduction(max : maxValue)
155#endif
156 for(ttk::SimplexId i = 1; i < vertexNumber; ++i) {
157 const dataType iter = abs_diff<dataType>(input1[i], input2[i]);
158 if(iter > maxValue)
159 maxValue = iter;
160
161 // Store absolute difference in output.
162 if(output)
163 output[i] = iter;
164 }
165
166 // Affect result.
167 result = (double)maxValue;
168 if(this->printRes) {
169 this->printMsg("Linf-distance: " + std::to_string(result));
170 }
171
172 return 0;
173}
