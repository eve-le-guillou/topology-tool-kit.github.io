178void PersistenceDiagramDistanceMatrix::getDiagramsDistMat(
179 const std::array<size_t, 2> &nInputs,
180 std::vector<std::vector<double>> &distanceMatrix,
181 const std::vector<BidderDiagram<double>> &diags_min,
182 const std::vector<BidderDiagram<double>> &diags_sad,
183 const std::vector<BidderDiagram<double>> &diags_max) const {
184
185 distanceMatrix.resize(nInputs[0]);
186
187#ifdef TTK_ENABLE_OPENMP
188#pragma omp parallel for schedule(dynamic) num_threads(this->threadNumber_)
189#endif // TTK_ENABLE_OPENMP
190 for(size_t i = 0; i < nInputs[0]; ++i) {
191
192 if(nInputs[1] == 0) {
193 distanceMatrix[i].resize(nInputs[0]);
194 // set the matrix diagonal
195 distanceMatrix[i][i] = 0.0;
196 } else {
197 distanceMatrix[i].resize(nInputs[1]);
198 }
199
200 const auto getDist = [&](const size_t a, const size_t b) -> double {
201 double distance{};
202 if(this->do_min_) {
203 auto &dimin = diags_min[a];
204 auto &djmin = diags_min[b];
205 distance += computePowerDistance(dimin, djmin);
206 }
207 if(this->do_sad_) {
208 auto &disad = diags_sad[a];
209 auto &djsad = diags_sad[b];
210 distance += computePowerDistance(disad, djsad);
211 }
212 if(this->do_max_) {
213 auto &dimax = diags_max[a];
214 auto &djmax = diags_max[b];
215 distance += computePowerDistance(dimax, djmax);
216 }
217 return Geometry::pow(distance, 1.0 / this->Wasserstein);
218 };
219
220 if(nInputs[1] == 0) {
221 // square matrix: only compute the upper triangle (i < j < nInputs[0])
222 for(size_t j = i + 1; j < nInputs[0]; ++j) {
223 distanceMatrix[i][j] = getDist(i, j);
224 }
225 } else {
226 // rectangular matrix: compute the whole line/column (0 <= j < nInputs[1])
227 for(size_t j = 0; j < nInputs[1]; ++j) {
228 distanceMatrix[i][j] = getDist(i, j + nInputs[0]);
229 }
230 }
231 }
232
233 if(nInputs[1] == 0) {
234 // square distance matrix is symmetric: complete the lower triangle
235 for(size_t i = 0; i < nInputs[0]; ++i) {
236 for(size_t j = i + 1; j < nInputs[0]; ++j) {
237 distanceMatrix[j][i] = distanceMatrix[i][j];
238 }
239 }
240 }
241}

267void PersistenceDiagramDistanceMatrix::enrichCurrentBidderDiagrams(
268 const std::vector<BidderDiagram<double>> &bidder_diags,
269 std::vector<BidderDiagram<double>> &current_bidder_diags,
270 const std::vector<double> &maxDiagPersistence) const {
271
272 current_bidder_diags.resize(bidder_diags.size());
273 const auto nInputs = current_bidder_diags.size();
274 const auto maxPersistence
275 = *std::max_element(maxDiagPersistence.begin(), maxDiagPersistence.end());
276
277 if(this->Constraint == ConstraintType::ABSOLUTE_PERSISTENCE
278 || this->Constraint == ConstraintType::RELATIVE_PERSISTENCE_PER_DIAG
279 || this->Constraint == ConstraintType::RELATIVE_PERSISTENCE_GLOBAL) {
280 for(size_t i = 0; i < nInputs; ++i) {
281 for(int j = 0; j < bidder_diags[i].size(); ++j) {
282 auto b = bidder_diags[i].get(j);
283
284 if( // filter out pairs below absolute persistence threshold
285 (this->Constraint == ConstraintType::ABSOLUTE_PERSISTENCE
286 && b.getPersistence() > this->MinPersistence)
287 || // filter out pairs below persistence threshold relative to
288 // the most persistent pair *of each diagrams*
289 (this->Constraint == ConstraintType::RELATIVE_PERSISTENCE_PER_DIAG
290 && b.getPersistence() > this->MinPersistence * maxDiagPersistence[i])
291 || // filter out pairs below persistence threshold relative to the
292 // most persistence pair *in all diagrams*
293 (this->Constraint == ConstraintType::RELATIVE_PERSISTENCE_GLOBAL
294 && b.getPersistence() > this->MinPersistence * maxPersistence)) {
295 b.id_ = current_bidder_diags[i].size();
296 b.setPositionInAuction(current_bidder_diags[i].size());
297 current_bidder_diags[i].addBidder(b);
298 }
299 }
300 }
301 return;
302 }
303
304 const double prev_min_persistence = 2.0 * getMostPersistent(bidder_diags);
305 double new_min_persistence = 0.0;
306
307 // 1. Get size of the largest current diagram, deduce the maximal number
308 // of points to append
309 size_t max_diagram_size = 0;
310 for(const auto &diag : current_bidder_diags) {
311 max_diagram_size
312 = std::max(static_cast<size_t>(diag.size()), max_diagram_size);
313 }
314 size_t max_points_to_add = std::max(
315 this->MaxNumberOfPairs, this->MaxNumberOfPairs + max_diagram_size / 10);
316 // 2. Get which points can be added, deduce the new minimal persistence
317 std::vector<std::vector<int>> candidates_to_be_added(nInputs);
318 std::vector<std::vector<size_t>> idx(nInputs);
319
320 for(size_t i = 0; i < nInputs; i++) {
321 double local_min_persistence = std::numeric_limits<double>::min();
322 std::vector<double> persistences;
323 for(int j = 0; j < bidder_diags[i].size(); j++) {
324 const auto b = bidder_diags[i].get(j);
325 double persistence = b.getPersistence();
326 if(persistence >= 0.0 && persistence <= prev_min_persistence) {
327 candidates_to_be_added[i].emplace_back(j);
328 idx[i].emplace_back(idx[i].size());
329 persistences.emplace_back(persistence);
330 }
331 }
332 const auto cmp = [&persistences](const size_t a, const size_t b) {
333 return ((persistences[a] > persistences[b])
334 || ((persistences[a] == persistences[b]) && (a > b)));
335 };
336 std::sort(idx[i].begin(), idx[i].end(), cmp);
337 const auto size = candidates_to_be_added[i].size();
338 if(size >= max_points_to_add) {
339 double last_persistence_added
340 = persistences[idx[i][max_points_to_add - 1]];
341 if(last_persistence_added > local_min_persistence) {
342 local_min_persistence = last_persistence_added;
343 }
344 }
345 if(i == 0) {
346 new_min_persistence = local_min_persistence;
347 } else {
348 if(local_min_persistence < new_min_persistence) {
349 new_min_persistence = local_min_persistence;
350 }
351 }
352 // 3. Add the points to the current diagrams
353 const auto s = candidates_to_be_added[i].size();
354 for(size_t j = 0; j < std::min(max_points_to_add, s); j++) {
355 auto b = bidder_diags[i].get(candidates_to_be_added[i][idx[i][j]]);
356 const double persistence = b.getPersistence();
357 if(persistence >= new_min_persistence) {
358 b.id_ = current_bidder_diags[i].size();
359 b.setPositionInAuction(current_bidder_diags[i].size());
360 current_bidder_diags[i].addBidder(b);
361 }
362 }
363 }
