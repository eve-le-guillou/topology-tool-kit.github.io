215 void *scalars,

218
219 if(!triangulation)

222 return -2;

224 _inpFldScalars = scalars;
225

227 if(_inpDimMax < 2 || _inpDimMax > 3)
228 return -3;

244 triangulation->preconditionEdges();
245 // for getEdgeTriangleNumber, getEdgeTriangle
246 triangulation->preconditionEdgeTriangles();
247 // for getTriangleEdge
248 triangulation->preconditionTriangleEdges();

302float ttk::ContourAroundPoint::compDist2(SimplexId v, SimplexId p) const {
303 float vx, vy, vz;
304 reinterpret_cast<Triang *>(_inpFldTriang)->getVertexPoint(v, vx, vy, vz);
305 const auto pCoords = &(_inpPtsCoords[p * 3]);
306 const float dx = pCoords[0] - vx;
307 const float dy = pCoords[1] - vy;
308 const float dz = pCoords[2] - vz;
309 return dx * dx + dy * dy + dz * dz;
310}

313template <class Triang>
314ttk::SimplexId ttk::ContourAroundPoint::findInpFldVert(SimplexId p) const {
315 // This implementation is based on a naive nearest neighbor search
316 SimplexId minv = 0;
317 float mind = compDist2(minv, p);
318 const auto nv
319 = reinterpret_cast<Triang *>(_inpFldTriang)->getNumberOfVertices();
320 for(SimplexId v = 1; v < nv; ++v) {
321 const auto d = compDist2(v, p);
322 if(d < mind) {
323 minv = v;
324 mind = d;
325 }
326 }
327 return minv;
328}

331template <typename scalarT, class Triang>
332void ttk::ContourAroundPoint::handleOneInpPt(SimplexId vBeg,
333 float isoval,
334 int flag,
335 float scalar) const {
336 auto triang = reinterpret_cast<Triang *>(_inpFldTriang);
337 auto inpScalars = reinterpret_cast<const scalarT *>(_inpFldScalars);
338 const bool vBegIsMin = flag == 0;
339
340 // The general idea of the algorithm is to start a (breadth-first) search
341 // on the triangulation starting at the input vertex.
342 // Until the isocontour is met, all vertices are collected to approximate
343 // the (weighted) centroid of the enclosed area/volume.
344 // The edges leading outside of the region are also collected, to compute
345 // the actual isocontour (that intersects these edges).
346
347 auto innerVerts = std::vector<SimplexId>{vBeg};
348 std::set<SimplexId> xEdges; // set data-structure is needed in addOutput
349
350 struct VertEdge {
351 SimplexId v; // where am i
352 SimplexId e; // how did i get here
353 };
354 std::vector<VertEdge> q;
355 std::set<SimplexId> handledEdges;
356
357 auto enqueueNeighbors = [triang, &q, &handledEdges](SimplexId vSrc) {
358 const auto n = triang->getVertexEdgeNumber(vSrc);
359 for(SimplexId i = 0; i < n; ++i) {
360 SimplexId e;
361 triang->getVertexEdge(vSrc, i, e);
362 if(handledEdges.count(e))
363 continue;
364 handledEdges.insert(e);
365 // We assume the local indexing of vertex neighbors is the same
366 // as for the edges (edge 0 leads to neighbor 0, etc.).
367 SimplexId vTgt;
368 triang->getVertexNeighbor(vSrc, i, vTgt);
369 q.push_back({vTgt, e});
370 }
371 };
372
373 enqueueNeighbors(vBeg);
374 while(!q.empty()) {
375 const auto ve = q.back();
376 q.pop_back();
377 const auto v = ve.v;
378 const bool vIsAboveIso = inpScalars[v] > isoval;
379 if(vBegIsMin == vIsAboveIso) { // crossed the contour
380 xEdges.insert(ve.e);
381 } else {
382 innerVerts.push_back(v);
383 enqueueNeighbors(v);
384 }
385 }
386
387 if(innerVerts.size() < _sizeMin)
388 return;
389 extendOutFld<scalarT>(xEdges, isoval, flag);
390 extendOutPts<scalarT>(innerVerts, isoval, flag, scalar);
391}

394template <typename scalarT, class Triang>
395void ttk::ContourAroundPoint::extendOutFld(const std::set<SimplexId> &inpEdges,
396 float isoval,
397 int flag) const {
398 auto triang = reinterpret_cast<Triang *>(_inpFldTriang);
399 auto inpScalars = reinterpret_cast<const scalarT *>(_inpFldScalars);
400
401 // Every edge e of the given edges maps to one output vertex v;
402 // v is connected to a number of other such vertices.
403 // Zero if e is part of an "antenna", one if e is part of the boundary
404 // of the grid, two in the default 2D case, possibly more in 3D.
405 // These neighbors of v are reached by looking at the triangles that e is a
406 // part of.
407
408 std::map<SimplexId, SimplexId> inpe2outv;
409
410 auto getOrMakeOutVert = [&](SimplexId e) {
411 if(inpe2outv.count(e))
412 return inpe2outv[e];
413
414 // geometry of the vertex (linear interpolation on e)
415 const SimplexId v = _outContoursScalars.size();
416 inpe2outv[e] = v;
417
418 SimplexId p;
419 triang->getEdgeVertex(e, 0, p);
420 SimplexId q;
421 triang->getEdgeVertex(e, 1, q);
422
423 const float pVal = static_cast<float>(inpScalars[p]);
424 const float qVal = static_cast<float>(inpScalars[q]);
425 const double pFac = (qVal - isoval) / (qVal - pVal);
426 const double qFac = 1 - pFac;
427
428 float px, py, pz;
429 triang->getVertexPoint(p, px, py, pz);
430 float qx, qy, qz;
431 triang->getVertexPoint(q, qx, qy, qz);
432 const float x = px * pFac + qx * qFac;
433 const float y = py * pFac + qy * qFac;
434 const float z = pz * pFac + qz * qFac;
435 _outContoursCoords.push_back(x);
436 _outContoursCoords.push_back(y);
437 _outContoursCoords.push_back(z);
438
439 _outContoursScalars.push_back(isoval);
440 _outContoursFlags.push_back(flag);
441 return v;
442 };
443
444 // Each input triangle can contribute to at most one output edge
445 std::set<SimplexId> handledTris;
446
447 if(_inpDimMax == 3)
448 printWrn("Currently only the *edges* of contour surfaces are computed");
449
450 for(const auto e : inpEdges) {
451 const auto nTriLoc = triang->getEdgeTriangleNumber(e);
452 if(nTriLoc == 0)
453 continue; // we do not output "contour points" (0D cells)
454
455 const auto v = getOrMakeOutVert(e);
456
457 for(SimplexId tLoc = 0; tLoc < nTriLoc; ++tLoc) {
458 SimplexId tGlo;
459 triang->getEdgeTriangle(e, tLoc, tGlo);
460 if(handledTris.count(tGlo))
461 continue;
462
463 handledTris.insert(tGlo);
464 // Find the other edge that contains the isovalue
465 for(SimplexId eLoc = 0; eLoc < 3; ++eLoc) {
466 SimplexId eGlo;
467 triang->getTriangleEdge(tGlo, eLoc, eGlo);
468 if(eGlo == e || !inpEdges.count(eGlo))
469 continue;
470
471 // new output edge from v(e) to v(eGlo)
472 _outContoursCinfos.push_back(2);
473 _outContoursCinfos.push_back(v);
474 _outContoursCinfos.push_back(getOrMakeOutVert(eGlo));
475 ++_outContoursNc;
476 } // END target edge
477 } // END triangle
478 } // END source edge
479}

482template <typename scalarT, class Triang>
483void ttk::ContourAroundPoint::extendOutPts(
484 const std::vector<SimplexId> &vertices,
485 float isoval,
486 int flag,
487 float extremeVal) const {
488
489 auto triang = reinterpret_cast<Triang *>(_inpFldTriang);
490 auto inpScalars = reinterpret_cast<const scalarT *>(_inpFldScalars);
491
492 const double radius = _radius;
493 const bool spherical = radius > 0.;
494
495 // We weight the vertices based on the difference to the extreme value.
496 // Vertices that are close to the extreme value get much weight,
497 // vertices that are close to the isovalue get little weight.
498 const double scalarDiffMax = std::abs(isoval - extremeVal);
499
500 // For now, we use a "full cosine" kernel
501 const double kernelInputScaler = M_PI / scalarDiffMax;
502 // d=0 ... cos(0)=1 ... w=1, d=scalarDiffMax ... cos(pi)=-1 ... w = 0
503 auto scalar_w = [kernelInputScaler](double d) {
504 return (std::cos(d * kernelInputScaler) + 1.) / 2.;
505 };
506
507 // Additionally, each vertex is weighted considering the size of the domain it
508 // "represents" (inversely proportional to the sampling density).
509 // NOTE Currently we assume that whenever we have a spherical domain,
510 // it is given on a regular lon-lat grid.
511 const bool regularLonLat = spherical;
512
513 // We collect the 4D samples and weights to compute the weighted average
514 // in a separate function
515 const auto numVerts = vertices.size();
516 auto pts4d = std::vector<std::array<float, 4>>(numVerts);
517 auto ws = std::vector<double>(numVerts);
518 for(std::size_t i = 0; i < numVerts; ++i) {
519
520 const auto v = vertices[i];
521 const scalarT vSca = inpScalars[v];
522 // Because we only use symmetric kernels for the weighting, the sign of
523 // the difference does not matter.
524 const double scalarDiff = vSca - extremeVal;
525 const double scalarW = scalar_w(scalarDiff);
526
527 float x, y, z;
528 triang->getVertexPoint(v, x, y, z);
529 pts4d[i] = {x, y, z, static_cast<float>(vSca)};
530
531 if(!regularLonLat) {
532 if(_inpDimMax > 2) {
533 // If someone has a lot of time, they can implement the case for a
534 // tetrahedralization.
535 printWrn("Vertex weighting does not consider vertex density of this "
536 "volumetric data");
537 ws[i] = scalarW;
538 } else {
539 // This is just *some* code to fill this gap until it is replaced by sth
540 // better; essentially this module was written for regular spherical
541 // grids.
542 // printMsg("Irregular triangulation");
543
544 // Compute the distance between points a and b.
545 static const auto compDist
546 = [](float ax, float ay, float az, float bx, float by, float bz) {
547 const auto dx = bx - ax;
548 const auto dy = by - ay;
549 const auto dz = bz - az;
550 return std::sqrt(dx * dx + dy * dy + dz * dz);
551 };
552
553 // We use Heron's formula for the area of a triangle, only using the
554 // side lengths a, b, c.
555 static const auto compArea = [](double a, double b, double c) {
556 const double s = (a + b + c) / 2.; // half the perimeter
557 // return std::sqrt(s*(s-a)*(s-b)*(s-c));
558 return std::sqrt(
559 3 * s * s - s * a - s * b
560 - s * c); // numerically more stable (with tested datasets)
561 };
562
563 // We assume a closed triangulation (i.e. no holes) and the vertex
564 // neighbors being ordered in a fan-like fashion.
565 double area = 0.;
566 float x2, y2, z2, x3, y3, z3;
567 SimplexId u;
568 triang->getVertexNeighbor(v, 0, u);
569 triang->getVertexPoint(u, x2, y2, z2);
570 double a = compDist(x, y, z, x2, y2, z2);
571
572 // Backup the first edge for the last triangle.
573 const float x2Bak = x2, y2Bak = y2, z2Bak = z2;
574 const double aBak = a;
575
576 const SimplexId n = triang->getVertexNeighborNumber(v);
577 for(SimplexId j = 1; j < n; ++j) {
578 triang->getVertexNeighbor(v, j, u);
579 triang->getVertexPoint(u, x3, y3, z3);
580 const double b = compDist(x, y, z, x3, y3, z3);
581 const double c = compDist(x2, y2, z2, x3, y3, z3);
582 area += compArea(a, b, c);
583 x2 = x3;
584 y2 = y3;
585 z2 = z3;
586 a = b;
587 }
588
589 const double b = aBak;
590 const double c = compDist(x2, y2, z2, x2Bak, y2Bak, z2Bak);
591 area += compArea(a, b, c);
592 // std::cout<<"area: "<<area<<std::endl;
593 ws[i] = scalarW * area;
594 }
595 } else {
596 // spatial weight 1 at equator, 0 at a pole
597 const double latInRad = std::asin(z / radius);
598 // std::asin is in [-pi/2,+pi/2] => equator at 0 => fits
599 const double spatialW = std::cos(latInRad);
600 ws[i] = scalarW * spatialW;
601 // Why product and not mean of the weights?
602 // -> The resulting weight should be 0 if the vertex is either outside
603 // of the sub/super-level set (scalar weight 0) or its cell size is 0
604 // (spatial weight 0)
605 }
606 }
607
608 const auto res = average(pts4d, ws);
609 auto x = res[0];
610 auto y = res[1];
611 auto z = res[2];
612
613 if(spherical) {
614 const auto radiusCur = std::sqrt(x * x + y * y + z * z);
615 const auto radiusScaler = radius / radiusCur;
616 x *= radiusScaler;
617 y *= radiusScaler;
618 z *= radiusScaler;
619 }
620
621 // std::cout<<x<<","<<y<<","<<z<<std::endl;
622 _outCentroidsCoords.push_back(static_cast<float>(x));
623 _outCentroidsCoords.push_back(static_cast<float>(y));
624 _outCentroidsCoords.push_back(static_cast<float>(z));
625 _outCentroidsScalars.push_back(static_cast<float>(res[3]));
626 _outCentroidsFlags.push_back(flag);
627}

630template <class PointIterable, class WeightIterable>
631std::array<double, std::tuple_size<typename PointIterable::value_type>::value>
632 ttk::ContourAroundPoint::average(const PointIterable &pts,
633 const WeightIterable &ws) {
634
635 constexpr auto numComponents
636 = std::tuple_size<typename PointIterable::value_type>::value;
637 std::array<double, numComponents> res{};
638 // as of C++14 there is no (simple) compile time "constexpr for loop"
639#ifndef NDEBUG
640 for(std::size_t j = 0; j < numComponents; ++j)
641 // assert(std::get<j>(res) == 0.);
642 assert(res[j] == 0.);
643#endif
644 double wSum = 0.;
645
646 const auto numPts = pts.size();
647 assert(ws.size() == numPts);
648
649 for(std::size_t i = 0; i < numPts; ++i) {
650 auto &pt = pts[i];
651 const auto w = ws[i];
652 wSum += w;
653 for(std::size_t j = 0; j < numComponents; ++j)
654 // std::get<j>(res) += std::get<j>(pt) * w;
655 res[j] += pt[j] * w;
656 }
657
658 assert(wSum != 0.);
659 for(std::size_t j = 0; j < numComponents; ++j)
660 // std::get<j>(res) /= wSum;
661 res[j] /= wSum;
662
663 return res;
664}
