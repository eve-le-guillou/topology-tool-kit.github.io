171 pairTypes->SetName("PairType");
172 pairTypes->SetNumberOfTuples(nCells);

174#ifdef TTK_ENABLE_OPENMP

176#endif // TTK_ENABLE_OPENMP
177 for(size_t i = 0; i < nCells; ++i) {
178 offsets->SetTuple1(i, 2 * i);
179 // each glyph/line has unique points
180 connectivity->SetTuple1(2 * i, 2 * i);
181 connectivity->SetTuple1(2 * i + 1, 2 * i + 1);
182 pairTypes->SetTuple1(i, gradientGlyphs_cells_pairTypes[i]);
183 }
184 offsets->SetTuple1(nCells, connectivity->GetNumberOfTuples());

192

195 this->printErr("In outputGradientGlyphs point or cell data");

197 }
198#endif
199
200 pointData->AddArray(pairOrigins);
201 cellData->SetScalars(pairTypes);
202
