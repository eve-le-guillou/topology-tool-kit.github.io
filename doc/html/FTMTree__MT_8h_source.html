897
898 MergeTree(const ftm::Scalars &scalarsT, ftm::Params paramsT)
899 : scalars(scalarsT), params(paramsT),
900 tree(&params, &scalars, params.treeType) {
901 tree.makeAlloc();
902 for(unsigned int i = 0; i < tree.getNumberOfNodes(); ++i)
903 scalarsValues.push_back(tree.getValue<dataType>(i));
904 scalars.values = (void *)(scalarsValues.data());

911 tree(&params, &scalars, params.treeType) {
912 tree.makeAlloc();
913 scalars.values = (void *)(scalarsValues.data());
914 }

917 // Copy scalars
918 scalars.size = mt.scalars.size;
919 scalarsValues = std::vector<dataType>(mt.scalarsValues);
920 scalars.values = (void *)(scalarsValues.data());
921
922 // Copy params
923 params.treeType = mt.params.treeType;
924
925 // Copy tree
926 tree.~FTMTree_MT();
927 tree.makeAlloc();
928 tree.copyMergeTreeStructure(const_cast<FTMTree_MT *>(&(mt.tree)));
929 }

932 : scalars(mt.scalars), scalarsValues(mt.scalarsValues),
933 params(mt.params), tree(&params, &scalars, params.treeType) {
934 copy(mt);
935 }
