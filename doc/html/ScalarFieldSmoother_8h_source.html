88template <class dataType, class triangulationType>
89int ttk::ScalarFieldSmoother::smooth(const triangulationType *triangulation,
90 const int &numberOfIterations) const {
91
92 Timer t;
93
94#ifndef TTK_ENABLE_KAMIKAZE
95 if(!triangulation)
96 return -1;
97 if(!dimensionNumber_)
98 return -2;
99 if(!inputData_)
100 return -3;
101 if(!outputData_)
102 return -4;
103#endif
104
105#if TTK_ENABLE_MPI
106 bool useMPI{false};
107 if(ttk::isRunningWithMPI() && triangulation->getVertRankArray() != nullptr
108 && triangulation->getVertsGlobalIds() != nullptr)
109 useMPI = true;
110#endif
111 SimplexId vertexNumber = triangulation->getNumberOfVertices();
112
113 std::vector<dataType> tmpData(vertexNumber * dimensionNumber_, 0);
114
115 dataType *outputData = (dataType *)outputData_;
116 dataType *inputData = (dataType *)inputData_;
117 // init the output
118 for(SimplexId i = 0; i < vertexNumber; i++) {
119 for(int j = 0; j < dimensionNumber_; j++) {
120 outputData[dimensionNumber_ * i + j]
121 = inputData[dimensionNumber_ * i + j];
122 }
123 }
124
125 printMsg("Smoothing " + std::to_string(vertexNumber) + " vertices", 0, 0,
126 threadNumber_, ttk::debug::LineMode::REPLACE);
127
128 int timeBuckets = 10;
129 if(numberOfIterations < timeBuckets)
130 timeBuckets = numberOfIterations;
131
132 for(int it = 0; it < numberOfIterations; it++) {
133#ifdef TTK_ENABLE_OPENMP
134#pragma omp parallel for num_threads(threadNumber_)
135#endif
136 for(SimplexId i = 0; i < vertexNumber; i++) {
137
138 // avoid to process masked vertices
139 if(mask_ != nullptr && mask_[i] == 0)
140 continue;
141
142 for(int j = 0; j < dimensionNumber_; j++) {
143 const auto curr{dimensionNumber_ * i + j};
144 tmpData[curr] = outputData[curr];
145
146 const auto neighborNumber = triangulation->getVertexNeighborNumber(i);
147 for(SimplexId k = 0; k < neighborNumber; k++) {
148 SimplexId neighborId = -1;
149 triangulation->getVertexNeighbor(i, k, neighborId);
150 tmpData[curr] += outputData[dimensionNumber_ * (neighborId) + j];
151 }
152 tmpData[curr] /= static_cast<double>(neighborNumber + 1);
153 }
154 }
155
156 if(numberOfIterations) {
157 // assign the tmpData back to the output
158 for(SimplexId i = 0; i < vertexNumber; i++) {
159 for(int j = 0; j < dimensionNumber_; j++) {
160 // only set value for unmasked points
161 if(mask_ == nullptr || mask_[i] != 0) {
162 outputData[dimensionNumber_ * i + j]
163 = tmpData[dimensionNumber_ * i + j];
164 }
165 }
166 }
167 }
168#if TTK_ENABLE_MPI
169 if(useMPI) {
170 // after each iteration we need to exchange the ghostcell values with our
171 // neighbors
172 exchangeGhostCells<dataType, SimplexId>(
173 outputData, triangulation->getVertRankArray(),
174 triangulation->getVertsGlobalIds(),
175 triangulation->getVertexGlobalIdMap(), vertexNumber, ttk::MPIcomm_,
176 dimensionNumber_);
177 }
178#endif
179
180 if(debugLevel_ >= (int)(debug::Priority::INFO)) {
181 if(!(it % ((numberOfIterations) / timeBuckets))) {
182 printMsg("Smoothing " + std::to_string(vertexNumber) + " vertices",
183 (it / (float)numberOfIterations), t.getElapsedTime(),
184 threadNumber_, debug::LineMode::REPLACE);
185 }
186 }
187 }
188
189 printMsg("Smoothed " + std::to_string(vertexNumber) + " vertices", 1,
190 t.getElapsedTime(), threadNumber_);
191
192 return 0;
193}
