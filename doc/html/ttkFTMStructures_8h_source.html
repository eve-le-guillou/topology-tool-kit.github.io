47 inline static CriticalType
48 getNodeType(FTMTree_MT &tree, const idNode nodeId, Params params) {
49 const Node *node = tree.getNode(nodeId);
50 int upDegree{};
51 int downDegree{};
52 if(params.treeType == TreeType::Join
53 or params.treeType == TreeType::Contour) {
54 upDegree = node->getNumberOfUpSuperArcs();
55 downDegree = node->getNumberOfDownSuperArcs();
56 } else {
57 downDegree = node->getNumberOfUpSuperArcs();
58 upDegree = node->getNumberOfDownSuperArcs();
59 }
60 int degree = upDegree + downDegree;
61
62 // saddle point
63 if(degree > 1) {
64 if(upDegree == 2 and downDegree == 1)
65 return CriticalType::Saddle2;
66 else if(upDegree == 1 and downDegree == 2)
67 return CriticalType::Saddle1;
68 else if(upDegree == 1 and downDegree == 1)
69 return CriticalType::Regular;
70 else
71 return CriticalType::Degenerate;
72 }
73 // local extremum
74 else {
75 if(upDegree)
76 return CriticalType::Local_minimum;
77 else
78 return CriticalType::Local_maximum;
79 }
80 }

382 void fillArrayPoint(const idSuperArc arcId,
383 LocalFTM &l_tree,
384 Triangulation *triangulation,
385 vtkDataArray *idMapper,
386 Params params) {
387 if(!params.segm)
388 return;
389
390 FTMTree_MT *tree = l_tree.tree.getTree(params.treeType);
391 const idNode idOffset = l_tree.offset;
392 SuperArc *arc = tree->getSuperArc(arcId);
393
394 const idNode upNodeId = arc->getUpNodeId();
395 const Node *upNode = tree->getNode(upNodeId);
396 const SimplexId l_upVertexId = upNode->getVertexId();
397 const SimplexId g_upVertexId = idMapper->GetTuple1(l_upVertexId);
398 const CriticalType upNodeType = getNodeType(*tree, upNodeId, params);
399 float coordUp[3];
400 triangulation->getVertexPoint(
401 l_upVertexId, coordUp[0], coordUp[1], coordUp[2]);
402
403 const idNode downNodeId = arc->getDownNodeId();
404 const Node *downNode = tree->getNode(downNodeId);
405 const SimplexId l_downVertexId = downNode->getVertexId();
406 const SimplexId g_downVertexId = idMapper->GetTuple1(l_downVertexId);
407 const CriticalType downNodeType
408 = getNodeType(*tree, downNodeId, params);
409 float coordDown[3];
410 triangulation->getVertexPoint(
411 l_downVertexId, coordDown[0], coordDown[1], coordDown[2]);
412
413 const SimplexId regionSize
414 = tree->getSuperArc(arcId)->getNumberOfRegularNodes();
415 const double regionSpan = Geometry::distance(coordUp, coordDown);
416
417 idSuperArc nid = arc->getNormalizedId();
418
419 ArcType regionType;
420 // RegionType
421 if((upNodeType == CriticalType::Local_minimum
422 && downNodeType == CriticalType::Local_maximum)
423 || (upNodeType == CriticalType::Local_minimum
424 || downNodeType == CriticalType::Local_minimum)) {
425 regionType = ArcType::Min_arc;
426 } else if(upNodeType == CriticalType::Local_maximum
427 || downNodeType == CriticalType::Local_maximum) {
428 regionType = ArcType::Max_arc;
429 } else if(upNodeType == CriticalType::Saddle1
430 && downNodeType == CriticalType::Saddle1) {
431 regionType = ArcType::Saddle1_arc;
432 } else if(upNodeType == CriticalType::Saddle2
433 && downNodeType == CriticalType::Saddle2) {
434 regionType = ArcType::Saddle2_arc;
435 } else {
436 regionType = ArcType::Saddle1_saddle2_arc;
437 }
438
439 // fill extrema and regular verts of this arc
440
441 // critical points
442 if(params.normalize) {
443 ids->SetTuple1(g_upVertexId, idOffset + nid);
444 ids->SetTuple1(g_downVertexId, idOffset + nid);
445 } else {
446 ids->SetTuple1(g_upVertexId, idOffset + arcId);
447 ids->SetTuple1(g_downVertexId, idOffset + arcId);
448 }
449
450 if(params.advStats) {
451 sizeRegion->SetTuple1(g_upVertexId, regionSize);
452 sizeRegion->SetTuple1(g_downVertexId, regionSize);
453 spanRegion->SetTuple1(g_upVertexId, regionSpan);
454 spanRegion->SetTuple1(g_downVertexId, regionSpan);
455 }
456 typeRegion->SetTuple1(g_upVertexId, static_cast<char>(regionType));
457 typeRegion->SetTuple1(g_downVertexId, static_cast<char>(regionType));
458
459 // regular nodes
460 for(const SimplexId l_vertexId : *arc) {
461 const SimplexId g_vertexId = idMapper->GetTuple1(l_vertexId);
462 if(params.normalize) {
463 ids->SetTuple1(g_vertexId, idOffset + nid);
464 } else {
465 ids->SetTuple1(g_vertexId, idOffset + arcId);
466 }
467 if(params.advStats) {
468 sizeRegion->SetTuple1(g_vertexId, regionSize);
469 spanRegion->SetTuple1(g_vertexId, regionSpan);
470 }
471 typeRegion->SetTuple1(g_vertexId, static_cast<char>(regionType));
472 }
