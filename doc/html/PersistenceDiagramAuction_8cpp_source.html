3void ttk::PersistenceDiagramAuction::runAuctionRound(int &n_biddings,
4 const int kdt_index) {
5 double max_price = getMaximalPrice();
6 double epsilon = epsilon_;
7 if(epsilon_ < 1e-6 * max_price) {
8 // Risks of floating point limits reached...
9 epsilon = 1e-6 * max_price;
10 }
11 while(unassignedBidders_.size() > 0) {
12 n_biddings++;
13 int pos = unassignedBidders_.front();
14 Bidder &b = this->bidders_[pos];
15 unassignedBidders_.pop();
16
17 GoodDiagram &all_goods = b.isDiagonal() ? diagonal_goods_ : goods_;
18 Good &twin_good = b.id_ >= 0 ? diagonal_goods_[b.id_] : goods_[-b.id_ - 1];
19 // double eps = epsilon_*(1+0.05*n_biddings/bidders_.size());
20 int idx_reassigned;
21 if(b.isDiagonal()) {
22 if(use_kdt_) {
23 idx_reassigned = b.runDiagonalKDTBidding(
24 &all_goods, twin_good, wasserstein_, epsilon, geometricalFactor_,
25 correspondance_kdt_map_, diagonal_queue_, kdt_index);
26 } else {
27 idx_reassigned
28 = b.runDiagonalBidding(&all_goods, twin_good, wasserstein_, epsilon,
29 geometricalFactor_, diagonal_queue_);
30 }
31 } else {
32 if(use_kdt_) {
33 // We can use the kd-tree to speed up the search
34 idx_reassigned
35 = b.runKDTBidding(&all_goods, twin_good, wasserstein_, epsilon,
36 geometricalFactor_, &kdt_, kdt_index);
37 } else {
38 idx_reassigned = b.runBidding(
39 &all_goods, twin_good, wasserstein_, epsilon, geometricalFactor_);
40 }
41 }
42 if(idx_reassigned >= 0) {
43 Bidder &reassigned = bidders_[idx_reassigned];
44 reassigned.resetProperty();
45 unassignedBidders_.push(idx_reassigned);
46 }
47 }
48}

73 double wassersteinDistance = 0;
74 for(size_t i = 0; i < bidders_.size(); i++) {
75 Bidder &b = bidders_[i];

77 int good_id = b.getProperty().id_;

79

91 MatchingType t = std::make_tuple(i, good_id, cost);

96 } else {
97 // b is diagonal
98 const Good &g = b.getProperty();
99 double cost
100 = 2.0 * Geometry::pow(std::abs((g.y_ - g.x_) / 2.0), wasserstein_);
101 if(get_diagonal_matches) {
102 MatchingType t = std::make_tuple(b.id_, g.id_, cost);
103 matchings.emplace_back(t);
104 }
105 wassersteinDistance += cost;
106 }

127double ttk::PersistenceDiagramAuctionActor::cost(
128 const PersistenceDiagramAuctionActor &g,
129 const int wasserstein,
130 const double geometricalFactor) const {
131
132 if(is_diagonal_ && g.isDiagonal()) {
133 return 0;
134 } else if(is_diagonal_) {
135 return geometricalFactor
136 * (2 * Geometry::pow(std::abs(g.y_ / 2 - g.x_ / 2), wasserstein))
137 + (1 - geometricalFactor) * getPairGeometricalLength(wasserstein);
138 } else if(g.isDiagonal()) {
139 return geometricalFactor
140 * (2 * Geometry::pow(std::abs(y_ / 2 - x_ / 2), wasserstein))
141 + (1 - geometricalFactor) * g.getPairGeometricalLength(wasserstein);
142 } else {
143 return geometricalFactor
144 * (Geometry::pow(std::abs(x_ - g.x_), wasserstein)
145 + Geometry::pow(std::abs(y_ - g.y_), wasserstein))
146 + (1 - geometricalFactor)
147 * (Geometry::pow(
148 std::abs(coords_[0] - g.coords_[0]), wasserstein)
149 + Geometry::pow(
150 std::abs(coords_[1] - g.coords_[1]), wasserstein)
151 + Geometry::pow(
152 std::abs(coords_[2] - g.coords_[2]), wasserstein));
153 }

156int ttk::Bidder::runBidding(GoodDiagram *goods,
157 Good &twinGood,
158 int wasserstein,
159 double epsilon,
160 double geometricalFactor) {
161 double best_val = std::numeric_limits<double>::lowest();
162 double second_val = std::numeric_limits<double>::lowest();
163 Good *best_good{};
164 for(size_t i = 0; i < goods->size(); i++) {
165 Good &g = (*goods)[i];
166 double val = -this->cost(g, wasserstein, geometricalFactor);
167 val -= g.getPrice();
168 if(val > best_val) {
169 second_val = best_val;
170 best_val = val;
171 best_good = &g;
172 } else if(val > second_val) {
173 second_val = val;
174 }
175 }
176 // And now check for the corresponding twin bidder
177 Good &g = twinGood;
178 double val = -this->cost(g, wasserstein, geometricalFactor);
179 val -= g.getPrice();
180 if(val > best_val) {
181 second_val = best_val;
182 best_val = val;
183 best_good = &g;
184 } else if(val > second_val) {
185 second_val = val;
186 }
187
188 if(second_val == std::numeric_limits<double>::lowest()) {
189 // There is only one acceptable good for the bidder
190 second_val = best_val;
191 }
192 if(best_good == nullptr) {
193 return -1;
194 }
195 double old_price = best_good->getPrice();
196 double new_price = old_price + best_val - second_val + epsilon;
197 if(new_price > std::numeric_limits<double>::max() / 2) {
198 new_price = old_price + epsilon;
199 }
200 // Assign bidder to best_good
201 this->setProperty(*best_good);
202 this->setPricePaid(new_price);
203
204 // Assign best_good to bidder and unassign the previous owner of best_good
205 // if need be
206 int idx_reassigned = best_good->getOwner();
207 best_good->assign(this->position_in_auction_, new_price);
208 return idx_reassigned;
209}

211int ttk::Bidder::runDiagonalBidding(
212 GoodDiagram *goods,
213 Good &twinGood,
214 int wasserstein,
215 double epsilon,
216 double geometricalFactor,
217 std::priority_queue<std::pair<int, double>,
218 std::vector<std::pair<int, double>>,
219 Compare> &diagonal_queue) {
220
221 // First, find the lowest and second lowest weights for diagonal goods
222 // Take this time to update the weights in the priority queue of the goods
223 // tested. It is not necessary to update weights for all diagonal bidders in
224 // the pririty queue since weights can only increase and we are interested
225 // only in the lowest ones
226 bool updated_top_pair
227 = false; // Boolean which equals true iff the top pair in the priority
228 // queue is given the good price
229 bool non_empty_goods = goods->size() > 0;
230 std::pair<int, double> best_pair;
231 if(non_empty_goods) {
232 while(!updated_top_pair) {
233 std::pair<int, double> top_pair = diagonal_queue.top();
234 diagonal_queue.pop();
235
236 double queue_weight = top_pair.second;
237 const auto &good = (*goods)[top_pair.first];
238 if(good.getPrice() > queue_weight) {
239 // If the weight in the priority queue is not the good one, update
240 std::get<1>(top_pair) = good.getPrice();
241 diagonal_queue.push(top_pair);
242 } else {
243 updated_top_pair = true;
244 best_pair = top_pair;
245 }
246 }
247 }
248
249 std::pair<int, double> second_pair;
250 if(non_empty_goods && diagonal_queue.size() > 0) {
251 bool updated_second_pair = false;
252 while(!updated_second_pair) {
253 second_pair = diagonal_queue.top();
254 double queue_weight = second_pair.second;
255 const auto &good = (*goods)[second_pair.first];
256 if(good.getPrice() != queue_weight) {
257 // If the weight in the priority queue is not the good one, update it
258 diagonal_queue.pop();
259 std::get<1>(second_pair) = good.getPrice();
260 diagonal_queue.push(second_pair);
261 } else {
262 updated_second_pair = true;
263 }
264 }
265 }
266
267 Good *best_good{};
268 double best_val = 0;
269 double second_val = 0;
270 if(non_empty_goods) {
271 best_val = -best_pair.second;
272 second_val = diagonal_queue.size() > 0 ? -second_pair.second : best_val;
273 best_good = &(*goods)[best_pair.first];
274 }
275
276 // And now check for the corresponding twin bidder
277 bool is_twin = false;
278 Good &g = twinGood;
279 double val = -this->cost(g, wasserstein, geometricalFactor);
280 val -= g.getPrice();
281 if(non_empty_goods) {
282 if(val > best_val) {
283 second_val = best_val;
284 best_val = val;
285 best_good = &g;
286 is_twin = true;
287 } else if(val > second_val || diagonal_queue.empty()) {
288 second_val = val;
289 }
290 } else {
291 best_val = val;
292 best_good = &g;
293 is_twin = true;
294 second_val = best_val;
295 }
296
297 if(second_val == std::numeric_limits<double>::lowest()) {
298 // There is only one acceptable good for the bidder
299 second_val = best_val;
300 }
301 double old_price = best_good->getPrice();
302 double new_price = old_price + best_val - second_val + epsilon;
303 if(new_price > std::numeric_limits<double>::max() / 2) {
304 new_price = old_price + epsilon;
305 }
306
307 // Assign bidder to best_good
308 this->setProperty(*best_good);
309 this->setPricePaid(new_price);
310
311 // Assign best_good to bidder and unassign the previous owner of best_good
312 // if need be
313 int idx_reassigned = best_good->getOwner();
314 best_good->assign(this->position_in_auction_, new_price);
315 if(!is_twin) {
316 std::get<1>(best_pair) = new_price;
317 }
318 if(non_empty_goods) {
319 diagonal_queue.push(best_pair);
320 }
321 return idx_reassigned;
322}

445int ttk::Bidder::runKDTBidding(GoodDiagram *goods,
446 Good &twinGood,
447 int wasserstein,
448 double epsilon,
449 double geometricalFactor,
450 KDT *kdt,
451 const int kdt_index) {
452
454 std::vector<KDT *> neighbours;
455 std::vector<double> costs;
456
457 std::array<double, 5> coordinates{
458 geometricalFactor * this->x_, geometricalFactor * this->y_, 0.0, 0.0, 0.0,
459 };
460 if(geometricalFactor < 1) {
461 coordinates[2] = (1 - geometricalFactor) * this->coords_[0];
462 coordinates[3] = (1 - geometricalFactor) * this->coords_[1];
463 coordinates[4] = (1 - geometricalFactor) * this->coords_[2];
464 }
465
466 kdt->getKClosest(2, coordinates, neighbours, costs, kdt_index);
467 double best_val, second_val;
468 KDT *closest_kdt;
469 Good *best_good{};
470 if(costs.size() == 2) {
471 std::array<int, 2> idx{0, 1};
472 std::sort(idx.begin(), idx.end(),
473 [&costs](int &a, int &b) { return costs[a] < costs[b]; });
474
475 closest_kdt = neighbours[idx[0]];
476 best_good = &(*goods)[closest_kdt->id_];
477 // Value is defined as the opposite of cost (each bidder aims at
478 // maximizing it)
479 best_val = -costs[idx[0]];
480 second_val = -costs[idx[1]];
481 } else {
482 // If the kdtree contains only one point
483 closest_kdt = neighbours[0];
484 best_good = &(*goods)[closest_kdt->id_];
485 best_val = -costs[0];
486 second_val = best_val;
487 }
488 // And now check for the corresponding twin bidder
489 bool twin_chosen = false;
490 Good &g = twinGood;
491 double val = -this->cost(g, wasserstein, geometricalFactor);
492 val -= g.getPrice();
493 if(val > best_val) {
494 second_val = best_val;
495 best_val = val;
496 best_good = &g;
497 twin_chosen = true;
498 } else if(val > second_val) {
499 second_val = val;
500 }
501
502 if(second_val == std::numeric_limits<double>::lowest()) {
503 // There is only one acceptable good for the bidder
504 second_val = best_val;
505 }
506 double old_price = best_good->getPrice();
507 double new_price = old_price + best_val - second_val + epsilon;
508 if(new_price > std::numeric_limits<double>::max() / 2) {
509 new_price = old_price + epsilon;
510 }
511 // Assign bidder to best_good
512 this->setProperty(*best_good);
513 this->setPricePaid(new_price);
514 // Assign best_good to bidder and unassign the previous owner of best_good
515 // if need be
516 int idx_reassigned = best_good->getOwner();
517
518 best_good->assign(this->position_in_auction_, new_price);
519 // Update the price in the KDTree
520 if(!twin_chosen) {
521 closest_kdt->updateWeight(new_price, kdt_index);
522 }
523 return idx_reassigned;
524}
