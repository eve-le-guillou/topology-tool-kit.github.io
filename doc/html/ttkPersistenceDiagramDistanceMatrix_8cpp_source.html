147double ttkPersistenceDiagramDistanceMatrix::getPersistenceDiagram(
148 ttk::Diagram &diagram, vtkUnstructuredGrid *CTPersistenceDiagram_) {
149
150 const auto pd = CTPersistenceDiagram_->GetPointData();
151 const auto cd = CTPersistenceDiagram_->GetCellData();
152 const auto points = CTPersistenceDiagram_->GetPoints();
153
154 if(pd == nullptr || cd == nullptr || points == nullptr) {
155 this->printErr("Missing Diagram PointData, CellData or Points");
156 return -1.0;
157 }
158
159 const auto vertexIdentifierScalars
160 = vtkIntArray::SafeDownCast(pd->GetArray(ttk::VertexScalarFieldName));
161 const auto nodeTypeScalars
162 = vtkIntArray::SafeDownCast(pd->GetArray("CriticalType"));
163 const auto pairIdentifierScalars
164 = vtkIntArray::SafeDownCast(cd->GetArray("PairIdentifier"));
165 const auto extremumIndexScalars
166 = vtkIntArray::SafeDownCast(cd->GetArray("PairType"));
167 const auto persistenceScalars = cd->GetArray("Persistence");
168 const auto birthScalars = cd->GetArray("Birth");
169 const auto critCoordinates
170 = vtkFloatArray::SafeDownCast(pd->GetArray("Coordinates"));
171
172 const bool embed = critCoordinates == nullptr;
173
174 int pairingsSize = (int)pairIdentifierScalars->GetNumberOfTuples();
175 // FIX : no more missed pairs
176 for(int pair_index = 0; pair_index < pairingsSize; pair_index++) {
177 const float index_of_pair = pair_index;
178 if(*pairIdentifierScalars->GetTuple(pair_index) != -1)
179 pairIdentifierScalars->SetTuple(pair_index, &index_of_pair);
180 }
181
182 // If diagram has the diagonal (we assume it is last)
183 if(*pairIdentifierScalars->GetTuple(pairingsSize - 1) == -1)
184 pairingsSize -= 1;
185
186 if(pairingsSize < 1 || !vertexIdentifierScalars || !pairIdentifierScalars
187 || !nodeTypeScalars || !persistenceScalars || !extremumIndexScalars
188 || !points) {
189 this->printErr("Missing Persistence Diagram data array");
190 return -3.0;
191 }
192
193 diagram.resize(pairingsSize + 1);
194 int nbNonCompact = 0;
195 double max_dimension = 0;
196
197 // skip diagonal cell (corresponding points already dealt with)
198 for(int i = 0; i < pairingsSize; ++i) {
199
200 int vertexId1 = vertexIdentifierScalars->GetValue(2 * i);
201 int vertexId2 = vertexIdentifierScalars->GetValue(2 * i + 1);
202 int nodeType1 = nodeTypeScalars->GetValue(2 * i);
203 int nodeType2 = nodeTypeScalars->GetValue(2 * i + 1);
204
205 int pairIdentifier = pairIdentifierScalars->GetValue(i);
206 int pairType = extremumIndexScalars->GetValue(i);
207 const auto persistence = persistenceScalars->GetTuple1(i);
208 const auto birth = birthScalars->GetTuple1(i);
209 const auto death = birth + persistence;
210
211 std::array<double, 3> coordsBirth{}, coordsDeath{};
212
213 const auto i0 = 2 * i;
214 const auto i1 = 2 * i + 1;
215
216 if(embed) {
217 points->GetPoint(i0, coordsBirth.data());
218 points->GetPoint(i1, coordsDeath.data());
219 } else {
220 critCoordinates->GetTuple(i0, coordsBirth.data());
221 critCoordinates->GetTuple(i1, coordsDeath.data());
222 }
223
224 if(pairIdentifier != -1 && pairIdentifier < pairingsSize) {
225 if(pairIdentifier == 0) {
226 max_dimension = persistence;
227
228 diagram[0] = std::make_tuple(
229 vertexId1, ttk::CriticalType::Local_minimum, vertexId2,
230 ttk::CriticalType::Saddle1, persistence, pairType, birth,
231 coordsBirth[0], coordsBirth[1], coordsBirth[2], death, coordsDeath[0],
232 coordsDeath[1], coordsDeath[2]);
233 diagram[pairingsSize] = std::make_tuple(
234 vertexId1, ttk::CriticalType::Saddle1, vertexId2,
235 ttk::CriticalType::Local_maximum, persistence, pairType, birth,
236 coordsBirth[0], coordsBirth[1], coordsBirth[2], death, coordsDeath[0],
237 coordsDeath[1], coordsDeath[2]);
238
239 } else {
240 diagram[pairIdentifier] = std::make_tuple(
241 vertexId1, static_cast<ttk::CriticalType>(nodeType1), vertexId2,
242 static_cast<ttk::CriticalType>(nodeType2), persistence, pairType,
243 birth, coordsBirth[0], coordsBirth[1], coordsBirth[2], death,
244 coordsDeath[0], coordsDeath[1], coordsDeath[2]);
245 }
246 }
247 if(pairIdentifier >= pairingsSize) {
248 nbNonCompact++;
249 if(nbNonCompact == 0) {
250 this->printWrn("Diagram pair identifiers must be compact (not exceed "
251 "the diagram size).");
252 }
253 }
254 }
255
256 if(nbNonCompact > 0) {
257 this->printWrn("Missed " + std::to_string(nbNonCompact)
258 + " pairs due to non-compactness.");
259 }
260
261 return max_dimension;
