250 vtkIdType p1 = points->InsertNextPoint(sa + delta, sb - delta, 0);

278 or (abs((double)(outputScalars[b] - inputScalars[b])) > 1e-6)) {
279 unfolded->InsertNextTuple1(0);
280 } else {

284 }

294 vtkIdType p1 = points->InsertNextPoint(s0, s0 + 2 * delta, 0);

300 quad->InsertNextId(p1);
301 quad->InsertNextId(p3);

303 bounds->InsertNextCell(VTK_QUAD, quad);
304 pairIdentifierScalars->InsertNextTuple1(-1);
305 BirthId->InsertNextTuple1(-1);
306 DeathId->InsertNextTuple1(-1);
307 persistenceScalars->InsertNextTuple1(Epsilon);
308 extremumIndexScalars->InsertNextTuple1(-1);
309 unfolded->InsertNextTuple1(-1);
310 // ____________________________________________
311
312 bounds->SetPoints(points);
313

315 bounds->GetCellData()->AddArray(DeathId);

317 bounds->GetCellData()->AddArray(pairIdentifierScalars);
