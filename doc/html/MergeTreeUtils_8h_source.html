64 template <class dataType>
65 std::tuple<double, double>
66 getNormalizedBirthDeathDouble(ftm::FTMTree_MT *tree,
67 ftm::idNode nodeId,
68 dataType newMin = 0.0,
69 dataType newMax = 1.0) {
70 auto birthDeath = tree->getBirthDeath<dataType>(nodeId);
71 double birth = std::get<0>(birthDeath);
72 double death = std::get<1>(birthDeath);
73 dataType shiftMin = getMinMaxLocal<dataType>(tree, nodeId);
74 dataType shiftMax = getMinMaxLocal<dataType>(tree, nodeId, false);
75 birth = (newMax - newMin) * (birth - shiftMin)
76 / (shiftMax - shiftMin); // + newMin;
77 death = (newMax - newMin) * (death - shiftMin)
78 / (shiftMax - shiftMin); // + newMin;
79 return std::make_tuple(birth, death);
80 }

127 template <class dataType>
128 dataType getMinMaxLocalFromVector(ftm::FTMTree_MT *tree,
129 ftm::idNode nodeId,
130 std::vector<dataType> &scalarsVector,
131 bool getMin = true) {
132 auto nodeIdParent = tree->getParentSafe(nodeId);
133
134 if(tree->notNeedToNormalize(nodeId))
135 return getMin ? 0.0 : 1.0;
136
137 // General case
138 dataType death = scalarsVector[nodeIdParent];
139 dataType birth = scalarsVector[tree->getNode(nodeIdParent)->getOrigin()];
140 if(death < birth) {
141 dataType temp = death;
142 death = birth;
143 birth = temp;
144 }
145
146 return getMin ? birth : death;
