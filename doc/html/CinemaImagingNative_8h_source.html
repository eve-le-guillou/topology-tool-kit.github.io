64 ttk::Timer timer;
65 int resX = resolution[0];
66 int resY = resolution[1];
67
68 this->printMsg("Rendering Image ("
69 + std::string(orthographicProjection ? "O" : "P") + "|"
70 + std::to_string(resX) + "x" + std::to_string(resY) + ")",
71 0, 0, this->threadNumber_, ttk::debug::LineMode::REPLACE);

73 // Compute camera size
74 const double aspect = resolution[0] / resolution[1];
75 const double camSize[2] = {aspect * camHeight, camHeight};
76
77 const auto normalize = [](double out[3], const double in[3]) {
78 double temp = sqrt(in[0] * in[0] + in[1] * in[1] + in[2] * in[2]);
79 out[0] = in[0] / temp;
80 out[1] = in[1] / temp;
81 out[2] = in[2] / temp;
82 };
83
84 double camDir[3]{0, 0, 0};
85 normalize(camDir, camDirRaw);
86
87 // Compute camRight = camDir x CamUp
88 double camRight[3]{camDir[1] * camUp[2] - camDir[2] * camUp[1],
89 camDir[2] * camUp[0] - camDir[0] * camUp[2],
90 camDir[0] * camUp[1] - camDir[1] * camUp[0]};
91 normalize(camRight, camRight);
92
93 // Compute true up std::vector
94 double camUpTrue[3]{camDir[1] * (-camRight[2]) - camDir[2] * (-camRight[1]),
95 camDir[2] * (-camRight[0]) - camDir[0] * (-camRight[2]),
96 camDir[0] * (-camRight[1]) - camDir[1] * (-camRight[0])};
97 normalize(camUpTrue, camUpTrue);
98
99 // Compute pixel size in world coordinates
100 double pixelWidthWorld = camSize[0] / resolution[0];
101 double pixelHeightWorld = camSize[1] / resolution[1];
102
103 // Optimization: precompute half of the camera size to reduce the number of
104 // operations in the for loop. Include a half pixel offset (-0.5) to center
105 // vertices at pixel centers
106 double camWidthWorldHalf = 0.5 * camSize[0] - 0.5 * pixelWidthWorld;
107 double camHeightWorldHalf = 0.5 * camSize[1] - 0.5 * pixelHeightWorld;
108
109 // Optimization: reorient camera model to bottom left corner to reduce
110 // operations in for loop
111 double camPosCorner[3] = {camPos[0] - camRight[0] * camWidthWorldHalf
112 - camUpTrue[0] * camHeightWorldHalf,
113 camPos[1] - camRight[1] * camWidthWorldHalf
114 - camUpTrue[1] * camHeightWorldHalf,
115 camPos[2] - camRight[2] * camWidthWorldHalf
116 - camUpTrue[2] * camHeightWorldHalf};
117
118 float nan = std::numeric_limits<float>::quiet_NaN();
119 if(orthographicProjection) {
120#ifdef TTK_ENABLE_OPENMP
121#pragma omp parallel for num_threads(this->threadNumber_)
122#endif
123 for(int y = 0; y < resY; y++) {
124 double v = ((double)y) * pixelHeightWorld;
125
126 size_t pixelIndex = y * resX;
127 size_t bcIndex = 2 * pixelIndex;
128
129 for(int x = 0; x < resX; x++) {
130 double u = ((double)x) * pixelWidthWorld;
131
132 depthBuffer[pixelIndex] = nan;
133 primitiveIds[pixelIndex] = CinemaImaging::INVALID_ID;
134 barycentricCoordinates[bcIndex] = nan;
135 barycentricCoordinates[bcIndex + 1] = nan;
136
137 // set origin
138 float org_x = camPosCorner[0] + u * camRight[0] + v * camUpTrue[0];
139 float org_y = camPosCorner[1] + u * camRight[1] + v * camUpTrue[1];
140 float org_z = camPosCorner[2] + u * camRight[2] + v * camUpTrue[2];
141
142 float ray_origin[3] = {org_x, org_y, org_z};
143
144 // set dir
145 float dir_x = camDir[0];
146 float dir_y = camDir[1];
147 float dir_z = camDir[2];
148
149 float ray_dir[3] = {dir_x, dir_y, dir_z};
150
151 Ray ray(ray_dir, ray_origin);
152 bool wasHit = false;
153 int triIdx;
154 float distance;
155 wasHit = bvh.intersect(
156 ray, connectivityList, vertexCoords, &triIdx, &distance);
157 if(wasHit) {
158 depthBuffer[pixelIndex] = distance;
159 primitiveIds[pixelIndex] = triIdx;
160 barycentricCoordinates[bcIndex] = ray.u;
161 barycentricCoordinates[bcIndex + 1] = ray.v;
162 }
163 pixelIndex++;
164 bcIndex += 2;
165 }
166 }
167 } else {
168 double factor = (viewAngle / 180.0 * 3.141592653589793) / resolution[0];
169
170#ifdef TTK_ENABLE_OPENMP
171#pragma omp parallel for num_threads(this->threadNumber_)
172#endif
173 for(int y = 0; y < resY; y++) {
174 double v = (y - resY * 0.5) * factor;
175 size_t pixelIndex = y * resX;
176 size_t bcIndex = 2 * pixelIndex;
177
178 for(int x = 0; x < resX; x++) {
179 double u = (x - resX * 0.5) * factor;
180
181 depthBuffer[pixelIndex] = nan;
182 primitiveIds[pixelIndex] = CinemaImaging::INVALID_ID;
183 barycentricCoordinates[bcIndex] = nan;
184 barycentricCoordinates[bcIndex + 1] = nan;
185
186 // set origin
187 float org_x = camPos[0];
188 float org_y = camPos[1];
189 float org_z = camPos[2];
190
191 float ray_origin[3] = {org_x, org_y, org_z};
192 // set dir
193 float dir_x = camDir[0] + u * camRight[0] + v * camUpTrue[0];
194 float dir_y = camDir[1] + u * camRight[1] + v * camUpTrue[1];
195 float dir_z = camDir[2] + u * camRight[2] + v * camUpTrue[2];
196
197 float ray_dir[3] = {dir_x, dir_y, dir_z};
198
199 Ray ray(ray_dir, ray_origin);
200 bool wasHit = false;
201 int triIdx;
202 float distance;
203 wasHit = bvh.intersect(
204 ray, connectivityList, vertexCoords, &triIdx, &distance);
205 if(wasHit) {
206 depthBuffer[pixelIndex] = distance;
207 primitiveIds[pixelIndex] = triIdx;
208 barycentricCoordinates[bcIndex] = ray.u;
209 barycentricCoordinates[bcIndex + 1] = ray.v;
210 }
211 pixelIndex++;
212 bcIndex += 2;
213 }
214 }
215 }
216 this->printMsg("Rendering Image ("
217 + std::string(orthographicProjection ? "O" : "P") + "|"
218 + std::to_string(resX) + "x" + std::to_string(resY) + ")",
219
220 1, timer.getElapsedTime(), this->threadNumber_);
221
222 return 1;
