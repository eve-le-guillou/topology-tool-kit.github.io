124 if(data != nullptr) {
125 const auto dim{data->getDimensionality()};
126
127 data->preconditionBoundaryVertices();
128 data->preconditionVertexNeighbors();
129 data->preconditionVertexEdges();
130 data->preconditionVertexStars();
131 data->preconditionEdges();
132 data->preconditionEdgeStars();

134 data->preconditionBoundaryEdges();

136 if(dim == 2) {
137 data->preconditionCellEdges();
138 } else if(dim == 3) {
139 data->preconditionBoundaryTriangles();
140 data->preconditionVertexTriangles();
141 data->preconditionEdgeTriangles();
142 data->preconditionTriangles();
143 data->preconditionTriangleEdges();
144 data->preconditionTriangleStars();
