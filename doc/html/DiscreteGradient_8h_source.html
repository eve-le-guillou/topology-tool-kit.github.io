253 template <typename dataType>
254 struct SaddleSaddleVPathComparator {
255 bool
256 operator()(const std::tuple<dataType, SimplexId, SimplexId> &v1,
257 const std::tuple<dataType, SimplexId, SimplexId> &v2) const {
258 const dataType persistence1 = std::get<0>(v1);
259 const dataType persistence2 = std::get<0>(v2);
260
261 const SimplexId vpathId1 = std::get<1>(v1);
262 const SimplexId vpathId2 = std::get<1>(v2);
263
264 const SimplexId saddleId1 = std::get<2>(v1);
265 const SimplexId saddleId2 = std::get<2>(v2);
266
267 if(persistence1 != persistence2) {
268 return (persistence1 < persistence2);
269 }
270
271 if(saddleId1 != saddleId2) {
272 return (saddleId1 < saddleId2);
273 }
274
275 return (vpathId1 < vpathId2);
276 }
277 };
