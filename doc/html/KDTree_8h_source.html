133 template <typename dataType>
134 typename KDTree<dataType>::KDTreeMap
135 KDTree<dataType>::build(dataType *data,
136 const int &ptNumber,
137 const int &dimension,
138 const int weight_number) {
139 KDTreeMap correspondance_map(ptNumber);
140 // First, perform a argsort on the data
141 // initialize original index locations
142 for(int axis = 0; axis < dimension; axis++) {
143 coords_min_.push_back(std::numeric_limits<dataType>::lowest());
144 coords_max_.push_back(std::numeric_limits<dataType>::max());
145 }
146 std::vector<int> idx(ptNumber);
147 for(int i = 0; i < ptNumber; i++) {
148 idx[i] = i;
149 }
150 // sort indexes based on comparing values in coordinates
151 sort(idx.begin(), idx.end(), [&](int i1, int i2) {
152 return data[dimension * i1 + coords_number_]
153 < data[dimension * i2 + coords_number_];
154 });
155 int median_loc = (int)(ptNumber - 1) / 2;
156 int median_idx = idx[median_loc];
157 correspondance_map[median_idx] = this;
158
159 for(int axis = 0; axis < dimension; axis++) {
160 coordinates_.push_back(data[dimension * median_idx + axis]);
161 }
162 for(int w = 0; w < weight_number; w++) {
163 weight_.push_back(0);
164 min_subweights_.push_back(0);
165 }
166 id_ = median_idx;
167 parent_ = nullptr;
168 level_ = 0;
169
170 if(idx.size() > 2) {
171 // Build left leaf
172 std::vector<int> idx_left(median_loc);
173 for(int i = 0; i < median_loc; i++) {
174 idx_left[i] = idx[i];
175 }
176
177 this->left_ = std::unique_ptr<KDTree>(
178 new KDTree(this, (coords_number_ + 1) % dimension, true));
179 this->left_->buildRecursive(data, idx_left, ptNumber, dimension, this,
180 correspondance_map, weight_number);
181 }
182
183 if(idx.size() > 1) {
184 // Build right leaf
185 std::vector<int> idx_right(ptNumber - median_loc - 1);
186 for(int i = 0; i < ptNumber - median_loc - 1; i++) {
187 idx_right[i] = idx[i + median_loc + 1];
188 }
189 this->right_ = std::unique_ptr<KDTree>(
190 new KDTree(this, (coords_number_ + 1) % dimension, false));
191 this->right_->buildRecursive(data, idx_right, ptNumber, dimension, this,
192 correspondance_map, weight_number);
193 }
194
195 return correspondance_map;
196 }

198 template <typename dataType>
199 void KDTree<dataType>::buildRecursive(dataType *data,
200 std::vector<int> idx_side,
201 const int &ptNumber,
202 const int &dimension,
203 KDTree<dataType> *parent,
204 KDTreeMap &correspondance_map,
205 const int weight_number) {
206
207 // First, perform a argsort on the data
208 sort(idx_side.begin(), idx_side.end(), [&](int i1, int i2) {
209 return data[dimension * i1 + coords_number_]
210 < data[dimension * i2 + coords_number_];
211 });
212 int median_loc = (int)(idx_side.size() - 1) / 2;
213 int median_idx = idx_side[median_loc];
214 correspondance_map[median_idx] = this;
215
216 for(int axis = 0; axis < dimension; axis++) {
217 coordinates_.push_back(data[dimension * median_idx + axis]);
218 }
219 for(int w = 0; w < weight_number; w++) {
220 weight_.push_back(0);
221 min_subweights_.push_back(0);
222 }
223 id_ = median_idx;
224 parent_ = parent;
225 level_ = parent->level_ + 1;
226
227 // Create bounding box
228 for(int axis = 0; axis < dimension; axis++) {
229 coords_min_.push_back(parent_->coords_min_[axis]);
230 coords_max_.push_back(parent_->coords_max_[axis]);
231 }
232 if(is_left_ && !this->isRoot()) {
233 coords_max_[parent_->coords_number_]
234 = parent_->coordinates_[parent_->coords_number_];
235 } else if(!is_left_ && !this->isRoot()) {
236 coords_min_[parent_->coords_number_]
237 = parent_->coordinates_[parent_->coords_number_];
238 }
239
240 if(idx_side.size() > 2) {
241 // Build left leaf
242 std::vector<int> idx_left(median_loc);
243 for(int i = 0; i < median_loc; i++) {
244 idx_left[i] = idx_side[i];
245 }
246
247 this->left_ = std::unique_ptr<KDTree>(
248 new KDTree(this, (coords_number_ + 1) % dimension, true));
249 this->left_->buildRecursive(data, idx_left, ptNumber, dimension, this,
250 correspondance_map, weight_number);
251 }
252
253 if(idx_side.size() > 1) {
254 // Build right leaf
255 std::vector<int> idx_right(idx_side.size() - median_loc - 1);
256 for(unsigned int i = 0; i < idx_side.size() - median_loc - 1; i++) {
257 idx_right[i] = idx_side[i + median_loc + 1];
258 }
259 this->right_ = std::unique_ptr<KDTree>(
260 new KDTree(this, (coords_number_ + 1) % dimension, false));
261 this->right_->buildRecursive(data, idx_right, ptNumber, dimension, this,
262 correspondance_map, weight_number);
263 }
264 return;
265 }

274 template <typename dataType>
275 void KDTree<dataType>::updateMinSubweight(const int weight_index) {
276 dataType new_min_subweight;
277 if(this->isLeaf()) {
278 new_min_subweight = weight_[weight_index];
279 } else if(!left_) {
280 new_min_subweight = std::min(
281 right_->min_subweights_[weight_index], weight_[weight_index]);
282 } else if(!right_) {
283 new_min_subweight
284 = std::min(left_->min_subweights_[weight_index], weight_[weight_index]);
285 } else {
286 new_min_subweight
287 = std::min(std::min(left_->min_subweights_[weight_index],
288 right_->min_subweights_[weight_index]),
289 weight_[weight_index]);
290 }
291
292 if(new_min_subweight != min_subweights_[weight_index]) {
293 min_subweights_[weight_index] = new_min_subweight;
294 if(!this->isRoot()) {
295 parent_->updateMinSubweight(weight_index);
296 }
297 }

323 template <typename dataType>
324 void KDTree<dataType>::recursiveGetKClosest(
325 const unsigned int k,
326 const std::vector<dataType> &coordinates,
327 KDTreeMap &neighbours,
328 std::vector<dataType> &costs,
329 const int weight_index) {
330 // 1- Look wether or not to include the current point in the nearest
331 // neighbours
332 dataType cost = this->cost(coordinates);
333 cost += weight_[weight_index];
334
335 if(costs.size() < k) {
336 neighbours.push_back(this);
337 costs.push_back(cost);
338 } else {
339 // 1.1- Find the most costly amongst neighbours
340 const auto idx_max_cost = std::distance(
341 costs.begin(), std::max_element(costs.begin(), costs.begin() + k));
342 dataType max_cost = costs[idx_max_cost];
343
344 // 1.2- If the current KDTree is less costly, put it in the neighbours and
345 // update costs.
346 if(cost < max_cost) {
347 costs[idx_max_cost] = cost;
348 neighbours[idx_max_cost] = this;
349 }
350 }
351
352 // 2- Recursively visit KDTrees that are worth it
353 if(left_) {
354 dataType max_cost = *std::max_element(costs.begin(), costs.end());
355 dataType &min_subweight = left_->min_subweights_[weight_index];
356 dataType d_min = this->distanceToBox(*left_, coordinates);
357 if(costs.size() < k || d_min + min_subweight < max_cost) {
358 // 2.2- It is possible that there exists a point in this subtree that is
359 // less costly than max_cost
360 left_->recursiveGetKClosest(
361 k, coordinates, neighbours, costs, weight_index);
362 }
363 }
364
365 if(right_) {
366 dataType max_cost = *std::max_element(costs.begin(), costs.end());
367 dataType &min_subweight = right_->min_subweights_[weight_index];
368 dataType d_min = this->distanceToBox(*right_, coordinates);
369 if(costs.size() < k || d_min + min_subweight < max_cost) {
370 // 2.2- It is possible that there exists a point in this subtree that is
371 // less costly than max_cost
372 right_->recursiveGetKClosest(
373 k, coordinates, neighbours, costs, weight_index);
374 }
375 }
376 return;
377 }

379 template <typename dataType>
380 dataType KDTree<dataType>::cost(const std::vector<dataType> &coordinates) {
381 dataType cost = 0;
382 for(size_t i = 0; i < coordinates.size(); i++) {
383 cost += Geometry::pow(abs(coordinates[i] - coordinates_[i]), p_);
384 }
385 return cost;

388 template <typename dataType>
389 dataType
390 KDTree<dataType>::distanceToBox(const KDTree<dataType> &subtree,
391 const std::vector<dataType> &coordinates) {
392 dataType d_min = 0;
393 for(size_t axis = 0; axis < coordinates.size(); axis++) {
394 if(subtree.coords_min_[axis] > coordinates[axis]) {
395 d_min
396 += Geometry::pow(subtree.coords_min_[axis] - coordinates[axis], p_);
397 } else if(subtree.coords_max_[axis] < coordinates[axis]) {
398 d_min
399 += Geometry::pow(coordinates[axis] - subtree.coords_max_[axis], p_);
400 }
401 }
402 return d_min;
403 }
