110 template <typename DT, typename IT>
111 int getGhostCellScalars(DT *scalarArray,
112 const int *const rankArray,
113 const LongSimplexId *const globalIds,
114 const std::unordered_map<IT, IT> &gidToLidMap,
115 const std::unordered_set<int> &neighbors,
116 const int rankToSend,
117 const IT nVerts,
118 MPI_Comm communicator) {
119
120 if(!ttk::isRunningWithMPI()) {
121 return -1;
122 }
123 MPI_Datatype MPI_DT = getMPIType(static_cast<DT>(0));
124 MPI_Datatype MPI_IT = getMPIType(static_cast<IT>(0));
125 // we need unique tags for each rankToSend, otherwise messages might become
126 // entangled
127 int tagMultiplier = rankToSend + 1;
128 int amountTag = 101 * tagMultiplier;
129 int idsTag = 102 * tagMultiplier;
130 int valuesTag = 103 * tagMultiplier;
131 if(rankToSend == ttk::MPIrank_) {
132 // initialize the inner vectors with size 0
133 std::vector<std::vector<IT>> rankVectors(
134 ttk::MPIsize_, std::vector<IT>(0));
135 // aggregate the needed ids
136
137 for(IT i = 0; i < nVerts; i++) {
138 if(ttk::MPIrank_ != rankArray[i]) {
139 rankVectors[rankArray[i]].push_back(globalIds[i]);
140 }
141 }
142 // send the amount of ids and the needed ids themselves
143 for(int r = 0; r < ttk::MPIsize_; r++) {
144 if(ttk::MPIrank_ != r && neighbors.find(r) != neighbors.end()) {
145 IT nValues = rankVectors[r].size();
146 MPI_Send(&nValues, 1, MPI_IT, r, amountTag, communicator);
147 if(nValues > 0) {
148 MPI_Send(
149 rankVectors[r].data(), nValues, MPI_IT, r, idsTag, communicator);
150 }
151 }
152 }
153 // receive the scalar values
154 for(int r = 0; r < ttk::MPIsize_; r++) {
155 if(ttk::MPIrank_ != r && neighbors.find(r) != neighbors.end()) {
156 IT nValues = rankVectors[r].size();
157 std::vector<DT> receivedValues(nValues);
158 if(nValues > 0) {
159 MPI_Recv(receivedValues.data(), nValues, MPI_DT, r, valuesTag,
160 communicator, MPI_STATUS_IGNORE);
161 for(IT i = 0; i < nValues; i++) {
162 DT receivedVal = receivedValues[i];
163 IT globalId = rankVectors[r][i];
164 IT localId = gidToLidMap.at(globalId);
165 scalarArray[localId] = receivedVal;
166 }
167 }
168 }
169 }
170
171 } else { // owner ranks
172 // if rankToSend is not the neighbor of the current rank, we do not need
173 // to do anything
174 if(neighbors.find(rankToSend) != neighbors.end()) {
175 // receive the amount of ids and the needed ids themselves
176 IT nValues;
177
178 MPI_Recv(&nValues, 1, MPI_IT, rankToSend, amountTag, communicator,
179 MPI_STATUS_IGNORE);
180
181 if(nValues > 0) {
182 std::vector<IT> receivedIds(nValues);
183 MPI_Recv(receivedIds.data(), nValues, MPI_IT, rankToSend, idsTag,
184 communicator, MPI_STATUS_IGNORE);
185
186 // assemble the scalar values
187 std::vector<DT> valuesToSend(nValues);
188 for(IT i = 0; i < nValues; i++) {
189 IT globalId = receivedIds[i];
190 IT localId = gidToLidMap.at(globalId);
191 valuesToSend[i] = scalarArray[localId];
192 }
193
194 // send the scalar values
195 MPI_Send(valuesToSend.data(), nValues, MPI_DT, rankToSend, valuesTag,
196 communicator);
197 }
198 }
199 }
200
201 return 0;
202 }

213 template <typename IT>
214 int getNeighbors(std::unordered_set<int> &neighbors,
215 const int *const rankArray,
216 const IT nVerts,
217 MPI_Comm communicator) {
218 for(IT i = 0; i < nVerts; i++) {
219 if(rankArray[i] != ttk::MPIrank_) {
220 neighbors.emplace(rankArray[i]);
221 }
222 }
223 std::vector<int> sendVector(neighbors.begin(), neighbors.end());
224 int localSize = neighbors.size();
225 int sizes[ttk::MPIsize_];
226 int displacements[ttk::MPIsize_];
227 MPI_Gather(&localSize, 1, MPI_INT, sizes, 1, MPI_INT, 0, communicator);
228 int totalSize = 0;
229 if(ttk::MPIrank_ == 0) {
230 for(int i = 0; i < ttk::MPIsize_; i++) {
231 totalSize += sizes[i];
232 if(i == 0) {
233 displacements[i] = 0;
234 } else {
235 displacements[i] = displacements[i - 1] + sizes[i - 1];
236 }
237 }
238 }
239 std::vector<int> rootVector(totalSize);
240
241 MPI_Gatherv(sendVector.data(), sendVector.size(), MPI_INT,
242 rootVector.data(), sizes, displacements, MPI_INT, 0,
243 communicator);
244 std::vector<int> scatterVector;
245
246 if(ttk::MPIrank_ == 0) {
247 // now we transform this 1d vector in a correct vector of sets which we
248 // will transform back to scatter it
249 std::vector<std::unordered_set<int>> setsFromRanks(ttk::MPIsize_);
250 auto begin = rootVector.begin();
251 auto end = rootVector.begin();
252 for(int i = 0; i < ttk::MPIsize_; i++) {
253 end = begin + sizes[i];
254 std::unordered_set<int> s(begin, end);
255 setsFromRanks[i] = s;
256 begin = end;
257 // std::cout << "R" << std::to_string(i) << " nr needs something from "
258 // << std::to_string(setsFromRanks[i].size()) << " neighbors." <<
259 // std::endl;
260 }
261 // now we need to check for each rank if they are a neighbor of any other
262 // rank. If so, we need to add those to the neighbors
263 for(int i = 0; i < ttk::MPIsize_; i++) {
264 for(int j = 0; j < ttk::MPIsize_; j++) {
265 if(setsFromRanks[j].find(i) != setsFromRanks[j].end()) {
266 setsFromRanks[i].emplace(j);
267 }
268 }
269 }
270 // now we transform this vector of sets back into a 1d vector to scatter
271 // it
272 for(int i = 0; i < ttk::MPIsize_; i++) {
273 sizes[i] = setsFromRanks[i].size();
274 if(i == 0) {
275 displacements[i] = 0;
276 } else {
277 displacements[i] = displacements[i - 1] + sizes[i - 1];
278 }
279 scatterVector.insert(scatterVector.end(), setsFromRanks[i].begin(),
280 setsFromRanks[i].end());
281 }
282 }
283
284 // scatter first the size and then the vector itself
285 int receivedSize;
286 MPI_Scatter(sizes, 1, MPI_INT, &receivedSize, 1, MPI_INT, 0, communicator);
287
288 // and then the actual neighbors
289 std::vector<int> receivedNeighbors(receivedSize);
290 MPI_Scatterv(scatterVector.data(), sizes, displacements, MPI_INT,
291 receivedNeighbors.data(), receivedSize, MPI_INT, 0,
292 communicator);
293 // then we turn the vector back into a set
294 std::unordered_set<int> finalSet(
295 receivedNeighbors.begin(), receivedNeighbors.end());
296 neighbors = finalSet;
297
298 return 0;
299 }

316 template <typename DT, typename IT>
317 int exchangeGhostCells(DT *scalarArray,
318 const int *const rankArray,
319 const LongSimplexId *const globalIds,
320 const std::unordered_map<IT, IT> &gidToLidMap,
321 const IT nVerts,
322 MPI_Comm communicator) {
323 if(!ttk::isRunningWithMPI()) {
324 return -1;
325 }
326 std::unordered_set<int> neighbors;
327 getNeighbors<IT>(neighbors, rankArray, nVerts, communicator);

337 // returns true if bounding boxes intersect, false if not
338 bool inline checkForIntersection(double *myBB, double *theirBB) {
339 return !(
340 myBB[0] > theirBB[1] // my left side is right of their right side
341 || myBB[1] < theirBB[0] // my right side is left of their left side
342 || myBB[2] > theirBB[3] // my bottom side is above their top side
343 || myBB[3] < theirBB[2] // my top side is under their bottom side
344 || myBB[4] > theirBB[5] // my front side is behind their back side
345 || myBB[5] < theirBB[4] // my back side is in front of their front side
346 );
347 }

357 void inline produceRankArray(std::vector<int> &rankArray,
358 LongSimplexId *globalIds,
359 unsigned char *ghostCells,
360 int nVertices,
361 double *boundingBox) {
362 std::vector<std::array<double, 6>> rankBoundingBoxes(ttk::MPIsize_);
363 std::copy(
364 boundingBox, boundingBox + 6, rankBoundingBoxes[ttk::MPIrank_].begin());
365 for(int r = 0; r < ttk::MPIsize_; r++) {
366 MPI_Bcast(rankBoundingBoxes[r].data(), 6, MPI_DOUBLE, r, ttk::MPIcomm_);
367 }
368
369 double epsilon = 0.00001;
370 // inflate our own bounding box by epsilon
371 for(int i = 0; i < 6; i++) {
372 if(i % 2 == 0)
373 boundingBox[i] -= epsilon;
374 if(i % 2 == 1)
375 boundingBox[i] += epsilon;
376 }
377 std::vector<int> neighbors;
378 for(int i = 0; i < ttk::MPIsize_; i++) {
379 if(i != ttk::MPIrank_) {
380 double *theirBoundingBox = rankBoundingBoxes[i].data();
381 if(checkForIntersection(boundingBox, theirBoundingBox)) {
382 neighbors.push_back(i);
383 }
384 }
385 }
386 MPI_Datatype MIT = ttk::getMPIType(ttk::SimplexId{});
387 std::vector<ttk::SimplexId> currentRankUnknownIds;
388 std::vector<std::vector<ttk::SimplexId>> allUnknownIds(ttk::MPIsize_);
389 std::unordered_set<ttk::SimplexId> gIdSet;
390 std::unordered_map<ttk::SimplexId, ttk::SimplexId> gIdToLocalMap;
391
392 for(int i = 0; i < nVertices; i++) {
393 int ghostCellVal = ghostCells[i];
394 ttk::SimplexId globalId = globalIds[i];
395 if(ghostCellVal == 0) {
396 // if the ghost cell value is 0, then this vertex mainly belongs to
397 // this rank
398 rankArray[i] = ttk::MPIrank_;
399 gIdSet.insert(globalId);
400 } else {
401 // otherwise the vertex belongs to another rank and we need to find
402 // out to which one this needs to be done by broadcasting the global
403 // id and hoping for some other rank to answer
404 currentRankUnknownIds.push_back(globalId);
405 gIdToLocalMap[globalId] = i;
406 }
407 }
408
409 ttk::SimplexId sizeOfCurrentRank = currentRankUnknownIds.size();
410 std::vector<ttk::SimplexId> gIdsToSend;
411 std::vector<ttk::SimplexId> receivedGlobals;
412 receivedGlobals.resize(sizeOfCurrentRank);
413 ttk::SimplexId sizeOfNeighbor;
414 std::vector<ttk::SimplexId> neighborUnknownIds;
415 for(int neighbor : neighbors) {
416 // we first send the size and then all needed ids to the neighbor
417 MPI_Sendrecv(&sizeOfCurrentRank, 1, MIT, neighbor, ttk::MPIrank_,
418 &sizeOfNeighbor, 1, MIT, neighbor, neighbor, ttk::MPIcomm_,
419 MPI_STATUS_IGNORE);
420 neighborUnknownIds.resize(sizeOfNeighbor);
421 gIdsToSend.reserve(sizeOfNeighbor);
422
423 MPI_Sendrecv(currentRankUnknownIds.data(), sizeOfCurrentRank, MIT,
424 neighbor, ttk::MPIrank_, neighborUnknownIds.data(),
425 sizeOfNeighbor, MIT, neighbor, neighbor, ttk::MPIcomm_,
426 MPI_STATUS_IGNORE);
427
428 // then we check if the needed globalid values are present in the local
429 // globalid set if so, we send the rank value to the requesting rank
430 for(ttk::SimplexId gId : neighborUnknownIds) {
431 if(gIdSet.count(gId)) {
432 // add the value to the vector which will be sent
433 gIdsToSend.push_back(gId);
434 }
435 }
436 MPI_Status status;
437 int amount;
438
439 MPI_Sendrecv(gIdsToSend.data(), gIdsToSend.size(), MIT, neighbor,
440 ttk::MPIrank_, receivedGlobals.data(),
441 currentRankUnknownIds.size(), MIT, neighbor, neighbor,
442 ttk::MPIcomm_, &status);
443
444 MPI_Get_count(&status, MIT, &amount);
445 receivedGlobals.resize(amount);
446
447 for(ttk::SimplexId receivedGlobal : receivedGlobals) {
448 ttk::SimplexId localVal = gIdToLocalMap[receivedGlobal];
449 rankArray[localVal] = neighbor;
450 }
451 // cleanup
452 gIdsToSend.clear();
453 receivedGlobals.resize(sizeOfCurrentRank);
454 }
455 }

466 void returnVectorForBurstsize(std::vector<value<DT, IT>> &outVector,
467 std::vector<value<DT, IT>> &values,
468 size_t burstSize) {
469
470 if(burstSize > values.size()) {
471 outVector.resize(values.size(), {0, 0});
472 outVector.assign(values.begin(), values.end());
473 values.clear();
474 } else {

530 template <typename DT, typename IT>
531 void getMax(int intTag,
532 int structTag,
533 IT &currentOrder,
534 int burstSize,
535 MPI_Datatype MPI_IT,
536 std::vector<value<DT, IT>> &finalValues,
537 std::vector<std::vector<value<DT, IT>>> &unsortedReceivedValues,
538 std::vector<std::vector<IT>> &orderResendValues,
539 std::vector<IT> &orderedValuesForRank,
540 std::vector<value<DT, IT>> &sortingValues) {
541 // take the current maximum scalar over all ranks
542 int rankIdOfMaxScalar = -1;
543 DT maxScalar = std::numeric_limits<DT>::lowest();
544 IT maxGId = -1;
545 for(size_t i = 0; i < unsortedReceivedValues.size(); i++) {
546 if(unsortedReceivedValues[i].size() > 0) {
547 const auto &v = unsortedReceivedValues[i].back();
548 if(v.scalar == maxScalar ? v.globalId > maxGId : v.scalar > maxScalar) {
549 maxScalar = v.scalar;
550 maxGId = v.globalId;
551 rankIdOfMaxScalar = i;
552 }
553 }
554 }
555
556 // move the struct from the unsortedReceivedValues subvector to the
557 // finalValues vector to get an ordering
558 value<DT, IT> currentValue
559 = unsortedReceivedValues[rankIdOfMaxScalar].back();
560 // we send the globalId and the the order via one send command,
561 // therefore we need to check two concurrent values at the same time
562 // later on
563 orderResendValues[rankIdOfMaxScalar].push_back(currentValue.globalId);
564 orderResendValues[rankIdOfMaxScalar].push_back(currentOrder);
565 currentOrder--;
566 finalValues.push_back(currentValue);
567 unsortedReceivedValues[rankIdOfMaxScalar].pop_back();
568 if(unsortedReceivedValues[rankIdOfMaxScalar].size() == 0) {
569 if(rankIdOfMaxScalar == 0) {
570 // append the ordered values to the correct vector
571 orderedValuesForRank.insert(
572 orderedValuesForRank.end(),
573 orderResendValues[rankIdOfMaxScalar].begin(),
574 orderResendValues[rankIdOfMaxScalar].end());
575 orderResendValues[rankIdOfMaxScalar].clear();
576
577 if(!sortingValues.empty()) {
578 std::vector<value<DT, IT>> ownValues;
579 returnVectorForBurstsize<DT, IT>(ownValues, sortingValues, burstSize);
580
581 unsortedReceivedValues[rankIdOfMaxScalar] = ownValues;
582 }
583 } else {
584 // receive more values from rank, send ordering to the rank
585 // send to the finished rank that we want more
586 MPI_Send(orderResendValues[rankIdOfMaxScalar].data(),
587 orderResendValues[rankIdOfMaxScalar].size(), MPI_IT,
588 rankIdOfMaxScalar, intTag * rankIdOfMaxScalar, ttk::MPIcomm_);
589 orderResendValues[rankIdOfMaxScalar].clear();
590 // we receive more vals, if the size is still zero afterwards, we are
591 // finished with this rank
592 ReceiveAndAddToVector(
593 rankIdOfMaxScalar, structTag, unsortedReceivedValues);
594 }
595 }

690 template <typename DT, typename IT>
691 void produceOrdering(SimplexId *orderArray,
692 const DT *scalarArray,
693 const IT *globalIds,
694 const int *rankArray,
695 const size_t nVerts,
696 const int burstSize) {
697 int intTag = 101;
698 int structTag = 102;
699
700 MPI_Barrier(ttk::MPIcomm_);
701
702 MPI_Datatype MPI_IT = ttk::getMPIType(static_cast<IT>(0));
703
704 ttk::Timer fillAndSortTimer;
705 std::vector<value<DT, IT>> sortingValues;
706 std::vector<IT> gidsToGetVector;
707 std::unordered_map<IT, IT> gidToLidMap;
708 populateVector<DT, IT>(sortingValues, gidsToGetVector, gidToLidMap, nVerts,
709 scalarArray, globalIds, rankArray);
710
711 // sort the scalar array distributed first by the scalar value itself,
712 // then by the global id
713 sortVerticesDistributed<DT, IT>(sortingValues, ttk::globalThreadNumber_);
714
715 // when all are done sorting, rank 0 requests the highest values and
716 // merges them
717 MPI_Barrier(ttk::MPIcomm_);
718
719 std::vector<IT> orderedValuesForRank;
720 std::vector<value<DT, IT>> finalValues;
721 ttk::Timer mergeTimer;
722 IT localSize = sortingValues.size();
723 IT totalSize;
724 // get the complete size of the dataset by summing up the local sizes
725 MPI_Reduce(&localSize, &totalSize, 1, MPI_IT, MPI_SUM, 0, ttk::MPIcomm_);
726 if(ttk::MPIrank_ == 0) {
727 finalValues.reserve(totalSize);
728 IT currentOrder = totalSize - 1;
729 std::vector<std::vector<value<DT, IT>>> unsortedReceivedValues;
730 unsortedReceivedValues.resize(ttk::MPIsize_);
731 std::vector<std::vector<IT>> orderResendValues;
732 orderResendValues.resize(ttk::MPIsize_);
733 for(int i = 0; i < ttk::MPIsize_; i++) {
734 orderResendValues[i].reserve(burstSize);
735 }
736
737 // receive the first batch of values
738 for(int i = 0; i < ttk::MPIsize_; i++) {
739 if(i == 0) {
740 std::vector<value<DT, IT>> ownValues;
741 returnVectorForBurstsize<DT, IT>(ownValues, sortingValues, burstSize);
742 unsortedReceivedValues[i] = ownValues;
743 } else {
744 ReceiveAndAddToVector<DT, IT>(i, structTag, unsortedReceivedValues);
745 }
746 }
747 while(finalValues.size() < (size_t)totalSize) {
748 getMax<DT, IT>(intTag, structTag, currentOrder, burstSize, MPI_IT,
749 finalValues, unsortedReceivedValues, orderResendValues,
750 orderedValuesForRank, sortingValues);
751 }
752
753 } else { // other Ranks
754 // send the next burstsize values and then wait for an answer from the
755 // root rank
756 while(!sortingValues.empty()) {
757 std::vector<value<DT, IT>> sendValues;
758 returnVectorForBurstsize<DT, IT>(sendValues, sortingValues, burstSize);
759 int size = sendValues.size();
760 MPI_Send(sendValues.data(), size * sizeof(value<DT, IT>), MPI_CHAR, 0,
761 structTag * ttk::MPIrank_, ttk::MPIcomm_);
762 std::vector<IT> receivedValues;
763
764 // be prepared to receive burstsize of elements, resize after
765 // receiving to the correct size
766 receivedValues.resize(burstSize * 2);
767 MPI_Status status;
768 int amount;
769 MPI_Recv(receivedValues.data(), burstSize * 2, MPI_IT, 0,
770 intTag * ttk::MPIrank_, ttk::MPIcomm_, &status);
771 MPI_Get_count(&status, MPI_IT, &amount);
772
773 receivedValues.resize(amount);
774 orderedValuesForRank.insert(orderedValuesForRank.end(),
775 receivedValues.begin(),
776 receivedValues.end());
777 }
778 // afterwards send once a message of length 0 to root to show that we
779 // are done
780 MPI_Send(sortingValues.data(), 0, MPI_CHAR, 0, structTag * ttk::MPIrank_,
781 ttk::MPIcomm_);
782 }
783
784 // all ranks do the following
785 MPI_Barrier(ttk::MPIcomm_);
786
