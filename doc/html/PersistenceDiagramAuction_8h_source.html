173 void buildKDTree() {
174 Timer t;
175 default_kdt_ = KDT{true, wasserstein_};
176 const int dimension
177 = geometricalFactor_ >= 1 ? (geometricalFactor_ <= 0 ? 3 : 2) : 5;
178 std::vector<double> coordinates;
179 for(size_t i = 0; i < goods_.size(); i++) {
180 Good &g = goods_[i];
181 if(geometricalFactor_ > 0) {
182 coordinates.push_back(geometricalFactor_ * g.x_);
183 coordinates.push_back(geometricalFactor_ * g.y_);
184 }
185 if(geometricalFactor_ < 1) {
186 coordinates.push_back((1 - geometricalFactor_) * g.coords_[0]);
187 coordinates.push_back((1 - geometricalFactor_) * g.coords_[1]);
188 coordinates.push_back((1 - geometricalFactor_) * g.coords_[2]);
189 }
190 }
191 correspondance_kdt_map_
192 = kdt_.build(coordinates.data(), goods_.size(), dimension);
