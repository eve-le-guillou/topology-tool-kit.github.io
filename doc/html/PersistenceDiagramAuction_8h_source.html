213 void buildKDTree() {
214 Timer t;
215 default_kdt_ = KDTree<dataType>(true, wasserstein_);
216 const int dimension
217 = geometricalFactor_ >= 1 ? (geometricalFactor_ <= 0 ? 3 : 2) : 5;
218 std::vector<dataType> coordinates;
219 for(int i = 0; i < goods_.size(); i++) {
220 Good<dataType> &g = goods_.get(i);
221 if(geometricalFactor_ > 0) {
222 coordinates.push_back(geometricalFactor_ * g.x_);
223 coordinates.push_back(geometricalFactor_ * g.y_);
224 }
225 if(geometricalFactor_ < 1) {
226 coordinates.push_back((1 - geometricalFactor_) * g.coords_x_);
227 coordinates.push_back((1 - geometricalFactor_) * g.coords_y_);
228 coordinates.push_back((1 - geometricalFactor_) * g.coords_z_);
229 }
230 }
231 correspondance_kdt_map_
232 = kdt_.build(coordinates.data(), goods_.size(), dimension);

239 void initializeEpsilon() {
240 dataType max_persistence = 0;
241 for(int i = 0; i < bidders_.size(); i++) {
242 Bidder<dataType> &b = bidders_.get(i);
243 dataType persistence = b.getPersistence();
244 if(persistence > max_persistence) {
245 max_persistence = persistence;

248
249 for(int i = 0; i < goods_.size(); i++) {
250 Good<dataType> &g = goods_.get(i);
251 dataType persistence = g.getPersistence();
252 if(persistence > max_persistence) {
253 max_persistence = persistence;
254 }
255 }
256 this->epsilon_ = 5.0 / 4 * Geometry::pow(max_persistence, wasserstein_);
