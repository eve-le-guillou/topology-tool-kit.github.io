373 template <class dataType>
374 void updateBarycenterTreeStructure(
375 std::vector<ftm::FTMTree_MT *> &trees,

377 std::vector<std::vector<std::tuple<ftm::idNode, ftm::idNode, double>>>

380 ftm::idNode baryTreeRoot = baryTree->getRoot();
381
382 // Init matching matrix
383 // m[i][j] contains the node in the barycenter matched to the jth node of
384 // the ith tree
385 std::vector<std::vector<ftm::idNode>> matrixMatchings(trees.size());
386 std::vector<bool> baryMatched(baryTree->getNumberOfNodes(), false);
387 for(unsigned int i = 0; i < matchings.size(); ++i) {
388 auto matching = matchings[i];
389 matrixMatchings[i]
390 = std::vector<ftm::idNode>(trees[i]->getNumberOfNodes(), -1);
391 for(auto match : matching) {
392 matrixMatchings[i][std::get<1>(match)] = std::get<0>(match);
393 baryMatched[std::get<0>(match)] = true;
394 }
395 }
396
397 // Iterate through trees to get the nodes to add in the barycenter
398 std::vector<std::vector<ftm::idNode>> nodesToAdd(trees.size());
399 for(unsigned int i = 0; i < trees.size(); ++i) {
400 ftm::idNode root = trees[i]->getRoot();
401 std::queue<ftm::idNode> queue;
402 queue.emplace(root);
403 while(!queue.empty()) {
404 ftm::idNode node = queue.front();
405 queue.pop();
406 bool processChildren = true;
407 if((int)matrixMatchings[i][node]
408 == -1) { // if node in trees[i] is not matched
409 if(not keepSubtree_) {
410 processChildren = false;
411 nodesToAdd[i].push_back(node);
412 } else {
413 // not todo manage if keepSubtree=true (not important since it is
414 // not a valid merge tree)
415 printErr(
416 "barycenter with keepSubtree_=true is not implemented yet");
417 }
418 }
419 if(processChildren) {
420 std::vector<ftm::idNode> children;
421 trees[i]->getChildren(node, children);
422 for(auto child : children)
423 if(not(trees[i]->isThereOnlyOnePersistencePair()
424 and trees[i]->isLeaf(child)))
425 queue.emplace(child);
426 }
427 }
428 }
429
430 bool foundRootNotMatched = false;
431 for(unsigned int i = 0; i < trees.size(); ++i)
432 foundRootNotMatched |= baryTree->isNodeIdInconsistent(
433 matrixMatchings[i][trees[i]->getRoot()]);
434 if(foundRootNotMatched)
435 printWrn("[updateBarycenterTreeStructure] an input tree has its root "
436 "not matched.");
437
438 // Delete nodes that are not matched in the barycenter
439 for(unsigned int i = 0; i < baryTree->getNumberOfNodes(); ++i)
440 if(not baryMatched[i])
441 baryTree->deleteNode(i);
442
443 if(not keepSubtree_) {
444 // Add scalars and nodes not present in the barycenter
445 ftm::idNode nodeCpt = baryTree->getNumberOfNodes();
446 std::vector<std::tuple<ftm::idNode, ftm::idNode, int>> nodesToProcess;
447 std::vector<dataType> newScalarsVector;
448 ftm::getTreeScalars<dataType>(baryMergeTree, newScalarsVector);
449 for(unsigned int i = 0; i < nodesToAdd.size(); ++i) {
450 for(auto node : nodesToAdd[i]) {
451 ftm::idNode parent
452 = matrixMatchings[i][trees[i]->getParentSafe(node)];
453 if(matchings[i].size() == 0)
454 parent = baryTreeRoot;
455
456 if((baryTree->isNodeIdInconsistent(parent)
457 or baryTree->isNodeAlone(parent))
458 and matchings[i].size() != 0) {
459 std::stringstream ss;
460 ss << trees[i]->getParentSafe(node) << " _ " << node;
461 printMsg(ss.str());
462 printMsg(trees[i]->printTree().str());
463 printMsg(trees[i]->printPairsFromTree<dataType>(true).str());
464 printMatching(matchings[i]);
465 std::stringstream ss2;
466 ss2 << "parent " << parent;
467 printMsg(ss2.str());
468 }
469 /*if(isRoot(trees[i], node))
470 parent = baryTree->getRoot();*/
471 std::vector<dataType> addedScalars;
472 nodeCpt = getNodesAndScalarsToAdd<dataType>(
473 baryMergeTree, parent, trees[i], node, addedScalars,
474 nodesToProcess, nodeCpt, i);
475 newScalarsVector.insert(
476 newScalarsVector.end(), addedScalars.begin(), addedScalars.end());
477 }
478 }
479 if(addNodes_) {
480 std::vector<std::vector<std::tuple<ftm::idNode, ftm::idNode>>>
481 nodesProcessed;
482 updateNodesAndScalars<dataType>(baryMergeTree, trees.size(),
483 nodesToProcess, newScalarsVector,
484 nodesProcessed);
485 for(unsigned int i = 0; i < matchings.size(); ++i) {
486 std::vector<std::tuple<ftm::idNode, ftm::idNode, double>>
487 nodesProcessedT;
488 for(auto tup : nodesProcessed[i])
489 nodesProcessedT.emplace_back(
490 std::get<0>(tup), std::get<1>(tup), -1);
491 matchings[i].insert(matchings[i].end(), nodesProcessedT.begin(),
492 nodesProcessedT.end());
493 }
494 }
495 } else {
496 // not todo manage if keepSubtree=true (not important since it is not a
497 // valid merge tree)
498 printErr("barycenter with keepSubtree_=true is not implemented yet");
499 }

536 template <class dataType>
537 std::tuple<dataType, dataType>
538 interpolation(ftm::MergeTree<dataType> &baryMergeTree,
539 ftm::idNode nodeId,
540 std::vector<dataType> &newScalarsVector,
541 std::vector<ftm::FTMTree_MT *> &trees,
542 std::vector<ftm::idNode> &nodes,
543 std::vector<double> &alphas) {
544 ftm::FTMTree_MT *baryTree = &(baryMergeTree.tree);
545 dataType mu_max = getMinMaxLocalFromVector<dataType>(
546 baryTree, nodeId, newScalarsVector, false);
547 dataType mu_min = getMinMaxLocalFromVector<dataType>(
548 baryTree, nodeId, newScalarsVector);
549 double newBirth = 0, newDeath = 0;
550
551 // Compute projection
552 double tempBirth = 0, tempDeath = 0;
553 int offDiagonal = 0;
554 double alphaSum = 0;
555 for(unsigned int i = 0; i < trees.size(); ++i)
556 if((int)nodes[i] != -1)
557 alphaSum += alphas[i];
558 for(unsigned int i = 0; i < trees.size(); ++i) {
559 if((int)nodes[i] != -1) { // if node is matched in trees[i]
560 auto iBirthDeath = getParametrizedBirthDeathFromVector<dataType>(
561 trees[i], nodes[i], baryTree, nodeId, newScalarsVector);
562 double tTempBirth = 0, tTempDeath = 0;
563 tTempBirth += std::get<0>(iBirthDeath);
564 tTempDeath += std::get<1>(iBirthDeath);
565 if(normalizedWasserstein_ and rescaledWasserstein_) {
566 auto newMinMax = getNewMinMaxFromVector<dataType>(
567 trees[i], nodes[i], baryTree, nodeId, newScalarsVector);
568 tTempBirth /= (std::get<1>(newMinMax) - std::get<0>(newMinMax));
569 tTempDeath /= (std::get<1>(newMinMax) - std::get<0>(newMinMax));
570 }
571 tempBirth += tTempBirth * alphas[i] / alphaSum;
572 tempDeath += tTempDeath * alphas[i] / alphaSum;
573 ++offDiagonal;
574 }
575 }
576 double projec = (tempBirth + tempDeath) / 2;
577
578 // Compute newBirth and newDeath
579 dataType divisor = 0;
580 for(unsigned int i = 0; i < trees.size(); ++i) {
581 double iBirth = projec, iDeath = projec;
582 if((int)nodes[i] != -1) { // if node is matched in trees[i]
583 auto iBirthDeath = getParametrizedBirthDeathFromVector<dataType>(
584 trees[i], nodes[i], baryTree, nodeId, newScalarsVector);
585 iBirth = std::get<0>(iBirthDeath);
586 iDeath = std::get<1>(iBirthDeath);
587 }
588 if(normalizedWasserstein_ and rescaledWasserstein_) {
589 dataType beta_max = 1, beta_min = 0;
590 auto newMinMax
591 = ((int)nodes[i] == -1)
592 ? getNewMinMax<dataType>(baryTree, nodeId, baryTree, nodeId)
593 : getNewMinMaxFromVector<dataType>(
594 trees[i], nodes[i], baryTree, nodeId, newScalarsVector);
595 if((int)nodes[i] == -1) {
596 beta_max = mu_max;
597 beta_min = mu_min;
598 iBirth *= (beta_max - beta_min);
599 iDeath *= (beta_max - beta_min);
600 } else {
601 beta_min = std::get<0>(newMinMax);
602 beta_max = std::get<1>(newMinMax);
603 }
604 iBirth *= (beta_max - beta_min);
605 iDeath *= (beta_max - beta_min);
606 divisor += alphas[i] * (beta_max - beta_min) * (beta_max - beta_min);
607 }
608 newBirth += alphas[i] * iBirth;
609 newDeath += alphas[i] * iDeath;
610 }
611 if(normalizedWasserstein_ and rescaledWasserstein_) {
612 newBirth /= divisor;
613 newDeath /= divisor;
614 }
615 if(normalizedWasserstein_ or rescaledWasserstein_) {
616 newBirth = newBirth * (mu_max - mu_min) + mu_min;
617 newDeath = newDeath * (mu_max - mu_min) + mu_min;
618 }
619
620 return std::make_tuple(newBirth, newDeath);
621 }

649 newBirth *= (beta_max - beta_min);
650 newDeath *= (beta_max - beta_min);
651 projec = projec * (mu_max - mu_min) * (mu_max - mu_min)
652 / (beta_max - beta_min);
653 divisor = alpha * (beta_max - beta_min) * (beta_max - beta_min)
654 + (1 - alpha) * (mu_max - mu_min) * (mu_max - mu_min);
655 }
656
657 newBirth = alpha * newBirth + (1 - alpha) * projec;
658 newDeath = alpha * newDeath + (1 - alpha) * projec;
659
660 if(normalizedWasserstein_ and rescaledWasserstein_) {
661 newBirth /= divisor;
662 newDeath /= divisor;
663 }
664
665 if(normalizedWasserstein_ or rescaledWasserstein_) {
666 newBirth = newBirth * (mu_max - mu_min) + mu_min;
667 newDeath = newDeath * (mu_max - mu_min) + mu_min;
668 }
669
670 dataType newBirthT = newBirth;
671 dataType newDeathT = newDeath;
672 return std::make_tuple(newBirthT, newDeathT);
673 }

675 template <class dataType>
676 void purgeBarycenter(ftm::MergeTree<dataType> &baryMergeTree,
677 std::vector<std::vector<ftm::idNode>> &baryMatching,
678 std::vector<ftm::FTMTree_MT *> &trees,
679 std::vector<double> &alphas) {
680 ftm::FTMTree_MT *baryTree = &(baryMergeTree.tree);
681 std::vector<bool> nodesProcessed(baryTree->getNumberOfNodes(), false);
682 std::vector<dataType> nodesMatchingCost(baryTree->getNumberOfNodes(), 0);
683 std::vector<dataType> nodesDestructCost(baryTree->getNumberOfNodes(), 0);
684 std::vector<ftm::idNode> leaves;
685 baryTree->getLeavesFromTree(leaves);
686 std::queue<ftm::idNode> queue;
687 for(auto leaf : leaves)
688 queue.emplace(leaf);
689 while(!queue.empty()) {
690 ftm::idNode node = queue.front();
691 queue.pop();
692 for(unsigned int i = 0; i < trees.size(); ++i) {
693 dataType newMatchingCost = alphas[i];
694 dataType newDestructCost = alphas[i];
695 if((int)baryMatching[node][i] != -1) {
696 newMatchingCost *= relabelCost<dataType>(
697 baryTree, node, trees[i], baryMatching[node][i]);
698 newDestructCost
699 *= deleteCost<dataType>(trees[i], baryMatching[node][i]);
700 } else {
701 newMatchingCost *= deleteCost<dataType>(baryTree, node);
702 newDestructCost *= 0;
703 }
704 nodesMatchingCost[node] += newMatchingCost;
705 nodesDestructCost[node] += newDestructCost;
706 }
707 std::vector<ftm::idNode> children;
708 baryTree->getChildren(node, children);
709 for(auto child : children) {
710 nodesMatchingCost[node] += nodesMatchingCost[child];
711 nodesDestructCost[node] += nodesDestructCost[child];
712 }
713 nodesProcessed[node] = true;
714 if(not nodesProcessed[baryTree->getParentSafe(node)])
715 queue.emplace(baryTree->getParentSafe(node));
716
717 // Destruct subtree if better
718 if(nodesDestructCost[node] < nodesMatchingCost[node]) {
719 baryTree->deleteSubtree(node);
720 nodesDestructCost[node] = 0;
721 nodesMatchingCost[node] = 0;
722 }
723 }

726 template <class dataType>
727 void updateBarycenterTreeScalars(
728 std::vector<ftm::FTMTree_MT *> &trees,
729 ftm::MergeTree<dataType> &baryMergeTree,
730 std::vector<double> &alphas,
731 int indexAddedNodes,
732 std::vector<std::vector<std::tuple<ftm::idNode, ftm::idNode, double>>>
733 &matchings) {
734 ftm::FTMTree_MT *baryTree = &(baryMergeTree.tree);
735 bool isJT = baryTree->isJoinTree<dataType>();
736
737 // Init matching matrix
738 // m[i][j] contains the node in trees[j] matched to the node i in the
739 // barycenter
740 std::vector<std::vector<ftm::idNode>> baryMatching(
741 baryTree->getNumberOfNodes(),
742 std::vector<ftm::idNode>(trees.size(), -1));
743 std::vector<int> nodesAddedTree(baryTree->getNumberOfNodes(), -1);
744 for(unsigned int i = 0; i < matchings.size(); ++i) {
745 auto matching = matchings[i];
746 for(auto match : matching) {
747 baryMatching[std::get<0>(match)][i] = std::get<1>(match);
748 if((int)std::get<0>(match)
749 >= indexAddedNodes) // get the tree of this added node
750 nodesAddedTree[std::get<0>(match)] = i;
751 }
752 }
753
754 // Interpolate scalars
755 std::vector<dataType> newScalarsVector(baryTree->getNumberOfNodes());
756 ftm::idNode root = baryTree->getRoot();
757 std::queue<ftm::idNode> queue;
758 queue.emplace(root);
759 while(!queue.empty()) {
760 ftm::idNode node = queue.front();
761 queue.pop();
762 std::tuple<dataType, dataType> newBirthDeath;
763 if((int)node < indexAddedNodes) {
764 newBirthDeath
765 = interpolation<dataType>(baryMergeTree, node, newScalarsVector,
766 trees, baryMatching[node], alphas);
767 } else {
768 int i = nodesAddedTree[node];
769 ftm::idNode nodeT = baryMatching[node][i];
770 newBirthDeath = interpolationAdded<dataType>(
771 trees[i], nodeT, alphas[i], baryMergeTree, node, newScalarsVector);
772 }
773 dataType nodeScalar
774 = (isJT ? std::get<1>(newBirthDeath) : std::get<0>(newBirthDeath));
775 dataType nodeOriginScalar
776 = (isJT ? std::get<0>(newBirthDeath) : std::get<1>(newBirthDeath));
777 newScalarsVector[node] = nodeScalar;
778 newScalarsVector[baryTree->getNode(node)->getOrigin()]
779 = nodeOriginScalar;
780 std::vector<ftm::idNode> children;
781 baryTree->getChildren(node, children);
782 for(auto child : children)
783 queue.emplace(child);
784 }
785
786 if(baryMergeTree.tree.isFullMerge()) {
787 auto mergedRootOrigin = baryTree->getMergedRootOrigin<dataType>();
788 dataType mergedRootOriginScalar = 0.0;
789 for(unsigned int i = 0; i < trees.size(); ++i)
790 mergedRootOriginScalar += trees[i]->getValue<dataType>(
791 trees[i]->getMergedRootOrigin<dataType>());
792 mergedRootOriginScalar /= trees.size();
793 newScalarsVector[mergedRootOrigin] = mergedRootOriginScalar;
794 }
795
796 setTreeScalars(baryMergeTree, newScalarsVector);
797
798 if(normalizedWasserstein_ and rescaledWasserstein_)
799 purgeBarycenter<dataType>(baryMergeTree, baryMatching, trees, alphas);
800 std::vector<ftm::idNode> deletedNodesT;
801 persistenceThresholding<dataType>(
802 &(baryMergeTree.tree), 0, deletedNodesT);
803 limitSizeBarycenter(baryMergeTree, trees);
804 ftm::cleanMergeTree<dataType>(baryMergeTree);
805 }

956 template <class dataType>
957 unsigned int
958 persistenceScaling(std::vector<ftm::FTMTree_MT *> &trees,
959 std::vector<ftm::MergeTree<dataType>> &mergeTrees,
960 std::vector<ftm::FTMTree_MT *> &oriTrees,
961 int iterationNumber,
962 std::vector<std::vector<ftm::idNode>> &deletedNodes) {
963 deletedNodes = std::vector<std::vector<ftm::idNode>>(oriTrees.size());
964 unsigned int noTreesUnscaled = 0;
965
966 // Scale trees
967 for(unsigned int i = 0; i < oriTrees.size(); ++i) {
968 double persistenceThreshold = 50.0;
969 if(iterationNumber != -1) {
970 // Get number of pairs in scaled merge tree
971 int noPairs = mergeTrees[i].tree.getRealNumberOfNodes();
972
973 // Get pairs in original merge tree
974 std::vector<std::tuple<ftm::idNode, ftm::idNode, dataType>> pairs;
975 oriTrees[i]->getPersistencePairsFromTree<dataType>(
976 pairs, branchDecomposition_);
977
978 // Compute new persistence threshold
979 double multiplier = (progressiveSpeedDivisor_ < 1e-6
980 ? 1.
981 : iterationNumber / progressiveSpeedDivisor_);
982 int decrement = multiplier * pairs.size() / 10;
983 int thresholdIndex = pairs.size() - noPairs - std::max(decrement, 2);
984 thresholdIndex = std::max(thresholdIndex, 0);
985 const double persistence = std::get<2>(pairs[thresholdIndex]);
986 persistenceThreshold
987 = persistence / std::get<2>(pairs.back()) * 100.0;
988 if(thresholdIndex == 0) {
989 persistenceThreshold = 0.;
990 ++noTreesUnscaled;
991 }
992 }
993 if(persistenceThreshold != 0.) {
994 ftm::MergeTree<dataType> mt
995 = ftm::copyMergeTree<dataType>(oriTrees[i]);
996 persistenceThresholding<dataType>(
997 &(mt.tree), persistenceThreshold, deletedNodes[i]);
998 if(mergeTrees.size() == 0)
999 mergeTrees = std::vector<ftm::MergeTree<dataType>>(oriTrees.size());
1000 mergeTrees[i] = mt;
1001 trees[i] = &(mt.tree);
1002 } else {
1003 trees[i] = oriTrees[i];
1004 }
1005 }
1006
1007 printTreesStats(trees);
1008
1009 return noTreesUnscaled;
1010 }

1050
1051 // Print bary stats
1052 printBaryStats(baryTree);
1053

1091

1093 dataType currentFrechetEnergy = 0;
1094 for(unsigned int i = 0; i < trees.size(); ++i)
1095 currentFrechetEnergy += alphas[i] * distances[i] * distances[i];

1098 converged = (frechetDiff <= tol_);
1099 converged = converged and (not progressiveBarycenter_ or treesUnscaled);
1100 frechetEnergy = currentFrechetEnergy;
1101 tol_ = frechetEnergy / 125.0;
1102
1103 std::stringstream ss4;
1104 auto barycenterTime = t_bary.getElapsedTime() - addDeletedNodesTime_;
1105 printMsg("Total", 1, barycenterTime, this->threadNumber_,
1106 debug::LineMode::NEW, debug::Priority::INFO);
1107 printBaryStats(baryTree);
1108 ss4 << "Frechet energy : " << frechetEnergy;
1109 printMsg(ss4.str());
1110
1111 minFrechet = std::min(minFrechet, frechetEnergy);
1112 if(not converged and (not progressiveBarycenter_ or treesUnscaled)) {
1113 cptBlocked = (minFrechet < frechetEnergy) ? cptBlocked + 1 : 0;
1114 converged = (cptBlocked >= 10);
1115 }
1116
1117 // --- Persistence scaling
1118 if(progressiveBarycenter_) {
1119 unsigned int noTreesUnscaled = persistenceScaling<dataType>(
1120 trees, scaledMergeTrees, oriTrees, NoIteration, deletedNodes);
1121 treesUnscaled = (noTreesUnscaled == oriTrees.size());
1122 }
1123 }
1124
1125 // Final processing
1126 printMsg(debug::Separator::L2);
1127 printMsg("Final assignment");
1128
1129 std::vector<dataType> distances(trees.size(), -1);
1130 assignment<dataType>(trees, baryMergeTree, finalMatchings, distances,
1131 finalAsgnDoubleInput, finalAsgnFirstInput);
1132 for(auto dist : distances)
1133 finalDistances_.push_back(dist);
1134 dataType currentFrechetEnergy = 0;
1135 for(unsigned int i = 0; i < trees.size(); ++i)
1136 currentFrechetEnergy += alphas[i] * distances[i] * distances[i];
1137
1138 std::stringstream ss, ss2;
1139 ss << "Frechet energy : " << currentFrechetEnergy;
1140 printMsg(ss.str());
1141 auto barycenterTime = t_bary.getElapsedTime() - addDeletedNodesTime_;
1142 printMsg("Total", 1, barycenterTime, this->threadNumber_,
1143 debug::LineMode::NEW, debug::Priority::INFO);
1144 // std::cout << "Bary Distance Time = " << allDistanceTime_ << std::endl;
1145
1146 if(trees.size() == 2 and not isCalled_)
1147 verifyBarycenterTwoTrees<dataType>(
1148 trees, baryMergeTree, finalMatchings, distances);
1149
1150 // Persistence (un)scaling
1151 if(progressiveBarycenter_) {
1152 scaledMergeTrees.clear();
1153 trees.clear();
1154 trees.insert(trees.end(), oriTrees.begin(), oriTrees.end());
1155 }
