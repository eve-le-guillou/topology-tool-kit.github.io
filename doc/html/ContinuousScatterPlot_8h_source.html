92template <typename dataType1, typename dataType2, class triangulationType>
93int ttk::ContinuousScatterPlot::execute(
94 const dataType1 *scalars1,
95 const dataType2 *scalars2,
96 const triangulationType *triangulation) const {
97#ifndef TTK_ENABLE_KAMIKAZE
98 if(!scalars1)
99 return -1;
100 if(!scalars2)
101 return -2;
102 if(!triangulation)
103 return -3;
104 if(!density_)
105 return -4;
106
107 if(triangulation->getNumberOfCells() <= 0) {
108 this->printErr("no cells.");
109 return -5;
110 }
111
112 if(triangulation->getCellVertexNumber(0) != 4) {
113 this->printErr("no tetrahedra.");
114 return -6;
115 }
116#endif
117
118 Timer t;
119
120 // helpers:
121 const SimplexId numberOfCells = triangulation->getNumberOfCells();
122
123 // rendering helpers:
124 // constant ray direction (ortho)
125 const double d[3]{0, 0, -1};
126 const double delta[2]{
127 scalarMax_[0] - scalarMin_[0], scalarMax_[1] - scalarMin_[1]};
128 const double sampling[2]{
129 delta[0] / resolutions_[0], delta[1] / resolutions_[1]};
130 const double epsilon{0.000001};
131
132 std::vector<std::array<SimplexId, 3>> triangles{};
133
134#ifdef TTK_ENABLE_OPENMP
135#pragma omp parallel for num_threads(threadNumber_) firstprivate(triangles)
136#endif
137 for(SimplexId cell = 0; cell < numberOfCells; ++cell) {
138 bool isDummy{};
139
140 // get tetrahedron info
141 SimplexId vertex[4];
142 double data[4][3];
143 float position[4][3];
144 double localScalarMin[2]{};
145 double localScalarMax[2]{};
146 // for each triangle
147 for(int k = 0; k < 4; ++k) {
148 // get indices
149 triangulation->getCellVertex(cell, k, vertex[k]);
150
151 // get scalars
152 data[k][0] = scalars1[vertex[k]];
153 data[k][1] = scalars2[vertex[k]];
154 data[k][2] = 0;
155
156 if(withDummyValue_
157 and (data[k][0] == dummyValue_ or data[k][1] == dummyValue_)) {
158 isDummy = true;
159 break;
160 }
161
162 // get local stats
163 if(!k or localScalarMin[0] > data[k][0])
164 localScalarMin[0] = data[k][0];
165 if(!k or localScalarMin[1] > data[k][1])
166 localScalarMin[1] = data[k][1];
167 if(!k or localScalarMax[0] < data[k][0])
168 localScalarMax[0] = data[k][0];
169 if(!k or localScalarMax[1] < data[k][1])
170 localScalarMax[1] = data[k][1];
171
172 // get positions
173
174 triangulation->getVertexPoint(
175 vertex[k], position[k][0], position[k][1], position[k][2]);
176 }
177 if(isDummy)
178 continue;
179
180 // gradient:
181 double g0[3];
182 double g1[3];
183 {
184 double v12[3];
185 double v13[3];
186 double v14[3];
187 double s12[3];
188 double s13[3];
189 double s14[3];
190 for(int k = 0; k < 3; ++k) {
191 v12[k] = position[1][k] - position[0][k];
192 v13[k] = position[2][k] - position[0][k];
193 v14[k] = position[3][k] - position[0][k];
194
195 s12[k] = data[1][k] - data[0][k];
196 s13[k] = data[2][k] - data[0][k];
197 s14[k] = data[3][k] - data[0][k];
198 }
199
200 double a[3];
201 double b[3];
202 double c[3];
203 Geometry::crossProduct(v13, v12, a);
204 Geometry::crossProduct(v12, v14, b);
205 Geometry::crossProduct(v14, v13, c);
206 double det = Geometry::dotProduct(v14, a);
207 if(det == 0.) {
208 for(int k = 0; k < 3; ++k) {
209 g0[k] = 0.0;
210 g1[k] = 0.0;
211 }
212 } else {
213 double invDet = 1.0 / det;
214 for(int k = 0; k < 3; ++k) {
215 g0[k] = (s14[0] * a[k] + s13[0] * b[k] + s12[0] * c[k]) * invDet;
216 g1[k] = (s14[1] * a[k] + s13[1] * b[k] + s12[1] * c[k]) * invDet;
217 }
218 }
219 }
220
221 // volume:
222 double volume;
223 bool isLimit{};
224 {
225 double cp[3];
226 Geometry::crossProduct(g0, g1, cp);
227 volume = Geometry::magnitude(cp);
228 if(volume == 0.)
229 isLimit = true;
230 }
231
232 // classification:
233 int index[4]{0, 1, 2, 3};
234 bool isInTriangle{};
235 if(Geometry::isPointInTriangle(data[0], data[1], data[2], data[3]))
236 isInTriangle = true;
237 else if(Geometry::isPointInTriangle(data[0], data[1], data[3], data[2])) {
238 isInTriangle = true;
239 index[0] = 0;
240 index[1] = 1;
241 index[2] = 3;
242 index[3] = 2;
243 } else if(Geometry::isPointInTriangle(data[0], data[2], data[3], data[1])) {
244 isInTriangle = true;
245 index[0] = 0;
246 index[1] = 2;
247 index[2] = 3;
248 index[3] = 1;
249 } else if(Geometry::isPointInTriangle(data[1], data[2], data[3], data[0])) {
250 isInTriangle = true;
251 index[0] = 1;
252 index[1] = 2;
253 index[2] = 3;
254 index[3] = 0;
255 }
256
257 // projection:
258 double density{};
259 double imaginaryPosition[3]{};
260 triangles.clear();
261 std::array<SimplexId, 3> triangle{};
262 // class 0
263 if(isInTriangle) {
264 // mass density
265 double massDensity{};
266 {
267 double A;
268
269 Geometry::computeTriangleArea(
270 data[index[0]], data[index[1]], data[index[2]], A);
271 double invA = 1.0 / A;
272 if(A == 0.) {
273 invA = 0.0;
274 isLimit = true;
275 }
276
277 double alpha, beta, gamma;
278
279 Geometry::computeTriangleArea(
280 data[index[1]], data[index[2]], data[index[3]], alpha);
281
282 Geometry::computeTriangleArea(
283 data[index[0]], data[index[2]], data[index[3]], beta);
284
285 Geometry::computeTriangleArea(
286 data[index[0]], data[index[1]], data[index[3]], gamma);
287
288 alpha *= invA;
289 beta *= invA;
290 gamma *= invA;
291
292 double p0[3];
293 double p1[3];
294 for(int k = 0; k < 3; ++k) {
295 p0[k] = position[index[3]][k];
296
297 p1[k] = alpha * position[index[0]][k] + beta * position[index[1]][k]
298 + gamma * position[index[2]][k];
299 }
300 massDensity = Geometry::distance(p0, p1);
301 }
302
303 if(isLimit)
304 density = std::numeric_limits<decltype(density)>::max();
305 else
306 density = massDensity / volume;
307
308 triangle[0] = vertex[index[3]];
309 triangle[1] = vertex[index[0]];
310 triangle[2] = vertex[index[1]];
311 triangles.push_back(triangle);
312
313 triangle[0] = vertex[index[3]];
314 triangle[1] = vertex[index[0]];
315 triangle[2] = vertex[index[2]];
316 triangles.push_back(triangle);
317
318 triangle[0] = vertex[index[3]];
319 triangle[1] = vertex[index[1]];
320 triangle[2] = vertex[index[2]];
321 triangles.push_back(triangle);
322 }
323 // class 1
324 else {
325 double massDensity{};
326 double p[3]{0, 0, 0};
327 if(Geometry::computeSegmentIntersection(
328 data[0][0], data[0][1], data[1][0], data[1][1], data[2][0],
329 data[2][1], data[3][0], data[3][1], p[0], p[1])) {
330 index[0] = 0;
331 index[1] = 1;
332 index[2] = 2;
333 index[3] = 3;
334 } else if(Geometry::computeSegmentIntersection(
335 data[0][0], data[0][1], data[2][0], data[2][1], data[1][0],
336 data[1][1], data[3][0], data[3][1], p[0], p[1])) {
337 index[0] = 0;
338 index[1] = 2;
339 index[2] = 1;
340 index[3] = 3;
341 } else if(Geometry::computeSegmentIntersection(
342 data[0][0], data[0][1], data[3][0], data[3][1], data[1][0],
343 data[1][1], data[2][0], data[2][1], p[0], p[1])) {
344 index[0] = 0;
345 index[1] = 3;
346 index[2] = 1;
347 index[3] = 2;
348 }
349
350 double a = Geometry::distance(data[index[0]], p);
351 double b = Geometry::distance(data[index[0]], data[index[1]]);
352 double r0 = a / b;
353
354 a = Geometry::distance(data[index[2]], p);
355 b = Geometry::distance(data[index[2]], data[index[3]]);
356 double r1 = a / b;
357
358 double p0[3];
359 double p1[3];
360 for(int k = 0; k < 3; ++k) {
361
362 p0[k] = position[index[0]][k]
363 + r0 * (position[index[1]][k] - position[index[0]][k]);
364
365 p1[k] = position[index[2]][k]
366 + r1 * (position[index[3]][k] - position[index[2]][k]);
367 }
368 massDensity = Geometry::distance(p0, p1);
369
370 if(isLimit)
371 density = std::numeric_limits<decltype(density)>::max();
372 else
373 density = massDensity / volume;
374
375 imaginaryPosition[0] = p[0];
376 imaginaryPosition[1] = p[1];
377 imaginaryPosition[2] = 0;
378
379 // four triangles projection
380 triangle[0] = -1; // new geometry
381 triangle[1] = vertex[index[0]];
382 triangle[2] = vertex[index[2]];
383 triangles.push_back(triangle);
384
385 triangle[1] = vertex[index[2]];
386 triangle[2] = vertex[index[1]];
387 triangles.push_back(triangle);
388
389 triangle[1] = vertex[index[1]];
390 triangle[2] = vertex[index[3]];
391 triangles.push_back(triangle);
392
393 triangle[1] = vertex[index[3]];
394 triangle[2] = vertex[index[0]];
395 triangles.push_back(triangle);
396 }
397
398 // rendering:
399 // "Fast, Minimum Storage Ray/Triangle Intersection", Tomas Moller & Ben
400 // Trumbore
401 {
402 const SimplexId minI
403 = floor((localScalarMin[0] - scalarMin_[0]) / sampling[0]);
404 const SimplexId minJ
405 = floor((localScalarMin[1] - scalarMin_[1]) / sampling[1]);
406 const SimplexId maxI
407 = ceil((localScalarMax[0] - scalarMin_[0]) / sampling[0]);
408 const SimplexId maxJ
409 = ceil((localScalarMax[1] - scalarMin_[1]) / sampling[1]);
410
411 for(SimplexId i = minI; i < maxI; ++i) {
412 for(SimplexId j = minJ; j < maxJ; ++j) {
413 // set ray origin
414 const double o[3]{scalarMin_[0] + i * sampling[0],
415 scalarMin_[1] + j * sampling[1], 1};
416 for(unsigned int k = 0; k < triangles.size(); ++k) {
417 const auto &tr = triangles[k];
418
419 // get triangle info
420 double p0[3];
421 if(isInTriangle) {
422 p0[0] = scalars1[tr[0]];
423 p0[1] = scalars2[tr[0]];
424 } else {
425 p0[0] = imaginaryPosition[0];
426 p0[1] = imaginaryPosition[1];
427 }
428 p0[2] = 0;
429
430 const double p1[3]{
431 (double)scalars1[tr[1]], (double)scalars2[tr[1]], 0};
432 const double p2[3]{
433 (double)scalars1[tr[2]], (double)scalars2[tr[2]], 0};
434 const double e1[3]{p1[0] - p0[0], p1[1] - p0[1], 0};
435 const double e2[3]{p2[0] - p0[0], p2[1] - p0[1], 0};
436
437 double q[3];
438 Geometry::crossProduct(d, e2, q);
439 const double a = Geometry::dotProduct(e1, q);
440 if(a > -epsilon and a < epsilon)
441 continue;
442
443 const double f = 1.0 / a;
444 const double s[3]{o[0] - p0[0], o[1] - p0[1], 1};
445 const double u = f * Geometry::dotProduct(s, q);
446 if(u < 0.0)
447 continue;
448
449 double r[3];
450 Geometry::crossProduct(s, e1, r);
451 const double v = f * Geometry::dotProduct(d, r);
452 if(v < 0.0 or (u + v) > 1.0)
453 continue;
454
455 // triangle/ray intersection below
456#ifdef TTK_ENABLE_OPENMP
457#ifdef _WIN32
458#pragma omp atomic
459#else
460#pragma omp atomic update
461#endif
462#endif
463 (*density_)[i][j] += (1.0 - u - v) * density;
464
465#ifdef TTK_ENABLE_OPENMP
466#ifdef _WIN32
467#pragma omp atomic
468 (*validPointMask_)[i][j] += 1;
469#else
470#pragma omp atomic write
471 (*validPointMask_)[i][j] = 1;
472#endif
473#else
474 (*validPointMask_)[i][j] = 1;
475#endif
476 break;
477 }
478 }
479 }
480 }
481 }
482
483 {
484 std::stringstream msg;
485 msg << "Processed " << numberOfCells << " tetrahedra";
486 this->printMsg(msg.str(), 1, t.getElapsedTime(), threadNumber_);
487 }
488
489 return 0;
