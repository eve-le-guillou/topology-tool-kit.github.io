196 template <typename T, typename Container, size_t dim>
197 int getBoundingBox(const Container &points,
198 std::array<std::pair<T, T>, dim> &bBox) {
199 if(points.empty()) {
200 return -1;
201 }
202
203 for(size_t i = 0; i < points.size(); i++) {
204
205 if(i == 0) {
206 for(size_t j = 0; j < dim; j++) {
207 bBox[j].first = points[i][j];
208 bBox[j].second = points[i][j];
209 }
210 } else {
211 for(size_t j = 0; j < dim; j++) {
212 if(points[i][j] < bBox[j].first) {
213 bBox[j].first = points[i][j];
214 }
215 if(points[i][j] > bBox[j].second) {
216 bBox[j].second = points[i][j];
217 }
218 }
219 }
220 }
221
222 return 0;
223 }

291 template <typename T>
292 inline T powInt(const T val, const int n) {
293 if(n < 0) {
294 return 1.0 / powInt(val, -n);
295 } else if(n == 0) {
296 return 1;
297 } else if(n == 1) {
298 return val;
299 } else if(n == 2) {
300 return val * val;
301 } else if(n == 3) {
302 return val * val * val;
303 } else {
304 T ret = val;
305 for(int i = 0; i < n - 1; ++i) {
306 ret *= val;
307 }
308 return ret;
309 }
310 }
