272 inline bool areSaddlesSwitchables(const TreeType treeType,
273 const int &firstId,
274 const int &secondId) const {
275 const double firstLower
276 = (treeType == TreeType::JoinTree)
277 ? lowerVertexScalars_[mandatoryJoinSaddleVertex_[firstId].first]
278 : lowerVertexScalars_[mandatorySplitSaddleVertex_[firstId].first];
279 const double firstUpper
280 = (treeType == TreeType::JoinTree)
281 ? upperVertexScalars_[mandatoryJoinSaddleVertex_[firstId].second]
282 : upperVertexScalars_[mandatorySplitSaddleVertex_[firstId].second];
283 const double secondLower
284 = (treeType == TreeType::JoinTree)
285 ? lowerVertexScalars_[mandatoryJoinSaddleVertex_[secondId].first]
286 : lowerVertexScalars_[mandatorySplitSaddleVertex_[secondId].first];
287 const double secondUpper
288 = (treeType == TreeType::JoinTree)
289 ? upperVertexScalars_[mandatoryJoinSaddleVertex_[secondId].second]
290 : upperVertexScalars_[mandatorySplitSaddleVertex_[secondId].second];
291 return !((secondUpper < firstLower) || (firstUpper < secondLower));
292 }
