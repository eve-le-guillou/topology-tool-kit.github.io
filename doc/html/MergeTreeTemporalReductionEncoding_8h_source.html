56 template <class dataType>
57 dataType computeL2Distance(std::vector<dataType> &img1,
58 std::vector<dataType> &img2,
59 bool emptyFieldDistance = false) {
60 size_t noPoints = img1.size();
61
62 std::vector<dataType> secondField = img2;
63 if(emptyFieldDistance)
64 secondField = std::vector<dataType>(noPoints, 0);
65
66 dataType distance = 0;
67
68 for(size_t i = 0; i < noPoints; ++i)
69 distance += std::pow((img1[i] - secondField[i]), 2);
70
71 distance = std::sqrt(distance);
72
73 return distance;
74 }

76 template <class dataType>
77 std::vector<dataType> computeL2Barycenter(std::vector<dataType> &img1,
78 std::vector<dataType> &img2,
79 double alpha) {
80
81 size_t noPoints = img1.size();
82
83 std::vector<dataType> barycenter(noPoints);
84 for(size_t i = 0; i < noPoints; ++i)
85 barycenter[i] = alpha * img1[i] * (1 - alpha) * img2[i];
86
87 return barycenter;
88 }

166 double computeAlpha(int index1, int middleIndex, int index2) {
167 index1 = timeVariable_[index1];
168 middleIndex = timeVariable_[middleIndex];
169 index2 = timeVariable_[index2];
170 return 1 - ((double)middleIndex - index1) / (index2 - index1);

173 template <class dataType>
174 void
175 temporalSubsampling(std::vector<ftm::MergeTree<dataType>> &mTrees,
176 std::vector<int> &removed,
177 std::vector<ftm::MergeTree<dataType>> &barycenters,
178 std::vector<std::vector<dataType>> &barycentersL2) {
179 std::vector<bool> treeRemoved(mTrees.size(), false);
180
181 int toRemoved = mTrees.size() * removalPercentage_ / 100.;
182 toRemoved = std::min(toRemoved, (int)(mTrees.size() - 3));
183
184 std::vector<std::vector<dataType>> images(fieldL2_.size());
185 for(size_t i = 0; i < fieldL2_.size(); ++i)
186 for(size_t j = 0; j < fieldL2_[i].size(); ++j)
187 images[i].push_back(static_cast<dataType>(fieldL2_[i][j]));
188
189 for(int iter = 0; iter < toRemoved; ++iter) {
190 dataType bestCost = std::numeric_limits<dataType>::max();
191 int bestMiddleIndex = -1;
192 ftm::MergeTree<dataType> bestBarycenter;
193 std::vector<std::tuple<ftm::MergeTree<dataType>, int>>
194 bestBarycentersOnPath;
195 std::vector<dataType> bestBarycenterL2;
196 std::vector<std::tuple<std::vector<dataType>, int>>
197 bestBarycentersL2OnPath;
198
199 // Compute barycenter for each pair of trees
200 printMsg("Compute barycenter for each pair of trees",
201 debug::Priority::VERBOSE);
202 unsigned int index1 = 0, index2 = 0;
203 while(index2 != mTrees.size() - 1) {
204
205 // Get index in the middle
206 int middleIndex = index1 + 1;
207 while(treeRemoved[middleIndex])
208 ++middleIndex;
209
210 // Get second index
211 index2 = middleIndex + 1;
212 while(treeRemoved[index2])
213 ++index2;
214
215 // Compute barycenter
216 printMsg("Compute barycenter", debug::Priority::VERBOSE);
217 double alpha = computeAlpha(index1, middleIndex, index2);
218 ftm::MergeTree<dataType> barycenter;
219 std::vector<dataType> barycenterL2;
220 if(not useL2Distance_)
221 barycenter = computeBarycenter<dataType>(
222 mTrees[index1], mTrees[index2], alpha);
223 else
224 barycenterL2 = computeL2Barycenter<dataType>(
225 images[index1], images[index2], alpha);
226
227 // - Compute cost
228 // Compute distance with middleIndex
229 printMsg(
230 "Compute distance with middleIndex", debug::Priority::VERBOSE);
231 dataType cost;
232 if(not useL2Distance_)
233 cost = computeDistance<dataType>(barycenter, mTrees[middleIndex]);
234 else
235 cost
236 = computeL2Distance<dataType>(barycenterL2, images[middleIndex]);
237
238 // Compute distances of previously removed trees on the path
239 printMsg("Compute distances of previously removed trees",
240 debug::Priority::VERBOSE);
241 std::vector<std::tuple<ftm::MergeTree<dataType>, int>>
242 barycentersOnPath;
243 std::vector<std::tuple<std::vector<dataType>, int>>
244 barycentersL2OnPath;
245 for(unsigned int i = 0; i < 2; ++i) {
246 int toReach = (i == 0 ? index1 : index2);
247 int offset = (i == 0 ? -1 : 1);
248 int tIndex = middleIndex + offset;
249 while(tIndex != toReach) {
250
251 // Compute barycenter
252 double alphaT = computeAlpha(index1, tIndex, index2);
253 ftm::MergeTree<dataType> barycenterP;
254 std::vector<dataType> barycenterPL2;
255 if(not useL2Distance_)
256 barycenterP = computeBarycenter<dataType>(
257 mTrees[index1], mTrees[index2], alphaT);
258 else
259 barycenterPL2 = computeL2Barycenter<dataType>(
260 images[index1], images[index2], alphaT);
261
262 // Compute distance
263 dataType costP;
264 if(not useL2Distance_)
265 costP = computeDistance<dataType>(barycenterP, mTrees[tIndex]);
266 else
267 costP
268 = computeL2Distance<dataType>(barycenterPL2, images[tIndex]);
269
270 // Save results
271 if(not useL2Distance_)
272 barycentersOnPath.push_back(
273 std::make_tuple(barycenterP, tIndex));
274 else
275 barycentersL2OnPath.push_back(
276 std::make_tuple(barycenterPL2, tIndex));
277 cost += costP;
278 tIndex += offset;
279 }
280 }
281
282 if(cost < bestCost) {
283 bestCost = cost;
284 bestMiddleIndex = middleIndex;
285 if(not useL2Distance_) {
286 bestBarycenter = barycenter;
287 bestBarycentersOnPath = barycentersOnPath;
288 } else {
289 bestBarycenterL2 = barycenterL2;
290 bestBarycentersL2OnPath = barycentersL2OnPath;
291 }
292 }
293
294 // Go to the next index
295 index1 = middleIndex;
296 }
297
298 // Removed the tree with the lowest cost
299 printMsg(
300 "Removed the tree with the lowest cost", debug::Priority::VERBOSE);
301 removed.push_back(bestMiddleIndex);
302 treeRemoved[bestMiddleIndex] = true;
303 if(not useL2Distance_) {
304 barycenters[bestMiddleIndex] = bestBarycenter;
305 for(auto &tup : bestBarycentersOnPath)
306 barycenters[std::get<1>(tup)] = std::get<0>(tup);
307 } else {
308 barycentersL2[bestMiddleIndex] = bestBarycenterL2;
309 for(auto &tup : bestBarycentersL2OnPath)
310 barycentersL2[std::get<1>(tup)] = std::get<0>(tup);
311 }
312 }
313 }

315 template <class dataType>
316 std::vector<int> execute(std::vector<ftm::MergeTree<dataType>> &mTrees,
317 std::vector<double> &emptyTreeDistances,
318 std::vector<ftm::MergeTree<dataType>> &allMT) {
319 Timer t_tempSub;
320
321 // --- Preprocessing
322 if(not useL2Distance_) {
323 treesNodeCorr_ = std::vector<std::vector<int>>(mTrees.size());
324 for(unsigned int i = 0; i < mTrees.size(); ++i) {
325 preprocessingPipeline<dataType>(mTrees[i], epsilonTree2_,
326 epsilon2Tree2_, epsilon3Tree2_,
327 branchDecomposition_, useMinMaxPair_,
328 cleanTree_, treesNodeCorr_[i]);
329 }
330 printTreesStats<dataType>(mTrees);
331 }
332
333 // --- Execute
334 std::vector<ftm::MergeTree<dataType>> barycenters(mTrees.size());
335 std::vector<std::vector<dataType>> barycentersL2(mTrees.size());
336 std::vector<int> removed;
337 if(not useCustomTimeVariable_) {
338 timeVariable_.clear();
339 for(size_t i = 0; i < mTrees.size(); ++i)
340 timeVariable_.push_back(i);
341 }
342 temporalSubsampling<dataType>(
343 mTrees, removed, barycenters, barycentersL2);
344
345 // --- Concatenate all trees/L2Images
346 std::vector<std::vector<dataType>> images(fieldL2_.size());
347 for(size_t i = 0; i < fieldL2_.size(); ++i)
348 for(size_t j = 0; j < fieldL2_[i].size(); ++j)
349 images[i].push_back(static_cast<dataType>(fieldL2_[i][j]));
350
351 for(auto &mt : mTrees)
352 allMT.push_back(mt);
353 std::vector<bool> removedB(mTrees.size(), false);
354 for(auto r : removed)
355 removedB[r] = true;
356 for(unsigned int i = 0; i < barycenters.size(); ++i)
357 if(removedB[i]) {
358 if(not useL2Distance_)
359 allMT.push_back(barycenters[i]);
360 else
361 images.push_back(barycentersL2[i]);
362 }
363
364 // --- Compute empty tree distances
365 unsigned int distMatSize
366 = (not useL2Distance_ ? allMT.size() : images.size());
367 for(unsigned int i = 0; i < distMatSize; ++i) {
368 dataType distance;
369 if(not useL2Distance_)
370 distance = computeDistance<dataType>(allMT[i], allMT[i], true);
371 else
372 distance = computeL2Distance<dataType>(images[i], images[i], true);
373 emptyTreeDistances.push_back(distance);
374 }
375
376 // --- Postprocessing
377 if(not useL2Distance_) {
378 for(unsigned int i = 0; i < allMT.size(); ++i)
379 postprocessingPipeline<dataType>(&(allMT[i].tree));
380 for(unsigned int i = 0; i < mTrees.size(); ++i)
381 postprocessingPipeline<dataType>(&(mTrees[i].tree));
382 }
383
384 // --- Print results
385 std::stringstream ss, ss2, ss3;
386 ss << "input size = " << mTrees.size();
387 printMsg(ss.str());
388 ss2 << "output size = "
389 << mTrees.size() - (distMatSize - mTrees.size());
390 printMsg(ss2.str());
391 ss3 << "removed : ";
392 for(unsigned int i = 0; i < removed.size(); ++i) {
393 auto r = removed[i];
394 ss3 << r;
395 if(i < removed.size() - 1)
396 ss3 << ", ";
397 }
398 printMsg(ss3.str());
399
400 sort(removed.begin(), removed.end());
401
402 printMsg("Encoding", 1, t_tempSub.getElapsedTime(), this->threadNumber_);
403
404 return removed;
405 }
