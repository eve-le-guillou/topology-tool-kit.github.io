149double ttk::PDBarycenter::updateBarycenter(
150 std::vector<std::vector<MatchingType>> &matchings) {
151 // 1. Initialize variables used in the sequel
152 Timer t_update;
153 size_t n_goods = barycenter_goods_[0].size();
154
155 size_t n_diagrams = current_bidder_diagrams_.size();
156 points_added_ = 0;
157 points_deleted_ = 0;
158 double max_shift = 0;
159
160 std::vector<size_t> count_diag_matchings(
161 n_goods); // Number of diagonal matchings for each point of the barycenter
162 std::vector<double> x(n_goods);
163 std::vector<double> y(n_goods);
164 std::vector<double> crit_coords_x(n_goods);
165 std::vector<double> crit_coords_y(n_goods);
166 std::vector<double> crit_coords_z(n_goods);
167 for(size_t i = 0; i < n_goods; i++) {
168 count_diag_matchings[i] = 0;
169 x[i] = 0;
170 y[i] = 0;
171 crit_coords_x[i] = 0;
172 crit_coords_y[i] = 0;
173 crit_coords_z[i] = 0;
174 }
175 std::vector<double> min_prices(n_diagrams);
176 for(size_t j = 0; j < n_diagrams; j++) {
177 min_prices[j] = std::numeric_limits<double>::max();
178 }
179
180 std::vector<Bidder *>
181 points_to_append; // Will collect bidders linked to diagonal
182 // 2. Preprocess the matchings
183 for(size_t j = 0; j < matchings.size(); j++) {
184 for(size_t i = 0; i < matchings[j].size(); i++) {
185 int bidder_id = std::get<0>(matchings[j][i]);
186 int good_id = std::get<1>(matchings[j][i]);
187 if(good_id < 0 && bidder_id >= 0) {
188 // Future new barycenter point
189 points_to_append.push_back(&current_bidder_diagrams_[j].at(bidder_id));
190 }
191
192 else if(good_id >= 0 && bidder_id >= 0) {
193 // Update coordinates (to be divided by the number of diagrams later on)
194 x[good_id] += current_bidder_diagrams_[j].at(bidder_id).x_;
195 y[good_id] += current_bidder_diagrams_[j].at(bidder_id).y_;
196 if(geometrical_factor_ < 1) {
197 const auto &critical_coordinates = current_bidder_diagrams_[j]
198 .at(bidder_id)
199 .GetCriticalCoordinates();
200 crit_coords_x[good_id] += critical_coordinates[0];
201 crit_coords_y[good_id] += critical_coordinates[1];
202 crit_coords_z[good_id] += critical_coordinates[2];
203 }
204 } else if(good_id >= 0 && bidder_id < 0) {
205 // Counting the number of times this barycenter point is linked to the
206 // diagonal
207 count_diag_matchings[good_id] = count_diag_matchings[good_id] + 1;
208 }
209 }
210 }
211
212 // 3. Update the previous points of the barycenter
213 for(size_t i = 0; i < n_goods; i++) {
214 if(count_diag_matchings[i] < n_diagrams) {
215 // Barycenter point i is matched at least to one off-diagonal bidder
216 // 3.1 Compute the arithmetic mean of off-diagonal bidders linked to it
217 double x_bar = x[i] / (double)(n_diagrams - count_diag_matchings[i]);
218 double y_bar = y[i] / (double)(n_diagrams - count_diag_matchings[i]);
219 // 3.2 Compute the new coordinates of the point (the more linked to the
220 // diagonal it was, the closer to the diagonal it'll be)
221 double new_x = ((double)(n_diagrams - count_diag_matchings[i]) * x_bar
222 + (double)count_diag_matchings[i] * (x_bar + y_bar) / 2.)
223 / (double)n_diagrams;
224 double new_y = ((double)(n_diagrams - count_diag_matchings[i]) * y_bar
225 + (double)count_diag_matchings[i] * (x_bar + y_bar) / 2.)
226 / (double)n_diagrams;
227 // TODO Weight by persistence
228 double new_crit_coord_x
229 = crit_coords_x[i] / (double)(n_diagrams - count_diag_matchings[i]);
230 double new_crit_coord_y
231 = crit_coords_y[i] / (double)(n_diagrams - count_diag_matchings[i]);
232 double new_crit_coord_z
233 = crit_coords_z[i] / (double)(n_diagrams - count_diag_matchings[i]);
234
235 // 3.3 Compute and store how much the point has shifted
236 // TODO adjust shift with geometrical_factor_
237 double dx = barycenter_goods_[0].at(i).x_ - new_x;
238 double dy = barycenter_goods_[0].at(i).y_ - new_y;
239 double shift = Geometry::pow(std::abs(dx), wasserstein_)
240 + Geometry::pow(std::abs(dy), wasserstein_);
241 if(shift > max_shift) {
242 max_shift = shift;
243 }
244 // 3.4 Update the position of the point
245 for(size_t j = 0; j < n_diagrams; j++) {
246 barycenter_goods_[j].at(i).SetCoordinates(new_x, new_y);
247 if(geometrical_factor_ < 1) {
248 barycenter_goods_[j].at(i).SetCriticalCoordinates(
249 new_crit_coord_x, new_crit_coord_y, new_crit_coord_z);
250 }
251 if(barycenter_goods_[j].at(i).getPrice() < min_prices[j]) {
252 min_prices[j] = barycenter_goods_[j].at(i).getPrice();
253 }
254 }
255 // TODO Reinitialize/play with prices here if you wish
256 }
257 }
258 for(size_t j = 0; j < n_diagrams; j++) {
259 if(min_prices[j] >= std::numeric_limits<double>::max() / 2.) {
260 min_prices[j] = 0;
261 }
262 }
263
264 // 4. Delete off-diagonal barycenter points not linked to any
265 // off-diagonal bidder
266 for(size_t i = 0; i < n_goods; i++) {
267 if(count_diag_matchings[i] == n_diagrams) {
268 points_deleted_ += 1;
269 double shift
270 = 2
271 * Geometry::pow(
272 barycenter_goods_[0].at(i).getPersistence() / 2., wasserstein_);
273 if(shift > max_shift) {
274 max_shift = shift;
275 }
276 for(size_t j = 0; j < n_diagrams; j++) {
277 barycenter_goods_[j].at(i).id_ = -1;
278 }
279 }
280 }
281
282 // 5. Append the new points to the barycenter
283 for(size_t k = 0; k < points_to_append.size(); k++) {
284 points_added_ += 1;
285 Bidder *b = points_to_append[k];
286 double gx
287 = (b->x_ + (n_diagrams - 1) * (b->x_ + b->y_) / 2.) / (n_diagrams);
288 double gy
289 = (b->y_ + (n_diagrams - 1) * (b->x_ + b->y_) / 2.) / (n_diagrams);
290 const auto &critical_coordinates = b->GetCriticalCoordinates();
291 for(size_t j = 0; j < n_diagrams; j++) {
292 Good g = Good(gx, gy, false, barycenter_goods_[j].size());
293 g.setPrice(min_prices[j]);
294 if(geometrical_factor_ < 1) {
295 g.SetCriticalCoordinates(std::get<0>(critical_coordinates),
296 std::get<1>(critical_coordinates),
297 std::get<2>(critical_coordinates));
298 }
299 barycenter_goods_[j].emplace_back(g);
300 double shift
301 = 2
302 * Geometry::pow(
303 barycenter_goods_[j].at(g.id_).getPersistence() / 2., wasserstein_);
304 if(shift > max_shift) {
305 max_shift = shift;
306 }
307 }
308 }
309
310 // 6. Finally, recreate barycenter_goods
311 for(size_t j = 0; j < n_diagrams; j++) {
312 int count = 0;
313 GoodDiagram new_barycenter;
314 for(size_t i = 0; i < barycenter_goods_[j].size(); i++) {
315 Good g = barycenter_goods_[j].at(i);
316 if(g.id_ != -1) {
317 g.id_ = count;
318 new_barycenter.emplace_back(g);
319 count++;
320 }
321 }
322 barycenter_goods_[j] = new_barycenter;
323 }
324
325 return max_shift;
326}

362double ttk::PDBarycenter::enrichCurrentBidderDiagrams(
363 double previous_min_persistence,
364 double min_persistence,
365 std::vector<double> &initial_diagonal_prices,
366 std::vector<double> &initial_off_diagonal_prices,
367 int min_points_to_add,
368 bool add_points_to_barycenter) {
369
370 double new_min_persistence = min_persistence;
371
372 // 1. Get size of the largest current diagram, deduce the maximal number of
373 // points to append
374 size_t max_diagram_size{};
375 for(int i = 0; i < numberOfInputs_; i++) {
376 if(current_bidder_diagrams_[i].size() > max_diagram_size) {
377 max_diagram_size = current_bidder_diagrams_[i].size();
378 }
379 }
380 int max_points_to_add = std::max(
381 min_points_to_add, min_points_to_add + (int)(max_diagram_size / 10));
382
383 // 2. Get which points can be added, deduce the new minimal persistence
384 std::vector<std::vector<int>> candidates_to_be_added(numberOfInputs_);
385 std::vector<std::vector<int>> idx(numberOfInputs_);
386 for(int i = 0; i < numberOfInputs_; i++) {
387
388 std::vector<double> persistences;
389 for(size_t j = 0; j < bidder_diagrams_[i].size(); j++) {
390 Bidder b = bidder_diagrams_[i].at(j);
391 double persistence = b.getPersistence();
392 if(persistence >= min_persistence
393 && persistence < previous_min_persistence) {
394 candidates_to_be_added[i].push_back(j);
395 idx[i].push_back(idx[i].size());
396 persistences.push_back(persistence);
397 }
398 }
399 sort(idx[i].begin(), idx[i].end(), [&persistences](int &a, int &b) {
400 return ((persistences[a] > persistences[b])
401 || ((persistences[a] == persistences[b]) && (a > b)));
402 });
403 int size = candidates_to_be_added[i].size();
404 if(size >= max_points_to_add) {
405 double last_persistence_added
406 = persistences[idx[i][max_points_to_add - 1]];
407 if(last_persistence_added > new_min_persistence) {
408 new_min_persistence = last_persistence_added;
409 }
410 }
411 }
412
413 // 3. Add the points to the current diagrams
414
415 // only to give determinism
416 int compteur_for_adding_points = 0;
417
418 for(int i = 0; i < numberOfInputs_; i++) {
419 int size = candidates_to_be_added[i].size();
420 for(int j = 0; j < std::min(max_points_to_add, size); j++) {
421 Bidder b = bidder_diagrams_[i].at(candidates_to_be_added[i][idx[i][j]]);
422 if(b.getPersistence() >= new_min_persistence) {
423 b.id_ = current_bidder_diagrams_[i].size();
424 b.setPositionInAuction(current_bidder_diagrams_[i].size());
425 b.setDiagonalPrice(initial_diagonal_prices[i]);
426 current_bidder_diagrams_[i].emplace_back(b);
427 // b.id_ --> position of b in current_bidder_diagrams_[i]
428 current_bidder_ids_[i][candidates_to_be_added[i][idx[i][j]]]
429 = current_bidder_diagrams_[i].size() - 1;
430
431 int to_be_added_to_barycenter
432 = deterministic_ ? compteur_for_adding_points % numberOfInputs_
433 : rand() % numberOfInputs_;
434 // We add the bidder as a good with probability 1/n_diagrams
435 if(to_be_added_to_barycenter == 0 && add_points_to_barycenter) {
436 for(int k = 0; k < numberOfInputs_; k++) {
437 Good g = Good(b.x_, b.y_, false, barycenter_goods_[k].size());
438 g.setPrice(initial_off_diagonal_prices[k]);
439 g.SetCriticalCoordinates(b.coords_[0], b.coords_[1], b.coords_[2]);
440 barycenter_goods_[k].emplace_back(g);
441 }
442 }
443 }
444 compteur_for_adding_points++;
445 }
446 }
447 return new_min_persistence;
448}

536typename ttk::PDBarycenter::KDTreePair ttk::PDBarycenter::getKDTree() const {
537 Timer tm;
538 auto kdt
539 = std::unique_ptr<KDTree<double>>(new KDTree<double>{true, wasserstein_});
540
541 const int dimension = geometrical_factor_ >= 1 ? 2 : 5;
542
543 std::vector<double> coordinates;
544 std::vector<std::vector<double>> weights;
545
546 for(size_t i = 0; i < barycenter_goods_[0].size(); i++) {
547 const Good &g = barycenter_goods_[0].at(i);
548 coordinates.push_back(geometrical_factor_ * g.x_);
549 coordinates.push_back(geometrical_factor_ * g.y_);
550 if(geometrical_factor_ < 1) {
551 coordinates.push_back((1 - geometrical_factor_) * g.coords_[0]);
552 coordinates.push_back((1 - geometrical_factor_) * g.coords_[1]);
553 coordinates.push_back((1 - geometrical_factor_) * g.coords_[2]);
554 }
555 }
556
557 for(size_t idx = 0; idx < barycenter_goods_.size(); idx++) {
558 std::vector<double> empty_weights;
559 weights.push_back(empty_weights);
560 for(size_t i = 0; i < barycenter_goods_[idx].size(); i++) {
561 const Good &g = barycenter_goods_[idx].at(i);
562 weights[idx].push_back(g.getPrice());
563 }
564 }
565 // Correspondance map : position in barycenter_goods_ --> KDT node
566
567 auto correspondance_kdt_map
568 = kdt->build(coordinates.data(), barycenter_goods_[0].size(), dimension,
569 weights, barycenter_goods_.size());
570 this->printMsg(" Building KDTree", 1, tm.getElapsedTime(),
571 debug::LineMode::NEW, debug::Priority::VERBOSE);
572 return std::make_pair(std::move(kdt), correspondance_kdt_map);
573}

575std::vector<std::vector<ttk::MatchingType>>
576 ttk::PDBarycenter::executeAuctionBarycenter(DiagramType &barycenter) {
577
578 std::vector<std::vector<MatchingType>> previous_matchings;
579 double min_persistence = 0;
580 double min_cost = std::numeric_limits<double>::max();
581 int last_min_cost_obtained = 0;
582
583 this->setBidderDiagrams();
584 this->setInitialBarycenter(
585 min_persistence); // false for a determinist initialization
586
587 double max_persistence = getMaxPersistence();
588
589 std::vector<double> min_diag_price(numberOfInputs_);
590 std::vector<double> min_price(numberOfInputs_);
591 for(int i = 0; i < numberOfInputs_; i++) {
592 min_diag_price[i] = 0;
593 min_price[i] = 0;
594 }
595
596 int min_points_to_add = std::numeric_limits<int>::max();
597 this->enrichCurrentBidderDiagrams(2 * max_persistence, min_persistence,
598 min_diag_price, min_price,
599 min_points_to_add, false);
600
601 bool converged = false;
602 bool finished = false;
603 double total_cost;
604
605 while(!finished) {
606 Timer tm;
607

622 }
623

639

642 }
643

649 last_min_cost_obtained = 0;
650 } else {
651 last_min_cost_obtained += 1;
652 }
653
654 converged = converged || last_min_cost_obtained > 1;
655 }

657 previous_matchings = std::move(all_matchings);
658 // END OF TIMER
659
660 for(size_t i = 0; i < barycenter_goods_.size(); ++i) {
661 for(size_t j = 0; j < barycenter_goods_[i].size(); ++j) {

667 current_bidder_diagrams_[i].at(j).setDiagonalPrice(0);
668 }
669 }
670 for(int i = 0; i < numberOfInputs_; i++) {
671 min_diag_price[i] = 0;
672 min_price[i] = 0;
673 }
674 }
675 barycenter.resize(0);
676 for(size_t j = 0; j < barycenter_goods_[0].size(); j++) {
677 Good &g = barycenter_goods_[0].at(j);
678 barycenter.emplace_back(PersistencePair{
679 CriticalVertex{0, nt1_, g.x_, {}}, CriticalVertex{0, nt2_, g.y_, {}},
680 g.getPersistence(), diagramType_, true});
681 }
682
683 cost_ = sqrt(total_cost);
684 std::vector<std::vector<MatchingType>> corrected_matchings
685 = correctMatchings(previous_matchings);
686 return corrected_matchings;

689double ttk::PDBarycenter::computeRealCost() {
690 double total_real_cost = 0;
691 std::vector<MatchingType> fake_matchings;
692 for(int i = 0; i < numberOfInputs_; i++) {
693 PersistenceDiagramAuction auction(
694 wasserstein_, geometrical_factor_, lambda_, 0.01, true);
695 GoodDiagram current_barycenter = barycenter_goods_[0];
696 BidderDiagram current_bidder_diagram = bidder_diagrams_[i];
697 auction.BuildAuctionDiagrams(current_bidder_diagram, current_barycenter);
698 double cost = auction.run(fake_matchings);
699 total_real_cost += cost * cost;
700 }
701 return sqrt(total_real_cost);
