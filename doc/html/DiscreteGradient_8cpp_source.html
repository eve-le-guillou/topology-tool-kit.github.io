72 DiscreteGradient::numUnpairedFacesTriangle(const CellExt &c,
73 const lowerStarType &ls) const {
74 // number of unpaired faces
75 std::pair<size_t, SimplexId> res{0, -1};
76
77 // loop over edge faces of triangle
78 // (2 edges per triangle in lower star)
79 for(size_t i = 0; i < 2; ++i) {
80 if(!ls[1][c.faces_[i]].paired_) {
81 res.first++;
82 res.second = c.faces_[i];

209int DiscreteGradient::setManifoldSize(
210 const size_t nCritPoints,
211 const std::vector<size_t> &nCriticalPointsByDim,
212 const SimplexId *const ascendingManifold,
213 const SimplexId *const descendingManifold,
214 std::vector<SimplexId> &manifoldSize) const {
215
216 // in the manifoldSize vector, critical points are sorted first by
217 // dim then by index
218
219 // ascendingManifold (resp. descendingManifold) region indices are
220 // numbered from 0 to #maxima == nCriticalPointsByDim.back()
221 // (resp. #minina == nCriticalPointsByDim[0])
222
223 if(nCritPoints == 0
224 || (nCriticalPointsByDim[0] == 0 && nCriticalPointsByDim.back() == 0)) {
225 // no critical points || no extrema
226 return 0;
227 }
228
229 manifoldSize.resize(nCritPoints, 0);
230
231 // descending manifold cells size
232 if(nCriticalPointsByDim[0] > 0) {
233 for(SimplexId i = 0; i < numberOfVertices_; ++i) {
234 if(descendingManifold[i] != -1) {
235 manifoldSize[descendingManifold[i]]++;
236 }
237 }
238 }
239
240 if(nCriticalPointsByDim.back() > 0) {
241 // index of first maximum in critical points array
242 const auto nFirstMaximum{nCritPoints - nCriticalPointsByDim.back()};
243
244 // ascending manifold cells size
245 for(SimplexId i = 0; i < numberOfVertices_; ++i) {
246 if(ascendingManifold[i] != -1) {
247 manifoldSize[ascendingManifold[i] + nFirstMaximum]++;
248 }
249 }
250 }
251
252 return 0;
253}
