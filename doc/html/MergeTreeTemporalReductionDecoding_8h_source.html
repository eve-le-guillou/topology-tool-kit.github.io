125 template <class dataType>
126 void execute(
127 std::vector<ftm::MergeTree<dataType>> &mTrees,
128 std::vector<std::tuple<double, int, int, int, int>> &coefs,
129 std::vector<ftm::MergeTree<dataType>> &allMT,
130 std::vector<std::vector<std::tuple<ftm::idNode, ftm::idNode, double>>>
131 &allMatching) {
132 Timer t_tempSub;
133
134 // --- Preprocessing
135 treesNodeCorr_ = std::vector<std::vector<int>>(mTrees.size());
136 for(unsigned int i = 0; i < mTrees.size(); ++i) {
137 preprocessingPipeline<dataType>(
138 mTrees[i], epsilonTree2_, epsilon2Tree2_, epsilon3Tree2_,
139 branchDecomposition_, useMinMaxPair_, cleanTree_, treesNodeCorr_[i]);
140 }
141 printTreesStats<dataType>(mTrees);
142
143 // --- Execute
144 distancesToKeyFrames_ = std::vector<double>(coefs.size() * 2);
145 int index = 0;
146 size_t cpt = 0;
147 while(cpt < coefs.size()) {
148 while(cpt < coefs.size() and std::get<2>(coefs[cpt]) <= index) {
149 double alpha = std::get<0>(coefs[cpt]);
150 int index1 = std::get<1>(coefs[cpt]);
151 int index2 = std::get<2>(coefs[cpt]);
152 ftm::MergeTree<dataType> tree = computeBarycenter<dataType>(
153 mTrees[index1], mTrees[index2], alpha);
154 allMT.push_back(tree);
155 distancesToKeyFrames_[cpt * 2]
156 = computeDistance<dataType>(mTrees[index1], tree);
157 distancesToKeyFrames_[cpt * 2 + 1]
158 = computeDistance<dataType>(tree, mTrees[index2]);
159 ++cpt;
160 }
161 allMT.push_back(mTrees[index]);
162 ++index;
163 }
164
165 allMatching = std::vector<
166 std::vector<std::tuple<ftm::idNode, ftm::idNode, double>>>(allMT.size()
167 - 1);
168 finalDistances_ = std::vector<double>(allMT.size() - 1);
169 for(unsigned int i = 0; i < allMT.size() - 1; ++i)
170 finalDistances_[i]
171 = computeDistance<dataType>(allMT[i], allMT[i + 1], allMatching[i]);
172
173 // --- Postprocessing
174 for(unsigned int i = 0; i < allMT.size(); ++i)
175 postprocessingPipeline<dataType>(&(allMT[i].tree));
176 for(unsigned int i = 0; i < mTrees.size(); ++i)
177 postprocessingPipeline<dataType>(&(mTrees[i].tree));
178
179 // --- Print results
180 std::stringstream ss, ss2, ss3;
181 ss << "input size = " << mTrees.size();
182 printMsg(ss.str());
183 ss2 << "output size = " << allMT.size();
184 printMsg(ss2.str());
185 ss3 << "reconstructed : " << allMT.size() - mTrees.size();
186 printMsg(ss3.str());
187 printMsg("Decoding", 1, t_tempSub.getElapsedTime(), this->threadNumber_);
188 }
