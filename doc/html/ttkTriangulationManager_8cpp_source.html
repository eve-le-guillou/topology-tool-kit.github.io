52void ttkTriangulationManager::processImplicit(
53 ttk::Triangulation &triangulation) const {
54
55 const bool prevPeriodic = triangulation.hasPeriodicBoundaries();
56 if(prevPeriodic != this->Periodicity) {
57 triangulation.setPeriodicBoundaryConditions(Periodicity);
58 printMsg("Switching regular grid periodicity from "
59 + (prevPeriodic ? std::string("ON") : std::string("OFF")) + " to "
60 + (Periodicity ? std::string("ON") : std::string("OFF")));
61 }
62
63 const bool prevPreconditions = triangulation.hasImplicitPreconditions();
64 if((this->PreconditioningStrategy == STRATEGY::NO_PRECONDITIONS
65 && !prevPreconditions)
66 || (this->PreconditioningStrategy == STRATEGY::WITH_PRECONDITIONS
67 && prevPreconditions)) {
68 return;
69 }
70
71 triangulation.setImplicitPreconditions(this->PreconditioningStrategy);
72 const auto newPreconditions = triangulation.hasImplicitPreconditions();
73 if(prevPreconditions != newPreconditions) {
74 printMsg("Switching regular grid preconditions from "
75 + (prevPreconditions ? std::string("ON") : std::string("OFF"))
76 + " to "

81int ttkTriangulationManager::processExplicit(
82 vtkUnstructuredGrid *const output,
83 vtkPointSet *const input,
84 ttk::Triangulation &triangulation) const {
85
86 if(output == nullptr || input == nullptr) {
87 this->printErr("Empty data-sets");
88 return 0;
89 }
90 ttk::Timer tm{};
91
92 // If all checks pass then log which array is going to be processed.
93 this->printMsg("Compact explicit triangulation...");
94 int status = 0; // this integer checks if the base code returns an error
95 ttk::CompactTriangulationPreconditioning worker{};
96 ttkTemplateMacro(
97 triangulation.getType(),
98 (status = worker.execute(
99 static_cast<TTK_TT *>(triangulation.getData()), this->Threshold)));
100
101 // On error cancel filter execution
102 if(status != 1)
103 return 0;
104
105 // Get input data array selection
106 std::vector<vtkDataArray *> pointDataArrays{};
107 std::vector<vtkDataArray *> cellDataArrays{};
108
109 vtkPointData *inPointData = input->GetPointData();
110 for(int i = 0; i < inPointData->GetNumberOfArrays(); i++) {
111 vtkDataArray *curArray = inPointData->GetArray(i);
112 if(curArray != nullptr && curArray->GetName() != nullptr
113 && ArraySelection->ArrayIsEnabled(curArray->GetName())) {
114 pointDataArrays.emplace_back(curArray);
115 }
116 }
117
118 vtkCellData *inCellData = input->GetCellData();
119 for(int i = 0; i < inCellData->GetNumberOfArrays(); i++) {
120 vtkDataArray *curArray = inCellData->GetArray(i);
121 if(curArray != nullptr && curArray->GetName() != nullptr
122 && ArraySelection->ArrayIsEnabled(curArray->GetName())) {
123 cellDataArrays.emplace_back(curArray);
124 }
125 }
126
127 output->Initialize();
128
129 const auto &vertices{worker.getVertices()};
130 const auto &nodes{worker.getNodes()};
131 const auto &cells{worker.getCells()};
132 std::vector<ttk::SimplexId> vertexMap(vertices.size());
133
134 vtkNew<vtkPoints> points{};
135 vtkNew<vtkIntArray> indices{};
136
137 indices->SetNumberOfComponents(1);
138 indices->SetName(ttk::compactTriangulationIndex);
139
140 // insert vertices in the output mesh
141 for(size_t i = 0; i < vertices.size(); i++) {
142 float x, y, z;
143 triangulation.getVertexPoint(vertices[i], x, y, z);
144 points->InsertNextPoint(x, y, z);
145 vertexMap[vertices[i]] = i;
146 indices->InsertNextTuple1(nodes[i]);
147 }
148 output->SetPoints(points);
149
150 output->GetPointData()->AddArray(indices);
151
152 // insert cells in the output mesh
153 output->Allocate(cells.size());
154 const size_t dimension = triangulation.getCellVertexNumber(0);
155
156 for(unsigned int i = 0; i < cells.size(); i++) {
157 std::array<vtkIdType, 4> cell{};
158 for(size_t j = 0; j < dimension; j++) {
159 ttk::SimplexId vertexId;
160 triangulation.getCellVertex(cells[i], j, vertexId);
161 cell[j] = vertexMap[vertexId];
162 }
163 std::sort(cell.begin(), cell.begin() + dimension);
164 if(dimension == 2) {
165 output->InsertNextCell(VTK_LINE, 2, cell.data());
166 } else if(dimension == 3) {
167 output->InsertNextCell(VTK_TRIANGLE, 3, cell.data());
168 } else if(dimension == 4) {
169 output->InsertNextCell(VTK_TETRA, 4, cell.data());
170 } else {
171 this->printErr("Should not get here!");
172 }
173 }
174
175 // Modify the selected point data arrays with new indices
176 for(vtkDataArray *scalarArray : pointDataArrays) {
177 vtkSmartPointer<vtkDataArray> updatedField(scalarArray->NewInstance());
178 updatedField->SetName(scalarArray->GetName());
179 for(size_t j = 0; j < vertices.size(); j++) {
180 updatedField->InsertTuple(j, scalarArray->GetTuple(vertices[j]));
181 }
182
183 output->GetPointData()->AddArray(updatedField);
184 }
185
186 // Modify the selected cell data arrays with new indices
187 for(vtkDataArray *scalarArray : cellDataArrays) {
188 vtkSmartPointer<vtkDataArray> updatedField(scalarArray->NewInstance());
189 updatedField->SetName(scalarArray->GetName());
190 for(size_t j = 0; j < cells.size(); j++) {
191 updatedField->InsertTuple(j, scalarArray->GetTuple(cells[j]));
192 }
193
194 output->GetCellData()->AddArray(updatedField);
195 }
196
197 this->printMsg("Done!", 1.0, tm.getElapsedTime(), 1);
198
199 // return success
200 return 1;
201}
