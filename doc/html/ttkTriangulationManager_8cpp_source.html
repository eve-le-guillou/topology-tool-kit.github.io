118static void switchPreconditions(ttk::Triangulation &triangulation,
119 const ttk::Triangulation::STRATEGY precStrategy,
120 const ttk::Debug &dbg) {
121 const bool prevPreconditions = triangulation.hasImplicitPreconditions();
122 if((precStrategy == ttk::Triangulation::STRATEGY::NO_PRECONDITIONS
123 && !prevPreconditions)
124 || (precStrategy == ttk::Triangulation::STRATEGY::WITH_PRECONDITIONS
125 && prevPreconditions)) {
126 return;
127 }
128
129 triangulation.setImplicitPreconditions(precStrategy);
130 const auto newPreconditions = triangulation.hasImplicitPreconditions();
131 if(prevPreconditions != newPreconditions) {
132 dbg.printMsg("Switching regular grid preconditions from "
133 + (prevPreconditions ? std::string("ON") : std::string("OFF"))
134 + " to "
135 + (newPreconditions ? std::string("ON") : std::string("OFF")));
136 }

157int ttkTriangulationManager::processExplicit(
158 vtkUnstructuredGrid *const output,
159 vtkPointSet *const input,
160 ttk::Triangulation &triangulation) const {
161
162 if(output == nullptr || input == nullptr) {
163 this->printErr("Empty data-sets");
164 return 0;
165 }
166 ttk::Timer tm{};
167
168#ifdef TTK_ENABLE_MPI
169 if(ttk::hasInitializedMPI()) {
170 this->printErr(
171 "Compact triangulation not (yet) supported in an MPI context!");
172 this->printErr("Keeping the Explicit triangulation.");
173 output->ShallowCopy(input);
174 return 0;
175 }
176#endif // TTK_ENABLE_MPI
177
178 // If all checks pass then log which array is going to be processed.
179 this->printMsg("Compact explicit triangulation...");
180 int status = 0; // this integer checks if the base code returns an error
181 ttk::CompactTriangulationPreconditioning worker{};
182 ttkTemplateMacro(
183 triangulation.getType(),
184 (status = worker.execute(
185 static_cast<TTK_TT *>(triangulation.getData()), this->Threshold)));
186
187 // On error cancel filter execution
188 if(status != 1)
189 return 0;
190
191 // Get input data array selection
192 std::vector<vtkDataArray *> pointDataArrays{};
193 std::vector<vtkDataArray *> cellDataArrays{};
194
195 vtkPointData *inPointData = input->GetPointData();
196 for(int i = 0; i < inPointData->GetNumberOfArrays(); i++) {
197 vtkDataArray *curArray = inPointData->GetArray(i);
198 if(curArray != nullptr && curArray->GetName() != nullptr
199 && ArraySelection->ArrayIsEnabled(curArray->GetName())) {
200 pointDataArrays.emplace_back(curArray);
201 }
202 }
203
204 vtkCellData *inCellData = input->GetCellData();
205 for(int i = 0; i < inCellData->GetNumberOfArrays(); i++) {
206 vtkDataArray *curArray = inCellData->GetArray(i);
207 if(curArray != nullptr && curArray->GetName() != nullptr
208 && ArraySelection->ArrayIsEnabled(curArray->GetName())) {
209 cellDataArrays.emplace_back(curArray);
210 }
211 }
212
213 output->Initialize();
214
215 const auto &vertices{worker.getVertices()};
216 const auto &nodes{worker.getNodes()};
217 const auto &cells{worker.getCells()};
218 std::vector<ttk::SimplexId> vertexMap(vertices.size());
219
220 vtkNew<vtkPoints> points{};
221 vtkNew<vtkIntArray> indices{};
222
223 indices->SetNumberOfComponents(1);
224 indices->SetName(ttk::compactTriangulationIndex);
225
226 // insert vertices in the output mesh
227 for(size_t i = 0; i < vertices.size(); i++) {
228 float x, y, z;
229 triangulation.getVertexPoint(vertices[i], x, y, z);
230 points->InsertNextPoint(x, y, z);
231 vertexMap[vertices[i]] = i;
232 indices->InsertNextTuple1(nodes[i]);
233 }
234 output->SetPoints(points);
235
236 output->GetPointData()->AddArray(indices);
237
238 // insert cells in the output mesh
239 output->Allocate(cells.size());
240 const size_t dimension = triangulation.getCellVertexNumber(0);
241 if(dimension > 4 || dimension < 2) {
242 this->printErr("Dimension not supported");
243 return 0;
244 }
245
246 for(unsigned int i = 0; i < cells.size(); i++) {
247 std::array<vtkIdType, 4> cell{};
248 for(size_t j = 0; j < dimension; j++) {
249 ttk::SimplexId vertexId;
250 triangulation.getCellVertex(cells[i], j, vertexId);
251 cell[j] = vertexMap[vertexId];
252 }
253 std::sort(cell.begin(), cell.begin() + dimension);
254 if(dimension == 2) {
255 output->InsertNextCell(VTK_LINE, 2, cell.data());
256 } else if(dimension == 3) {
257 output->InsertNextCell(VTK_TRIANGLE, 3, cell.data());
258 } else if(dimension == 4) {
259 output->InsertNextCell(VTK_TETRA, 4, cell.data());
260 } else {
261 this->printErr("Should not get here!");
262 }
263 }
264
265 // Modify the selected point data arrays with new indices
266 for(vtkDataArray *scalarArray : pointDataArrays) {
267 vtkSmartPointer<vtkDataArray> updatedField(scalarArray->NewInstance());
268 updatedField->SetName(scalarArray->GetName());
269 for(size_t j = 0; j < vertices.size(); j++) {
270 updatedField->InsertTuple(j, scalarArray->GetTuple(vertices[j]));
271 }
272
273 output->GetPointData()->AddArray(updatedField);
274 }
275
276 // Modify the selected cell data arrays with new indices
277 for(vtkDataArray *scalarArray : cellDataArrays) {
278 vtkSmartPointer<vtkDataArray> updatedField(scalarArray->NewInstance());
279 updatedField->SetName(scalarArray->GetName());
280 for(size_t j = 0; j < cells.size(); j++) {
281 updatedField->InsertTuple(j, scalarArray->GetTuple(cells[j]));
282 }
283
284 output->GetCellData()->AddArray(updatedField);
285 }
286
287 this->printMsg("Done!", 1.0, tm.getElapsedTime(), 1);
288
289 // return success
290 return 1;
291}

293int ttkTriangulationManager::RequestData(vtkInformation *ttkNotUsed(request),
294 vtkInformationVector **inputVector,
295 vtkInformationVector *outputVector) {
296
297 auto *inputDataSet = vtkDataSet::GetData(inputVector[0]);
298 auto *triangulation = ttkAlgorithm::GetTriangulation(inputDataSet);
299
300 if(triangulation == nullptr) {
301 this->printErr("Triangulation is NULL");
302 return 0;
303 }
304
305 if(inputDataSet->IsA("vtkImageData")) {
306#ifdef TTK_ENABLE_MPI
307 vtkImageData *imageIn = vtkImageData::GetData(inputVector[0]);
308 vtkImageData *imageOut = vtkImageData::GetData(outputVector);
309 imageOut->ShallowCopy(imageIn);
310 this->processImplicit(*triangulation, imageIn, imageOut);
311#else
312 this->processImplicit(*triangulation);
313 auto *outputDataSet = vtkDataSet::GetData(outputVector, 0);
314 outputDataSet->ShallowCopy(inputDataSet);
315#endif
316 return 1;
317 } else if(inputDataSet->IsA("vtkUnstructuredGrid")
318 || inputDataSet->IsA("vtkPolyData")) {
319 return this->processExplicit(vtkUnstructuredGrid::GetData(outputVector, 0),
320 vtkPointSet::SafeDownCast(inputDataSet),
321 *triangulation);
322 }
323
324 return 1;
