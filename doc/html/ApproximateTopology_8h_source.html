519template <typename ScalarType, typename OffsetType>
520void ttk::ApproximateTopology::sortTriplets(std::vector<triplet> &triplets,
521 const ScalarType *const fakeScalars,
522 const OffsetType *const offsets,
523 const int *const monotonyOffsets,
524 const bool splitTree) const {
525 if(triplets.empty())
526 return;
527
528 const auto lt = [=](const SimplexId a, const SimplexId b) -> bool {
529 return ((fakeScalars[a] < fakeScalars[b])
530 || (fakeScalars[a] == fakeScalars[b]
531 && ((monotonyOffsets[a] < monotonyOffsets[b])
532 || (monotonyOffsets[a] == monotonyOffsets[b]
533 && offsets[a] < offsets[b]))));
534 };
535
536 // Sorting step
537 const auto cmp = [=](const triplet &t1, const triplet &t2) {
538 const SimplexId s1 = std::get<0>(t1);
539 const SimplexId s2 = std::get<0>(t2);
540 const SimplexId m1 = std::get<2>(t1);
541 const SimplexId m2 = std::get<2>(t2);
542 if(s1 != s2)
543 return lt(s1, s2) != splitTree;
544 else // s1 == s2
545 return lt(m1, m2) == splitTree;
546 };
547
548 TTK_PSORT(this->threadNumber_, triplets.begin(), triplets.end(), cmp);

674template <typename scalarType, typename offsetType>
675int ttk::ApproximateTopology::getMonotonyChangeByOldPointCPApproximate(
676 const SimplexId vertexId,
677 double eps,
678 const std::vector<polarity> &isNew,
679 std::vector<polarity> &toProcess,
680 std::vector<polarity> &toReprocess,
681 std::vector<std::pair<polarity, polarity>> &vlp,
682 scalarType *fakeScalars,
683 const offsetType *const offsets,
684 int *monotonyOffsets) const {
685
686 int hasMonotonyChanged = 0;
687 const SimplexId neighborNumber
688 = multiresTriangulation_.getVertexNeighborNumber(vertexId);
689 for(SimplexId i = 0; i < neighborNumber; i++) {
690 SimplexId neighborId = -1;
691 multiresTriangulation_.getVertexNeighbor(vertexId, i, neighborId);
692
693 // check for monotony changes
694 // const bool lowerStatic
695 // = (scalarField[neighborId] < fakeScalars[vertexId])
696 // || ((scalarField[neighborId] == fakeScalars[vertexId])
697 // && (offsets[neighborId] < offsets[vertexId]));
698 const bool lowerDynamic
699 = ((fakeScalars[neighborId] < fakeScalars[vertexId])
700 || (fakeScalars[neighborId] == fakeScalars[vertexId]
701 && ((monotonyOffsets[neighborId] < monotonyOffsets[vertexId])
702 || (monotonyOffsets[neighborId] == monotonyOffsets[vertexId]
703 && offsets[neighborId] < offsets[vertexId]))));
704
705 const polarity isUpperDynamic = lowerDynamic ? 0 : 255;
706 // const polarity isUpperStatic = lowerStatic ? 0 : 255;
707 const polarity isUpperOld = vlp[i].first;
708
709 if(isUpperDynamic != isUpperOld) { // change of monotony
710 SimplexId oldNeighbor = -1;
711 int oldDecimation = pow(2, decimationLevel_ + 1);
712 multiresTriangulation_.getVertexNeighborAtDecimation(
713 vertexId, i, oldNeighbor, oldDecimation);
714
715 double replacementValueDynamic
716 = (0.5 * (double)fakeScalars[oldNeighbor]
717 + .5 * (double)fakeScalars[vertexId]); // depends on epsilon
718 double deltaDynamic
719 = fabs((double)fakeScalars[neighborId] - replacementValueDynamic);
720
721 //=====================
722 SimplexId oldNeighNumber = 0;
723 SimplexId nnumber
724 = multiresTriangulation_.getVertexNeighborNumber(neighborId);
725 for(SimplexId iii = 0; iii < nnumber; iii++) {
726 SimplexId neighborId2 = -1;
727 multiresTriangulation_.getVertexNeighbor(neighborId, iii, neighborId2);
728 if(!isNew[neighborId2]) {
729 oldNeighNumber++;
730 }
731 }
732
733 if(deltaDynamic > eps or !isNew[neighborId] or oldNeighNumber > 2) {
734 hasMonotonyChanged = 1;
735
736 toReprocess[vertexId] = 255;
737 if(isNew[neighborId]) {
738 toProcess[neighborId] = 255;
739 } else {
740 toReprocess[neighborId] = 255;
741 }
742 const SimplexId neighborNumberNew
743 = multiresTriangulation_.getVertexNeighborNumber(neighborId);
744 for(SimplexId j = 0; j < neighborNumberNew; j++) {
745 SimplexId neighborIdNew = -1;
746 multiresTriangulation_.getVertexNeighbor(
747 neighborId, j, neighborIdNew);
748 if(isNew[neighborIdNew])
749 toProcess[neighborIdNew] = 255;
750 }
751 vlp[i].second = 255;
752 } else {
753 fakeScalars[neighborId] = replacementValueDynamic;
754
755 // corrects rounding error when we process an integer scalar field
756 if(fakeScalars[neighborId] == fakeScalars[oldNeighbor]) {
757 fakeScalars[neighborId] = fakeScalars[vertexId];
758 }
759
760 // change monotony offset
761 // The monotony must be preserved, meaning that
762 // if we had f(vertexId)>f(oldNeighbor)
763 // we should have f(vertexId)>f(neighborId)
764 // which is enforced when
765 // monotonyOffsets[vertexId]>monotonyOffsets[neighborId]
766 if(isUpperOld) { // we should enforce f(vertexId)<f(neighborId)
767 if(offsets[vertexId] > offsets[neighborId]) {
768 monotonyOffsets[neighborId]
769 = monotonyOffsets[vertexId] + pow(2, decimationLevel_);
770 if(monotonyOffsets[vertexId] == monotonyOffsets[oldNeighbor]
771 and fakeScalars[vertexId] == fakeScalars[oldNeighbor]) {
772 std::cout << "THIS IS AN ISSUE" << std::endl;
773 }
774 } else {
775 monotonyOffsets[neighborId] = monotonyOffsets[vertexId];
776 }
777 } else { // we should enforce f(vertexId)>f(neighborId)
778 if(offsets[vertexId] < offsets[neighborId]) {
779 monotonyOffsets[neighborId]
780 = monotonyOffsets[vertexId] - pow(2, decimationLevel_);
781 if(monotonyOffsets[vertexId] == monotonyOffsets[oldNeighbor]
782 and fakeScalars[vertexId] == fakeScalars[oldNeighbor]) {
783 std::cout << "THIS IS AN ISSUE" << std::endl;
784 }
785 } else {
786 monotonyOffsets[neighborId] = monotonyOffsets[vertexId];
787 }
788 }
789 }
790 } // end if change of monotony
791 } // end for neighbors
792 return hasMonotonyChanged;
793}
