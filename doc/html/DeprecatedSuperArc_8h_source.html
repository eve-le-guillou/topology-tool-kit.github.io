283 // From array : alloc should be already done
284 inline void appendVertLists(
285 const std::list<std::pair<SimplexId, bool> *> &vertLists,
286 std::list<SimplexId> vertSizes,
287 const SimplexId &totalSize) {
288 // size local
289 SimplexId newSize = sizeVertList_;
290 if(newSize == -1)
291 newSize = 0;
292
293 // size added
294 newSize += totalSize;
295
296 // alloc
297 std::pair<SimplexId, bool> *tmpVert
298 = new std::pair<SimplexId, bool>[newSize];
299 SimplexId pos = 0;
300
301 // values local
302 for(SimplexId i = 0; i < sizeVertList_; ++i) {
303 tmpVert[pos++] = vertList_[i];
304 }
305
306 // values added
307 for(std::pair<SimplexId, bool> *vertices : vertLists) {
308 const SimplexId size = vertSizes.front();
309 vertSizes.pop_front();
310 for(SimplexId i = 0; i < size; ++i) {
311 tmpVert[pos++] = vertices[i];
312 }
313 }
314
315 // set values
316 vertList_ = tmpVert;
317 sizeVertList_ = newSize;
318 }
