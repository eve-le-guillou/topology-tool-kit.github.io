170 inline int printMsg(const std::string &msg,
171 const double &progress,
172 const double &time,
173 const int &threads,
174 const double &memory,
175 const debug::LineMode &lineMode = debug::LineMode::NEW,
176 const debug::Priority &priority
177 = debug::Priority::PERFORMANCE,
178 std::ostream &stream = std::cout) const {
179 if((this->debugLevel_ < (int)priority)
180 && (globalDebugLevel_ < (int)priority))
181 return 0;
182
183 std::vector<std::string> chunks(4);
184 size_t q = 0;
185
186 if(memory >= 0)
187 chunks[q++] = std::to_string(static_cast<int>(memory)) + "MB";
188 if(time >= 0) {
189 std::stringstream sStream;
190 sStream.precision(3);
191 sStream << std::fixed;
192 sStream << time;
193 chunks[q++] = sStream.str() + "s";
194 }
195 if(threads >= 0)
196 chunks[q++] = std::to_string(threads) + "T";
197 if(progress >= 0)
198 chunks[q++] = std::to_string((int)(progress * 100)) + "%";
199
200 std::string stats = "";
201 if(q > 0) {
202 stats += " [";
203 stats += chunks[0];
204 for(size_t i = 1; i < q; i++)
205 stats += "|" + chunks[i];
206
207 stats += "]";
208 }
209
210 return this->printMsgInternal(
211 msg, stats, msg.length() < 1 ? ">" : ".", priority, lineMode, stream);
212 }

272 inline int printMsg(const std::vector<std::vector<std::string>> &rows,
273 const debug::Priority &priority = debug::Priority::INFO,
274 const bool hasHeader = true,
275 const debug::LineMode &lineMode = debug::LineMode::NEW,
276 std::ostream &stream = std::cout) const {
277 if((this->debugLevel_ < (int)priority)
278 && (globalDebugLevel_ < (int)priority))
279 return 0;
280
281 int nRows = rows.size();
282 int nCols = nRows > 0 ? rows[0].size() : 0;
283 if(nCols < 1)
284 return 0;
285
286 std::vector<std::string> formatedRows(nRows);
287 std::vector<size_t> colSizes(nCols, 0);
288 for(int i = 0; i < nRows; i++)
289 for(int j = 0; j < nCols; j++)
290 colSizes[j] = std::max(colSizes[j], rows[i][j].size());
291
292 auto formatCell = [](const std::string &value, const size_t &width,
293 const std::string &fillSymbol) {
294 std::string cell = value;
295 int diff = width - cell.size();
296 for(int i = 0; i < diff; i++)
297 cell += fillSymbol;
298 return cell;
299 };
300
301 // Values
302 int resultIndex = 0;
303 for(int i = 0; i < nRows; i++) {
304 auto &row = formatedRows[resultIndex++];
305 row
306 = formatCell(rows[i][0], colSizes[0], " ") + (hasHeader ? ": " : "");
307 if(nCols > 1)
308 row += formatCell(rows[i][1], colSizes[1], " ");
309 for(int j = 2; j < nCols; j++)
310 row += "," + formatCell(rows[i][j], colSizes[j], " ");
311 }
312
313 return this->printMsg(formatedRows, priority, lineMode, stream);
314 }

382 inline int
383 printMsgInternal(const std::string &msg,
384 const std::string &right,
385 const std::string &filler,
386 const debug::Priority &priority = debug::Priority::INFO,
387 const debug::LineMode &lineMode = debug::LineMode::NEW,
388 std::ostream &stream = std::cout) const {
389
390 std::string combinedMsg = msg;
391
392 if(filler.length() > 0) {
393 if(msg.length() > 0)
394 combinedMsg += " ";
395
396 int gapWidth = debug::LINEWIDTH - this->debugMsgPrefix_.length()
397 - combinedMsg.length() - right.length();
398 gapWidth = std::max(gapWidth / filler.length(), (size_t)1);
399
400 for(int i = 0; i < gapWidth; i++)
401 combinedMsg += filler;
402
403 combinedMsg += debug::output::BLUE + right + debug::output::ENDCOLOR;
404 }
405
406 return this->printMsgInternal(combinedMsg, priority, lineMode, stream);
407 }

412 inline int printMsgInternal(const std::string &msg,
413 const debug::Priority &priority,
414 const debug::LineMode &lineMode,
415 std::ostream &stream = std::cout) const {
416
417 if((this->debugLevel_ < (int)priority)
418 && (globalDebugLevel_ < (int)priority))
419 return 0;
420
421 // on error or warning print end of line
422 if((int)priority < 2 && this->lastLineMode == debug::LineMode::REPLACE)
423 stream << "\n";
424
425 // print prefix
426 if(lineMode != debug::LineMode::APPEND)
427 stream << debug::output::GREEN << this->debugMsgPrefix_
428 << debug::output::ENDCOLOR;
429
430 // print error or warning prefix
431 if((int)priority == 0)
432 stream << debug::output::RED << "[ERROR]" << debug::output::ENDCOLOR
433 << " ";
434 else if((int)priority == 1)
435 stream << debug::output::YELLOW << "[WARNING]"
436 << debug::output::ENDCOLOR << " ";
437
438 // print msg
439 stream << msg.data();
440
441 // go either into new line or replace current line
442 if(lineMode == debug::LineMode::NEW)
443 stream << "\n";
444 else if(lineMode == debug::LineMode::REPLACE)
445 stream << "\r";
446
447 // flush stream
448 stream.flush();
449
450 this->lastLineMode = lineMode;
451
452 return 1;
453 }
