95 template <class dataType>
96 void initCentroids(
97 std::vector<ftm::FTMTree_MT *> &trees,
98 std::vector<ftm::FTMTree_MT *> &trees2,
99 std::vector<std::vector<ftm::MergeTree<dataType>>> &allCentroids) {
100 allCentroids = std::vector<std::vector<ftm::MergeTree<dataType>>>(
101 2, std::vector<ftm::MergeTree<dataType>>(noCentroids_));
102 std::vector<dataType> distances(
103 trees.size(), std::numeric_limits<dataType>::max());
104
105 // Manage size limited trees
106 double limitPercent = barycenterSizeLimitPercent_ / noCentroids_;
107 std::vector<ftm::MergeTree<dataType>> mTreesLimited, mTrees2Limited;
108 bool doSizeLimit
109 = (limitPercent > 0.0 or barycenterMaximumNumberOfPairs_ > 0);
110 if(doSizeLimit) {
111 getSizeLimitedTrees<dataType>(
112 trees, barycenterMaximumNumberOfPairs_, limitPercent, mTreesLimited);
113 if(trees2.size() != 0)
114 getSizeLimitedTrees<dataType>(trees2, barycenterMaximumNumberOfPairs_,
115 limitPercent, mTrees2Limited);
116 }
117
118 // Init centroids
119 for(unsigned int i = 0; i < noCentroids_; ++i) {
120 int bestIndex = -1;
121 if(i == 0) {
122 bestIndex = getBestInitTreeIndex<dataType>(
123 trees, trees2, limitPercent, false);
124 } else {
125 // Create vector of probabilities
126 double sum = 0;
127 for(auto val : distances)
128 sum += val;
129 double bestValue = std::numeric_limits<double>::lowest();
130 std::vector<double> probabilities(trees.size());
131 for(unsigned int j = 0; j < distances.size(); ++j) {
132 probabilities[j]
133 = (sum != 0 ? distances[j] / sum : 1.0 / distances.size());
134 if(probabilities[j] > bestValue) {
135 bestValue = probabilities[j];
136 bestIndex = j;
137 }
138 }
139 if(not deterministic_) {
140 std::random_device rd;
141 std::default_random_engine generator(rd());
142 std::discrete_distribution<int> distribution(
143 probabilities.begin(), probabilities.end());
144 bestIndex = distribution(generator);
145 }
146 }
147 printMsg(
148 "Init index : " + std::to_string(bestIndex), debug::Priority::DETAIL);
149 // Create new centroid
150 allCentroids[0][i]
151 = ftm::copyMergeTree<dataType>(trees[bestIndex], true);
152 limitSizeBarycenter(allCentroids[0][i], trees, limitPercent);
153 ftm::cleanMergeTree<dataType>(allCentroids[0][i]);
154 if(trees2.size() != 0) {
155 allCentroids[1][i]
156 = ftm::copyMergeTree<dataType>(trees2[bestIndex], true);
157 limitSizeBarycenter(allCentroids[1][i], trees2, limitPercent);
158 ftm::cleanMergeTree<dataType>(allCentroids[1][i]);
159 }
160
161 if(i == noCentroids_ - 1)
162 continue;
163#ifdef TTK_ENABLE_OPENMP
164#pragma omp parallel for schedule(dynamic) shared(allCentroids) \
165 num_threads(this->threadNumber_) if(parallelize_)
166#endif
167 for(unsigned int j = 0; j < trees.size(); ++j) {
168 std::vector<std::tuple<ftm::idNode, ftm::idNode, double>> matching,
169 matching2;
170 dataType distanceT, distanceT2;
171 ftm::FTMTree_MT *treeToUse
172 = (doSizeLimit ? &(mTreesLimited[j].tree) : trees[j]);
173 computeOneDistance<dataType>(treeToUse, allCentroids[0][i], matching,
174 distanceT, useDoubleInput_);
175 if(trees2.size() != 0) {
176 ftm::FTMTree_MT *tree2ToUse
177 = (doSizeLimit ? &(mTrees2Limited[j].tree) : trees2[j]);
178 computeOneDistance<dataType>(tree2ToUse, allCentroids[1][i],
179 matching2, distanceT2, useDoubleInput_,
180 false);
181 distanceT = mixDistances<dataType>(distanceT, distanceT2);
182 }
183 distances[j] = std::min(distances[j], distanceT);
184 }
185 }

250 template <class dataType>
251 void assignmentCentroidsAccelerated(
252 std::vector<ftm::FTMTree_MT *> &trees,
253 std::vector<ftm::MergeTree<dataType>> &centroids,
254 std::vector<std::tuple<int, int>> &assignmentC,
255 std::vector<dataType> &bestDistanceT,
256 std::vector<ftm::FTMTree_MT *> &trees2,
257 std::vector<ftm::MergeTree<dataType>> &centroids2) {
258 if(not acceleratedInitialized_) {
259 initAcceleratedKMeans<dataType>(trees, centroids, trees2, centroids2);
260 } else {
261 // Compute distance between old and new corresponding centroids
262 std::vector<dataType> distanceShift(centroids.size()),
263 distanceShift2(centroids2.size());
264#ifdef TTK_ENABLE_OPENMP
265#pragma omp parallel for schedule(dynamic) \
266 shared(centroids, centroids2, oldCentroids_, oldCentroids2_) \
267 num_threads(this->threadNumber_) if(parallelize_)
268#endif
269 for(unsigned int i = 0; i < centroids.size(); ++i) {
270 std::vector<std::tuple<ftm::idNode, ftm::idNode, double>> matching,
271 matching2;
272 computeOneDistance<dataType>(centroids[i], oldCentroids_[i], matching,
273 distanceShift[i], useDoubleInput_);
274 if(trees2.size() != 0) {
275 computeOneDistance<dataType>(centroids2[i], oldCentroids2_[i],
276 matching2, distanceShift2[i],
277 useDoubleInput_, false);
278 distanceShift[i]
279 = mixDistances<dataType>(distanceShift[i], distanceShift2[i]);
280 }
281 }
282
283 // Step 5
284 for(unsigned int i = 0; i < trees.size(); ++i)
285 for(unsigned int c = 0; c < centroids.size(); ++c)
286 lowerBound_[i][c]
287 = std::max(lowerBound_[i][c] - distanceShift[c], 0.0);
288
289 // Step 6
290 for(unsigned int i = 0; i < trees.size(); ++i) {
291 upperBound_[i] = upperBound_[i] + distanceShift[bestCentroid_[i]];
292 recompute_[i] = true;
293 }
294 }
295
296 // Step 1
297 std::vector<std::vector<double>> centroidsDistance, centroidsDistance2;
298 getCentroidsDistanceMatrix<dataType>(
299 centroids, centroidsDistance, useDoubleInput_);
300 if(trees2.size() != 0) {
301 getCentroidsDistanceMatrix<dataType>(
302 centroids2, centroidsDistance2, useDoubleInput_, false);
303 mixDistancesMatrix(centroidsDistance, centroidsDistance2);
304 }
305 std::vector<double> centroidScore(
306 centroids.size(), std::numeric_limits<double>::max());
307 for(unsigned int i = 0; i < centroids.size(); ++i)
308 for(unsigned int j = i + 1; j < centroids.size(); ++j) {
309 if(0.5 * centroidsDistance[i][j] < centroidScore[i])
310 centroidScore[i] = 0.5 * centroidsDistance[i][j];
311 if(0.5 * centroidsDistance[i][j] < centroidScore[j])
312 centroidScore[j] = 0.5 * centroidsDistance[i][j];
313 }
314
315 // Step 2
316 std::vector<bool> identified(trees.size());
317 for(unsigned int i = 0; i < trees.size(); ++i)
318 identified[i] = (upperBound_[i] <= centroidScore[bestCentroid_[i]]);
319
320 // Step 3
321#ifdef TTK_ENABLE_OPENMP
322#pragma omp parallel for schedule(dynamic) shared(centroids, centroids2) \
323 num_threads(this->threadNumber_) if(parallelize_)
324#endif
325 for(unsigned int i = 0; i < trees.size(); ++i)
326 for(unsigned int c = 0; c < centroids.size(); ++c) {
327 if(not identified[i] and (int) c != bestCentroid_[i]
328 and upperBound_[i] > lowerBound_[i][c]
329 and upperBound_[i]
330 > 0.5 * centroidsDistance[bestCentroid_[i]][c]) {
331 // Step 3a
332 if(recompute_[i]) {
333 std::vector<std::tuple<ftm::idNode, ftm::idNode, double>>
334 matching, matching2;
335 dataType distance, distance2;
336 computeOneDistance<dataType>(trees[i],
337 centroids[bestCentroid_[i]],
338 matching, distance, useDoubleInput_);
339 if(trees2.size() != 0) {
340 computeOneDistance<dataType>(
341 trees2[i], centroids2[bestCentroid_[i]], matching2, distance2,
342 useDoubleInput_, false);
343 distance = mixDistances<dataType>(distance, distance2);
344 }
345 recompute_[i] = false;
346 lowerBound_[i][bestCentroid_[i]] = distance;
347 upperBound_[i] = distance;
348 bestDistance_[i] = distance;
349 } else {
350 bestDistance_[i] = upperBound_[i];
351 }
352 // Step 3b
353 if(bestDistance_[i] > lowerBound_[i][c]
354 and bestDistance_[i]
355 > 0.5 * centroidsDistance[bestCentroid_[i]][c]) {
356 std::vector<std::tuple<ftm::idNode, ftm::idNode, double>>
357 matching, matching2;
358 dataType distance, distance2;
359 computeOneDistance<dataType>(
360 trees[i], centroids[c], matching, distance, useDoubleInput_);
361 if(trees2.size() != 0) {
362 computeOneDistance<dataType>(trees2[i], centroids2[c],
363 matching2, distance2,
364 useDoubleInput_, false);
365 distance = mixDistances<dataType>(distance, distance2);
366 }
367 lowerBound_[i][c] = distance;
368 if(distance < bestDistance_[i]) {
369 bestCentroid_[i] = c;
370 upperBound_[i] = distance;
371 bestDistance_[i] = distance;
372 }
373 }
374 }
375 }
376
377 // Copy centroids for next step
378 copyCentroids<dataType>(centroids, oldCentroids_);
379 if(trees2.size() != 0)
380 copyCentroids<dataType>(centroids2, oldCentroids2_);
381
382 // Manage output
383 for(unsigned int i = 0; i < bestDistance_.size(); ++i)
384 bestDistanceT[i] = bestDistance_[i];
385 for(unsigned int i = 0; i < bestCentroid_.size(); ++i)
386 assignmentC.emplace_back(bestCentroid_[i], i);
387 }

541 for(auto asgn : assignmentC) {
542 assignedTrees[std::get<0>(asgn)].push_back(trees[std::get<1>(asgn)]);
543 assignedTreesIndex[std::get<0>(asgn)].push_back(std::get<1>(asgn));
544 assignedAlphas[std::get<0>(asgn)].push_back(alphas[std::get<1>(asgn)]);

557 shared(centroids) if(parallelize_ and parallelizeUpdate_)
558 {
559#pragma omp single nowait

561#endif

566#endif

576 limitSizeBarycenter(centroids[i], assignedTrees[i]);
577 ftm::cleanMergeTree<dataType>(centroids[i]);
578 } else if(not samePreviousAssignment(i)) {
579 // Do not update if same previous assignment

581 oneCentroidUpdated = true;
582 double alphasSum = 0;
583 for(unsigned int j = 0; j < assignedAlphas[i].size(); ++j)
584 alphasSum += assignedAlphas[i][j];
585 for(unsigned int j = 0; j < assignedAlphas[i].size(); ++j)

589 assignedTrees[i], centroids[i], assignedAlphas[i], matching);
590 std::vector<ftm::idNode> deletedNodesT;

592 &(centroids[i].tree), 0, deletedNodesT);
593 ftm::cleanMergeTree<dataType>(centroids[i]);
594 }
595#ifdef TTK_ENABLE_OPENMP
596 } // pragma omp task
597#endif
598 }
599#ifdef TTK_ENABLE_OPENMP
600#pragma omp taskwait
601 } // pragma omp single nowait
602 } // pragma omp parallel
603#endif
604 return oneCentroidUpdated;
605 }

642 template <class dataType>
643 void computeCentroids(std::vector<ftm::FTMTree_MT *> &trees,
644 std::vector<ftm::MergeTree<dataType>> &centroids,
645 matchingVector &outputMatching,
646 std::vector<double> &alphas,
647 std::vector<int> &clusteringAssignment,
648 std::vector<ftm::FTMTree_MT *> &trees2,
649 std::vector<ftm::MergeTree<dataType>> &centroids2,
650 matchingVector &outputMatching2) {
651 Timer t_clust;
652
653 printCentroidsStats(centroids, centroids2);
654
655 // Run
656 int noCentroidsT = centroids.size();
657 bool converged = false;
658 dataType inertia = -1;
659 dataType minInertia = std::numeric_limits<dataType>::max();
660 int cptBlocked = 0;
661 noIterationC_ = 0;
662 std::vector<std::tuple<int, int>> assignmentC;
663 std::vector<dataType> bestDistanceT(
664 trees.size(), std::numeric_limits<dataType>::max());
665 while(not converged) {
666 ++noIterationC_;
667
668 printMsg(debug::Separator::L1);
669 std::stringstream ssIter;
670 ssIter << "Iteration " << noIterationC_;
671 printMsg(ssIter.str());
672
673 // --- Assignment
674 Timer t_assignment;
675 assignmentCentroids<dataType>(
676 trees, centroids, assignmentC, bestDistanceT, trees2, centroids2);
677 auto t_assignment_time = t_assignment.getElapsedTime();
678 printMsg("Assignment", 1, t_assignment_time, this->threadNumber_);
679
680 // --- Update
681 Timer t_update;
682 bool trees1Updated = true, trees2Updated = true;
683 trees1Updated
684 = updateCentroids<dataType>(trees, centroids, alphas, assignmentC);
685 if(trees2.size() != 0)
686 trees2Updated = updateCentroids<dataType>(
687 trees2, centroids2, alphas, assignmentC);
688 auto t_update_time = t_update.getElapsedTime();
689 printMsg("Update", 1, t_update_time, this->threadNumber_);
690 printCentroidsStats(centroids, centroids2);
691
692 // --- Check convergence
693 dataType currentInertia = 0;
694 for(auto distance : bestDistanceT)
695 currentInertia += distance * distance;
696 converged = std::abs((double)(inertia - currentInertia)) < 0.01;
697 inertia = currentInertia;
698 std::stringstream ss3;
699 ss3 << "Inertia : " << inertia;
700 printMsg(ss3.str());
701
702 minInertia = std::min(minInertia, inertia);
703 if(not converged) {
704 cptBlocked += (minInertia < inertia) ? 1 : 0;
705 converged = (cptBlocked >= 10);
706 }
707
708 // Converged if barycenters were not updated (same assignment than last
709 // iteration)
710 converged = converged or (not trees1Updated and not trees2Updated);
711
712 // --- Reset vectors
713 if(not converged) {
714 assignmentC.clear();
715 bestDistanceT = std::vector<dataType>(
716 trees.size(), std::numeric_limits<dataType>::max());
717 }
718 }
719
720 // Final processing
721 printMsg(debug::Separator::L1);
722 printMsg("Final assignment");
723 matchingVector matchingsC(noCentroidsT);
724 matchingVector matchingsC2(noCentroidsT);
725 finalAssignmentCentroids<dataType>(trees, centroids, matchingsC,
726 assignmentC, bestDistanceT, trees2,
727 centroids2, matchingsC2);
728 for(auto dist : bestDistanceT)
729 finalDistances_.push_back(dist);
730 dataType currentInertia = 0;
731 for(auto distance : bestDistanceT)
732 currentInertia += distance * distance;
733 std::stringstream ss;
734 ss << "Inertia : " << currentInertia;
735 printMsg(ss.str());
736
737 // Manage output
738 std::vector<int> cptCentroid(centroids.size(), 0);
739 for(auto asgn : assignmentC) {
740 int centroid = std::get<0>(asgn);
741 int tree = std::get<1>(asgn);
742 // std::cout << centroid << " " << tree << std::endl;
743 clusteringAssignment[tree] = centroid;
744 outputMatching[centroid][tree]
745 = matchingsC[centroid][cptCentroid[centroid]];
746 if(trees2.size() != 0)
747 outputMatching2[centroid][tree]
748 = matchingsC2[centroid][cptCentroid[centroid]];
749 ++cptCentroid[centroid];
750 }
751
752 auto clusteringTime = t_clust.getElapsedTime() - addDeletedNodesTime_;
753 printMsg("Total", 1, clusteringTime, this->threadNumber_);
754 }
