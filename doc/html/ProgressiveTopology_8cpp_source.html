363void ttk::ProgressiveTopology::sortTriplets(std::vector<triplet> &triplets,
364 const SimplexId *const offsets,
365 const bool splitTree) const {
366 if(triplets.empty())
367 return;
368
369 const auto lt = [=](const SimplexId a, const SimplexId b) -> bool {
370 return offsets[a] < offsets[b];
371 };
372
373 // Sorting step
374 const auto cmp = [=](const triplet &t1, const triplet &t2) {
375 const SimplexId s1 = std::get<0>(t1);
376 const SimplexId s2 = std::get<0>(t2);
377 const SimplexId m1 = std::get<2>(t1);
378 const SimplexId m2 = std::get<2>(t2);
379 if(s1 != s2)
380 return lt(s1, s2) != splitTree;
381 else // s1 == s2
382 return lt(m1, m2) == splitTree;
383 };
384
385 TTK_PSORT(this->threadNumber_, triplets.begin(), triplets.end(), cmp);
386}

388void ttk::ProgressiveTopology::tripletsToPersistencePairs(
389 std::vector<PersistencePair> &pairs,
390 std::vector<std::vector<SimplexId>> &vertexRepresentatives,
391 std::vector<triplet> &triplets,
392 const SimplexId *const offsets,
393 const bool splitTree) const {
394
395 Timer tm;
396 if(triplets.empty())
397 return;
398 size_t numberOfPairs = 0;
399
400 const auto lt = [=](const SimplexId a, const SimplexId b) -> bool {
401 return offsets[a] < offsets[b];
402 };
403
404 const auto getRep = [&](SimplexId v) -> SimplexId {
405 auto r = vertexRepresentatives[v][0];
406 while(r != v) {
407 v = r;
408 r = vertexRepresentatives[v][0];
409 }
410 return r;
411 };
412
413 for(const auto &t : triplets) {
414 SimplexId r1 = getRep(std::get<1>(t));
415 SimplexId r2 = getRep(std::get<2>(t));
416 if(r1 != r2) {
417 SimplexId s = std::get<0>(t);
418 numberOfPairs++;
419
420 // Add pair
421 if(splitTree) {
422 // r1 = min(r1, r2), r2 = max(r1, r2)
423 if(lt(r2, r1)) {
424 std::swap(r1, r2);
425 }
426 // pair saddle-max: s -> min(r1, r2);
427 pairs.emplace_back(s, r1, 2);
428
429 } else {
430 // r1 = max(r1, r2), r2 = min(r1, r2)
431 if(lt(r1, r2)) {
432 std::swap(r1, r2);
433 }
434 // pair min-saddle: max(r1, r2) -> s;
435 pairs.emplace_back(r1, s, 0);
436 }
437
438 vertexRepresentatives[std::get<1>(t)][0] = r2;
439 vertexRepresentatives[r1][0] = r2;
440 }
441 }
442
443 const std::string ptype = splitTree ? "sad-max" : "min-sad";
444 this->printMsg("found all " + ptype + " pairs", 1.0, tm.getElapsedTime(),
445 this->threadNumber_, debug::LineMode::NEW,
446 debug::Priority::DETAIL);
447}

704ttk::SimplexId ttk::ProgressiveTopology::propageFromSaddles(
705 const SimplexId vertexId,
706 std::vector<Lock> &vertLock,
707 std::vector<polarity> &toPropage,
708 std::vector<std::vector<SimplexId>> &vertexRepresentatives,
709 std::vector<std::vector<SimplexId>> &saddleCC,
710 std::vector<polarity> &isUpdated,
711 std::vector<SimplexId> &globalExtremum,
712 const SimplexId *const offsets,
713 const bool splitTree) const {
714
715 auto &toProp = toPropage[vertexId];
716 auto &reps = vertexRepresentatives[vertexId];
717 auto &updated = isUpdated[vertexId];
718
719 const auto gt = [=](const SimplexId a, const SimplexId b) {
720 return (offsets[a] > offsets[b]) == splitTree;
721 };
722
723 if(updated) {
724 return reps[0];
725 }
726
727 if(this->threadNumber_ > 1) {
728 vertLock[vertexId].lock();
729 }
730
731 if(toProp) { // SADDLE POINT
732 const auto &CC = saddleCC[vertexId];
733 reps.clear();
734 reps.reserve(CC.size());
735 for(size_t r = 0; r < CC.size(); r++) {
736 SimplexId neighborId = -1;
737 SimplexId localId = CC[r];
738 multiresTriangulation_.getVertexNeighbor(vertexId, localId, neighborId);
739 SimplexId ret = propageFromSaddles(
740 neighborId, vertLock, toPropage, vertexRepresentatives, saddleCC,
741 isUpdated, globalExtremum, offsets, splitTree);
742 reps.emplace_back(ret);
743 }
744
745 if(reps.size() > 1) {
746 // sort & remove duplicate elements
747 std::sort(reps.begin(), reps.end(), gt);
748 const auto last = std::unique(reps.begin(), reps.end());
749 reps.erase(last, reps.end());
750 }
751
752 updated = 255;
753 if(this->threadNumber_ > 1) {
754 vertLock[vertexId].unlock();
755 }
756
757 return reps[0];
758
759 } else {
760
761 SimplexId ret = vertexId;
762 SimplexId neighborNumber
763 = multiresTriangulation_.getVertexNeighborNumber(vertexId);
764 SimplexId maxNeighbor = vertexId;
765 for(SimplexId i = 0; i < neighborNumber; i++) {
766 SimplexId neighborId = -1;
767 multiresTriangulation_.getVertexNeighbor(vertexId, i, neighborId);
768 if(gt(neighborId, maxNeighbor)) {
769 maxNeighbor = neighborId;
770 }
771 }
772 if(maxNeighbor != vertexId) { // not an extremum
773 ret = propageFromSaddles(maxNeighbor, vertLock, toPropage,
774 vertexRepresentatives, saddleCC, isUpdated,
775 globalExtremum, offsets, splitTree);
776 } else {
777#ifdef TTK_ENABLE_OPENMP
778 const auto tid = omp_get_thread_num();
779#else
780 const auto tid = 0;
781#endif // TTK_ENABLE_OPENMP
782 if(gt(vertexId, globalExtremum[tid])) {
783 globalExtremum[tid] = vertexId;
784 }
785 }
786 reps.resize(1);
787 reps[0] = ret;
788 updated = 255;
789 if(this->threadNumber_ > 1) {
790 vertLock[vertexId].unlock();
791 }
792 return ret;
793 }
794}

980void ttk::ProgressiveTopology::updateDynamicLink(
981 DynamicTree &link,
982 std::vector<std::pair<polarity, polarity>> &vlp,
983 std::vector<std::pair<SimplexId, SimplexId>> &vl) const {
984
985 std::vector<SimplexId> monotony_changes_list{};
986
987 for(size_t neighborId = 0; neighborId < vlp.size(); neighborId++) {
988 const polarity isBroken = vlp[neighborId].second;
989 if(isBroken != 0) {
990 monotony_changes_list.emplace_back(neighborId);
991 }
992 }
993
994 // loop on the link
995 // for each edge that shares n0
996 // if only one break and different polarity : remove
997 // else if only one break and same polarity : insert
998 // else : do nothing
999 std::vector<std::vector<std::pair<SimplexId, SimplexId>>> edgesToInsertLater(
1000 vlp.size());
1001 std::vector<std::vector<std::pair<SimplexId, SimplexId>>> edgesToRemoveLater(
1002 vlp.size());
1003
1004 for(size_t e = 0; e < vl.size(); e++) {
1005 const SimplexId n0 = vl[e].first;
1006 const SimplexId n1 = vl[e].second;
1007 const polarity isBroken0 = vlp[n0].second;
1008 const polarity isBroken1 = vlp[n1].second;
1009
1010 if(isBroken0 != 0 and isBroken1 == 0) {
1011 if(vlp[n0].first != vlp[n1].first) {
1012 edgesToInsertLater[n0].emplace_back(n0, n1);
1013 } else {
1014 edgesToRemoveLater[n0].emplace_back(n0, n1);
1015 }
1016 } else if(isBroken0 == 0 and isBroken1 != 0) {
1017 if(vlp[n0].first != vlp[n1].first) {
1018 edgesToInsertLater[n1].emplace_back(n1, n0);
1019 } else {
1020 edgesToRemoveLater[n1].emplace_back(n1, n0);
1021 }
1022 }
1023 }
1024
1025 // REMOVE EDGES:
1026 for(const auto brokenNode : monotony_changes_list) {
1027 vlp[brokenNode].first = ~vlp[brokenNode].first;
1028 vlp[brokenNode].second = 0;
1029
1030 link.getNode(brokenNode)->evert();
1031 for(const auto &edge : edgesToRemoveLater[brokenNode]) {
1032 link.removeEdge(edge.first, edge.second);
1033 }
1034 }
1035 if(!edgesToRemoveLater.empty()) {
1036 // reconnect link
1037 for(const auto &edge : vl) {
1038 if(vlp[edge.first].first == vlp[edge.second].first) {
1039 link.insertEdge(edge.first, edge.second);
1040 }
1041 }
1042 }
1043
1044 // INSERT EDGES
1045 for(const auto brokenNode : monotony_changes_list) {
1046 for(const auto &edge : edgesToInsertLater[brokenNode]) {
1047 link.insertEdge(edge.first, edge.second);
1048 }
1049 }
1050}

1090char ttk::ProgressiveTopology::getCriticalTypeFromLink(
1091 const SimplexId globalId,
1092 const std::vector<std::pair<polarity, polarity>> &vlp,
1093 DynamicTree &link,
1094 polarity &reprocess) const {
1095
1096 const auto nbCC = link.getNbCC();
1097
1098 bool isVertexOnBoundary
1099 = multiresTriangulation_.getTriangulation()->isVertexOnBoundary(globalId);
1100
1101 int dimensionality = multiresTriangulation_.getDimensionality();
1102 SimplexId downValence = 0, upValence = 0;
1103
1104 std::vector<size_t> CCIds;
1105 CCIds.reserve(nbCC);
1106 link.retrieveNbCC(CCIds);
1107 for(size_t i = 0; i < CCIds.size(); i++) {
1108 const SimplexId neighbor = CCIds[i];
1109 const polarity isUpper = vlp[neighbor].first;
1110 if(isUpper) {
1111 upValence++;
1112 } else {
1113 downValence++;
1114 }
1115 }
1116
1117 if(downValence == -1 && upValence == -1) {
1118 return -1;
1119 } else if(downValence == 0 && upValence == 1) {
1120 return static_cast<char>(CriticalType::Local_minimum);
1121 } else if(downValence == 1 && upValence == 0) {
1122 return static_cast<char>(CriticalType::Local_maximum);
1123 } else if(downValence == 1 && upValence == 1) {
1124
1125 // special case of boundary saddles
1126 if(dimensionality == 3 && isVertexOnBoundary) {
1127 reprocess = 255; // need to reprocess at finer resolutions
1128
1129 // find whether the upper and lower links touch the boundary
1130 bool isUpperOnBoundary = false, isLowerOnBoundary = false;
1131 const SimplexId neighborNumber = vlp.size();
1132 SimplexId neighborId = -1;
1133 for(int ineigh = 0; ineigh < neighborNumber; ineigh++) {
1134 multiresTriangulation_.getVertexNeighbor(globalId, ineigh, neighborId);
1135 if(multiresTriangulation_.getTriangulation()->isVertexOnBoundary(
1136 neighborId)) {
1137 if(vlp[ineigh].first) {
1138 isUpperOnBoundary = true;
1139 } else {
1140 isLowerOnBoundary = true;
1141 }
1142 }
1143 }
1144
1145 if((isUpperOnBoundary) && (!isLowerOnBoundary)) {
1146 return (char)(CriticalType::Saddle1);
1147 }
1148 if((!isUpperOnBoundary) && (isLowerOnBoundary)) {
1149 return (char)(CriticalType::Saddle2);
1150 }
1151 }
1152
1153 // regular point
1154 return static_cast<char>(CriticalType::Regular);
1155
1156 } else {
1157 // saddles
1158 if(dimensionality == 2) {
1159 if((downValence == 2 && upValence == 1)
1160 || (downValence == 1 && upValence == 2)
1161 || (downValence == 2 && upValence == 2)) {
1162 // regular saddle
1163 return static_cast<char>(CriticalType::Saddle1);
1164 } else {
1165 // monkey saddle, saddle + extremum
1166 return static_cast<char>(CriticalType::Degenerate);
1167 // NOTE: you may have multi-saddles on the boundary in that
1168 // configuration
1169 // to make this computation 100% correct, one would need to
1170 // disambiguate boundary from interior vertices
1171 }
1172 } else if(dimensionality == 3) {
1173 if(downValence == 2 && upValence == 1) {
1174 return static_cast<char>(CriticalType::Saddle1);
1175 } else if(downValence == 1 && upValence == 2) {
1176 return static_cast<char>(CriticalType::Saddle2);
1177 } else {
1178 // monkey saddle, saddle + extremum
1179 return static_cast<char>(CriticalType::Degenerate);
1180 // NOTE: we may have a similar effect in 3D (TODO)
1181 }
1182 }
1183 }
1184
1185 // -2: regular points
1186 return static_cast<char>(CriticalType::Regular);
1187}
