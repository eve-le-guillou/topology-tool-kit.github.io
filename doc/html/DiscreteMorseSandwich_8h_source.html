451template <typename triangulationType>
452std::vector<std::vector<SimplexId>>
453 ttk::DiscreteMorseSandwich::getSaddle1ToMinima(
454 const std::vector<SimplexId> &criticalEdges,
455 const triangulationType &triangulation) const {
456
457 Timer tm{};
458
459 std::vector<std::vector<SimplexId>> res(criticalEdges.size());
460
461 // follow vpaths from 1-saddles to minima
462#ifdef TTK_ENABLE_OPENMP
463#pragma omp parallel for num_threads(threadNumber_)
464#endif
465 for(size_t i = 0; i < criticalEdges.size(); ++i) {
466 auto &mins = res[i];
467
468 const auto followVPath = [this, &mins, &triangulation](const SimplexId v) {
469 std::vector<Cell> vpath{};
470 this->dg_.getDescendingPath(Cell{0, v}, vpath, triangulation);
471 Cell &lastCell = vpath.back();
472 if(lastCell.dim_ == 0 && this->dg_.isCellCritical(lastCell)) {
473 mins.emplace_back(lastCell.id_);
474 }
475 };
476
477 // critical edge vertices
478 SimplexId v0{}, v1{};
479 triangulation.getEdgeVertex(criticalEdges[i], 0, v0);
480 triangulation.getEdgeVertex(criticalEdges[i], 1, v1);
481
482 // follow vpath from each vertex of the critical edge
483 followVPath(v0);
484 followVPath(v1);
485 }
486
487 this->printMsg("Computed the descending 1-separatrices", 1.0,
488 tm.getElapsedTime(), this->threadNumber_, debug::LineMode::NEW,
489 debug::Priority::DETAIL);
490
491 return res;
492}

494template <typename triangulationType, typename GFS, typename GFSN, typename OB>
495std::vector<std::vector<SimplexId>>
496 ttk::DiscreteMorseSandwich::getSaddle2ToMaxima(
497 const std::vector<SimplexId> &criticalCells,
498 const GFS &getFaceStar,
499 const GFSN &getFaceStarNumber,
500 const OB &isOnBoundary,
501 const triangulationType &triangulation) const {
502
503 Timer tm{};
504
505 const auto dim = this->dg_.getDimensionality();
506 std::vector<std::vector<SimplexId>> res(criticalCells.size());
507
508 // follow vpaths from 2-saddles to maxima
509#ifdef TTK_ENABLE_OPENMP
510#pragma omp parallel for num_threads(threadNumber_)
511#endif
512 for(size_t i = 0; i < criticalCells.size(); ++i) {
513 const auto sid = criticalCells[i];
514 auto &maxs = res[i];
515
516 const auto followVPath
517 = [this, dim, &maxs, &triangulation](const SimplexId v) {
518 std::vector<Cell> vpath{};
519 this->dg_.getAscendingPath(Cell{dim, v}, vpath, triangulation);
520 Cell &lastCell = vpath.back();
521 if(lastCell.dim_ == dim && this->dg_.isCellCritical(lastCell)) {
522 maxs.emplace_back(lastCell.id_);
523 } else if(lastCell.dim_ == dim - 1) {
524 maxs.emplace_back(-1);
525 }
526 };
527
528 const auto starNumber = getFaceStarNumber(sid);
529
530 for(SimplexId j = 0; j < starNumber; ++j) {
531 SimplexId cellId{};
532 getFaceStar(sid, j, cellId);
533 followVPath(cellId);
534 }
535
536 if(isOnBoundary(sid)) {
537 // critical saddle is on boundary
538 maxs.emplace_back(-1);
539 }
540 }
541
542 this->printMsg("Computed the ascending 1-separatrices", 1.0,
543 tm.getElapsedTime(), this->threadNumber_, debug::LineMode::NEW,
544 debug::Priority::DETAIL);
545
546 return res;
547}

549template <typename triangulationType>
550void ttk::DiscreteMorseSandwich::getMinSaddlePairs(
551 std::vector<PersistencePair> &pairs,
552 std::vector<bool> &pairedMinima,
553 std::vector<bool> &paired1Saddles,
554 const std::vector<SimplexId> &criticalEdges,
555 const std::vector<SimplexId> &critEdgesOrder,
556 const SimplexId *const offsets,
557 const triangulationType &triangulation) const {
558
559 Timer tm{};
560
561 auto saddle1ToMinima = getSaddle1ToMinima(criticalEdges, triangulation);
562
563 Timer tmseq{};
564
565 auto &firstRep{this->firstRepMin_};
566 std::iota(firstRep.begin(), firstRep.end(), 0);
567 std::vector<tripletType> sadMinTriplets{};
568
569 for(size_t i = 0; i < saddle1ToMinima.size(); ++i) {
570 auto &mins = saddle1ToMinima[i];
571 const auto s1 = criticalEdges[i];
572 // remove duplicates
573 TTK_PSORT(this->threadNumber_, mins.begin(), mins.end());
574 const auto last = std::unique(mins.begin(), mins.end());
575 mins.erase(last, mins.end());
576 if(mins.size() != 2) {
577 continue;
578 }
579 sadMinTriplets.emplace_back(tripletType{s1, mins[0], mins[1]});
580 }
581
582 tripletsToPersistencePairs(pairs, pairedMinima, paired1Saddles, firstRep,
583 sadMinTriplets, critEdgesOrder.data(), offsets, 0);
584
585 const auto nMinSadPairs = pairs.size();
586
587 this->printMsg(
588 "Computed " + std::to_string(nMinSadPairs) + " min-saddle pairs", 1.0,
589 tm.getElapsedTime(), this->threadNumber_);
590
591 this->printMsg("min-saddle pairs sequential part", 1.0,
592 tmseq.getElapsedTime(), 1, debug::LineMode::NEW,
593 debug::Priority::VERBOSE);
594}

596template <typename triangulationType>
597void ttk::DiscreteMorseSandwich::getMaxSaddlePairs(
598 std::vector<PersistencePair> &pairs,
599 std::vector<bool> &pairedMaxima,
600 std::vector<bool> &pairedSaddles,
601 const std::vector<SimplexId> &criticalSaddles,
602 const std::vector<SimplexId> &critSaddlesOrder,
603 const std::vector<SimplexId> &critMaxsOrder,
604 const triangulationType &triangulation) const {
605
606 Timer tm{};
607
608 const auto dim = this->dg_.getDimensionality();
609
610 auto saddle2ToMaxima
611 = dim == 3
612 ? getSaddle2ToMaxima(
613 criticalSaddles,
614 [&triangulation](const SimplexId a, const SimplexId i, SimplexId &r) {
615 return triangulation.getTriangleStar(a, i, r);
616 },
617 [&triangulation](const SimplexId a) {
618 return triangulation.getTriangleStarNumber(a);
619 },
620 [&triangulation](const SimplexId a) {
621 return triangulation.isTriangleOnBoundary(a);
622 },
623 triangulation)
624 : getSaddle2ToMaxima(
625 criticalSaddles,
626 [&triangulation](const SimplexId a, const SimplexId i, SimplexId &r) {
627 return triangulation.getEdgeStar(a, i, r);
628 },
629 [&triangulation](const SimplexId a) {
630 return triangulation.getEdgeStarNumber(a);
631 },
632 [&triangulation](const SimplexId a) {
633 return triangulation.isEdgeOnBoundary(a);
634 },
635 triangulation);
636
637 Timer tmseq{};
638
639 auto &firstRep{this->firstRepMax_};
640 std::iota(firstRep.begin(), firstRep.end(), 0);
641 std::vector<tripletType> sadMaxTriplets{};
642
643 for(size_t i = 0; i < saddle2ToMaxima.size(); ++i) {
644 auto &maxs = saddle2ToMaxima[i];
645 // remove duplicates
646 TTK_PSORT(this->threadNumber_, maxs.begin(), maxs.end(),
647 [](const SimplexId a, const SimplexId b) {
648 // positive values (actual maxima) before negative ones
649 // (boundary component id)
650 if(a * b >= 0) {
651 return a < b;
652 } else {
653 return a > b;
654 }
655 });
656 const auto last = std::unique(maxs.begin(), maxs.end());
657 maxs.erase(last, maxs.end());
658
659 // remove "doughnut" configurations: two ascending separatrices
660 // leading to the same maximum/boundary component
661 if(maxs.size() != 2) {
662 continue;
663 }
664
665 const auto s2 = criticalSaddles[i];
666 if(!pairedSaddles[s2]) {
667 sadMaxTriplets.emplace_back(tripletType{s2, maxs[0], maxs[1]});
668 }
669 }
670
671 const auto nMinSadPairs = pairs.size();
672
673 tripletsToPersistencePairs(pairs, pairedMaxima, pairedSaddles, firstRep,
674 sadMaxTriplets, critSaddlesOrder.data(),
675 critMaxsOrder.data(), dim - 1);
676
677 const auto nSadMaxPairs = pairs.size() - nMinSadPairs;
678
679 this->printMsg(
680 "Computed " + std::to_string(nSadMaxPairs) + " saddle-max pairs", 1.0,
681 tm.getElapsedTime(), this->threadNumber_);
682
683 this->printMsg("saddle-max pairs sequential part", 1.0,
684 tmseq.getElapsedTime(), 1, debug::LineMode::NEW,
685 debug::Priority::VERBOSE);
686}

688template <typename triangulationType, typename Container>
689SimplexId ttk::DiscreteMorseSandwich::eliminateBoundariesSandwich(
690 const SimplexId s2,
691 std::vector<bool> &onBoundary,
692 std::vector<Container> &s2Boundaries,
693 const std::vector<SimplexId> &s2Mapping,
694 const std::vector<SimplexId> &partners,
695 const triangulationType &triangulation) const {
696
697 auto &boundaryIds{s2Boundaries[s2Mapping[s2]]};
698 const auto addBoundary = [&boundaryIds, &onBoundary](const SimplexId e) {
699 // add edge e to boundaryIds/onBoundary modulo 2
700 if(!onBoundary[e]) {
701 boundaryIds.emplace(e);
702 onBoundary[e] = true;
703 } else {
704 const auto it = boundaryIds.find(e);
705 boundaryIds.erase(it);
706 onBoundary[e] = false;
707 }
708 };
709
710 if(!boundaryIds.empty()) {
711 // restore previously computed s2 boundary
712 for(const auto e : boundaryIds) {
713 onBoundary[e] = true;
714 }
715 } else {
716 // init cascade with s2 triangle boundary (3 edges)
717 for(SimplexId i = 0; i < 3; ++i) {
718 SimplexId e{};
719 triangulation.getTriangleEdge(s2, i, e);
720 addBoundary(e);
721 }
722 }
723
724 while(!boundaryIds.empty()) {
725 // tau: youngest edge on boundary
726 const auto tau{*boundaryIds.begin()};
727 // use the Discrete Gradient to find a triangle paired to tau
728 auto pTau{this->dg_.getPairedCell(Cell{1, tau}, triangulation)};
729 bool critical{false};
730 if(pTau == -1) {
731 // maybe tau is critical and paired to a critical triangle
732 pTau = partners[tau];
733 critical = true;
734 }
735 if(pTau == -1) {
736 // tau is critical and not paired
737 return tau;
738 }
739 // expand boundary
740 if(critical && s2Mapping[pTau] != -1) { // pTau is a non-paired 2-saddle
741 // merge pTau boundary into s2 boundary
742 for(const auto e : s2Boundaries[s2Mapping[pTau]]) {
743 addBoundary(e);
744 }
745 } else { // pTau is a regular triangle
746 // add pTau triangle boundary (3 edges)
747 for(SimplexId i = 0; i < 3; ++i) {
748 SimplexId e{};
749 triangulation.getTriangleEdge(pTau, i, e);
750 addBoundary(e);
751 }
752 }
753 }
754
755 return -1;
756}

758template <typename triangulationType>
759void ttk::DiscreteMorseSandwich::getSaddleSaddlePairs(
760 std::vector<PersistencePair> &pairs,
761 std::vector<bool> &paired1Saddles,
762 std::vector<bool> &paired2Saddles,
763 const bool exportBoundaries,
764 std::vector<GeneratorType> &boundaries,
765 const std::vector<SimplexId> &critical1Saddles,
766 const std::vector<SimplexId> &critical2Saddles,
767 const std::vector<SimplexId> &crit1SaddlesOrder,
768 const triangulationType &triangulation) const {
769
770 Timer tm2{};
771 const auto nSadExtrPairs = pairs.size();
772
773 // 1- and 2-saddles yet to be paired
774 std::vector<SimplexId> saddles1{}, saddles2{};
775 // filter out already paired 1-saddles (edge id)
776 for(const auto s1 : critical1Saddles) {
777 if(!paired1Saddles[s1]) {
778 saddles1.emplace_back(s1);
779 }
780 }
781 // filter out already paired 2-saddles (triangle id)
782 for(const auto s2 : critical2Saddles) {
783 if(!paired2Saddles[s2]) {
784 saddles2.emplace_back(s2);
785 }
786 }
787
788 Timer tmpar{};
789
790 // sort every triangulation edges by filtration order
791 const auto &edgesFiltrOrder{crit1SaddlesOrder};
792
793 auto &onBoundary{this->onBoundary_};
794 auto &edgeTrianglePartner{this->edgeTrianglePartner_};
795
796 const auto cmpEdges
797 = [&edgesFiltrOrder](const SimplexId a, const SimplexId b) {
798 return edgesFiltrOrder[a] > edgesFiltrOrder[b];
799 };
800 using Container = std::set<SimplexId, decltype(cmpEdges)>;
801 std::vector<Container> s2Boundaries(saddles2.size(), Container(cmpEdges));
802 // unpaired critical triangle id -> index in saddle2 vector
803 auto &s2Mapping{this->s2Mapping_};
804#ifdef TTK_ENABLE_OPENMP
805#pragma omp parallel for num_threads(threadNumber_)
806#endif // TTK_ENABLE_OPENMP
807 for(size_t i = 0; i < saddles2.size(); ++i) {
808 s2Mapping[saddles2[i]] = i;
809 }
810
811 // first parallel pass to pre-determine the boundary of every
812 // unpaired 2-saddle to the first unpaired 1-saddle on its wall
813#ifdef TTK_ENABLE_OPENMP
814#pragma omp parallel for num_threads(threadNumber_) schedule(dynamic) \
815 firstprivate(onBoundary)
816#endif // TTK_ENABLE_OPENMP
817 for(size_t i = 0; i < saddles2.size(); ++i) {
818 // 2-saddles sorted in increasing order
819 const auto s2 = saddles2[i];
820 this->eliminateBoundariesSandwich(s2, onBoundary, s2Boundaries, s2Mapping,
821 edgeTrianglePartner, triangulation);
822 // reset onBoundary to false
823 for(const auto e : s2Boundaries[i]) {
824 onBoundary[e] = false;
825 }
826 }
827
828 this->printMsg("Precomputed 2-saddle boundaries", 1.0, tmpar.getElapsedTime(),
829 this->threadNumber_, debug::LineMode::NEW,
830 debug::Priority::DETAIL);
831
832 Timer tmseq{};
833
834 for(size_t i = 0; i < saddles2.size(); ++i) {
835 // 2-saddles sorted in increasing order
836 const auto s2 = saddles2[i];
837 SimplexId s1{-1};
838
839 if(!s2Boundaries[i].empty()
840 && edgeTrianglePartner[*s2Boundaries[i].begin()] == -1) {
841 // use shortcut if first 1-saddle on wall is non-paired
842 s1 = *s2Boundaries[i].begin();
843 } else {
844 s1 = this->eliminateBoundariesSandwich(s2, onBoundary, s2Boundaries,
845 s2Mapping, edgeTrianglePartner,
846 triangulation);
847 }
848
849 if(s1 != -1) {
850 // we found a pair
851 pairs.emplace_back(s1, s2, 1);
852 paired1Saddles[s1] = true;
853 paired2Saddles[s2] = true;
854 edgeTrianglePartner[s1] = s2;
855 // reset onBoundary to false
856 for(const auto e : s2Boundaries[i]) {
857 onBoundary[e] = false;
858 }
859 }
860 }
861
862 if(exportBoundaries) {
863 boundaries.resize(s2Boundaries.size());
864 for(size_t i = 0; i < boundaries.size(); ++i) {
865 const auto &boundSet{s2Boundaries[i]};
866 if(boundSet.empty()) {
867 continue;
868 }
869 boundaries[i] = {
870 {boundSet.begin(), boundSet.end()},
871 saddles2[i],
872 std::array<SimplexId, 2>{
873 this->dg_.getCellGreaterVertex(Cell{2, saddles2[i]}, triangulation),
874 this->dg_.getCellGreaterVertex(
875 Cell{1, *boundSet.begin()}, triangulation),
876 }};
877 }
878 }
879
880 const auto nSadSadPairs = pairs.size() - nSadExtrPairs;
881
882 this->printMsg(
883 "Computed " + std::to_string(nSadSadPairs) + " saddle-saddle pairs", 1.0,
884 tm2.getElapsedTime(), this->threadNumber_);
885
886 this->printMsg("saddle-saddle pairs sequential part", 1.0,
887 tmseq.getElapsedTime(), 1, debug::LineMode::NEW,
888 debug::Priority::VERBOSE);

891template <typename triangulationType>
892void ttk::DiscreteMorseSandwich::extractCriticalCells(
893 std::array<std::vector<SimplexId>, 4> &criticalCellsByDim,
894 std::array<std::vector<SimplexId>, 4> &critCellsOrder,
895 const SimplexId *const offsets,
896 const triangulationType &triangulation,
897 const bool sortEdges) const {
898
899 Timer tm{};
900 const auto dim = this->dg_.getDimensionality();
901
902 for(int i = 0; i < dim + 1; ++i) {
903
904 // map: store critical cell per dimension per thread
905 std::vector<std::vector<SimplexId>> critCellsPerThread(this->threadNumber_);
906
907 const SimplexId numberOfCells
908 = this->dg_.getNumberOfCells(i, triangulation);
909#ifdef TTK_ENABLE_OPENMP
910#pragma omp parallel for num_threads(this->threadNumber_)
911#endif // TTK_ENABLE_OPENMP
912 for(SimplexId j = 0; j < numberOfCells; ++j) {
913#ifdef TTK_ENABLE_OPENMP
914 const auto tid = omp_get_thread_num();
915#else
916 const auto tid = 0;
917#endif // TTK_ENABLE_OPENMP
918 if(this->dg_.isCellCritical(i, j)) {
919 critCellsPerThread[tid].emplace_back(j);
920 }
921 }
922
923 // reduce: aggregate critical cells per thread
924 criticalCellsByDim[i] = std::move(critCellsPerThread[0]);
925 for(size_t j = 1; j < critCellsPerThread.size(); ++j) {
926 const auto &vec{critCellsPerThread[j]};
927 criticalCellsByDim[i].insert(
928 criticalCellsByDim[i].end(), vec.begin(), vec.end());
929 }
930 }
931
932 this->printMsg("Extracted critical cells", 1.0, tm.getElapsedTime(),
933 this->threadNumber_, debug::LineMode::NEW,
934 debug::Priority::VERBOSE);
935
936 // memory allocations
937 auto &critEdges{this->critEdges_};
938 if(!sortEdges) {
939 critEdges.resize(criticalCellsByDim[1].size());
940 }
941 std::vector<TriangleSimplex> critTriangles(criticalCellsByDim[2].size());
942 std::vector<TetraSimplex> critTetras(criticalCellsByDim[3].size());
943
944#ifdef TTK_ENABLE_OPENMP
945#pragma omp parallel num_threads(threadNumber_)
946#endif // TTK_ENABLE_OPENMP
947 {
948 if(sortEdges) {
949#ifdef TTK_ENABLE_OPENMP
950#pragma omp for nowait
951#endif // TTK_ENABLE_OPENMP
952 for(size_t i = 0; i < critEdges.size(); ++i) {
953 critEdges[i].fillEdge(i, offsets, triangulation);
954 }
955 } else {
956#ifdef TTK_ENABLE_OPENMP
957#pragma omp for nowait
958#endif // TTK_ENABLE_OPENMP
959 for(size_t i = 0; i < critEdges.size(); ++i) {
960 critEdges[i].fillEdge(criticalCellsByDim[1][i], offsets, triangulation);
961 }
962 }
963
964#ifdef TTK_ENABLE_OPENMP
965#pragma omp for nowait
966#endif // TTK_ENABLE_OPENMP
967 for(size_t i = 0; i < critTriangles.size(); ++i) {
968 critTriangles[i].fillTriangle(
969 criticalCellsByDim[2][i], offsets, triangulation);
970 }
971
972#ifdef TTK_ENABLE_OPENMP
973#pragma omp for
974#endif // TTK_ENABLE_OPENMP
975 for(size_t i = 0; i < critTetras.size(); ++i) {
976 critTetras[i].fillTetra(criticalCellsByDim[3][i], offsets, triangulation);
977 }
978 }
979
980 TTK_PSORT(this->threadNumber_, critEdges.begin(), critEdges.end());
981 TTK_PSORT(this->threadNumber_, critTriangles.begin(), critTriangles.end());
982 TTK_PSORT(this->threadNumber_, critTetras.begin(), critTetras.end());
983
984 if(sortEdges) {
985 TTK_PSORT(this->threadNumber_, criticalCellsByDim[1].begin(),
986 criticalCellsByDim[1].end(),
987 [&critCellsOrder](const SimplexId a, const SimplexId b) {
988 return critCellsOrder[1][a] < critCellsOrder[1][b];
989 });
990 } else {
991#ifdef TTK_ENABLE_OPENMP
992#pragma omp parallel for num_threads(threadNumber_)
993#endif // TTK_ENABLE_OPENMP
994 for(size_t i = 0; i < critEdges.size(); ++i) {
995 criticalCellsByDim[1][i] = critEdges[i].id_;
996 }
997 }
998
999#ifdef TTK_ENABLE_OPENMP
1000#pragma omp parallel num_threads(threadNumber_)
1001#endif // TTK_ENABLE_OPENMP
1002 {
1003#ifdef TTK_ENABLE_OPENMP
1004#pragma omp for nowait
1005#endif // TTK_ENABLE_OPENMP
1006 for(size_t i = 0; i < critEdges.size(); ++i) {
1007 critCellsOrder[1][critEdges[i].id_] = i;
1008 }
1009
1010#ifdef TTK_ENABLE_OPENMP
1011#pragma omp for nowait
1012#endif // TTK_ENABLE_OPENMP
1013 for(size_t i = 0; i < critTriangles.size(); ++i) {
1014 criticalCellsByDim[2][i] = critTriangles[i].id_;
1015 critCellsOrder[2][critTriangles[i].id_] = i;
1016 }
1017
1018#ifdef TTK_ENABLE_OPENMP
1019#pragma omp for
1020#endif // TTK_ENABLE_OPENMP
1021 for(size_t i = 0; i < critTetras.size(); ++i) {
1022 criticalCellsByDim[3][i] = critTetras[i].id_;
1023 critCellsOrder[3][critTetras[i].id_] = i;
1024 }
1025 }
1026
1027 this->printMsg("Extracted & sorted critical cells", 1.0, tm.getElapsedTime(),
1028 this->threadNumber_, debug::LineMode::NEW,
1029 debug::Priority::DETAIL);
1030}

1032template <typename triangulationType>
1033int ttk::DiscreteMorseSandwich::computePersistencePairs(
1034 std::vector<PersistencePair> &pairs,
1035 const SimplexId *const offsets,
1036 const triangulationType &triangulation,
1037 const bool ignoreBoundary) {
1038
1039 // allocate memory
1040 this->alloc(triangulation);
1041
1042 Timer tm{};
1043 pairs.clear();
1044 const auto dim = this->dg_.getDimensionality();
1045
1046 // get every critical cell sorted them by dimension
1047 std::array<std::vector<SimplexId>, 4> criticalCellsByDim{};
1048 // holds the critical cells order
1049 auto &critCellsOrder{this->critCellsOrder_};
1050
1051 this->extractCriticalCells(
1052 criticalCellsByDim, critCellsOrder, offsets, triangulation, dim == 3);
1053
1054 // if minima are paired
1055 auto &pairedMinima{this->pairedCritCells_[0]};
1056 // if 1-saddles are paired
1057 auto &paired1Saddles{this->pairedCritCells_[1]};
1058 // if 2-saddles are paired
1059 auto &paired2Saddles{this->pairedCritCells_[dim - 1]};
1060 // if maxima are paired
1061 auto &pairedMaxima{this->pairedCritCells_[dim]};
1062
1063 // connected components (global min/max pair)
1064 size_t nConnComp{};
1065
1066 if(this->ComputeMinSad) {
1067 // minima - saddle pairs
1068 this->getMinSaddlePairs(pairs, pairedMinima, paired1Saddles,
1069 criticalCellsByDim[1], critCellsOrder[1], offsets,
1070 triangulation);
1071
1072 // non-paired minima
1073 for(const auto min : criticalCellsByDim[0]) {
1074 if(!pairedMinima[min]) {
1075 pairs.emplace_back(min, -1, 0);
1076 pairedMinima[min] = true;
1077 nConnComp++;
1078 }
1079 }
1080 } else {
1081 // still extract the global pair
1082 const auto globMin{*std::min_element(
1083 criticalCellsByDim[0].begin(), criticalCellsByDim[0].end(),
1084 [offsets](const SimplexId a, const SimplexId b) {
1085 return offsets[a] < offsets[b];
1086 })};
1087 pairs.emplace_back(globMin, -1, 0);
1088 pairedMinima[globMin] = true;
1089 nConnComp++;
1090 }
1091
1092 if(dim > 1 && this->ComputeSadMax) {
1093 // saddle - maxima pairs
1094 this->getMaxSaddlePairs(
1095 pairs, pairedMaxima, paired2Saddles, criticalCellsByDim[dim - 1],
1096 critCellsOrder[dim - 1], critCellsOrder[dim], triangulation);
1097 }
1098
1099 if(ignoreBoundary) {
1100 // post-process saddle-max pairs: remove the one with the global
1101 // maximum (if it exists) to be (more) compatible with FTM
1102 const auto it
1103 = std::find_if(pairs.begin(), pairs.end(), [&](const PersistencePair &p) {
1104 if(p.type < dim - 1) {
1105 return false;
1106 }
1107 const Cell cmax{dim, p.death};
1108 const auto vmax{this->getCellGreaterVertex(cmax, triangulation)};
1109 return offsets[vmax] == triangulation.getNumberOfVertices() - 1;
1110 });
1111
1112 if(it != pairs.end()) {
1113 // remove saddle-max pair with global maximum
1114 paired2Saddles[it->birth] = false;
1115 pairedMaxima[it->death] = false;
1116 pairs.erase(it);
1117 }
1118 }
1119
1120 // saddle - saddle pairs
1121 if(dim == 3 && !criticalCellsByDim[1].empty()
1122 && !criticalCellsByDim[2].empty() && this->ComputeSadSad) {
1123 std::vector<GeneratorType> tmp{};
1124 this->getSaddleSaddlePairs(
1125 pairs, paired1Saddles, paired2Saddles, false, tmp, criticalCellsByDim[1],
1126 criticalCellsByDim[2], critCellsOrder[1], triangulation);
1127 }
1128
1129 if(std::is_same<triangulationType, ttk::ExplicitTriangulation>::value) {
1130 // create infinite pairs from non-paired 1-saddles, 2-saddles and maxima
1131 size_t nHandles{}, nCavities{}, nNonPairedMax{};
1132 if((dim == 2 && !ignoreBoundary && this->ComputeMinSad
1133 && this->ComputeSadMax)
1134 || (dim == 3 && this->ComputeMinSad && this->ComputeSadSad)) {
1135 // non-paired 1-saddles
1136 for(const auto s1 : criticalCellsByDim[1]) {
1137 if(!paired1Saddles[s1]) {
1138 paired1Saddles[s1] = true;
1139 // topological handles
1140 pairs.emplace_back(s1, -1, 1);
1141 nHandles++;
1142 }
1143 }
1144 }
1145 if(dim == 3 && !ignoreBoundary && this->ComputeSadMax
1146 && this->ComputeSadSad) {
1147 // non-paired 2-saddles
1148 for(const auto s2 : criticalCellsByDim[2]) {
1149 if(!paired2Saddles[s2]) {
1150 paired2Saddles[s2] = true;
1151 // cavities
1152 pairs.emplace_back(s2, -1, 2);
1153 nCavities++;
1154 }
1155 }
1156 }
1157 if(dim == 2 && !ignoreBoundary && this->ComputeSadMax) {
1158 // non-paired maxima
1159 for(const auto max : criticalCellsByDim[dim]) {
1160 if(!pairedMaxima[max]) {
1161 pairs.emplace_back(max, -1, 2);
1162 nNonPairedMax++;
1163 }
1164 }
1165 }
1166
1167 // print Betti numbers
1168 std::vector<std::vector<std::string>> rows{
1169 {" #Connected components", std::to_string(nConnComp)},
1170 {" #Topological handles", std::to_string(nHandles)},
1171 {" #Cavities", std::to_string(nCavities)},
1172 {" #Boundary components", std::to_string((dim == 3 ? nCavities : nHandles)
1173 + nConnComp - nNonPairedMax)},
1174 };
1175
1176 this->printMsg(rows, debug::Priority::DETAIL);
1177 }
1178
1179 this->printMsg(
1180 "Computed " + std::to_string(pairs.size()) + " persistence pairs", 1.0,
1181 tm.getElapsedTime(), this->threadNumber_);
1182
1183 this->displayStats(pairs, criticalCellsByDim, pairedMinima, paired1Saddles,
1184 paired2Saddles, pairedMaxima);
1185
1186 // free memory
1187 this->clear();
1188
1189 return 0;
1190}
