10template <typename triangulationType>
11int ttk::TopologicalCompression::ReadPersistenceGeometry(
12 FILE *fm, const triangulationType &triangulation) {
13
14 std::vector<std::tuple<double, int>> mappingsSortedPerValue;
15
16 double min = 0;
17 double max = 0;
18 int nbConstraints = 0;
19
20 int numberOfBytesRead = 0;
21 if(!ZFPOnly) {
22 numberOfBytesRead
23 += ReadPersistenceIndex(fm, mapping_, mappingsSortedPerValue,
24 criticalConstraints_, min, max, nbConstraints);
25 this->printMsg("Successfully read geomap.");
26 }
27
28 // Prepare array reconstruction.
29 int nx = 1 + dataExtent_[1] - dataExtent_[0];
30 int ny = 1 + dataExtent_[3] - dataExtent_[2];
31 int nz = 1 + dataExtent_[5] - dataExtent_[4];
32 int vertexNumber = nx * ny * nz;
33
34 decompressedData_.resize(vertexNumber);
35 if(ZFPTolerance < 0.0) {
36
37 // 2.a. (2.) Assign values to points thanks to topology indices.
38 for(int i = 0; i < vertexNumber; ++i) {
39 int seg = segmentation_[i];
40 auto end = mapping_.end();
41 auto it = std::lower_bound(
42 mapping_.begin(), mapping_.end(), std::make_tuple(0, seg), cmp);
43 if(it != end) {
44 std::tuple<double, int> tt = *it;
45 double value = std::get<0>(tt);
46 int sseg = std::get<1>(tt);
47 if(seg != sseg) {
48 this->printErr("Decompression mismatch (" + std::to_string(seg) + ", "
49 + std::to_string(sseg) + ")");
50 }
51 decompressedData_[i] = value;
52 } else {
53 this->printErr("Could not find " + std::to_string(seg) + " index.");
54 std::tuple<double, int> tt = *it;
55 double value = std::get<0>(tt);
56 decompressedData_[i] = value;
57 }
58 }
59
60 this->printMsg("Successfully assigned geomap.");
61
62 } else {
63#ifdef TTK_ENABLE_ZFP
64 // 2.b. (2.) Read with ZFP.
65 numberOfBytesRead
66 += CompressWithZFP(fm, true, decompressedData_, nx, ny, nz, ZFPTolerance);
67 this->printMsg("Successfully read with ZFP.");
68#else
69 this->printErr(
70 "Attempted to read with ZFP a ZFP block but ZFP is not installed.");
71 return -5;
72#endif
73 }
74
75 // No SQ.
76 if(SQMethodInt == 0 || SQMethodInt == 3) {
77 for(int i = 0; i < (int)criticalConstraints_.size(); ++i) {
78 std::tuple<int, double, int> t = criticalConstraints_[i];
79 int id = std::get<0>(t);
80 double val = std::get<1>(t);
81 decompressedData_[id] = val;
82 }
83 }
84
85 // double tolerance = Tolerance;
86 if(min == max) {
87 this->printWrn("Empty scalar field range.");
88 } else {
89 // tolerance *= (max - min);
90 }
91
92 if(SQMethodInt == 1 || SQMethodInt == 2)
93 return 0;
94
95 if(ZFPOnly)
96 return 0;
97
98 // 2.b. (3.) Crop whatever doesn't fit in topological intervals.
99 CropIntervals(mapping_, mappingsSortedPerValue, min, max, vertexNumber,
100 decompressedData_.data(), segmentation_);
101 this->printMsg("Successfully cropped bad intervals.");
102
103 // 2.b. (4.) Apply topological simplification with min/max constraints
104 PerformSimplification<double>(criticalConstraints_, nbConstraints,
105 vertexNumber, decompressedData_.data(),
106 triangulation);
107 this->printMsg("Successfully performed simplification.");
108
109 this->rawFileLength += numberOfBytesRead;
110
111 return 0;
112}

114template <typename dataType, typename triangulationType>
115int ttk::TopologicalCompression::PerformSimplification(
116 const std::vector<std::tuple<int, double, int>> &constraints,
117 int nbConstraints,
118 int vertexNumber,
119 double *array,
120 const triangulationType &triangulation) {
121
122 std::vector<int> inputOffsets(vertexNumber);
123 std::vector<SimplexId> critConstraints(nbConstraints);
124 std::vector<double> inArray(vertexNumber);
125 // std::vector<int> *oo = new std::vector<int>(vertexNumber);
126 decompressedOffsets_.resize(vertexNumber); // oo->data();
127 int status = 0;
128
129 // Offsets
130 for(int i = 0; i < vertexNumber; ++i)
131 inputOffsets[i] = i;
132
133 // Preprocess simplification.
134 // std::vector<int>* authorizedSaddles = new std::vector<int>();
135 for(int i = 0; i < nbConstraints; ++i) {
136 std::tuple<int, double, int> t = constraints[i];
137 int id = std::get<0>(t);
138 double val = std::get<1>(t);
139 int type = std::get<2>(t);
140
141 // if (type == 0)
142 // authorizedSaddles->push_back(id);
143
144 array[id] = val;
145
146 // Smoothe neighborhood (along with offsets).
147 SimplexId neighborNumber = triangulation.getVertexNeighborNumber(id);
148 for(SimplexId j = 0; j < neighborNumber; ++j) {
149 SimplexId neighbor{-1};
150 triangulation.getVertexNeighbor(id, j, neighbor);
151
152 if(type == 1) { // Local_maximum.
153 if(array[neighbor] > val)
154 array[neighbor] = val;
155 if(array[neighbor] == val
156 && inputOffsets[neighbor] > inputOffsets[id]) {
157 std::swap(inputOffsets[id], inputOffsets[neighbor]);
158 }
159 } else if(type == -1) { // Local_minimum.
160 if(array[neighbor] < val)
161 array[neighbor] = val;
162 if(array[neighbor] == val
163 && inputOffsets[neighbor] < inputOffsets[id]) {
164 std::swap(inputOffsets[id], inputOffsets[neighbor]);
165 }
166 } else if(type == 0) { // Saddle
167 }
168 }
169
170 critConstraints[i] = id;
171 }
172
173 for(int i = 0; i < vertexNumber; ++i) {
174 inArray[i] = array[i];
175 }
176 for(int i = 0; i < vertexNumber; ++i)
177 decompressedOffsets_[i] = 0;
178
179 std::vector<SimplexId> vertsOrder(vertexNumber);
180 sortVertices(vertexNumber, array, inputOffsets.data(), vertsOrder.data(),
181 this->threadNumber_);
182
183 status = topologicalSimplification.execute<double>(
184 inArray.data(), array, critConstraints.data(), vertsOrder.data(),
185 decompressedOffsets_.data(), nbConstraints, triangulation);
186
187 return status;
188}

190template <typename dataType>
191void ttk::TopologicalCompression::CropIntervals(
192 std::vector<std::tuple<dataType, int>> &mappings,
193 std::vector<std::tuple<dataType, int>> &mappingsSortedPerValue,
194 double min,
195 double max,
196 int vertexNumber,
197 double *array,
198 std::vector<int> &segmentation) const {
199
200 int numberOfMisses = 0;
201 for(int i = 0; i < vertexNumber; ++i) {
202 int seg = segmentation[i];
203 auto end = mappings.end();
204 auto it = lower_bound(
205 mappings.begin(), mappings.end(), std::make_tuple(0, seg), cmp);
206 if(it != end) {
207 std::tuple<dataType, int> tt = *it;
208 double value = std::get<0>(tt);
209 int sseg = std::get<1>(tt);
210 if(seg != sseg) {
211 this->printErr("Decompression mismatch.");
212 }
213
214 auto it2 = lower_bound(mappingsSortedPerValue.begin(),
215 mappingsSortedPerValue.end(),
216 std::make_tuple(value, 0), cmp2);
217
218 if(it2 != mappingsSortedPerValue.end()
219 && it2 != mappingsSortedPerValue.begin()
220 && (it2 + 1) != mappingsSortedPerValue.end()) {
221 dataType vv0 = std::get<0>(*(it2 - 1));
222 // double vv1 = std::get<0>(*(it2));
223 dataType vv2 = std::get<0>(*(++it2));
224 // double m0 = (vv0 + vv1) / 2;
225 // double m1 = (vv1 + vv2) / 2;
226
227 if(array[i] < vv0) {
228 numberOfMisses++;
229 array[i] = vv0;
230 } else if(array[i] > vv2) {
231 numberOfMisses++;
232 array[i] = vv2;
233 }
234 } else {
235 if((it2 == mappingsSortedPerValue.end()
236 || (it2 + 1) == mappingsSortedPerValue.end())
237 && array[i] > max) {
238 numberOfMisses++;
239 array[i] = max;
240 } else if((it2 == mappingsSortedPerValue.begin()
241 || (it2 - 1) == mappingsSortedPerValue.begin())
242 && array[i] < min) {
243 numberOfMisses++;
244 array[i] = min;
245 }
246 }
247 } else {
248 this->printErr("Error looking for topo index.");
249 }
250 }
251
252 if(numberOfMisses > 0) {
253 this->printWrn("Missed " + std::to_string(numberOfMisses) + " values.");
254 }

257template <typename dataType, typename triangulationType>
258int ttk::TopologicalCompression::computePersistencePairs(
259 std::vector<std::tuple<SimplexId, SimplexId, dataType>> &JTPairs,
260 std::vector<std::tuple<SimplexId, SimplexId, dataType>> &STPairs,
261 const dataType *const inputScalars_,
262 const SimplexId *const inputOffsets,
263 const triangulationType &triangulation) {
264
265 // Compute offsets
266 const SimplexId numberOfVertices = triangulation.getNumberOfVertices();
267 std::vector<SimplexId> voffsets((unsigned long)numberOfVertices);
268 std::copy(inputOffsets, inputOffsets + numberOfVertices, voffsets.begin());
269
270 // Get contour tree
271 ftmTreePP.setVertexScalars(inputScalars_);
272 ftmTreePP.setTreeType(ftm::TreeType::Join_Split);
273 ftmTreePP.setVertexSoSoffsets(voffsets.data());
274 ftmTreePP.setThreadNumber(threadNumber_);
275 ftmTreePP.build<dataType>(&triangulation);
276 ftmTreePP.setSegmentation(false);
277 ftmTreePP.computePersistencePairs<dataType>(JTPairs, true);
278 ftmTreePP.computePersistencePairs<dataType>(STPairs, false);
279
280 return 0;
281}

283template <typename dataType, typename triangulationType>
284int ttk::TopologicalCompression::compressForPersistenceDiagram(
285 int vertexNumber,
286 const dataType *const inputData,
287 const SimplexId *const inputOffsets,
288 dataType *outputData,
289 const double &tol,
290 const triangulationType &triangulation) {
291
292 ttk::Timer t;
293 ttk::Timer t1;
294
295 // 1. Compute persistence pairs.
296 std::vector<std::tuple<dataType, int>> topoIndices;
297
298 // Compute global min & max
299 dataType iter;
300 dataType maxValue = inputData[0];
301 int maxIndex = 0;
302 for(int i = 1; i < vertexNumber; ++i) {
303 iter = inputData[i];
304 if(iter > maxValue) {
305 maxIndex = i;
306 maxValue = iter;
307 }
308 }
309
310 dataType minValue = inputData[0];
311 int minIndex = 0;
312 for(int i = 1; i < vertexNumber; ++i) {
313 iter = inputData[i];
314 if(iter < minValue) {
315 minIndex = i;
316 minValue = iter;
317 }
318 }
319
320 topoIndices.push_back(std::make_tuple(maxValue, maxIndex));
321 topoIndices.push_back(std::make_tuple(minValue, minIndex));
322 double tolerance = 0.01 * tol * (maxValue - minValue);
323 double maxError = 0.01 * MaximumError * (maxValue - minValue);
324
325 this->printMsg(
326 "Computed min/max", 1.0, t.getElapsedTime(), this->threadNumber_);
327 t.reStart();
328
329 bool sqDomain = false;
330 bool sqRange = false;
331
332 const char *sq = SQMethod.c_str();
333 int nbCrit = 0;
334 std::vector<SimplexId> simplifiedConstraints;
335 if(strcmp(sq, "") == 0 && !ZFPOnly) {
336 // No SQ: perform topological control
337
338 std::vector<std::tuple<SimplexId, SimplexId, dataType>> JTPairs;
339 std::vector<std::tuple<SimplexId, SimplexId, dataType>> STPairs;
340 computePersistencePairs<dataType>(
341 JTPairs, STPairs, inputData, inputOffsets, triangulation);
342
343 this->printMsg("Computed persistence pairs", 1.0, t.getElapsedTime(),
344 this->threadNumber_);
345 t.reStart();
346
347 int nbJ = JTPairs.size();
348 int nbS = STPairs.size();
349 std::vector<int> critConstraints(2 * nbJ + 2 * nbS);
350
351 // auto* authorizedSaddles = new std::vector<int>();
352
353 dataType maxEpsilon = 0;
354 dataType epsilonSum2 = 0;
355 dataType epsilonSum1 = 0;
356 dataType persistentSum2 = 0;
357 dataType persistentSum1 = 0;
358
359 // Join
360 for(int i = 0; i < nbJ; ++i) {
361 SimplexId cp1 = std::get<0>(JTPairs[i]);
362 SimplexId cp2 = std::get<1>(JTPairs[i]);
363 dataType idt1 = inputData[cp1];
364 dataType idt2 = inputData[cp2];
365 dataType p1 = std::max(idt2, idt1) - std::min(idt2, idt1);
366 if(p1 > tolerance) {
367 persistentSum2 += (p1 * p1);
368 persistentSum1 += abs<dataType>(p1);
369 int type1 = topologicalSimplification.getCriticalType(
370 cp1, inputOffsets, triangulation);
371 int type2 = topologicalSimplification.getCriticalType(
372 cp2, inputOffsets, triangulation);
373 if(type1 == 0) {
374 // authorizedSaddles->push_back(cp1);
375 topoIndices.push_back(std::make_tuple(idt1, cp1));
376 }
377 if(type2 == 0) {
378 // authorizedSaddles->push_back(cp2);
379 topoIndices.push_back(std::make_tuple(idt2, cp2));
380 }
381
382 nbCrit += 2;
383 critConstraints[2 * i] = cp1;
384 critConstraints[2 * i + 1] = cp2;
385 } else {
386 if(maxEpsilon < abs<dataType>(p1))
387 maxEpsilon = abs<dataType>(p1);
388 epsilonSum2 += (p1 * p1);
389 epsilonSum1 += abs<dataType>(p1);
390 critConstraints[2 * i] = -1;
391 critConstraints[2 * i + 1] = -1;
392 }
393 }
394
395 this->printMsg("Post-processed join pairs", 1.0, t.getElapsedTime(),
396 this->threadNumber_);
397 t.reStart();
398
399 // Split
400 for(int i = nbJ; i < nbJ + nbS; ++i) {
401 int si = i - nbJ;
402 SimplexId cp1 = std::get<0>(STPairs[si]);
403 SimplexId cp2 = std::get<1>(STPairs[si]);
404 dataType idt1 = inputData[cp1];
405 dataType idt2 = inputData[cp2];
406 dataType p1 = std::max(idt2, idt1) - std::min(idt2, idt1);
407 if(p1 > tolerance) {
408 persistentSum2 += (p1 * p1);
409 persistentSum1 += abs<dataType>(p1);
410 // Saddle selection.
411 int type1 = topologicalSimplification.getCriticalType(
412 cp1, inputOffsets, triangulation);
413 int type2 = topologicalSimplification.getCriticalType(
414 cp2, inputOffsets, triangulation);
415 if(type1 == 0) {
416 // authorizedSaddles->push_back(cp1);
417 topoIndices.push_back(std::make_tuple(idt1, cp1));
418 }
419 if(type2 == 0) {
420 // authorizedSaddles->push_back(cp2);
421 topoIndices.push_back(std::make_tuple(idt2, cp2));
422 }
423
424 nbCrit += 2;
425 critConstraints[2 * i] = cp1;
426 critConstraints[2 * i + 1] = cp2;
427 } else {
428 if(maxEpsilon < p1)
429 maxEpsilon = abs<dataType>(p1);
430 epsilonSum2 += (p1 * p1);
431 epsilonSum1 += abs<dataType>(p1);
432 critConstraints[2 * i] = -1;
433 critConstraints[2 * i + 1] = -1;
434 }
435 }
436
437 this->printMsg("Post-processed split pairs", 1.0, t.getElapsedTime(),
438 this->threadNumber_);
439 t.reStart();
440
441 simplifiedConstraints.resize(nbCrit);
442 {
443 int j = 0;
444 for(int i = 0; i < 2 * nbJ + 2 * nbS; ++i) {
445 int c = critConstraints[i];
446 if(c != -1)
447 simplifiedConstraints[j++] = c;
448 }
449 }
450
451 this->printMsg("Got pairs", 1.0, t.getElapsedTime(), this->threadNumber_);
452 t.reStart();
453
454 // 2. Perform topological simplification with constraints.
455 if(UseTopologicalSimplification) {
456 compressedOffsets_.resize(vertexNumber);
457 int status = 0;
458 status = topologicalSimplification.execute<dataType>(
459 inputData, outputData, simplifiedConstraints.data(), inputOffsets,
460 compressedOffsets_.data(), nbCrit, triangulation);
461 if(status != 0) {
462 return status;
463 }
464 }
465
466 this->printMsg(
467 "Performed simplification", 1.0, t.getElapsedTime(), this->threadNumber_);
468 t.reStart();
469
470 } else if(strcmp(sq, "r") == 0 || strcmp(sq, "R") == 0) {
471 // Range-based SQ: simple range quantization (automatic)
472 sqRange = true;
473 } else if(strcmp(sq, "d") == 0 || strcmp(sq, "D") == 0) {
474 // Domain-based SQ: range quantization + later domain control
475 sqDomain = true;
476 } else if(ZFPOnly) {
477 return 0;
478 } else {
479 this->printErr("Unrecognized SQ option (" + SQMethod + ").");
480 return -3;
481 }
482
483 // 3. Subdivision and attribution of a topological index
484
485 auto cmp = [](const std::tuple<dataType, int> &a,
486 const std::tuple<dataType, int> &b) {
487 return std::get<0>(a) < std::get<0>(b);
488 };
489 std::sort(topoIndices.begin(), topoIndices.end(), cmp);
490 std::vector<std::tuple<dataType, int>> segments;
491 // ith rank = ith segment (from bottom)
492 // 0: value
493 // 1: critical point index or -1 if !critical
494 auto l = (int)topoIndices.size();
495 if(l < 1) {
496 this->printMsg("Trivial subdivision performed.");
497 segments.push_back(topoIndices[l - 1]);
498 } else {
499 for(int i = 0; i < l; ++i) {
500 dataType v0 = std::get<0>(topoIndices[i]);
501 if(i == l - 1) {
502 segments.push_back(topoIndices[i]);
503 break;
504 }
505
506 dataType v1 = std::get<0>(topoIndices[i + 1]);
507 int i1 = std::get<1>(topoIndices[i + 1]);
508 auto diff = (double)(v1 - v0);
509
510 if(diff == 0)
511 continue;
512 if(!Subdivide || (diff < (dataType)maxError)) {
513 segments.push_back(std::make_tuple(v1, i1));
514 } else {
515 // Subdivide.
516 double nSegments = std::ceil(diff / maxError);
517 for(int j = 0, nbs = (int)nSegments; j < nbs; ++j) {
518 dataType sample = v0 + j * maxError;
519 int int1 = i1;
520 segments.push_back(std::make_tuple(sample, int1));
521 }
522 }
523 }
524 }
525
526 this->printMsg("Subdivision/topological index attribution", 1.0,
527 t.getElapsedTime(), this->threadNumber_);
528 t.reStart();
529
530 // 4. Affect segment value to all points.
531
532 // (pad buffer to avoid an out-of-bounds access/buffer overflow in
533 // WriteCompactSegmentation while loop)
534 segmentation_.resize(vertexNumber + 32);
535
536 std::sort(segments.begin(), segments.end(), cmp);
537 for(int i = 0; i < vertexNumber; ++i) {
538 dataType scalar = inputData[i];
539
540 auto begin = segments.begin();
541 auto end = segments.end();
542 auto it = std::lower_bound(
543 segments.begin(), segments.end(), std::make_tuple(scalar, -1), cmp);
544
545 if(it != end) {
546 std::tuple<dataType, int> tt = *it;
547 int j = it - begin;
548 int last = (int)segments.size() - 1;
549 if(j < last) {
550 dataType dtv = std::get<0>(tt);
551 if(j > 0) {
552 dtv = 0.5 * (dtv + std::get<0>(*(it - 1)));
553 }
554
555 outputData[i] = dtv;
556 int seg = j;
557 segmentation_[i] = seg;
558 } else {
559 segmentation_[i] = last;
560 outputData[i] = std::get<0>(tt); // maxValue;
561 }
562 }
563 }
564
565 this->printMsg(
566 "Assigned values", 1.0, t.getElapsedTime(), this->threadNumber_);
567 t.reStart();
568
569 // 4.1. Simplify mapping
570 auto segmentsSize = (int)segments.size();
571 std::vector<bool> affectedSegments;
572 affectedSegments.resize(segmentsSize);
573 std::vector<int> oob;
574 for(int i = 0; i < vertexNumber; ++i) {
575 int seg = segmentation_[i];
576 if(seg >= segmentsSize
577 && std::find(oob.begin(), oob.end(), seg) == oob.end())
578 oob.push_back(seg);
579 else if(seg >= 0)
580 affectedSegments[seg] = true;
581 else {
582 this->printErr("Negative segment encoutered (" + std::to_string(i) + ")");
583 }
584 }
585 std::vector<int> empty;
586 for(int i = 0; i < segmentsSize; ++i) {
587 if(!affectedSegments[i])
588 empty.push_back(i);
589 }
590
591 std::sort(oob.begin(), oob.end());
592 std::sort(empty.begin(), empty.end());
593 int indexLast = -1;
594 if(oob.size() > empty.size()) {
595 this->printWrn("oob size > empty size: " + std::to_string(oob.size()) + ", "
596 + std::to_string(empty.size()));
597 } else {
598 // Replace
599 for(int i = 0; i < vertexNumber; ++i) {
600 int seg = segmentation_[i];
601 if(seg >= segmentsSize) {
602 auto begin = oob.begin();
603 auto end = oob.end();
604 auto it = std::lower_bound(begin, end, seg);
605 if(it != end) {
606 int j = (int)(it - begin);
607 segmentation_[i] = empty[j];
608 affectedSegments[j] = true;
609 }
610 } else if(!affectedSegments[seg]) {
611 this->printErr("Something impossible happened");
612 }
613 }
614
615 if(empty.size() > oob.size()) {
616 std::vector<int> map2(segmentsSize, -1);
617 for(int i = 0; i < segmentsSize; ++i) {
618 if(affectedSegments[i])
619 continue;
620 for(int j = (segmentsSize - 1); j > i; --j) {
621 if(!affectedSegments[j] || map2[j] > 0)
622 continue;
623 map2[j] = i;
624 affectedSegments[j] = false;
625 affectedSegments[i] = true;
626 break;
627 }
628 }
629
630 bool doneAffecting = false;
631 for(int i = 0; i < segmentsSize; ++i) {
632 if(!affectedSegments[i]) {
633 doneAffecting = true;
634 } else {
635 if(doneAffecting) {
636 this->printWrn("Hole detected at " + std::to_string(i)
637 + "th segment.");
638 } else {
639 indexLast = i;
640 }
641 }
642 }
643
644 for(int i = 0; i < vertexNumber; ++i) {
645 int seg = segmentation_[i];
646 if(map2[seg] > 0)
647 segmentation_[i] = map2[seg];
648 }
649 }
650 }
651
652 this->printMsg(
653 "Simplified mapping", 1.0, t.getElapsedTime(), this->threadNumber_);
654 t.reStart();
655
656 // 5. Expose mapping.
657 std::vector<bool> already(vertexNumber);
658 for(int i = 0; i < vertexNumber; ++i)
659 already[i] = false; // init
660
661 for(int i = 0; i < vertexNumber; ++i) {
662 int vert = segmentation_[i];
663 if(!already[vert]) {
664 already[vert] = true;
665 dataType dttt = outputData[i];
666 mapping_.emplace_back((double)dttt, vert);
667 }
668 }
669
670 this->printMsg(
671 "Exposed mapping", 1.0, t.getElapsedTime(), this->threadNumber_);
672 t.reStart();
673
674 // 6. SQ-D correction step.
675 // Indices stay compact.
676 if(sqDomain && !sqRange) {
677 std::vector<bool> markedVertices(vertexNumber); // false by default
678 std::vector<bool> markedSegments(segmentsSize); // false by default
679 int newIndex = segmentsSize;
680
681 for(int i = 0; i < vertexNumber; ++i) {
682 // Skip processed vertices.
683 if(markedVertices[i])
684 continue;
685
686 int seg = segmentation_[i];
687 bool newSegment = markedSegments[seg];
688 dataType minNewSegment = inputData[i];
689 dataType maxNewSegment = minNewSegment;
690
691 if(!newSegment) {
692 markedSegments[seg] = true;
693 }
694
695 // Neighborhood search with seed = i
696 std::stack<int> s;
697 s.push(i);
698
699 while(!s.empty()) {
700 // Get next element.
701 int vertex = s.top();
702 s.pop();
703
704 // Mark vertex as processed.
705 markedVertices[vertex] = true;
706
707 // New region was detected.
708 if(newSegment) {
709 // Affect new segmentation id to all vertices in region.
710 segmentation_[vertex] = newIndex;
711 // Progressively compute min & max on local region.
712 dataType currentValue = inputData[vertex];
713 if(currentValue > maxNewSegment)
714 maxNewSegment = currentValue;
715 if(currentValue < minNewSegment)
716 minNewSegment = currentValue;
717 }
718
719 // Get neighbors.
720 SimplexId neighborNumber
721 = triangulation.getVertexNeighborNumber(vertex);
722 for(SimplexId j = 0; j < neighborNumber; ++j) {
723 SimplexId neighbor{-1};
724 triangulation.getVertexNeighbor(vertex, j, neighbor);
725
726 // Add current neighbor to processing stack.
727 if(!markedVertices[neighbor] && segmentation_[neighbor] == seg) {
728 s.push(neighbor);
729 }
730 }
731 }
732
733 // Affect a value to the new segmentation index.
734 if(newSegment) {
735 dataType result = (maxNewSegment + minNewSegment) / 2;
736 segments.push_back(std::make_tuple(result, newIndex));
737 mapping_.emplace_back((double)result, newIndex);
738 newIndex++; // Set index for next potential segment.
739 }
740 }
741
742 this->printMsg(
743 "SQ-D correction step", 1.0, t.getElapsedTime(), this->threadNumber_);
744 t.reStart();
745 }
746
747 // 7. [ZFP]: max constraints, min constraints
748 if(!sqDomain && !sqRange && !ZFPOnly) {
749 for(int i = 0; i < nbCrit; ++i) {
750 SimplexId id = simplifiedConstraints[i];
751 dataType val = inputData[id];
752 int type = topologicalSimplification.getCriticalType(
753 id, inputOffsets, triangulation);
754 if(type == -1 // Local_minimum
755 || type == 1 // Local_maximum
756 || type == 0) {
757 criticalConstraints_.emplace_back(id, (double)val, type);
758 } else { // 0 -> Saddle
759 }
760 }
761
762 this->printMsg(
763 "Exposed constraints", 1.0, t.getElapsedTime(), this->threadNumber_);
764 t.reStart();
765 }
766
767 {
768 if(indexLast > -1 && indexLast + 1 != (int)mapping_.size()) {
769 this->printWrn("Possible affectation mismatch ("
770 + std::to_string(indexLast + 1) + ", "
771 + std::to_string(mapping_.size()) + ")");
772 }
773
774 int nSegments = indexLast > -1 ? indexLast + 1 : (int)segments.size() - 1;
775 this->NbSegments = nSegments;
776 this->NbVertices = vertexNumber;
777
778 this->printMsg("Assigned " + std::to_string(nSegments) + " segment(s).",
779 1.0, t1.getElapsedTime(), this->threadNumber_);
780 }
781
782 return 0;
783}
