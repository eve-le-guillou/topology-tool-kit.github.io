116template <typename dataType, class triangulationType>
117int ttk::IntegralLines::execute(const triangulationType *triangulation) const {
118 const auto offsets = inputOffsets_;
119 SimplexId *identifiers = vertexIdentifierScalarField_;
120 dataType *scalars = static_cast<dataType *>(inputScalarField_);
121 std::vector<std::vector<SimplexId>> *trajectories = outputTrajectories_;
122
123 Timer t;
124
125 // get the seeds
126 std::unordered_set<SimplexId> isSeed;
127 for(SimplexId k = 0; k < seedNumber_; ++k)
128 isSeed.insert(identifiers[k]);
129 std::vector<SimplexId> seeds(isSeed.begin(), isSeed.end());
130 isSeed.clear();
131
132 trajectories->resize(seeds.size());
133 for(SimplexId i = 0; i < (SimplexId)seeds.size(); ++i) {
134 SimplexId v{seeds[i]};
135 (*trajectories)[i].push_back(v);
136
137 bool isMax{};
138 while(!isMax) {
139 SimplexId vnext{-1};
140 float fnext = std::numeric_limits<float>::min();
141 SimplexId neighborNumber = triangulation->getVertexNeighborNumber(v);
142 for(SimplexId k = 0; k < neighborNumber; ++k) {
143 SimplexId n{-1};
144 triangulation->getVertexNeighbor(v, k, n);
145
146 if((direction_ == static_cast<int>(Direction::Forward))
147 xor (offsets[n] < offsets[v])) {
148 const float f = getGradient<dataType, triangulationType>(
149 triangulation, v, n, scalars);
150 if(f > fnext) {
151 vnext = n;
152 fnext = f;
153 }
154 }
155 }
156
157 if(vnext == -1)
158 isMax = true;
159 else {
160 v = vnext;
161 (*trajectories)[i].push_back(v);
162 }
163 }
164 }
165
166 {
167 std::stringstream msg;
168 msg << "Processed " << vertexNumber_ << " points";
169 this->printMsg(msg.str(), 1, t.getElapsedTime(), 1);
170 }
171
172 return 0;
173}
