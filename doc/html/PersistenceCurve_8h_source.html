98template <typename scalarType>
99int ttk::PersistenceCurve::computePersistencePlot(
100 const std::vector<std::tuple<SimplexId, SimplexId, scalarType>> &pairs,
101 std::vector<std::pair<scalarType, SimplexId>> &plot) const {
102
103 SimplexId nbElmnt = pairs.size();
104 plot.resize(nbElmnt);
105
106 // build curve
107 const scalarType epsilon
108 = static_cast<scalarType>(Geometry::powIntTen(-REAL_SIGNIFICANT_DIGITS));
109 for(SimplexId i = 0; i < nbElmnt; ++i) {
110 plot[i].first = std::max(std::get<2>(pairs[i]), epsilon);
111 plot[i].second = pairs.size() - i;
112 }
113
114 return 0;
115}

118int ttk::PersistenceCurve::execute(const scalarType *inputScalars,
119 const size_t scalarsMTime,
120 const SimplexId *inputOffsets,
121 const triangulationType *triangulation) {
122
123 printMsg(ttk::debug::Separator::L1);
124
125 Timer timer;
126
127 using plotType = std::vector<std::pair<scalarType, SimplexId>>;
128
129 auto JTPlot = static_cast<plotType *>(JTPlot_);
130 auto STPlot = static_cast<plotType *>(STPlot_);
131 auto CTPlot = static_cast<plotType *>(CTPlot_);
132 auto MSCPlot = static_cast<plotType *>(MSCPlot_);
133
134 contourTree_.setVertexScalars(inputScalars);
135 contourTree_.setTreeType(ftm::TreeType::Join_Split);
136 contourTree_.setVertexSoSoffsets(inputOffsets);
137 contourTree_.setSegmentation(false);
138 contourTree_.setThreadNumber(threadNumber_);
139 contourTree_.build<scalarType>(triangulation);
140
141 // get persistence pairs
142 std::vector<std::tuple<SimplexId, SimplexId, scalarType>> JTPairs;
143 std::vector<std::tuple<SimplexId, SimplexId, scalarType>> STPairs;
144 contourTree_.computePersistencePairs<scalarType>(JTPairs, true);
145 contourTree_.computePersistencePairs<scalarType>(STPairs, false);
146
147 // merge pairs
148 std::vector<std::tuple<SimplexId, SimplexId, scalarType>> CTPairs(
149 JTPairs.size() + STPairs.size());
150 std::copy(JTPairs.begin(), JTPairs.end(), CTPairs.begin());
151 std::copy(STPairs.begin(), STPairs.end(), CTPairs.begin() + JTPairs.size());
152 {
153 auto cmp = [](const std::tuple<SimplexId, SimplexId, scalarType> &a,
154 const std::tuple<SimplexId, SimplexId, scalarType> &b) {
155 return std::get<2>(a) < std::get<2>(b);
156 };
157 std::sort(CTPairs.begin(), CTPairs.end(), cmp);
158 }
159
160 // get the saddle-saddle pairs
161 const int dimensionality = triangulation->getDimensionality();
162 if(dimensionality == 3 and ComputeSaddleConnectors and MSCPlot_ != nullptr) {
163 std::vector<std::tuple<SimplexId, SimplexId, scalarType>>
164 pl_saddleSaddlePairs;
165 dcg_.setInputScalarField(inputScalars, scalarsMTime);
166 dcg_.setInputOffsets(inputOffsets);
167 dcg_.computeSaddleSaddlePersistencePairs<scalarType>(
168 pl_saddleSaddlePairs, *triangulation);
169
170 // sort the saddle-saddle pairs by persistence value and compute curve
171 {
172 auto cmp = [](const std::tuple<SimplexId, SimplexId, scalarType> &a,
173 const std::tuple<SimplexId, SimplexId, scalarType> &b) {
174 return std::get<2>(a) < std::get<2>(b);
175 };
176 std::sort(pl_saddleSaddlePairs.begin(), pl_saddleSaddlePairs.end(), cmp);
177 }
178
179 computePersistencePlot<scalarType>(pl_saddleSaddlePairs, *MSCPlot);
180 }
181
182 // get persistence curves
183#ifdef TTK_ENABLE_OPENMP
184#pragma omp parallel sections
185#endif
186 {
187#ifdef TTK_ENABLE_OPENMP
188#pragma omp section
189#endif
190 if(JTPlot_ != nullptr) {
191 computePersistencePlot<scalarType>(JTPairs, *JTPlot);
192 }
193#ifdef TTK_ENABLE_OPENMP
194#pragma omp section
195#endif
196 if(STPlot_ != nullptr) {
197 computePersistencePlot<scalarType>(STPairs, *STPlot);
198 }
199#ifdef TTK_ENABLE_OPENMP
200#pragma omp section
201#endif
202 if(CTPlot_ != nullptr) {
203 computePersistencePlot<scalarType>(CTPairs, *CTPlot);
204 }
205 }
206
207 printMsg(
208 "Base execution completed", 1, timer.getElapsedTime(), threadNumber_);
209 printMsg(ttk::debug::Separator::L1);
210
211 return 0;
