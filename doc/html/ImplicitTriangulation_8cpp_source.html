229int ImplicitTriangulation::checkAcceleration() {
230 isAccelerated_ = false;
231
232 unsigned long long int msb[3];
233 if(dimensionality_ == 3) {
234 bool allDimensionsArePowerOfTwo = true;
235 for(int k = 0; k < 3; ++k)
236 if(!isPowerOfTwo(dimensions_[k], msb[k]))
237 allDimensionsArePowerOfTwo = false;
238
239 if(allDimensionsArePowerOfTwo) {
240 mod_[0] = dimensions_[0] - 1;
241 mod_[1] = dimensions_[0] * dimensions_[1] - 1;
242 div_[0] = msb[0];
243 div_[1] = msb[0] + msb[1];
244 isAccelerated_ = true;
245 }
246 } else if(dimensionality_ == 2) {
247 bool isDi = isPowerOfTwo(dimensions_[Di_], msb[Di_]);
248 bool isDj = isPowerOfTwo(dimensions_[Dj_], msb[Dj_]);
249 bool allDimensionsArePowerOfTwo = (isDi and isDj);
250
251 if(allDimensionsArePowerOfTwo) {
252 mod_[0] = dimensions_[Di_] - 1;
253 div_[0] = msb[Di_];
254 isAccelerated_ = true;
255 }
256 }
257
258 if(isAccelerated_) {
259 printMsg("Accelerated getVertex*() requests.", debug::Priority::INFO);
260 }
261
262 return 0;

1250template <typename Derived>
1251int ImplicitTriangulationCRTP<Derived>::getEdgeVertexInternal(
1252 const SimplexId &edgeId,
1253 const int &localVertexId,
1254 SimplexId &vertexId) const {
1255#ifndef TTK_ENABLE_KAMIKAZE
1256 if(edgeId < 0 or edgeId >= edgeNumber_)
1257 return -1;
1258 if(localVertexId < 0 or localVertexId >= 2)
1259 return -2;
1260#endif
1261
1262 const auto &p = this->underlying().getEdgeCoords(edgeId);
1263
1264 const auto helper3d = [&](const SimplexId a, const SimplexId b) -> SimplexId {
1265 if(isAccelerated_) {
1266 const auto tmp = p[0] + (p[1] << div_[0]) + (p[2] << div_[1]);
1267 return (localVertexId == 0) ? tmp + a : tmp + b;
1268 } else {
1269 const auto tmp = p[0] + (p[1] * vshift_[0]) + (p[2] * vshift_[1]);
1270 return (localVertexId == 0) ? tmp + a : tmp + b;
1271 }
1272 };
1273
1274 const auto helper2d = [&](const SimplexId a, const SimplexId b) -> SimplexId {
1275 if(isAccelerated_) {
1276 const auto tmp = p[0] + (p[1] << div_[0]);
1277 return localVertexId == 0 ? tmp + a : tmp + b;
1278 } else {
1279 const auto tmp = p[0] + (p[1] * vshift_[0]);
1280 return localVertexId == 0 ? tmp + a : tmp + b;
1281 }
1282 };
1283
1284 switch(this->underlying().getEdgePosition(edgeId)) {
1285 CASE_EDGE_POSITION_L_3D:
1286 vertexId = helper3d(0, 1);
1287 break;
1288 CASE_EDGE_POSITION_H_3D:
1289 vertexId = helper3d(0, vshift_[0]);
1290 break;
1291 CASE_EDGE_POSITION_P_3D:
1292 vertexId = helper3d(0, vshift_[1]);
1293 break;
1294 CASE_EDGE_POSITION_D1_3D:
1295 vertexId = helper3d(1, vshift_[0]);
1296 break;
1297 CASE_EDGE_POSITION_D2_3D:
1298 vertexId = helper3d(0, vshift_[0] + vshift_[1]);
1299 break;
1300 CASE_EDGE_POSITION_D3_3D:
1301 vertexId = helper3d(1, vshift_[1]);
1302 break;
1303 case EdgePosition::D4_3D:
1304 vertexId = helper3d(1, vshift_[0] + vshift_[1]);
1305 break;
1306
1307 CASE_EDGE_POSITION_L_2D:
1308 vertexId = helper2d(0, 1);
1309 break;
1310 CASE_EDGE_POSITION_H_2D:
1311 vertexId = helper2d(0, vshift_[0]);
1312 break;
1313 case EdgePosition::D1_2D:
1314 vertexId = helper2d(1, vshift_[0]);
1315 break;
1316
1317 case EdgePosition::FIRST_EDGE_1D:
1318 vertexId = localVertexId == 0 ? 0 : 1;
1319 break;
1320 case EdgePosition::LAST_EDGE_1D:
1321 vertexId = localVertexId == 0 ? edgeNumber_ - 1 : edgeNumber_;
1322 break;
1323 case EdgePosition::CENTER_1D:
1324 vertexId = localVertexId == 0 ? edgeId : edgeId + 1;
1325 break;
1326 }
1327
1328 return 0;
1329}
