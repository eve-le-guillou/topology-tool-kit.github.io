44int FTMTree_CT::combine() {
45 Timer stepTime;
46 queue<pair<bool, idNode>> growingNodes, remainingNodes;
47
48 const bool DEBUG = false;
49
50 // Reserve
51 mt_data_.nodes->reserve(jt_->getNumberOfNodes());
52 mt_data_.superArcs->reserve(jt_->getNumberOfSuperArcs() + 2);
53 mt_data_.leaves->reserve(jt_->getNumberOfLeaves() + st_->getNumberOfLeaves());
54
55 // Add JT & ST Leaves to growingNodes
56
57 // Add leves to growing nodes
58 const auto &nbSTLeaves = st_->getNumberOfLeaves();
59 if(nbSTLeaves > 1) {
60 for(idNode n = 0; n < nbSTLeaves; ++n) {
61 const auto &nId = st_->getLeave(n);
62 growingNodes.emplace(false, nId);
63 }
64 } else {
65 move(jt_);
66 return 0;
67 }
68
69 // count how many leaves can be added, if more than one : ok!
70 const auto &nbJTLeaves = jt_->getNumberOfLeaves();
71 if(nbJTLeaves > 1) {
72 for(idNode n = 0; n < nbJTLeaves; ++n) {
73 const auto &nId = jt_->getLeave(n);
74 growingNodes.emplace(true, nId);
75 }
76 } // else can't clone, not same up and down
77
78 if(DEBUG) {
79 cout << "growingNodes : " << growingNodes.size()
80 << " in : " << stepTime.getElapsedTime() << endl;
81 }
82
83 // Warning, have a reserve here, can't make it at the begnining, need build
84 // output
85 mt_data_.leaves->reserve(jt_->getLeaves().size() + st_->getLeaves().size());
86 mt_data_.superArcs->reserve(jt_->getNumberOfSuperArcs());
87 mt_data_.nodes->reserve(jt_->getNumberOfNodes());
88
89 if(growingNodes.empty()) {
90 cout << "[FTMTree_CT::combine ] Nothing to combine" << endl;
91 }
92
93#ifdef TTK_ENABLE_FTM_TREE_DUAL_QUEUE_COMBINE
94 do {
95 while(!remainingNodes.empty()) {
96 bool isJT;
97 idNode currentNodeId;
98 FTMTree_MT *xt;
99
100 tie(isJT, currentNodeId) = remainingNodes.front();
101 remainingNodes.pop();
102 if(isJT) {
103 // node come frome jt
104 xt = jt_;
105 } else {
106 // node come from st
107 xt = st_;
108 }
109 if(xt->getNode(currentNodeId)->getNumberOfUpSuperArcs() == 1) {
110 growingNodes.emplace(isJT, currentNodeId);
111 if(DEBUG) {
112 cout << "repush in growing:" << isJT
113 << "::" << xt->printNode(currentNodeId) << endl;
114 }
115 }
116 }
117#endif
118
119 while(!growingNodes.empty()) {
120 idNode currentNodeId;
121 bool isJT;
122
123 // INFO QUEUE
124
125 tie(isJT, currentNodeId) = growingNodes.front();
126 growingNodes.pop();
127
128 FTMTree_MT *xt = (isJT) ? jt_ : st_;
129 FTMTree_MT *yt = (isJT) ? st_ : jt_;
130
131 // INFO JT / ST
132
133 // i <- Get(Q)
134 const Node *currentNode = xt->getNode(currentNodeId);
135
136 if(DEBUG) {
137 if(xt == jt_)
138 cout << endl << "JT ";
139 else
140 cout << endl << "ST ";
141 cout << "node : " << currentNode->getVertexId() << endl;
142 }
143
144 // "choose a non-root leaf that is not a split in ST" so we ignore such
145 // nodes
146 if(currentNode->getNumberOfUpSuperArcs() == 0) {
147 if(DEBUG) {
148 cout << " ignore already processed" << endl;
149 }
150 continue;
151 }
152
153 idNode correspondingNodeId
154 = yt->getCorrespondingNodeId(currentNode->getVertexId());
155
156 if(yt->getNode(correspondingNodeId)->getNumberOfDownSuperArcs() > 1) {
157 if(DEBUG) {
158 cout << "put remain:" << isJT << "::" << xt->printNode(currentNodeId)
159 << endl;
160 cout << " which is in yt : " << yt->printNode(correspondingNodeId)
161 << endl;
162 }
163#ifdef TTK_ENABLE_FTM_TREE_DUAL_QUEUE_COMBINE
164 remainingNodes.emplace(isJT, currentNodeId);
165#else
166 growingNodes.emplace(isJT, currentNodeId);
167#endif
168 continue;
169 }
170
171 // NODES IN CT
172
173 idNode node1, node2;
174 SimplexId curVert = currentNode->getVertexId();
175 // NODE1
176 if(isCorrespondingNode(curVert)) {
177 // already a node in the tree
178 node1 = getCorrespondingNodeId(curVert);
179 } else {
180 // create a new node
181 node1 = makeNode(currentNode);
182
183 // check if leaf
184 if(!currentNode->getNumberOfDownSuperArcs()
185 || !currentNode->getNumberOfUpSuperArcs())
186 mt_data_.leaves->emplace_back(node1);
187 }
188
189 // j <- GetAdj(XT, i)
190 idSuperArc curUpArc = currentNode->getUpSuperArcId(0);
191 idNode parentId = xt->getSuperArc(curUpArc)->getUpNodeId();
192 const Node *parentNode = xt->getNode(parentId);
193
194 if(DEBUG) {
195 cout << " parent node :" << parentNode->getVertexId() << endl;
196 }
197
198 SimplexId parVert = parentNode->getVertexId();
199 // NODE2
200 if(isCorrespondingNode(parVert)) {
201 // already a node in the tree
202 node2 = getCorrespondingNodeId(parVert);
203 } else {
204 // create a new node
205 node2 = makeNode(parentNode);
206 if(!parentNode->getNumberOfUpSuperArcs())
207 mt_data_.leaves->emplace_back(node2);
208 }
209
210 // CREATE ARC
211
212 idSuperArc processArc = currentNode->getUpSuperArcId(0);
213
214 // create the arc in in the good direction
215 // and add it to crossing if needed
216 idSuperArc createdArc;
217 if(scalars_->isLower(
218 currentNode->getVertexId(),
219 parentNode->getVertexId())) { // take care of the order
220 createdArc = makeSuperArc(node1, node2);
221 } else {
222 createdArc = makeSuperArc(node2, node1);
223 }
224
225 // Segmentation
226 if(params_->segm) {
227 createCTArcSegmentation(createdArc, isJT, processArc);
228 }
229
230 if(DEBUG) {
231 cout << "create arc : " << printArc(createdArc) << endl;
232 }
233
234 // DEL NODES
235
236 // DelNode(XT, i)
237 {
238 if(DEBUG) {
239 cout << " delete xt (" << (xt == jt_) << ") ";
240 cout << "node :" << xt->printNode(currentNodeId) << endl;
241 }
242
243 xt->delNode(currentNodeId);
244 }
245
246 // DelNode(YT, i)
247 {
248 if(DEBUG) {
249 cout << " delete yt (" << isJT << ") node :";
250 cout << yt->printNode(correspondingNodeId) << endl;
251 }
252
253 yt->delNode(correspondingNodeId);
254 }
255
256 // PROCESS QUEUE
257
258 if(parentNode->getNumberOfDownSuperArcs() == 0
259 && parentNode->getNumberOfUpSuperArcs()) {
260 growingNodes.emplace(isJT, parentId);
261
262 if(DEBUG) {
263 cout << "will see : " << parentNode->getVertexId() << endl;
264 }
265 }
266 }
267#ifdef TTK_ENABLE_FTM_TREE_DUAL_QUEUE_COMBINE
268 } while(!remainingNodes.empty());
269#endif
270
271 if(DEBUG) {
272 printTree2();
273 }
274
275 return 0;

278void FTMTree_CT::createCTArcSegmentation(idSuperArc ctArc,
279 const bool isJT,
280 idSuperArc xtArc) {
281 const FTMTree_MT *xt = (isJT) ? jt_ : st_;
282
283 /*Here we prefere to create lots of small region, each arc having its own
284 * segmentation with no overlap instead of having a same vertice in several
285 * arc and using vert2tree to decide because we do not want to maintain
286 * vert2tree information during the whole computation*/
287 const list<Region> &xtRegions = xt->getSuperArc(xtArc)->getRegions();
288 for(const Region &reg : xtRegions) {
289 segm_it cur = reg.segmentBegin;
290 segm_it end = reg.segmentEnd;
291 segm_it tmpBeg = reg.segmentBegin;
292 // each element inside this region
293 for(; cur != end; ++cur) {
294 if(isCorrespondingNull(*cur)) {
295 updateCorrespondingArc(*cur, ctArc);
296 } else {
297 // already set, we finish a region
298 if(cur != tmpBeg) {
299 getSuperArc(ctArc)->concat(tmpBeg, cur);
300 }
301 // if several contiguous vertices are discarded
302 // cur will be equals to tmpBeg and we will not create empty regions
303 tmpBeg = cur + 1;
304 }
305 }
306 // close last region
307 if(cur != tmpBeg) {
308 getSuperArc(ctArc)->concat(tmpBeg, cur);
309 }
310 }
