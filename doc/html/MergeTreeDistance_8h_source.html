151 template <class dataType>
152 void createCostMatrix(std::vector<std::vector<dataType>> &treeTable,
153 std::vector<ftm::idNode> &children1,
154 std::vector<ftm::idNode> &children2,
155 std::vector<std::vector<dataType>> &costMatrix) {
156 unsigned int nRows = children1.size(), nCols = children2.size();
157 for(unsigned int i = 0; i < nRows; ++i) {
158 int forestTableI = children1[i] + 1;
159 for(unsigned int j = 0; j < nCols; ++j) {
160 int forestTableJ = children2[j] + 1;
161 // Cost of assigning i and j
162 costMatrix[i][j] = treeTable[forestTableI][forestTableJ];
163 if(tree1Level_[children1[i]] != tree2Level_[children2[j]]
164 and not keepSubtree_)
165 printErr("different levels!"); // should be impossible
166 }
167 // Cost of not assigning i
168 costMatrix[i][nCols] = treeTable[forestTableI][0];
169 }
170 for(unsigned int j = 0; j < nCols; ++j) {
171 int forestTableJ = children2[j] + 1;
172 // Cost of not assigning j
173 costMatrix[nRows][j] = treeTable[0][forestTableJ];
174 }
175 costMatrix[nRows][nCols] = 0;
176 }

178 template <class dataType>
179 dataType postprocessAssignment(
180 std::vector<MatchingType> &matchings,
181 std::vector<ftm::idNode> &children1,
182 std::vector<ftm::idNode> &children2,
183 std::vector<std::tuple<int, int>> &forestAssignment) {
184 dataType cost = 0;
185 for(const auto &mTuple : matchings) {
186 cost += std::get<2>(mTuple);
187 if(std::get<0>(mTuple) >= (int)children1.size()

189 continue;
190 int tableId1 = children1[std::get<0>(mTuple)] + 1;
191 int tableId2 = children2[std::get<1>(mTuple)] + 1;
192 forestAssignment.emplace_back(tableId1, tableId2);
193 }
194 return cost;
195 }

235 std::vector<ftm::idNode> &children2) {

237 dataType forestTerm1, forestTerm2, forestTerm3;
238 std::tuple<dataType, ftm::idNode> forestCoTerm1, forestCoTerm2;
239 // Term 1
240 forestCoTerm1
241 = computeTerm1_2<dataType>(children2, i, forestTable, true);
242 forestTerm1 = forestTable[0][j] + std::get<0>(forestCoTerm1);
243
244 // Term2
245 forestCoTerm2

247 forestTerm2 = forestTable[i][0] + std::get<0>(forestCoTerm2);

252 forestTerm3 = forestAssignmentProblem<dataType>(
253 tree1, tree2, treeTable, children1, children2, forestAssignment);
254 if(not parallelize_)
255 t_assignment_time_ += t_assignment.getElapsedTime();
256
257 // Compute table value
258 forestTable[i][j] = keepSubtree_ ? std::min(
259 std::min(forestTerm1, forestTerm2), forestTerm3)
260 : forestTerm3;
261
262 // Add backtracking information
263 if(forestTable[i][j] == forestTerm3) {
264 forestBackTable[i][j] = forestAssignment;
265 } else if(forestTable[i][j] == forestTerm2) {
266 forestBackTable[i][j].push_back(
267 std::make_tuple(std::get<1>(forestCoTerm2), j));
268 } else {
269 forestBackTable[i][j].push_back(
270 std::make_tuple(i, std::get<1>(forestCoTerm1)));
271 }
272 } else {
273 // If one of the forest is empty we get back to equation 8 or 10
274 forestTable[i][j]
275 = (children1.size() == 0) ? forestTable[0][j] : forestTable[i][0];

331 template <class dataType>
332 std::tuple<dataType, ftm::idNode>
333 computeTerm1_2(std::vector<ftm::idNode> &childrens,
334 int ind,
335 std::vector<std::vector<dataType>> &table,
336 bool computeTerm1) {
337 dataType tempMin = (childrens.size() == 0)
338 ? ((computeTerm1) ? table[ind][0] : table[0][ind])
339 : std::numeric_limits<dataType>::max();
340 ftm::idNode bestIdNode = 0;
341 for(ftm::idNode children : childrens) {
342 children += 1;
343 dataType temp;
344 if(computeTerm1) {
345 temp = table[ind][children] - table[0][children];
346 } else {
347 temp = table[children][ind] - table[children][0];
348 }
349 if(temp < tempMin) {
350 tempMin = temp;
351 bestIdNode = children;
352 }
353 }
354 return std::make_tuple(tempMin, bestIdNode);
355 }

400 // Output Matching
401 // --------------------------------------------------------------------------------
402 template <class dataType>
403 void computeMatching(
404 ftm::FTMTree_MT *tree1,
405 ftm::FTMTree_MT *tree2,
406 std::vector<std::vector<std::tuple<int, int>>> &treeBackTable,
407 std::vector<std::vector<std::vector<std::tuple<int, int>>>>
408 &forestBackTable,
409 std::vector<std::tuple<ftm::idNode, ftm::idNode, double>> &outputMatching,
410 int startR,
411 int startC) {
412 outputMatching.clear();
413 std::queue<std::tuple<int, int, bool>> backQueue;
414 backQueue.emplace(std::make_tuple(startR, startC, true));
415 while(!backQueue.empty()) {
416 std::tuple<int, int, bool> elem = backQueue.front();
417 backQueue.pop();
418 bool useTreeTable = std::get<2>(elem);
419 int i = std::get<0>(elem);
420 int j = std::get<1>(elem);
421
422 if(useTreeTable) {
423 int tupleI = std::get<0>(treeBackTable[i][j]);
424 int tupleJ = std::get<1>(treeBackTable[i][j]);
425 if(tupleI != 0 && tupleJ != 0) {
426 useTreeTable = (tupleI != i || tupleJ != j);
427 backQueue.emplace(std::make_tuple(tupleI, tupleJ, useTreeTable));
428 if(not useTreeTable) { // We have matched i and j
429 ftm::idNode tree1Node = tupleI - 1;
430 ftm::idNode tree2Node = tupleJ - 1;
431 double cost = 0;
432 dataType costT
433 = relabelCost<dataType>(tree1, tree1Node, tree2, tree2Node);
434 cost = static_cast<double>(costT);
435 outputMatching.emplace_back(tree1Node, tree2Node, cost);
436 }
437 }
438 } else {
439 for(std::tuple<int, int> forestBackElem : forestBackTable[i][j]) {
440 int tupleI = std::get<0>(forestBackElem);
441 int tupleJ = std::get<1>(forestBackElem);
442 if(tupleI != 0 && tupleJ != 0) {
443 useTreeTable = (tupleI != i && tupleJ != j);
444 backQueue.emplace(std::make_tuple(tupleI, tupleJ, useTreeTable));
445 }
446 }
447 }
448 }

454 template <class dataType>
455 dataType
456 computeDistance(ftm::FTMTree_MT *tree1,
457 ftm::FTMTree_MT *tree2,
458 std::vector<std::tuple<ftm::idNode, ftm::idNode, double>>
459 &outputMatching) {
460 // ---------------------
461 // ----- Init dynamic progamming tables
462 // --------------------
463 size_t nRows = tree1->getNumberOfNodes() + 1;
464 size_t nCols = tree2->getNumberOfNodes() + 1;
465 std::vector<std::vector<dataType>> treeTable(
466 nRows, std::vector<dataType>(nCols));
467 std::vector<std::vector<dataType>> forestTable(
468 nRows, std::vector<dataType>(nCols));
469
470 // Backtracking tables (output matching)
471 std::vector<std::vector<std::tuple<int, int>>> treeBackTable(
472 nRows, std::vector<std::tuple<int, int>>(nCols));
473 std::vector<std::vector<std::vector<std::tuple<int, int>>>>
474 forestBackTable(
475 nRows, std::vector<std::vector<std::tuple<int, int>>>(nCols));
476
477 int indR = tree1->getRoot() + 1;
478 int indC = tree2->getRoot() + 1;
479
480 tree1->getAllNodeLevel(tree1Level_);
481 tree2->getAllNodeLevel(tree2Level_);
482 tree2->getLevelToNode(tree2LevelToNode_);
483
484 // ---------------------
485 // ----- Compute edit distance
486 // --------------------
487 if(progressiveComputation_) {
488 /*MergeTreeDistanceProgressive<dataType> editDistanceProgressive;
489 editDistanceProgressive.computeEditDistanceProgressive(tree1, tree2,
490 treeTable, forestTable, treeBackTable,
491 forestBackTable);*/
492 } else {
493 computeEditDistance(tree1, tree2, treeTable, forestTable, treeBackTable,
494 forestBackTable, nRows, nCols);
495 }
496 dataType distance = treeTable[indR][indC];
497 if(onlyEmptyTreeDistance_)
498 distance = treeTable[indR][0];
499 if(branchDecomposition_) {
500 if(not useMinMaxPair_) {
501 if(onlyEmptyTreeDistance_)
502 distance -= deleteCost<dataType>(tree1, tree1->getRoot());
503 else
504 distance -= relabelCost<dataType>(
505 tree1, tree1->getRoot(), tree2, tree2->getRoot());
506 } else {
507 if(minMaxPairWeight_ != 1.0) {
508 auto cost = relabelCost<dataType>(
509 tree1, tree1->getRoot(), tree2, tree2->getRoot());
510 distance = distance - cost + minMaxPairWeight_ * cost;
511 }
512 }
513 }
514 if(distanceSquared_)
515 distance = std::sqrt(distance);
516
517 // ---------------------
518 // ----- Compute matching
519 // --------------------
520 computeMatching<dataType>(tree1, tree2, treeBackTable, forestBackTable,
521 outputMatching, indR, indC);
522
523 return distance;
524 }
