24 template <class dataType>
25 MergeTree<dataType> makeTree(vtkUnstructuredGrid *treeNodes,
26 vtkUnstructuredGrid *treeArcs) {
27 auto treeNodeIdArray = treeNodes->GetPointData()->GetArray("TreeNodeId");
28
29 // Init Scalars
30 Scalars scalars;
31 vtkSmartPointer<vtkDataArray> nodesScalar
32 = treeNodes->GetPointData()->GetArray("Scalar"); // 1: Scalar
33 scalars.size = nodesScalar->GetNumberOfTuples();
34 std::vector<dataType> scalarsValues(nodesScalar->GetNumberOfTuples());
35 for(int i = 0; i < nodesScalar->GetNumberOfTuples(); ++i) {
36 int index = (treeNodeIdArray ? treeNodeIdArray->GetTuple1(i) : i);
37 scalarsValues[index] = nodesScalar->GetTuple1(i);
38 }
39 scalars.values = (void *)(scalarsValues.data());
40
41 // Init Tree
42 Params params;
43 params.treeType = Join_Split;
44 MergeTree<dataType> mergeTree(scalars, scalarsValues, params);
45
46 // Add Nodes
47 vtkSmartPointer<vtkDataArray> nodesId
48 = treeNodes->GetPointData()->GetArray("NodeId"); // 0: NodeId
49 vtkIdType nodesNumTuples = nodesId->GetNumberOfTuples();
50 for(vtkIdType i = 0; i < nodesNumTuples; ++i) {
51 mergeTree.tree.makeNode(i);
52 }
53
54 // Add Arcs
55 vtkSmartPointer<vtkDataArray> arcsUp
56 = treeArcs->GetCellData()->GetArray("upNodeId"); // 1: upNodeId
57 vtkSmartPointer<vtkDataArray> arcsDown
58 = treeArcs->GetCellData()->GetArray("downNodeId"); // 2: downNodeId
59 vtkIdType arcsNumTuples = arcsUp->GetNumberOfTuples();
60 vtkSmartPointer<vtkDataArray> dummyArcArray
61 = treeArcs->GetCellData()->GetArray("isDummyArc");
62 std::set<std::tuple<double, double>> added_arcs; // Avoid duplicates
63 for(vtkIdType i = 0; i < arcsNumTuples; ++i) {
64 if(dummyArcArray != nullptr and dummyArcArray->GetTuple1(i) == 1)
65 continue;
66 double downId = arcsDown->GetTuple1(i);
67 double upId = arcsUp->GetTuple1(i);
68 if(treeNodeIdArray) {
69 downId = treeNodeIdArray->GetTuple1(downId);
70 upId = treeNodeIdArray->GetTuple1(upId);
71 }
72 auto it = added_arcs.find(std::make_tuple(downId, upId));
73 if(it == added_arcs.end()) { // arc not added yet
74 mergeTree.tree.makeSuperArc(downId, upId); // (down, Up)
75 added_arcs.insert(std::make_tuple(downId, upId));
76 }
77 }
78
79 // Manage inconsistent arcs
80 manageInconsistentArcsMultiParent(&(mergeTree.tree));
81
82 // Remove self link
83 removeSelfLink(&(mergeTree.tree));
84
85 return mergeTree;
86 }

88 inline void
89 loadBlocks(std::vector<vtkSmartPointer<vtkMultiBlockDataSet>> &inputTrees,
90 vtkMultiBlockDataSet *blocks) {
91 if(blocks != nullptr) {
92 inputTrees.resize(
93 vtkMultiBlockDataSet::SafeDownCast(blocks->GetBlock(0))
94 ->GetNumberOfBlocks());
95 for(size_t i = 0; i < inputTrees.size(); ++i) {
96 vtkSmartPointer<vtkMultiBlockDataSet> vtkBlock
97 = vtkSmartPointer<vtkMultiBlockDataSet>::New();
98 vtkBlock->SetNumberOfBlocks(blocks->GetNumberOfBlocks());
99 for(unsigned int j = 0; j < blocks->GetNumberOfBlocks(); ++j)
100 vtkBlock->SetBlock(
101 j, vtkMultiBlockDataSet::SafeDownCast(blocks->GetBlock(j))
102 ->GetBlock(i));
103 inputTrees[i] = vtkBlock;
104 }
105 }

109 void constructTrees(
110 std::vector<vtkSmartPointer<vtkMultiBlockDataSet>> &inputTrees,
111 std::vector<MergeTree<dataType>> &intermediateTrees,
112 std::vector<vtkUnstructuredGrid *> &treesNodes,
113 std::vector<vtkUnstructuredGrid *> &treesArcs,
114 std::vector<vtkDataSet *> &treesSegmentation) {
115 const int numInputs = inputTrees.size();
116 intermediateTrees = std::vector<MergeTree<dataType>>(numInputs);
117 treesNodes = std::vector<vtkUnstructuredGrid *>(numInputs);
118 treesArcs = std::vector<vtkUnstructuredGrid *>(numInputs);
119 treesSegmentation = std::vector<vtkDataSet *>(numInputs);
120 for(int i = 0; i < numInputs; i++) {
121 treesNodes[i]
122 = vtkUnstructuredGrid::SafeDownCast(inputTrees[i]->GetBlock(0));
123 treesArcs[i]
124 = vtkUnstructuredGrid::SafeDownCast(inputTrees[i]->GetBlock(1));
125 if(inputTrees[i]->GetNumberOfBlocks() > 2)
126 treesSegmentation[i]
127 = vtkDataSet::SafeDownCast(inputTrees[i]->GetBlock(2));
128 intermediateTrees[i] = makeTree<dataType>(treesNodes[i], treesArcs[i]);
129 }
130 }
