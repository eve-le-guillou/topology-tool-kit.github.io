288 template <typename dataType>
289 dataType AssignmentExhaustive<dataType>::tryAssignment(
290 std::vector<int> &asgn, std::vector<MatchingType> &matchings) {
291 unsigned int nRows = this->costMatrix.size() - 1;
292 unsigned int nCols = this->costMatrix[0].size() - 1;
293 // int max_dim = std::max(nRows, nCols);
294 unsigned int min_dim = std::min(nRows, nCols);
295 bool transpose = nRows > nCols;
296
297 dataType cost = 0;
298 for(unsigned int ind = 0; ind < asgn.size(); ++ind) {
299 int indMatrix = std::min(ind, min_dim);
300 int i = (!transpose) ? indMatrix : asgn[ind];
301 int j = (!transpose) ? asgn[ind] : indMatrix;
302 cost += this->costMatrix[i][j];
303 matchings.push_back(std::make_tuple(i, j, this->costMatrix[i][j]));
304 }
305 return cost;
306 }

308 template <typename dataType>
309 int AssignmentExhaustive<dataType>::run(
310 std::vector<MatchingType> &matchings) {
311 int nRows = this->costMatrix.size() - 1;
312 int nCols = this->costMatrix[0].size() - 1;
313 int max_dim = std::max(nRows, nCols);
314 int min_dim = std::min(nRows, nCols);
315
316 // --- Construct all possible assignments
317 std::vector<std::vector<int>> allAsgn;
318 // We hard write the basic assignments to avoid the call of
319 // constructAssignments These assignments are, of course, automatically
320 // generated by constructAssignments
321 if(min_dim == 1 and max_dim == 1)
322 allAsgn = {{0}, {1, 0}};
323 else if(min_dim == 1 and max_dim == 2)
324 allAsgn = {{0, 1}, {2, 0, 1}, {1, 0}};
325 else if(min_dim == 1 and max_dim == 3)
326 allAsgn = {{0, 1, 2}, {3, 0, 1, 2}, {1, 0, 2}, {2, 0, 1}};
327 else if(min_dim == 2 and max_dim == 2)
328 allAsgn = {{0, 1}, {0, 2, 1}, {2, 1, 0}, {2, 2, 0, 1},
329 {1, 0}, {1, 2, 0}, {2, 0, 1}};
330 else {
331 std::stringstream ss;
332 ss << min_dim << "_" << max_dim;
333 std::string asgnName = ss.str();
334 auto it = savedAsgn.find(asgnName);
335 if(it == savedAsgn.end()) { // not found
336 printMsg(asgnName, debug::Priority::VERBOSE);
337 // allAsgn = constructAssignments(min_dim, max_dim);
338 // allAsgn = constructAssignments2(min_dim, max_dim);
339 allAsgn = constructAssignments3(min_dim, max_dim);
340 savedAsgn[asgnName] = allAsgn;
341 std::stringstream ss2;
342 ss2 << allAsgn.size() << " done";
343 printMsg(ss2.str(), debug::Priority::VERBOSE);
344 } else
345 allAsgn = savedAsgn[asgnName];
346 }
347
348 // --- Try these assignments and get the better
349 dataType bestCost = std::numeric_limits<dataType>::max();
350 std::vector<MatchingType> bestMatching;
351 for(std::vector<int> asgn : allAsgn) {
352 std::vector<MatchingType> tempMatching;
353 dataType cost = tryAssignment(asgn, tempMatching);
354 if(bestCost > cost) {
355 bestCost = cost;
356 bestMatching = tempMatching;
357 }
358 }
359 matchings = bestMatching;
360
361 return 0;
362 }
