158void Triangulation::switchGrid(const bool usePeriodic,
159 const bool usePreconditions) {
160 if(abstractTriangulation_ != nullptr
161 && abstractTriangulation_ != &implicitTriangulation_
162 && abstractTriangulation_ != &implicitPreconditionsTriangulation_
163 && abstractTriangulation_ != &periodicImplicitTriangulation_
164 && abstractTriangulation_ != &periodicPreconditionsTriangulation_) {
165 return;
166 }
167
168 if(abstractTriangulation_ != nullptr) {
169 // clear preconditions when switching triangulation type
170 if(abstractTriangulation_ == &implicitPreconditionsTriangulation_
171 && (usePeriodic || !usePreconditions)) {
172 implicitPreconditionsTriangulation_.clear();
173 } else if(abstractTriangulation_ == &periodicPreconditionsTriangulation_
174 && (!usePeriodic || !usePreconditions)) {
175 periodicPreconditionsTriangulation_.clear();
176 }
177 }
178
179 if(!usePeriodic && !usePreconditions) {
180 abstractTriangulation_ = &implicitTriangulation_;
181 implicitTriangulation_.preconditionVerticesAndCells();
182 } else if(!usePeriodic && usePreconditions) {
183 abstractTriangulation_ = &implicitPreconditionsTriangulation_;
184 implicitPreconditionsTriangulation_.preconditionVerticesAndCells();
185 } else if(usePeriodic && !usePreconditions) {
186 abstractTriangulation_ = &periodicImplicitTriangulation_;
187 periodicImplicitTriangulation_.preconditionVerticesAndCells();
188 } else if(usePeriodic && usePreconditions) {
189 abstractTriangulation_ = &periodicPreconditionsTriangulation_;
190 periodicPreconditionsTriangulation_.preconditionVerticesAndCells();
191 }
192}

206 // disable preconditioning above TTK_IMPLICIT_PRECONDITIONS_THRESHOLD
207 const auto doPreconditioning = nVerts <= threshold;
208
209 if(!doPreconditioning) {
210

219 this->printWrn("Defaulting to the fully implicit triangulation");
220 if(prevDebugLevel != -1) {
221 this->debugLevel_ = prevDebugLevel;
222 }
