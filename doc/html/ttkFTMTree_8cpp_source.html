43int ttkFTMTree::RequestData(vtkInformation *ttkNotUsed(request),
44 vtkInformationVector **inputVector,
45 vtkInformationVector *outputVector) {
46
47 const auto input = vtkDataSet::GetData(inputVector[0]);
48 auto outputSkeletonNodes = vtkUnstructuredGrid::GetData(outputVector, 0);
49 auto outputSkeletonArcs = vtkUnstructuredGrid::GetData(outputVector, 1);
50 auto outputSegmentation = vtkDataSet::GetData(outputVector, 2);
51
52#ifndef TTK_ENABLE_KAMIKAZE
53 if(!input) {
54 this->printErr("Error: input pointer is NULL.");
55 return 0;
56 }
57
58 if(!input->GetNumberOfPoints()) {
59 this->printErr("Error: input has no point.");
60 return 0;
61 }
62
63 if(!outputSkeletonNodes || !outputSkeletonArcs || !outputSegmentation) {
64 this->printErr("Error: output pointer is NULL.");
65 return 0;
66 }
67#endif
68
69 // Arrays
70
71 vtkDataArray *inputArray = this->GetInputArrayToProcess(0, inputVector);
72 if(!inputArray)
73 return 0;
74
75 // Connected components
76 if(input->IsA("vtkUnstructuredGrid")) {
77 // This data set may have several connected components,
78 // we need to apply the FTM Tree for each one of these components
79 // We then reconstruct the global tree using an offest mecanism
80 auto inputWithId = vtkSmartPointer<vtkUnstructuredGrid>::New();
81 inputWithId->ShallowCopy(input);
82 identify(inputWithId);
83
84 vtkNew<vtkConnectivityFilter> connectivity{};
85 connectivity->SetInputData(inputWithId);
86 connectivity->SetExtractionModeToAllRegions();
87 connectivity->ColorRegionsOn();
88 connectivity->Update();
89
90 nbCC_ = connectivity->GetOutput()
91 ->GetCellData()
92 ->GetArray("RegionId")
93 ->GetRange()[1]
94 + 1;
95 connected_components_.resize(nbCC_);
96
97 if(nbCC_ > 1) {
98 // Warning, in case of several connected components, the ids seen by
99 // the base code will not be consistent with those of the original
100 // mesh
101 for(int cc = 0; cc < nbCC_; cc++) {
102 vtkNew<vtkThreshold> threshold{};
103 threshold->SetInputConnection(connectivity->GetOutputPort());
104 threshold->SetInputArrayToProcess(
105 0, 0, 0, vtkDataObject::FIELD_ASSOCIATION_CELLS, "RegionId");
106#if VTK_VERSION_NUMBER < VTK_VERSION_CHECK(9, 2, 0)
107 threshold->ThresholdBetween(cc, cc);
108#else
109 threshold->SetThresholdFunction(vtkThreshold::THRESHOLD_BETWEEN);
110 threshold->SetLowerThreshold(cc);
111 threshold->SetUpperThreshold(cc);
112#endif
113 threshold->Update();
114 connected_components_[cc] = vtkSmartPointer<vtkUnstructuredGrid>::New();
115 connected_components_[cc]->ShallowCopy(threshold->GetOutput());
116 }
117 } else {
118 connected_components_[0] = inputWithId;
119 }
120 } else if(input->IsA("vtkPolyData")) {
121 // NOTE: CC check should not be implemented on a per vtk module layer.
122 nbCC_ = 1;
123 connected_components_.resize(nbCC_);
124 connected_components_[0] = vtkSmartPointer<vtkPolyData>::New();
125 connected_components_[0]->ShallowCopy(input);
126 identify(connected_components_[0]);
127 } else {
128 nbCC_ = 1;
129 connected_components_.resize(nbCC_);
130 connected_components_[0] = vtkSmartPointer<vtkImageData>::New();
131 connected_components_[0]->ShallowCopy(input);
132 identify(connected_components_[0]);
133 }
134
135 // now proceed for each triangulation obtained.
136
137 if(preconditionTriangulation() == 0) {
138#ifndef TTK_ENABLE_KAMIKAZE
139 this->printErr("Error : wrong triangulation.");
140 return 0;
141#endif
142 }
143
144 // Fill the vector of scalar/offset, cut the array in pieces if needed
145 if(getScalars() == 0) {
146#ifndef TTK_ENABLE_KAMIKAZE
147 this->printErr("Error : wrong input scalars.");
148 return 0;
149#endif
150 }
151 getOffsets();
152
153 this->printMsg("Launching on field "
154 + std::string{inputScalars_[0]->GetName()});
155
156 ttk::ftm::idNode acc_nbNodes = 0;
157
158 // Build tree
159 for(int cc = 0; cc < nbCC_; cc++) {
160 ftmTree_[cc].tree.setVertexScalars(
161 ttkUtils::GetVoidPointer(inputScalars_[cc]));
162 ftmTree_[cc].tree.setVertexSoSoffsets(offsets_[cc].data());
163 ftmTree_[cc].tree.setTreeType(GetTreeType());
164 ftmTree_[cc].tree.setSegmentation(GetWithSegmentation());
165 ftmTree_[cc].tree.setNormalizeIds(GetWithNormalize());
166
167 ttkVtkTemplateMacro(inputArray->GetDataType(),
168 triangulation_[cc]->getType(),
169 (ftmTree_[cc].tree.build<VTK_TT, TTK_TT>(
170 (TTK_TT *)triangulation_[cc]->getData())));
171
172 ftmTree_[cc].offset = acc_nbNodes;
173 acc_nbNodes += ftmTree_[cc].tree.getTree(GetTreeType())->getNumberOfNodes();
174 }
175
176 UpdateProgress(0.50);
177
178 // Construct output
179 if(getSkeletonNodes(outputSkeletonNodes) == 0) {
180#ifndef TTK_ENABLE_KAMIKAZE
181 this->printErr("Error : wrong properties on skeleton nodes.");
182 return 0;
183#endif
184 }
185
186 if(getSkeletonArcs(outputSkeletonArcs) == 0) {
187#ifndef TTK_ENABLE_KAMIKAZE
188 this->printErr("Error : wrong properties on skeleton arcs.");
189 return 0;
190#endif
191 }
192
193 if(GetWithSegmentation()) {
194 outputSegmentation->ShallowCopy(input);
195 if(getSegmentation(outputSegmentation) == 0) {
196#ifndef TTK_ENABLE_KAMIKAZE
197 this->printErr("Error : wrong properties on segmentation.");
198 return 0;
199#endif
200 }
201 }
202
203 UpdateProgress(1);
204
205#ifdef TTK_ENABLE_FTM_TREE_STATS_TIME
206 printCSVStats();
207#endif
208
209 return 1;
210}

212int ttkFTMTree::addCompleteSkeletonArc(const ttk::ftm::idSuperArc arcId,
213 const int cc,
214 vtkPoints *points,
215 vtkUnstructuredGrid *skeletonArcs,
216 ttk::ftm::ArcData &arcData) {
217 auto tree = ftmTree_[cc].tree.getTree(GetTreeType());
218 vtkDataArray *idMapper = connected_components_[cc]->GetPointData()->GetArray(
219 ttk::VertexScalarFieldName);
220 auto arc = tree->getSuperArc(arcId);
221 float point[3];
222 vtkIdType pointIds[2];
223
224 using ttk::SimplexId;
225 const SimplexId downNodeId = tree->getLowerNodeId(arc);
226 const SimplexId l_downVertexId = tree->getNode(downNodeId)->getVertexId();
227 const SimplexId g_downVertexId = idMapper->GetTuple1(l_downVertexId);
228 triangulation_[cc]->getVertexPoint(
229 l_downVertexId, point[0], point[1], point[2]);
230 const double scalarMin = inputScalars_[cc]->GetTuple1(l_downVertexId);
231
232 // Get or create first point of the arc
233 SimplexId nextPointId;
234 if(!arcData.hasPoint(g_downVertexId)) {
235 nextPointId = points->InsertNextPoint(point);
236 arcData.addPoint(g_downVertexId, nextPointId, scalarMin, false);
237 } else {
238 nextPointId = arcData.point_ids[g_downVertexId];
239 }
240
241 pointIds[0] = nextPointId;
242
243 for(const SimplexId vertexId : *arc) {
244 triangulation_[cc]->getVertexPoint(vertexId, point[0], point[1], point[2]);
245 pointIds[1] = points->InsertNextPoint(point);
246 const double scalar = inputScalars_[cc]->GetTuple1(vertexId);
247 arcData.setPoint(pointIds[1], scalar, true);
248
249 const SimplexId nextCell
250 = skeletonArcs->InsertNextCell(VTK_LINE, 2, pointIds);
251 arcData.fillArrayCell(
252 nextCell, arcId, ftmTree_[cc], triangulation_[cc], params_);
253
254 pointIds[0] = pointIds[1];
255 }
256
257 const SimplexId upNodeId = tree->getUpperNodeId(arc);
258 const SimplexId l_upVertexId = tree->getNode(upNodeId)->getVertexId();
259 const SimplexId g_upVertexId = idMapper->GetTuple1(l_upVertexId);
260 triangulation_[cc]->getVertexPoint(
261 l_upVertexId, point[0], point[1], point[2]);
262 const double scalarMax = inputScalars_[cc]->GetTuple1(l_upVertexId);
263
264 // Get or create last point of the arc
265 if(!arcData.hasPoint(g_upVertexId)) {
266 nextPointId = points->InsertNextPoint(point);
267 arcData.addPoint(g_upVertexId, nextPointId, scalarMax, false);
268 } else {
269 nextPointId = arcData.point_ids[g_upVertexId];
270 }
271
272 pointIds[1] = nextPointId;
273
274 const SimplexId nextCell
275 = skeletonArcs->InsertNextCell(VTK_LINE, 2, pointIds);
276 arcData.fillArrayCell(
277 nextCell, arcId, ftmTree_[cc], triangulation_[cc], params_);
278
279 return 1;
280}

333int ttkFTMTree::addSampledSkeletonArc(const ttk::ftm::idSuperArc arcId,
334 const int cc,
335 vtkPoints *points,
336 vtkUnstructuredGrid *skeletonArcs,
337 ttk::ftm::ArcData &arcData) {
338 auto tree = ftmTree_[cc].tree.getTree(GetTreeType());
339 vtkDataArray *idMapper = connected_components_[cc]->GetPointData()->GetArray(
340 ttk::VertexScalarFieldName);
341 auto arc = tree->getSuperArc(arcId);
342 float point[3];
343 vtkIdType pointIds[2];
344
345 using ttk::SimplexId;
346 const SimplexId downNodeId = tree->getLowerNodeId(arc);
347 const SimplexId l_downVertexId = tree->getNode(downNodeId)->getVertexId();
348 const SimplexId g_downVertexId = idMapper->GetTuple1(l_downVertexId);
349 triangulation_[cc]->getVertexPoint(
350 l_downVertexId, point[0], point[1], point[2]);
351 const double scalarMin = inputScalars_[cc]->GetTuple1(l_downVertexId);
352
353 // Get or create first point of the arc
354 SimplexId nextPointId;
355 if(!arcData.hasPoint(g_downVertexId)) {
356 nextPointId = points->InsertNextPoint(point);
357 arcData.addPoint(g_downVertexId, nextPointId, scalarMin, false);
358 } else {
359 nextPointId = arcData.point_ids[g_downVertexId];
360 }
361
362 pointIds[0] = nextPointId;
363
364 const SimplexId upNodeId = tree->getUpperNodeId(arc);
365 const SimplexId l_upVertexId = tree->getNode(upNodeId)->getVertexId();
366 const SimplexId g_upVertexId = idMapper->GetTuple1(l_upVertexId);
367 triangulation_[cc]->getVertexPoint(
368 l_upVertexId, point[0], point[1], point[2]);
369 const double scalarMax = inputScalars_[cc]->GetTuple1(l_upVertexId);
370
371 const double delta = (scalarMax - scalarMin) / (params_.samplingLvl + 1);
372 double scalarLimit = scalarMin + delta;
373 double scalarAvg = 0;
374
375 // Get or create last point of the arc
376 if(!arcData.hasPoint(g_upVertexId)) {
377 nextPointId = points->InsertNextPoint(point);
378 arcData.addPoint(g_upVertexId, nextPointId, scalarMax, false);
379 } else {
380 nextPointId = arcData.point_ids[g_upVertexId];
381 }
382
383 SimplexId c = 0;
384 float sum[3]{0, 0, 0};
385 for(const SimplexId vertexId : *arc) {
386 triangulation_[cc]->getVertexPoint(vertexId, point[0], point[1], point[2]);
387 const double scalarVertex = inputScalars_[cc]->GetTuple1(vertexId);
388
389 if(scalarVertex < scalarLimit) {
390 sum[0] += point[0];
391 sum[1] += point[1];
392 sum[2] += point[2];
393 scalarAvg += scalarVertex;
394 ++c;
395 } else {
396 if(c) {
397 sum[0] /= c;
398 sum[1] /= c;
399 sum[2] /= c;
400 scalarAvg /= c;
401
402 pointIds[1] = points->InsertNextPoint(sum);
403 arcData.setPoint(pointIds[1], scalarAvg, true);
404 const SimplexId nextCell
405 = skeletonArcs->InsertNextCell(VTK_LINE, 2, pointIds);
406 arcData.fillArrayCell(
407 nextCell, arcId, ftmTree_[cc], triangulation_[cc], params_);
408
409 pointIds[0] = pointIds[1];
410 }
411
412 scalarLimit += delta;
413 sum[0] = 0;
414 sum[1] = 0;
415 sum[2] = 0;
416 scalarAvg = 0;
417 c = 0;
418 }
419 }
420
421 // The up id
422 pointIds[1] = nextPointId;
423
424 const SimplexId nextCell
425 = skeletonArcs->InsertNextCell(VTK_LINE, 2, pointIds);
426 arcData.fillArrayCell(
427 nextCell, arcId, ftmTree_[cc], triangulation_[cc], params_);
428
429 return 1;
430}

556 ttk::ftm::NodeData nodeData;
557 nodeData.init(ftmTree_, params_);
558 nodeData.setScalarType(inputScalars_[0]->GetDataType());
559
560 for(int cc = 0; cc < nbCC_; cc++) {
561 auto tree = ftmTree_[cc].tree.getTree(GetTreeType());
562 vtkDataArray *idMapper

565

567#ifndef TTK_ENABLE_KAMIKAZE
568 if(!numberOfNodes) {
569 this->printErr("Error : tree has no nodes.");
570 return 0;
571 }
572#endif
573
574 for(ttk::ftm::idNode nodeId = 0; nodeId < numberOfNodes; ++nodeId) {
575 const auto node = tree->getNode(nodeId);
576#ifndef TTK_ENABLE_KAMIKAZE
577 if(!node) {
578 this->printMsg({"Error : node ", std::to_string(nodeId), " is null."},
579 ttk::debug::Priority::ERROR);
580 return 0;
581 }
582#endif
583 const auto local_vertId = node->getVertexId();
584 float point[3];
585 triangulation_[cc]->getVertexPoint(
586 local_vertId, point[0], point[1], point[2]);
587 const auto nextPoint = points->InsertNextPoint(point);
588 nodeData.fillArrayPoint(
589 nextPoint, nodeId, ftmTree_[cc], idMapper, triangulation_[cc], params_);
590 }
591 }
592
593 ttkUtils::CellVertexFromPoints(skeletonNodes, points);
594
595 vtkPointData *pointData = skeletonNodes->GetPointData();
596 nodeData.addArray(pointData, params_);
597
598 outputSkeletonNodes->ShallowCopy(skeletonNodes);
