126 int totalInputs = inputTrees.size() + table->GetNumberOfRows();

130 ->GetTuple1(i);

136 ->GetTuple1(i);

138 ->GetTuple1(i);
139
140 coefs.emplace_back(alpha, index1R, index2R, index1, index2);
141 for(int j = index1 + 1; j < index2; ++j)
